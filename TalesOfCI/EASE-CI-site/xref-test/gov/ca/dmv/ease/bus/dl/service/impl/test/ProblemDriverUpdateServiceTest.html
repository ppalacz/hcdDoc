37  public class ProblemDriverUpdateServiceTest extends AbstractBusTestCase {
38  	private ProblemDriverUpdateService fixture;
39  	/*** The userContext */
40  	private UserContext userContext;
41  	EcsRequestFactory mockEcsRequestFactory;
42  	private ProblemDriverUpdateRequestEcs mockProblemDriverUpdateRequestEcs;
43  
44  	@Override
45  	protected void setUp() {
46  		mockProblemDriverUpdateRequestEcs = createMock(ProblemDriverUpdateRequestEcs.class);
47  		mockEcsRequestFactory = createMock(EcsRequestFactory.class);
48  		fixture = new ProblemDriverUpdateService();
49  		fixture.setEcsRequestFactory(mockEcsRequestFactory);
50  		userContext = UserContextFixtureFactory
51  				.createUserContextForValidateLegalPresence();
52  	}
53  
54  	@Override
55  	protected void tearDown() {
56  		mockProblemDriverUpdateRequestEcs = null;
57  		mockEcsRequestFactory = null;
58  		userContext = null;
59  		fixture = null;
60  	}
61  
62  	public void testExecute() {
63  		FireAndForgetReceipt ecsResponse = new FireAndForgetReceipt(
64  				new ErrorCollector());
65  		expect(
66  				mockEcsRequestFactory.createProblemDriverUpdateRequestFor(
67  						isA(DlApplication.class), isA(IUserContext.class)))
68  				.andReturn(mockProblemDriverUpdateRequestEcs);
69  		expect(mockProblemDriverUpdateRequestEcs.execute()).andReturn(
70  				ecsResponse);
71  		replay(mockEcsRequestFactory);
72  		replay(mockProblemDriverUpdateRequestEcs);
73  		ProblemDriverUpdateRequestBus busRequest = new ProblemDriverUpdateRequestBus(
74  				userContext, new DlApplication());
75  		ProblemDriverUpdateResponseBus busResponse = fixture
76  				.execute(busRequest);
77  		assertFalse(busResponse.hasErrors());
78  		verify(mockEcsRequestFactory);
79  		verify(mockProblemDriverUpdateRequestEcs);
80  	}
81  
82  	public void testExecuteELBNull() {
83  		ProblemDriverUpdateRequestBus busRequest = null;
84  		ProblemDriverUpdateResponseBus busResponse = fixture
85  				.execute(busRequest);
86  		assertTrue(busResponse.hasErrors());
87  	}
88  }

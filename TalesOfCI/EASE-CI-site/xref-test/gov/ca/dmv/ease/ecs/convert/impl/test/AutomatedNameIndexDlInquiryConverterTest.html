50  	/***
51  	 * Set the mailing address of a person to an existing list.
52  	 * 
53  	 * @return the mailing address
54  	 */
55  	public static void setMailingAddress(Person person, Address mailingAddress) {
56  		// If the mailing address is null then do nothing
57  		if (mailingAddress == null) {
58  			return;
59  		}
60  		List <Locator> locators = person.getLocators();
61  		if (locators != null && locators.size() > 0) {
62  			boolean exists = false;
63  			for (int index = 0; index < locators.size(); index++) {
64  				Address address = (Address) locators.get(index);
65  				if (address != null) {
66  					AddressType addressType = address.getAddressType();
67  					if (addressType != null
68  							&& AddressType.MAILING.equals(addressType)) {
69  						locators.remove(index);
70  						locators.add(index, mailingAddress);
71  						exists = true;
72  						break;
73  					}
74  				} // if loop
75  			} // for loop
76  			if (!exists) {
77  				locators.add(mailingAddress);
78  			}
79  		}
80  		else {
81  			locators.add(mailingAddress);
82  		}
83  	}

132 		List <SectionViolated> sectionViolatedList = new ArrayList <SectionViolated>();
133 		SectionViolated sectionViolated1 = new SectionViolated();
134 		sectionViolated1.setSectionViolatedNumber("12500A");
135 		sectionViolated1.setStatuteCode(new CodeSetElement("A", "A", "A"));
136 		SectionViolated sectionViolated2 = new SectionViolated();
137 		sectionViolated2.setStatuteCode(new CodeSetElement("A", "A", "A"));
138 		sectionViolated2.setSectionViolatedNumber("A405095");
139 		sectionViolatedList.add(sectionViolated1);
140 		sectionViolatedList.add(sectionViolated2);
141 		List <CodeSetElement> dispositionCodeList = new ArrayList <CodeSetElement>();
142 		dispositionCodeList.add(new CodeSetElement("F", "F", "F"));
143 		this.initData("I1472071", "TEM", birthCal.getTime(), "*CMZ019",
144 				violationCal.getTime(), convictionCal.getTime(), "30460",
145 				sectionViolatedList, "4APE032", dispositionCodeList);
146 		this.request = this.ecsRequestFactory
147 				.createFailureToAppearDCARequestFor(this.dlApplicant,
148 						this.citationDetail, this.userContext);
149 		FailureToAppearDCAResponseEcs response = this.request.execute();

61  	 * @param date the date
62  	 * @param format the format
63  	 * 
64  	 * @return the date
65  	 */
66  	public static Date convertStringToDate(String date, String format) {
67  		if (isNullOrBlank(date) || (date = date.trim()).matches("[0]*")) {
68  			return null;
69  		}
70  		DateFormat formatter = new SimpleDateFormat(format);
71  		try {
72  			Date returnDate = formatter.parse(date);
73  			return returnDate;
74  		}
75  		catch (ParseException e) {
76  		}
77  		return null;
78  	}

100 			format = EcsConverterHelper.FORMATTER_YY_MM_DD;

103 			format = EcsConverterHelper.FORMATTER_HH_MM;
104 		}

173 		}

176 		return returnString;

179 	/***
180 	 * Removes the duplicate header.
181 	 * 
182 	 * @param string the string
183 	 * 
184 	 * @return returnString the string
185 	 */
186 	public static String removeDuplicateHeader(String string) {
187 		StringTokenizer tokenizer = new StringTokenizer(string, LINE_SEPARATOR);
188 		int headerLengh = tokenizer.nextToken().length();
189 		String header = string.substring(0, headerLengh);
190 		String body = string.replace(header, "");
191 		StringBuffer stringBuffer = new StringBuffer();
192 		if (!body.contains(header)) {
193 			stringBuffer.append(header).append(LINE_SEPARATOR);
194 		}
195 		stringBuffer.append(body);
196 		String returnString = stringBuffer.toString();
197 		return returnString;
198 	}

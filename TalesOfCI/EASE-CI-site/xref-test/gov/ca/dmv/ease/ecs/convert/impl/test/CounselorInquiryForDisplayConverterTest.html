191 	/***
192 	 * Formats the expected result.
193 	 * @param expectedResult
194 	 * @return
195 	 */
196 	private String formatExpectedResult(String expectedResult) {
197 		StringBuffer strBuff = new StringBuffer();
198 		String ebcidic = null;
199 		String screen = null;
200 		String formattedScreen = null;
201 		try {
202 			ebcidic = new String(expectedResult.getBytes("Cp1047"));
203 			screen = Decoder3270.decode(ebcidic);
204 			formattedScreen = Decoder3270.formatScreen(screen);
205 			strBuff.append(formattedScreen).append(LINE_SEPARATOR);
206 		}
207 		catch (UnsupportedEncodingException e) {
208 			throw new MessageConversionException(e);
209 		}
210 		String formatResponse = StringUtils
211 				.trimTrailingWhitespace(removeDuplicateHeader(strBuff));
212 		return formatResponse;
213 	}

215 	/***
216 	 * This method removes duplicate header in given input string. first line of input string treated as a header.
217 	 * 
218 	 * @param buffer the buffer
219 	 * 
220 	 * @return returnString the string
221 	 */
222 	private String removeDuplicateHeader(StringBuffer buffer) {
223 		String string = buffer.toString();
224 		if (isNullOrBlank(string)) {
225 			return string;
226 		}
227 		StringTokenizer tokenizer = new StringTokenizer(string, LINE_SEPARATOR);
228 		int headerLengh = tokenizer.nextToken().length();
229 		String header = string.substring(0, headerLengh);
230 		String body = string.replace(header, "");
231 		StringBuffer stringBuffer = new StringBuffer();
232 		if (!body.contains(header)) {
233 			stringBuffer.append(header).append(LINE_SEPARATOR);
234 		}
235 		stringBuffer.append(body);
236 		String returnString = stringBuffer.toString();
237 		return returnString;
238 	}

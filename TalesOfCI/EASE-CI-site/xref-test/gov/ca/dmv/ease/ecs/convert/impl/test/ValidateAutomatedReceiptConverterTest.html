155 	@Test
156 	public void testCreateResponseWithCorrectSize() {
157 		initUserContextFixture();
158 		String ciaMessage = "CIA130091116BT1744130091116BT0015130091116BT0005E0275M92CIB   0000045991130               67  000000000000000                   P95                 ";
159 		converter.setUserContext(userContextFixture);
160 		ValidateAutomatedReceiptResponseEcs response = (ValidateAutomatedReceiptResponseEcs) converter
161 				.createResponse(ciaMessage);
162 		// Check sequence number
163 		assertEquals("1744", response.getInquiredSequenceNumber());
164 		List <VoucherReturnCode> actualVoucherReturnCodes = response
165 				.getVoucherReturnCodes();
166 		// Check total number for VoucherReturnCode
167 		assertEquals(2, actualVoucherReturnCodes.size());
168 		// Check the parameters inside VoucherReturnCode
169 		for (int i = 0; i < 2; i++) {
170 			VoucherReturnCode voucherReturnCode = actualVoucherReturnCodes
171 					.get(i);
172 			Voucher voucher = voucherReturnCode.getValidatedVoucher();
173 			Office office = voucher.getOffice();
174 			Employee employee = voucher.getEmployee();
175 			EmployeeWorkdateControl employeeWorkdateControl = employee
176 					.getEmployeeWorkdateControl();
177 			ReturnCode returnCode = voucherReturnCode.getReturnCode();
178 			ReturnCode errorCode = voucherReturnCode.getErrorCode();
179 			if (i == 0) {
180 				assertEquals("130", office.getOfficeId());
181 				assertEquals("BT", employee.getEmployeeTechId());
182 				assertEquals("091116", convertDateToString(
183 						employeeWorkdateControl.getAuthorizedWorkDate(),
184 						"yymmdd"));
185 				assertEquals(15, voucher.getSequenceNumber().intValue());
186 				assertEquals("E02", returnCode.getCode());
187 				assertEquals("75", errorCode.getCode());
188 				assertEquals(getExpectedErrorMessage(errorCode.getCode()),
189 						voucherReturnCode.getErrorCode().getMessage());
190 				// Below parameters should be null for error response
191 				assertEquals(null, voucher.getAmount());
192 				assertEquals(null, voucher.getDate());
193 				assertEquals(null, voucher.getLicenseNumber());
194 				assertEquals(null, voucher.getNameOrBundleId());
195 				assertEquals(null, voucher.getLast3Vin());
196 				assertEquals(null, voucher.getComment());
197 			}
198 			if (i == 1) {
199 				assertEquals("130", office.getOfficeId());
200 				assertEquals("BT", employee.getEmployeeTechId());
201 				assertEquals("091116", convertDateToString(
202 						employeeWorkdateControl.getAuthorizedWorkDate(),
203 						"yymmdd"));
204 				assertEquals(5, voucher.getSequenceNumber().intValue());
205 				assertEquals("M92", returnCode.getCode());
206 				assertEquals("45.67", voucher.getAmount().toString());
207 				assertEquals("991130", convertDateToString(voucher.getDate(),
208 						"yymmdd"));
209 				assertEquals("", voucher.getLicenseNumber());
210 				assertEquals("", voucher.getNameOrBundleId());
211 				assertEquals(null, voucher.getLast3Vin());
212 				assertEquals("", voucher.getComment());
213 			}
214 		}
215 	}

351 	/***
352 	 * Get the expected error message.
353 	 * 
354 	 * @param errorCode the error code
355 	 * 
356 	 * @return the expected error message
357 	 */
358 	private String getExpectedErrorMessage(String errorCode) {
359 		if (errorCode.equals("73") || errorCode.equals("74")) {
360 			return "F057 - CNA-CANNOT PROCESS Z97 - CANCEL TRANS";
361 		}
362 		else if (errorCode.equals("75")) {
363 			return "F068 - RECEIPT NOT ON FILE - CANNOT PROCESS Z97";
364 		}
365 		else if (errorCode.equals("TR")) {
366 			return "AR NOT FOUND";
367 		}
368 		else {
369 			return BLANK;
370 		}
371 	}

373 	/***
374 	 * Set the status for voucher.
375 	 * 
376 	 * @param errorCode the error code
377 	 * @param voucher the voucher
378 	 */
379 	private String getExpectedVoucherStatus(String errorCode) {
380 		String voucherStatus = BLANK;
381 		String functionalMode = userContextFixture.getFunctionalMode()
382 				.getCode();
383 		if (errorCode.equals("73") || errorCode.equals("74")) {
384 			if (functionalMode.equals("D") || functionalMode.equals("V")) {
385 				voucherStatus = "I";
386 			}
387 			else {
388 				voucherStatus = "S";
389 			}
390 		}
391 		return voucherStatus;

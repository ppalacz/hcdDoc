147 	/***
148 	 * Set the mailing address of a person to an existing list.
149 	 * 
150 	 * @return the mailing address
151 	 */
152 	public static void setMailingAddress(Person person, Address mailingAddress) {
153 		// If the mailing address is null then do nothing
154 		if (mailingAddress == null) {
155 			return;
156 		}
157 		List <Locator> locators = person.getLocators();
158 		if (locators != null && locators.size() > 0) {
159 			boolean exists = false;
160 			for (int index = 0; index < locators.size(); index++) {
161 				Address address = (Address) locators.get(index);
162 				if (address != null) {
163 					AddressType addressType = address.getAddressType();
164 					if (addressType != null
165 							&& AddressType.MAILING.equals(addressType)) {
166 						locators.remove(index);
167 						locators.add(index, mailingAddress);
168 						exists = true;
169 						break;
170 					}
171 				} // if loop
172 			} // for loop
173 			if (!exists) {
174 				locators.add(mailingAddress);
175 			}
176 		}
177 		else {
178 			locators.add(mailingAddress);
179 		}
180 	}

130 		List <SectionViolated> sectionViolatedList = new ArrayList <SectionViolated>();
131 		SectionViolated sectionViolated1 = new SectionViolated();
132 		sectionViolated1.setSectionViolatedNumber("12500A");
133 		sectionViolated1.setStatuteCode(new CodeSetElement("A", "A", "A"));
134 		SectionViolated sectionViolated2 = new SectionViolated();
135 		sectionViolated2.setStatuteCode(new CodeSetElement("A", "A", "A"));
136 		sectionViolated2.setSectionViolatedNumber("A405095");
137 		sectionViolatedList.add(sectionViolated1);
138 		sectionViolatedList.add(sectionViolated2);
139 		List <CodeSetElement> dispositionCodeList = new ArrayList <CodeSetElement>();
140 		dispositionCodeList.add(new CodeSetElement("F", "F", "F"));
141 		this.initData("I1472071", "TEM", birthCal.getTime(), "*CMZ019",
142 				violationCal.getTime(), convictionCal.getTime(), "30460",
143 				sectionViolatedList, dispositionCodeList);
144 		this.request = this.ecsRequestFactory
145 				.createUpdateFailureToAppearDCARequestFor(this.dlApplicant,
146 						this.citationDetail, this.userContext);
147 		UpdateFailureToAppearDCAResponseEcs response = this.request.execute();
148 		Assert.assertFalse("Response Contains Errors", response
149 				.getErrorCollector().hasErrors());

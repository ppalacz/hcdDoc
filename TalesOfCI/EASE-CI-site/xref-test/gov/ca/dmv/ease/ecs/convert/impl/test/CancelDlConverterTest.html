89  	/***
90  	 * This method creates response suffix.
91  	 * 
92  	 * @param personName the person name
93  	 * @return the response suffix
94  	 */
95  	private String responseSuffix(PersonName personName) {
96  		StringBuffer responseSuffix = new StringBuffer();
97  		responseSuffix.append(personName.getLastName()).append(DELIMITER)
98  				.append(personName.getFirstName()).append(DELIMITER).append(
99  						personName.getMiddleName()).append(DELIMITER).append(
100 						personName.getSuffix());
101 		return responseSuffix.toString();
102 	}

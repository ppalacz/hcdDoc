246 	 * The purpose of this method is to create a clone object or the replica of
247 	 * the passed object or another instance of passed object.
248 	 * 
249 	 * @param obj
250 	 *            the object that needs to be cloned
251 	 * @return the cloned object or the replica of the passed object or another
252 	 *         instance of passed object
253 	 */
254 	private Object deepClone(Object obj) {
255 		Object deepCopy = null;
256 		ObjectInputStream ois = null;
257 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
258 		try {
259 			ObjectOutputStream oos = new ObjectOutputStream(baos);
260 			oos.writeObject(obj);
261 			ByteArrayInputStream bais = new ByteArrayInputStream(baos
262 					.toByteArray());
263 			ois = new ObjectInputStream(bais);
264 			deepCopy = ois.readObject();
265 		}
266 		catch (IOException e) {
267 			e.printStackTrace();
268 		}
269 		catch (ClassNotFoundException e) {
270 			e.printStackTrace();
271 		}
272 		return deepCopy;
273 	}

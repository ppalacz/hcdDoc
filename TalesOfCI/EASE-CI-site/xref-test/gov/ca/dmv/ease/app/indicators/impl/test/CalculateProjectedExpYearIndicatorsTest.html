53  		Date applicationDate = fixture.calculateProjectedExpYearForDl("IRT",
54  				null, null, null, 0, null, null, 0);
55  		assertNull(applicationDate);
56  		//Scenario 2 - Original Application Date less than 6 months from DL Expiration Date 
57  		String ttc = "DLA";
58  		int limitedTermYears = 4;
59  		List <CodeSetElement> attachments = new ArrayList <CodeSetElement>();
60  		attachments.add(new CodeSetElement("10", "10", "10"));
61  		CodeSetElement feeReasonCode = new CodeSetElement("L", "L", "L");
62  		int dlFeeDueYear = 10;
63  		Date originalAppDate = new Date();
64  		Calendar dlExpCal = Calendar.getInstance();
65  		dlExpCal.add(Calendar.MONTH, 3);
66  		Calendar birthCal = Calendar.getInstance();
67  		birthCal.set(1980, Calendar.JANUARY, 01);
68  		applicationDate = fixture.calculateProjectedExpYearForDl(ttc,
69  				originalAppDate, dlExpCal.getTime(), birthCal.getTime(),
70  				limitedTermYears, attachments, feeReasonCode, dlFeeDueYear);
71  		Calendar appCal = Calendar.getInstance();
72  		appCal.setTime(applicationDate);
73  		assertEquals(appCal.get(Calendar.YEAR), 2014);
74  		assertEquals(appCal.get(Calendar.MONTH), Calendar.JANUARY);
75  		//Scenario 3 - Original Application Date more than 6 months from DL Expiration Date
76  		dlExpCal.add(Calendar.MONTH, 9);
77  		applicationDate = fixture.calculateProjectedExpYearForDl(ttc,
78  				originalAppDate, dlExpCal.getTime(), birthCal.getTime(),
79  				limitedTermYears, attachments, feeReasonCode, dlFeeDueYear);
80  		appCal = Calendar.getInstance();
81  		appCal.setTime(applicationDate);
82  		assertEquals(appCal.get(Calendar.YEAR), 2014);
83  		assertEquals(appCal.get(Calendar.MONTH), Calendar.JANUARY);

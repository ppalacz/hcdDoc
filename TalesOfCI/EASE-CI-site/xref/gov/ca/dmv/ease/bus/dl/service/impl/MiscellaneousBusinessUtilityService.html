56  	/***
57  	 * This method sends DLM TCODE to the back-end system and verifies if the given
58  	 * driver license number/id number is valid
59  	 * 
60  	 * @param addressChangeDlIdInfoRequestBus
61  	 * @return
62  	 */
63  	public AddressChangeDlIdInfoResponseBus execute(
64  			AddressChangeDlIdInfoRequestBus addressChangeDlIdInfoRequestBus) {
65  		AddressChangeDlIdInfoResponseBus busResponse = null;
66  		SearchPersonByNumberAndLastNameRequestEcs ecsRequest = null;
67  		SearchPersonByNumberAndLastNameResponseEcs ecsResponse = null;
68  		IErrorCollector errorCollector = null;
69  		try {
70  			errorCollector = addressChangeDlIdInfoRequestBus.validate();
71  			if (errorCollector.hasErrors()) {
72  				// Set the Validation Messages to BusinessServiceResponse
73  				busResponse = new AddressChangeDlIdInfoResponseBus(
74  						errorCollector);
75  			}
76  			else {
77  				ecsRequest = ecsRequestFactory
78  						.createSearchPersonByNumberAndLastNameRequestForEcs(addressChangeDlIdInfoRequestBus
79  								.getPerson());
80  				ecsResponse = ecsRequest.execute();
81  				if (ecsResponse.hasErrors()) {
82  					busResponse = new AddressChangeDlIdInfoResponseBus(
83  							errorCollector);
84  				}
85  				else {
86  					busResponse = new AddressChangeDlIdInfoResponseBus(
87  							ecsResponse.getDlmRecord());
88  				}
89  			}
90  		}
91  		catch (Exception e) {
92  			errorCollector.register(e);
93  			LOGGER.error(e);
94  			return busResponse = new AddressChangeDlIdInfoResponseBus(
95  					errorCollector);
96  		}
97  		return busResponse;
98  	}

167 	/***
168 	 * This method search the driver license number of the person by calling ECS
169 	 * Service to search the number on DL Master.
170 	 * 
171 	 * @param busRequest
172 	 *            the person number request
173 	 * 
174 	 * @return Response returns information associated with the call to DL
175 	 *         Master by ECS Service. The Response object contains the Person
176 	 *         and license objects.
177 	 */
178 	public CancelDlResponseBus execute(CancelDlRequestBus cancelDlBusRequest) {
179 		CancelDlResponseBus cancelDlbusResponse = null;
180 		CancelDlRequestEcs ecsRequest = null;
181 		CancelDlResponseEcs ecsResponse = null;
182 		IErrorCollector errorCollector = new ErrorCollector();
183 		try {
184 			//Validate the request
185 			//TODO: is it necessary to validate the request at this step? 
186 			errorCollector = cancelDlBusRequest.validate();
187 			if (errorCollector.hasErrors()) {
188 				// Set the Validation Messages to Business Service Response
189 				cancelDlbusResponse = new CancelDlResponseBus(errorCollector);
190 			}
191 			else {
192 				Person person = cancelDlBusRequest.getPerson();
193 				DriverLicense driverLicense = person.getDriverLicense();
194 				ecsRequest = ecsRequestFactory.createCancelDlRequestForEcs(
195 						driverLicense.getLicenseNumber(), person
196 								.getCurrentName().getLastName(), person
197 								.getBirthDate(), cancelDlBusRequest
198 								.getSubCode(), driverLicense
199 								.getLicenseLocation().getCode(), driverLicense
200 								.getCancellationEffectiveDate(),
201 						cancelDlBusRequest.getUserContext());
202 				ecsResponse = ecsRequest.execute();
203 				if (ecsResponse.getErrorCollector().hasErrors()) {
204 					cancelDlbusResponse = new CancelDlResponseBus(ecsResponse
205 							.getErrorCollector());
206 				}
207 				else {
208 					cancelDlbusResponse = new CancelDlResponseBus(ecsResponse
209 							.getNameOfRecord(), ecsResponse
210 							.getResponseInfoCode());
211 				}
212 			}
213 		}
214 		catch (Exception e) {
215 			errorCollector.register(e);
216 			return cancelDlbusResponse = new CancelDlResponseBus(errorCollector);
217 		}
218 		return cancelDlbusResponse;
219 	}

240 								.getLastName(), busRequest
241 								.getTransactionSubCode(), busRequest
242 								.getLicenseLocation(), busRequest
243 								.getProcessingDate(), busRequest
244 								.getUserContext());
245 				ecsResponse = (CancelIdCardResponseEcs) ecsRequest.execute();
246 				if (ecsResponse.hasErrors()) {
247 					busResponse = new CancelIdCardResponseBus(errorCollector);
248 				}
249 				else {
250 					busResponse = new CancelIdCardResponseBus(ecsResponse
251 							.getNameOfRecord(), ecsResponse
252 							.getResponseInfoCode(), ecsResponse
253 							.isIdOrDlOperationSuccess());
254 				}
255 			}
256 		}
257 		catch (Exception e) {
258 			LOGGER.error(e);
259 			errorCollector.register(e);
260 			return busResponse = new CancelIdCardResponseBus(errorCollector);

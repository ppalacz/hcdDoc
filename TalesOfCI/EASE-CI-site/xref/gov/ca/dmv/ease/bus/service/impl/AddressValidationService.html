26  public class AddressValidationService {
27  	/*** The ECS request factory */
28  	IEcsRequestFactory ecsRequestFactory = null;
29  
30  	/***
31  	 * This method formats the address and then calls DCS Service to validate
32  	 * the address with ZIPMOD
33  	 * 
34  	 * @param ValidateAddressRequestBus
35  	 *            -This request object contains the domain object Address.
36  	 * @return Response returns information associated with the call to ZIPMOD
37  	 *         by DCS Service. The Response object contains status of
38  	 *         communication success or failure with CAMVDCS, and if the
39  	 *         communication is successful, contains the Address object with
40  	 *         missing data such as the Zip Code and the County.
41  	 */
42  	public ValidateAddressResponseBus execute(
43  			ValidateAddressRequestBus busRequest) {
44  		ValidateAddressResponseBus busResponse = null;
45  		ValidateAddressRequestEcs ecsRequest = null;
46  		ValidateAddressResponseEcs ecsResponse = null;
47  		IErrorCollector errorCollector = new ErrorCollector();
48  		try {
49  			// errorCollector = busRequest.validate();
50  			if (errorCollector.hasErrors()) {
51  				// Set the Validation Messages to Response
52  				// by getting the response from factory with overloaded method
53  				// for errors
54  				busResponse = new ValidateAddressResponseBus(errorCollector);
55  			}
56  			else {
57  				ecsRequest = ecsRequestFactory.createValidateAddressRequestFor(
58  						busRequest.getAddressesForValidation(), busRequest
59  								.getUserContext());
60  				ecsResponse = ecsRequest.execute();
61  				if (ecsResponse.hasErrors()) {
62  					busResponse = new ValidateAddressResponseBus(ecsResponse
63  							.getErrorCollector());
64  				}
65  				else {
66  					busResponse = new ValidateAddressResponseBus(ecsRequest
67  							.getAddresses(), ecsResponse
68  							.getValidatedAddresses());
69  				}
70  			}
71  		}
72  		catch (Exception e) {
73  			// Register exception in error collector and assign it to response
74  			// object.
75  			errorCollector.register(e);
76  			return busResponse = new ValidateAddressResponseBus(errorCollector);
77  		}
78  		return busResponse;
79  	}
80  
81  	/***
82  	 * Sets the ECS request factory.
83  	 * 
84  	 * @param ecsRequestFactory
85  	 *            the ecsRequestFactory to set
86  	 */
87  	public void setRequestFactory(IEcsRequestFactory requestFactory) {
88  		this.ecsRequestFactory = requestFactory;
89  	}

51  	 * @param userContext
52  	 *            the user context
53  	 */
54  	public ValidateSsnRequestBus(
55  			Person aPerson, CodeSetElement appliedLicenseClass,
56  			CodeSetElement licenseCategory, IUserContext userContext) {

60  		this.licenseCategory = licenseCategory;
61  	}

95  				.validate(this.getPerson());
96  		//User Context Validation

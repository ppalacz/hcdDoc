41  	/***
42  	 * This method provides an inquiry of the driving history(problems of
43  	 * drivers) by calling DCS service to provide an inquiry against CDLIS/PDPS
44  	 * when the request contains the Token Identification value and Time out
45  	 * Period. This method is called by the client to poll for the actual
46  	 * response.
47  	 * 
48  	 * @param busRequest
49  	 *            the request
50  	 * 
51  	 * @return - Response returns information associated with the call to
52  	 *         CDLIS/PDPS by DCS Service. The Response object contains the
53  	 *         driving history of the person, total amount of CDLIS and PDPS
54  	 *         records.
55  	 */
56  	public CdlisAndPdpsHistoryResponseBus execute(
57  			GetCdlisAndPdpsResponseRequestBus busRequest) {
58  		CdlisAndPdpsHistoryResponseBus busResponse = null;
59  		CashPromiseRequestEcs ecsRequest = null;
60  		CdlisAndPdpsResponseEcs ecsResponse = null;
61  		IBusDlResponseFactory responseFactory = new BusDlResponseFactory();
62  		IErrorCollector errorCollector = null;
63  		try {
64  			errorCollector = busRequest.validate();
65  			if (errorCollector.hasErrors()) {
66  				busResponse = responseFactory
67  						.createCdlisAndPdpsHistoryResponseForBus(errorCollector);
68  			} else {
69  				ecsRequest = ecsRequestFactory
70  						.createSendCdlisAndPdpsRequestForEcs(busRequest
71  								.getPromise());
72  				ecsResponse = (CdlisAndPdpsResponseEcs) ecsRequest.execute();
73  				if (ecsResponse != null && ecsResponse.hasError()) {
74  					busResponse = responseFactory
75  							.createCdlisAndPdpsHistoryResponseForBus(ecsResponse
76  									.getErrorCollector());
77  				} else {
78  					busResponse = responseFactory
79  							.createCdlisAndPdpsHistoryResponseForBus(ecsResponse);
80  				}
81  			}
82  
83  		} catch (Exception e) {
84  			// Register exception in error collector and assign it to response
85  			// object.
86  			errorCollector.register(e);
87  			return busResponse = responseFactory
88  					.createCdlisAndPdpsHistoryResponseForBus(errorCollector);
89  		}
90  		return busResponse;
91  	}

93  	/***
94  	 * This method provides the initial inquiry of the driving history(problems
95  	 * of drivers) by calling the DCS service to provide an inquiry against
96  	 * CDLIS/PDPS.
97  	 * 
98  	 * @param busRequest
99  	 *            the request
100 	 * 
101 	 * @return - Response returns a Token representing acknowledgment of the
102 	 *         initial request from the DCS Service. This token is prepared by
103 	 *         the DCS service and contains the timeout end-time along with a
104 	 *         request identifier.
105 	 */
106 	public CdlisAndPdpsInquiryResponseBus execute(
107 			SendCdlisAndPdpsRequestBus busRequest) {
108 		SendCdlisAndPdpsRequestEcs ecsRequest = null;
109 		CdlisAndPdpsInquiryResponseBus busResponse = null;
110 		IBusDlResponseFactory responseFactory = new BusDlResponseFactory();
111 		IErrorCollector errorCollector =null;;
112 		try {
113 			errorCollector = busRequest.validate();
114 			if (errorCollector.hasErrors()) {
115 				busResponse = responseFactory
116 						.createSendCdlisAndPdpsResponseForTs(errorCollector);
117 			} else {
118 
119 				ecsRequest = ecsRequestFactory
120 						.createSendCdlisAndPdpsRequestForBus(busRequest
121 								.getPerson(), busRequest.getUserContext());
122 
123 				AsynchronousResponsePromise ecsResponse = ecsRequest.execute();
124 				busResponse = responseFactory
125 						.createSendCdlisAndPdpsResponseForTs(ecsResponse);
126 				return CdlisAndPdpsMockResponse.createPromise(busRequest,
127 						ecsResponse.getCorrelationId());
128 			}
129 		} catch (Exception e) {
130 			errorCollector.register(e);
131 			return busResponse = responseFactory
132 					.createSendCdlisAndPdpsResponseForTs(errorCollector);
133 		}
134 		return busResponse;
135 	}

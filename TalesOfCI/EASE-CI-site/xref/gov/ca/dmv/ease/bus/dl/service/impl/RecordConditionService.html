47  	/***
48  	 * Finds list of record conditions satisfy rules
49  	 *@param busRequest
50  	 * @return
51  	 */
52  	public RecordConditionResponseBus execute(
53  			RecordConditionRequestBus busRequest) {
54  		RecordConditionResponseBus busResponse = null;
55  		try {
56  			IPersistenceServiceRequest request = persistenceServiceRequestFactory
57  					.createRetrieveCriteriaObjectRequest(busRequest
58  							.getUserContext(), RecordConditionRule.class);
59  			RetrieveCriteriaObjectResponse perResponse = 
60  				(RetrieveCriteriaObjectResponse) request.execute();
61  			if (perResponse.hasErrors()) {
62  				busResponse = new RecordConditionResponseBus(perResponse
63  						.getErrorCollector());
64  			}
65  			else {
66  				Criteria criteria = perResponse.getCriteriaInstance();
67  				criteria.add(Restrictions.eq("tcode", getTcode(
68  						busRequest.getTcode()))); //Select rules for TCode DLA or DLM.
69  				criteria.addOrder(Order.asc("executionSequence"));
70  				List listOfBusinessObjects = criteria.list();
71  				List <RecordConditionRule> recordConditions = 
72  					new ArrayList <RecordConditionRule>();
73  				ProcessContext processContext = 
74  					busRequest.getProcessContext();
75  				if (processContext instanceof HqInquiryProcessContext) {
76  					HqInquiryProcessContext hqInquiryProcessContext = 
77  						(HqInquiryProcessContext)busRequest.getProcessContext();
78  					EliRecord eliRecord = hqInquiryProcessContext.getEliRecord();
79  					busResponse = getEliResponseBus(eliRecord, 
80  							hqInquiryProcessContext, listOfBusinessObjects, 
81  							recordConditions);
82  				}
83  				else if (processContext instanceof MiscDlProcessContext) {
84  					MiscDlProcessContext miscProcessContext = 
85  						(MiscDlProcessContext)busRequest.getProcessContext();
86  					DlmRecord dlmRecord = miscProcessContext.getDlmRecord();
87  					busResponse = getDlmResponseBus(dlmRecord, 
88  							miscProcessContext, listOfBusinessObjects, 
89  							recordConditions);
90  				}
91  			}
92  		}
93  		catch (Exception e) {
94  			ErrorCollector errorCollector = new ErrorCollector();
95  			errorCollector.register(e);
96  			busResponse = new RecordConditionResponseBus(errorCollector);
97  		}
98  		return busResponse;
99  	}

110 	 * Gets the eli response bus.
111 	 * 
112 	 * @param eliRecord the eli record
113 	 * @param hqInquiryProcessContext the hq inquiry process context
114 	 * @param listOfBusinessObjects the list of business objects
115 	 * @param recordConditions the record conditions
116 	 * 
117 	 * @return the eli response bus
118 	 */
119 	private RecordConditionResponseBus getEliResponseBus(
120 			EliRecord eliRecord, HqInquiryProcessContext 
121 			hqInquiryProcessContext, List listOfBusinessObjects, 
122 			List <RecordConditionRule> recordConditions) {
123 		String transactionCode = getTransactionCode(eliRecord, 
124 				hqInquiryProcessContext); //Transaction codes: ATL, DLI, DLA, DLM.
125 		//Check all Action Messages rules for a selected tcode i.e. DLA or DLM.
126 		checkAndAddDlaMessages(listOfBusinessObjects, 
127 				recordConditions, eliRecord, 
128 				transactionCode, hqInquiryProcessContext);
129 		processActionsForMessages(hqInquiryProcessContext, 
130 				recordConditions, eliRecord);
131 		RecordConditionRulesUtil.setRecordConditionIndicators(
132 				hqInquiryProcessContext, recordConditions);
133 		return new RecordConditionResponseBus(
134 				recordConditions);
135 	}

162 	/***
163 	 * Check and add DLA TCode messages.
164 	 * 
165 	 * @param listOfBusinessObjects the list of business objects
166 	 * @param recordConditions the record conditions
167 	 * @param eliRecord the eli record
168 	 * @param transactionCode the transaction code
169 	 * @param hqInquiryProcessContext the hq inquiry process context
170 	 */
171 	private void checkAndAddDlaMessages(List listOfBusinessObjects, 
172 			List <RecordConditionRule> recordConditions, 
173 			EliRecord eliRecord, String transactionCode, 
174 			HqInquiryProcessContext hqInquiryProcessContext) {
175 		for (Object condition : listOfBusinessObjects) {	
176 			RecordConditionRule recordCondition = 
177 				(RecordConditionRule) condition;
178 			if (recordCondition.execute(eliRecord,
179 					hqInquiryProcessContext, transactionCode)) {
180 				//If this rule satisfy condition then add this rule to the list.
181 				if (ArrayUtils.contains(RecordConditionConstants
182 						.DYNAMIC_MSG_LIST, 
183 						recordCondition.getActionMessage())) {
184 					//If this is message has dynamic values to populate.
185 					String dynamicMsg = RecordConditionRulesUtil
186 					.getDynamicMessage(recordCondition
187 							.getActionMessage(), eliRecord);
188 					if (EaseUtil.isNullOrBlank(dynamicMsg)) {
189 						//If dynamic message does not have required values then do not add that message.
190 						continue;
191 					}
192 					recordCondition.setActionMessage(dynamicMsg);
193 				}
194 				recordConditions.add(recordCondition);
195 			}
196 		}
197 	}

235 	/***
236 	 * Gets the transaction code.
237 	 * 
238 	 * @param eliRecord the eli record
239 	 * @param hqInquiryProcessContext the hq inquiry process context
240 	 * 
241 	 * @return the transaction code
242 	 */
243 	private String getTransactionCode(EliRecord eliRecord, 
244 			HqInquiryProcessContext hqInquiryProcessContext) {
245 		String transactionCode = null;
246 		if (eliRecord != null) {
247 			if (eliRecord.getTransactionCode() != null) {
248 				//Transaction code could be DLA, DLI, ATL, DLM etc.
249 				transactionCode = eliRecord.getTransactionCode().getCode();
250 			}
251 		}
252 		else if (RecordConditionRulesUtil.hasAnyMultipleAniHitsPresent(
253 				hqInquiryProcessContext)) {
254 			//If no ELI record exists and Multiple ANI hits are present. 
255 			//i.e. 'No More Tries' selected and no record was selected as 'Primary'.
256 			transactionCode = null;
257 		}
258 		else {
259 			//If no ELI record exists and no multiple hits.
260 			//i.e. Initial ANI inquiry with 'No Hits'.
261 			transactionCode = RecordConditionConstants.ATL;
262 		}
263 		return transactionCode;
264 	}

270 	 * @param recordConditions the record conditions

272 	private void processActionsForMessages(
273 			HqInquiryProcessContext hqInquiryProcessContext, 
274 			List <RecordConditionRule> recordConditions,
275 			EliRecord eliRecord) {
276 		if (RecordConditionRulesUtil.isSingleHit(eliRecord)) {
277 			//Process Can not continue message only for Single hit response. 
278 			RecordConditionRulesUtil.processCannotContinueMsg(
279 					recordConditions, hqInquiryProcessContext);
280 			//Remove contradictory messages.
281 			if (RecordConditionRulesUtil
282 					.hasAnyCannotContMessage(
283 					recordConditions)) {
284 				//If list of action messages contains any Cannot Continue type message.
285 				//Then list should not contain Can Continue type messages.
286 				if (RecordConditionRulesUtil.hasAnyCanContMessage(
287 						recordConditions)) {
288 					//If list contains any Can Continue type message remove those messages.
289 					RecordConditionRulesUtil.removeAllCanContMessages(
290 							recordConditions);
291 				}
292 			}
293 			else {
294 				//If list of action messages doesn't contain any Cannot Continue type message.
295 				if (RecordConditionRulesUtil.hasAnyCanContMessage(
296 						recordConditions)) {
297 				}
298 			}
299 		}
300 	}

325 		if (ArrayUtils.contains(RecordConditionConstants.NON_MISC_APP_TTC,

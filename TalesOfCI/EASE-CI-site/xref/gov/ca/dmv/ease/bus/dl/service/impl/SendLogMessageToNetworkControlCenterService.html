34  	public SendLogMessageToNetworkControlCenterResponseBus execute(
35  			SendLogMessageToNetworkControlCenterRequestBus busRequest) {
36  		SendLogMessageToNetworkControlCenterResponseBus busResponse = null;
37  		SendLogMessageToNetworkControlCenterRequestEcs ecsRequest = null;
38  		SendLogMessageToNetworkControlCenterResponseEcs ecsResponse = null;
39  		IErrorCollector errorCollector = new ErrorCollector();
40  		try {
41  			errorCollector = busRequest.validate();
42  			if (errorCollector.hasErrors()) {
43  				// Set the Validation Messages to Business Service Response
44  				busResponse = new SendLogMessageToNetworkControlCenterResponseBus(
45  						errorCollector);
46  			}
47  			else {
48  				ecsRequest = ecsRequestFactory
49  						.createSendLogMessageToNetworkControlCenterRequestFor(busRequest
50  								.getUserContext());
51  				ecsResponse = ecsRequest.execute();
52  				if (ecsResponse.hasErrors()) {
53  					busResponse = new SendLogMessageToNetworkControlCenterResponseBus(
54  							errorCollector);
55  				}
56  				else {
57  					busResponse = new SendLogMessageToNetworkControlCenterResponseBus();
58  				}
59  			}
60  		}
61  		catch (Exception e) {
62  			//Register exception in error collector and assign it to response object.
63  			errorCollector.register(e);
64  			return busResponse = new SendLogMessageToNetworkControlCenterResponseBus(
65  					errorCollector);
66  		}
67  		return busResponse;
68  	}

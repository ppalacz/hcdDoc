48  	/***
49  	 * This method calculate the fee of DriverLicense.
50  	 * 
51  	 * @param busRequest
52  	 *            the calculate dl fee request bus
53  	 * 
54  	 * @return Response returns the calculated Fee.
55  	 */
56  	public CalculateDlFeeResponseBus execute(CalculateDlFeeRequestBus busRequest) {
57  		CalculateDlFeeResponseBus busResponse = null;
58  		IBusDlResponseFactory responseFactory = new BusDlResponseFactory();
59  		DlApplication application = busRequest.getDlApplication();
60  		try {
61  			dlaFeeRequest.setTtc(busRequest.getUserContext().getTtc());
62  			
63  			dlaFeeRequest.setDateFeesReceived(getPaymentDate(application));
64  			dlaFeeRequest
65  					.setOriginalApplicationDate(getApplciationDate(application));
66  			dlaFeeRequest.setLicenseClass1(getPrimaryLicenseClass(application));
67  			dlaFeeRequest.setLicenseClass2(getSecondaryLicenseClass(application));
68  			dlaFeeRequest.setOrginalRenewal(getOriginalRenewal(application));
69  			dlaFeeRequest.setCategory(getLicenseCategory(application));
70  			List<ApplicableFee> applicableFees = dlaFeeRequest.computeFees();
71  			double totalFeeFromEngine = 0.0;
72  			for (ApplicableFee applicableFee : applicableFees) {
73  				totalFeeFromEngine += applicableFee.getFee();
74  			}
75  			Fee fee = new Fee();
76  			fee.setAmount(new BigDecimal(totalFeeFromEngine));
77  			busResponse = responseFactory
78  					.createCalculateDlFeeResponseForBus(fee);
79  		} catch (Exception e) {
80  			LOGGER.error(e);
81  			IErrorCollector errorCollector = new ErrorCollector();
82  //			errorCollector.register(e);
83  			return busResponse = responseFactory
84  					.createCalculateDlFeeResponseBus(errorCollector);
85  		}
86  		return busResponse;
87  	}

109 		if (application.getInvoice() != null
110 				&& application.getInvoice().getPayments() != null
111 				&& application.getInvoice().getPayments().size() > 0
112 				&& application.getInvoice().getPayments().get(0) != null) {
113 			Date feePaidDate = application.getInvoice().getPayments().get(0)
114 					.getDate();
115 			return (feePaidDate != null) ? feePaidDate : new Date();
116 		} else {
117 			return new Date();
118 		}

201 	/***
202 	 * This method returns the Fee Engine's representation of the license class
203 	 * 
204 	 * @param appliedLicense
205 	 *            CodeSetElement with license class
206 	 * @return Fee Engine representation of the license class
207 	 */
208 	private LICENSECLASS getLicenseClass(CodeSetElement appliedLicense) {
209 		if (appliedLicense == null) {
210 			return LICENSECLASS.CLASSBLANK;
211 		} else {
212 			String licenseCode = appliedLicense.getCode();
213 			if (licenseCode == null) {
214 				return LICENSECLASS.CLASSBLANK;
215 			} else {
216 				licenseCode = licenseCode.trim();
217 				if (licenseCode.length() == 0) {
218 					return LICENSECLASS.CLASSBLANK;
219 				} else if (licenseCode.equalsIgnoreCase("A")) {
220 					return LICENSECLASS.CLASSA;
221 				} else if (licenseCode.equalsIgnoreCase("B")) {
222 					return LICENSECLASS.CLASSB;
223 				} else if (licenseCode.equalsIgnoreCase("C")) {
224 					return LICENSECLASS.CLASSC;
225 				} else if (licenseCode.equalsIgnoreCase("D")) {
226 					return LICENSECLASS.CLASSD;
227 				} else if (licenseCode.equalsIgnoreCase("E")) {
228 					return LICENSECLASS.CLASSE;
229 				} else if (licenseCode.equalsIgnoreCase("F")) {
230 					return LICENSECLASS.CLASSF;
231 				} else if (licenseCode.equalsIgnoreCase("M")) {
232 					return LICENSECLASS.CLASSM;
233 				}
234 			}
235 		}
236 		return LICENSECLASS.CLASSBLANK;
237 	}

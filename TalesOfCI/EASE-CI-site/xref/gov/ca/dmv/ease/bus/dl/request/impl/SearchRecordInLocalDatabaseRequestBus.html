34  public class SearchRecordInLocalDatabaseRequestBus extends AbstractRequest {
35  
36  	/*** The Constant serialVersionUID. */
37  	private static final long serialVersionUID = -4110380904568339906L;
38  
39  	/*** The license. */
40  	private License license;
41  
42  	/***
43  	 * Instantiates a new search record in local database request bus.
44  	 * 
45  	 * @param license the license
46  	 * @param userContext the user context
47  	 */
48  	public SearchRecordInLocalDatabaseRequestBus(License license,
49  			IUserContext userContext) {
50  		super(userContext);
51  		this.license = license;
52  	}
53  	
54  	/***
55  	 * Gets the person.
56  	 * 
57  	 * @return the person
58  	 */
59  	public License getLicense() {
60  		return license;
61  	}
62  	
63  	/* (non-Javadoc)
64  	 * @see gov.ca.dmv.ease.fw.AbstractRequest#getValidator()
65  	 */
66  	@Override
67  	protected IValidator getValidator() {
68  		return new PersonInquiryRequestValidator();
69  	}
70  	
71  	/*
72  	 * (non-Javadoc)
73  	 * 
74  	 * @see gov.ca.dmv.ease.fw.AbstractRequest#validate()
75  	 */
76  	@Override
77  	public IErrorCollector validate() {
78  		IErrorCollector aCollector = new ErrorCollector();
79  		DlAgeDataSingleton dlAgeDataSingleton = DlAgeDataSingleton
80  				.getInstance();
81  		List<DlAgeRequirement> dlAgeRequirements = dlAgeDataSingleton
82  				.getDlAgeData(getUserContext());
83  		if (dlAgeRequirements != null) {
84  			List<IValidatable> iValidatables = new ArrayList<IValidatable>();
85  			for (DlAgeRequirement dlAgeRequirement : dlAgeRequirements) {
86  				iValidatables.add(dlAgeRequirement);
87  			}
88  			aCollector = ((PersonInquiryRequestValidator) getValidator())
89  					.validate(this.getLicense(), iValidatables);
90  		}
91  		// User Context Validation
92  		UserContextValidator contextValidator = new UserContextValidator();
93  		contextValidator.validate(this.getUserContext(), aCollector);
94  		return aCollector;
95  	}
96  }

132 	 * Checks if is comm drive test fee not used.
133 	 * 
134 	 * @param eliRecord the eli record
135 	 * 
136 	 * @return true, if is comm drive test fee not used
137 	 */
138 	protected boolean isCommDriveTestFeeNotUsed(
139 			EliRecord eliRecord) {
140 		if (eliRecord.getInvoiceItem() != null 
141 				&& eliRecord.getInvoiceItem().getDriverLicenceFee() != null 
142 				&& eliRecord.getInvoiceItem().getDriverLicenceFee()
143 				.getDriverLicenseRateStatusForExpectedFeeRateCode(
144 						COML_DL_DRIVE_TEST_FEE_IND) != null) {
145 			DriverLicenseRateStatus rateStatus = 
146 				eliRecord.getInvoiceItem()
147 				.getDriverLicenceFee()
148 			.getDriverLicenseRateStatusForExpectedFeeRateCode(
149 					COML_DL_DRIVE_TEST_FEE_IND);
150 			for (FeeStatusType fs 
151 					: rateStatus.getFeeStatusTypes()) {
152 				if (fs.equals(
153 						FeeStatusType.FEE_NOT_USED)) {
154 					return Boolean.TRUE;
155 				}
156 			}
157 		}
158 		return Boolean.FALSE;
159 	}

165 	 * @param eliRecord the eli record
166 	 * 
167 	 * @return true, if is current and record name match
168 	 */
169 	protected boolean isCurrentAndRecordNameMatch(
170 			ProcessContext processContext, 
171 			EliRecord eliRecord) {
172 		if (isCurrentNamePresent(
173 				processContext)) {
174 			//If current name is present.
175 			if (eliRecord.getPerson() != null 
176 					&& eliRecord.getPerson()
177 					.getPreviousName() != null) {
178 				//If record name is present.
179 				if (eliRecord.getPerson().getPreviousName()
180 						.equals(((HqInquiryProcessContext)
181 								processContext).getDlApplicant()
182 								.getCurrentName())) {
183 					//If both current name and record name matches.
184 					return Boolean.TRUE;
185 				}
186 			}
187 		}
188 		return Boolean.FALSE;
189 	}

428 	 * Checks if is age between 15 to21 years.
429 	 * 
430 	 * @param eliRecord the eli record
431 	 * @param processContext the process context
432 	 * 
433 	 * @return true, if is age between15 to21 yrs
434 	 */
435 	protected boolean isAgeBetween15To21Yrs(EliRecord eliRecord,
436 			ProcessContext processContext) {
437 		if (eliRecord.getPerson() != null
438 				&& eliRecord.getPerson().getBirthDate() != null) {
439 			// If birth date on record present
440 			Date birthDate = eliRecord.getPerson().getBirthDate();
441 			Calendar calendar = Calendar.getInstance();
442 			calendar.setTime(birthDate);
443 			calendar.add(Calendar.YEAR, 21);
444 			Date processDate = getDlProcessDate(processContext);
445 			if (processDate != null) {
446 				if (calendar.getTime().compareTo(processDate) > 0) {
447 					// If birth date + 21 years is after
448 					// process date.
449 					calendar.setTime(birthDate);
450 					calendar.add(Calendar.YEAR, 15);
451 					if (calendar.getTime().compareTo(processDate) <= 0) {
452 						// If birth date + 15 years is on or
453 						// before process date.
454 						return Boolean.TRUE;
455 					}
456 				}
457 			}
458 		}
459 		return Boolean.FALSE;
460 	}

464 	 * Checks for class code changed.
465 	 * 
466 	 * @param eliRecord
467 	 *            the eli record
468 	 * 
469 	 * @return true, if successful
470 	 */
471 	protected boolean hasClassCodeChanged(EliRecord eliRecord) {
472 		DriverLicense recordLicense = eliRecord.getDriverLicense();
473 		if (!EaseUtil.isNullOrBlank(recordLicense.getExistingLicenseClasses())) {
474 			// If Class of Record is present.
475 			if (EaseUtil.isNullOrBlank(eliRecord.getEchoedDriverLicense()
476 					.getAppliedLicenseClasses())) {
477 				// If Class of Inquiry is not present.
478 				return Boolean.TRUE; // Class of Record and class of inquiry not
479 				// match.
480 			} else {
481 				// Class of Inquiry is present.
482 				if (recordLicense.getExistingLicenseClasses().get(0) != null
483 						&& eliRecord.getEchoedDriverLicense()
484 								.getAppliedLicenseClasses().get(0) != null) {
485 					if (recordLicense.getExistingLicenseClasses().get(0)
486 							.hasCodeNotEqualTo(
487 									eliRecord.getEchoedDriverLicense()
488 											.getAppliedLicenseClasses().get(0)
489 											.getCode())) {
490 						// If class of record is not same as class of inquiry.
491 						return Boolean.TRUE; // Class of Record and class of
492 						// inquiry not match.
493 					}
494 				}
495 			}
496 		} else {
497 			// Class of record is not present.
498 			if (!EaseUtil.isNullOrBlank(eliRecord.getEchoedDriverLicense()
499 					.getAppliedLicenseClasses())) {
500 				// If Class of Inquiry is present.
501 				return Boolean.TRUE; // Class of Record and class of inquiry not
502 				// match.
503 			} else {
504 				return Boolean.FALSE; // Class of Record and inquiry not
505 				// present.
506 			}
507 		}
508 		return Boolean.FALSE;
509 	}

511 	/***
512 	 * Checks for endorsements changed.
513 	 * 
514 	 * @param eliRecord
515 	 *            the eli record
516 	 * @param newLicense
517 	 *            the new license
518 	 * 
519 	 * @return true, if successful
520 	 */
521 	protected boolean hasEndorsementsChanged(EliRecord eliRecord,
522 			DriverLicense newLicense) {
523 		DriverLicense recordLicense = eliRecord.getDriverLicense();
524 		if (!EaseUtil.isNullOrBlank(recordLicense.getExistingEndorsements())) {
525 			// If Endorsements of Record is present.
526 			if (EaseUtil.isNullOrBlank(newLicense.getNewEndorsements())) {
527 				// If new endorsements is not present.
528 				return Boolean.TRUE; // Endorsement of Record and inquiry not
529 				// match.
530 			} else {
531 				// Endorsement of Inquiry and record is present.
532 				for (CodeSetElement recordEndorsement : recordLicense
533 						.getExistingEndorsements()) {
534 					if (recordEndorsement != null) {
535 						boolean found = false;
536 						for (CodeSetElement newEndorsement : newLicense
537 								.getNewEndorsements()) {
538 							if (newEndorsement != null) {
539 								if (recordEndorsement
540 										.hasCodeEqualTo(newEndorsement
541 												.getCode())) {
542 									// If Endorsement of record match with new
543 									// Endorsement.
544 									found = true;
545 								}
546 							}
547 						}
548 						if (!found) {
549 							return Boolean.TRUE; // If not found, endorsements
550 							// does not match.
551 						}
552 					}
553 				}
554 			}
555 		} else {
556 			// If Endorsements of Record is not present.
557 			if (!EaseUtil.isNullOrBlank(newLicense.getNewEndorsements())) {
558 				// If new endorsements is present.
559 				return Boolean.TRUE; // Endorsement of Record and inquiry not
560 				// match.
561 			} else {
562 				return Boolean.FALSE; // Endorsement of Record and inquiry not
563 				// present.
564 			}
565 		}
566 		return Boolean.FALSE;
567 	}

570 	 * Checks for categories changed.
571 	 * 
572 	 * @param eliRecord
573 	 *            the eli record
574 	 * @param newLicense
575 	 *            the new license
576 	 * 
577 	 * @return true, if successful
578 	 */
579 	protected boolean hasCategoriesChanged(EliRecord eliRecord,
580 			DriverLicense newLicense) {
581 		DriverLicense recordLicense = eliRecord.getDriverLicense();
582 		if (!EaseUtil.isNullOrBlank(recordLicense.getExistingCategories())) {
583 			// If Categories of Record is present.
584 			if (EaseUtil.isNullOrBlank(newLicense.getNewCategories())) {
585 				// If new Categories is not present.
586 				return Boolean.TRUE; // Categories of Record and inquiry not
587 				// match.
588 			} else {
589 				// Categories of Inquiry and record is present.
590 				for (CodeSetElement recordCategory : recordLicense
591 						.getExistingCategories()) {
592 					if (recordCategory != null) {
593 						boolean found = false;
594 						for (CodeSetElement newCategory : newLicense
595 								.getNewCategories()) {
596 							if (newCategory != null) {
597 								if (recordCategory.hasCodeEqualTo(newCategory
598 										.getCode())) {
599 									// If Category of record match with new
600 									// Categories.
601 									found = true;
602 								}
603 							}
604 						}
605 						if (!found) {
606 							return Boolean.TRUE; // If not found, Category does
607 							// not match.
608 						}
609 					}
610 				}
611 			}
612 		} else {
613 			// If Categories of Record is not present.
614 			if (!EaseUtil.isNullOrBlank(newLicense.getNewCategories())) {
615 				// If new Categories is present.
616 				return Boolean.TRUE; // Categories of Record and inquiry not
617 				// match.
618 			} else {
619 				return Boolean.FALSE; // Categories of Record and inquiry not
620 				// present.
621 			}
622 		}
623 		return Boolean.FALSE;
624 	}

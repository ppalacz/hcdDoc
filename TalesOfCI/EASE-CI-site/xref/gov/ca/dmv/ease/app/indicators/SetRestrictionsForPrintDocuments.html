44  	 * Adds the year in date.
45  	 * 
46  	 * @param date the date
47  	 * @param numberOfYearsToAdd the number of years to add
48  	 * 
49  	 * @return the date
50  	 */
51  	private Date addYearInDate(Date date, int numberOfYearsToAdd) {
52  		Calendar calendar = Calendar.getInstance();
53  		calendar.setTime(date);
54  		calendar.add(Calendar.YEAR, numberOfYearsToAdd);
55  		return calendar.getTime();
56  	}

58  	/***
59  	 * Calculate age.
60  	 * 
61  	 * @param birthDate the birth date
62  	 * @param dateFeeReceived the date fee received
63  	 * 
64  	 * @return the int
65  	 */
66  	private int calculateAge(Date birthDate, Date dateFeeReceived) {
67  		int age = 0;
68  		if (isNotNull(birthDate) && isNotNull(dateFeeReceived)) {
69  			Calendar paymentDateInstance = Calendar.getInstance();
70  			Calendar dobInstance = Calendar.getInstance();
71  			dobInstance.setTime(birthDate);
72  			paymentDateInstance.setTime(dateFeeReceived);
73  			int paymentYear = paymentDateInstance.get(Calendar.YEAR);
74  			int dobYear = dobInstance.get(Calendar.YEAR);
75  			age = paymentYear - dobYear;
76  			int paymentMonth = paymentDateInstance.get(Calendar.MONTH);
77  			int dobMonth = dobInstance.get(Calendar.MONTH);
78  			if (dobMonth > paymentMonth) {
79  				age--;
80  			}
81  			else if (paymentMonth == dobMonth) {
82  				int paymentDay = paymentDateInstance.get(Calendar.DAY_OF_MONTH);
83  				int dobDay = dobInstance.get(Calendar.DAY_OF_MONTH);
84  				if (dobDay > paymentDay) {
85  					age--;
86  				}
87  			}
88  		}
89  		return age;
90  	}

92  	/***
93  	 * Execute.
94  	 * 
95  	 * @param dlApplication the dl application
96  	 * 
97  	 * @return the i error collector
98  	 */
99  	public IErrorCollector execute(DlApplication dlApplication) {
100 		Date applicantBirthDate = null;
101 		Date previousAppAppDate = null;
102 		Date dateFeeReceived = null;
103 		List <DrivingRestriction> restrictions = null;
104 		// Retrieving required attributes from BOs. 
105 		if (isNotNull(dlApplication)) {
106 			// get applicant's birth date
107 			Person applicant = dlApplication.getApplicant();
108 			if (isNotNull(applicant)) {
109 				applicantBirthDate = applicant.getBirthDate();
110 			}
111 			// get previous application date
112 			IProcessHistory previousProcessHistory = dlApplication
113 					.getPreviousProcessHistory();
114 			if (isNotNull(previousProcessHistory)) {
115 				previousAppAppDate = previousProcessHistory
116 						.getApplicationDate();
117 			}
118 			List <Payment> fees = dlApplication.getInvoice().getPayments();
119 			if (isNotNull(fees)) {
120 				Payment fee = fees.get(0);
121 				if (isNotNull(fee)) {
122 					dateFeeReceived = fee.getDate();
123 				}
124 			}
125 			restrictions = dlApplication.getLicense().getNewRestrictions();
126 		}
127 		IErrorCollector errorCollector = new ErrorCollector();
128 		//If restriction 47 is found and if the applicant is 18 or older, remove restriction 47	
129 		executeCondition1(restrictions, applicantBirthDate, dateFeeReceived);
130 		try {
131 			// Need Restriction 47 for younger than 18
132 			executeCondition2(restrictions, applicantBirthDate,
133 					dateFeeReceived, previousAppAppDate);
134 		}
135 		catch (ApplicationException e) {
136 			errorCollector.register(e);
137 		}
138 		try {
139 			// Need Restriction 59 for younger than 18
140 			executeCondition3(restrictions, applicantBirthDate,
141 					dateFeeReceived, previousAppAppDate);
142 		}
143 		catch (ApplicationException e) {
144 			errorCollector.register(e);
145 		}
146 		//If restriction 59 is found and if the applicant is 18 or older, remove restriction 59	
147 		executeCondition4(restrictions, applicantBirthDate, dateFeeReceived);
148 		return errorCollector;
149 	}

158 	private void executeCondition1(List <DrivingRestriction> restrictions,
159 			Date applicantBirthDate, Date dateFeeReceived) {
160 		if (isApplicantEighteenOrOlder(applicantBirthDate, dateFeeReceived)) {
161 			if (isNotNull(restrictions)) {
162 				for (DrivingRestriction restriction : restrictions) {
163 					CodeSetElement restrictionTypeCode = restriction
164 							.getRestrictionTypeCode();
165 					if (!isNullOrBlank(restrictionTypeCode)) {
166 						String aCode = restrictionTypeCode.getCode();
167 						if (aCode
168 								.equals(IRestrictionCodesConstants.RESTRICTIONS_12_MONTHS)) {
169 							restrictions.remove(restriction);
170 							break;
171 						}
172 					}
173 				}
174 			}
175 		}
176 	}

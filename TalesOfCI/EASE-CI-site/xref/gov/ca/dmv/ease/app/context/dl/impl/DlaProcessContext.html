117 	 * Empty constructor.

119 	public DlaProcessContext() {
120 		this(new DlApplication());
121 		getDlApplication().setNewLicenseNumberRequired(false);
122 		aniResults = new ArrayList <AniRecord>();
123 		Person person = new Person();
124 		person.setCurrentName(new PersonName());
125 		List <PersonName> personNames = new ArrayList <PersonName>();
126 		person.setPersonNames(personNames);
127 		person.setDriverLicense(new DriverLicense());
128 		person.getDriverLicense().setAppliedLicenseClasses(
129 				new ArrayList <CodeSetElement>());
130 		OutOfStateLicense outOfStateLicense = new OutOfStateLicense();
131 		outOfStateLicense.setName(new PersonName());
132 		person.setPreviousLicenses(new ArrayList <OutOfStateLicense>());
133 		person.addPreviousLicense(outOfStateLicense);
134 		getDlApplication().setApplicant(person);
135 		AniRecord aniRecord = new AniRecord();
136 		aniResults.add(aniRecord);
137 		lpvRecord = new LpvRecord();
138 		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
139 		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
140 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
141 		legalPresenceDocument.setVerificationCode(new CodeSetElement());
142 		List <Document> list = new ArrayList <Document>();
143 		Passport passport = new Passport();
144 		list.add(ssnDocument);
145 		list.add(legalPresenceDocument);
146 		list.add(passport);
147 		person.setDocuments(list);
148 		getDlApplication().setApplicationTypeCode(
149 				new CodeSetElement("", "", ""));
150 		Address mailingAddress = new Address(AddressType.MAILING);
151 		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
152 		person.addAddress(mailingAddress);
153 		person.addAddress(residenceAddress);
154 		Invoice invoice = new Invoice();
155 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
156 		invoice.setInvoiceItems(items);
157 		List <Payment> payments = new ArrayList <Payment>();
158 		invoice.setPayments(payments);
159 		CashPayment cashPayment = new CashPayment();
160 		payments.add(cashPayment);
161 		CheckPayment checkPayment = new CheckPayment();
162 		payments.add(checkPayment);
163 		DebitPayment debitPayment = new DebitPayment();
164 		payments.add(debitPayment);
165 		List <IProcessHistory> processHistory = new ArrayList <IProcessHistory>();
166 		getDlApplication().setProcessHistory(processHistory);
167 		getDlApplication().setInvoice(invoice);
168 		setChangedName(new PersonName());
169 	}

446 	/***
447 	 * This method returns Person object of the primary record from ANI response.
448 	 * 
449 	 * @return the primary ani record
450 	 */
451 	public AniRecord getPrimaryAniRecord() {
452 		Iterator <AniRecord> iterator = getAniRecords().iterator();
453 		while (iterator.hasNext()) {
454 			AniRecord aniRecord = iterator.next();
455 			if (aniRecord.isPrimary()) {
456 				return aniRecord;
457 			}
458 		}
459 		return null;
460 	}

24  public class NewInventoryRequiredIndicators {
25  	/***
26  	 * This method checks if New Inventory is Required.
27  	 * 
28  	 * @param dlApplication the DL Application
29  	 * @param ttc the Current TTC
30  	 * 
31  	 * @return true if New Inventory is Required
32  	 */
33  	public boolean isNewInventoryRequired(final DlApplication dlApplication,
34  			String ttc) {
35  		boolean isNewInventoryRequired = true;
36  		//If DL is Applied then check if any Issued ID Card Exists
37  		if (IndicatorsConstants.TTC_DLA.equalsIgnoreCase(ttc)) {
38  			//Check if ID Card Exists and is Issued
39  			if (dlApplication.getApplicant().getIdCard().getIssued()
40  					&& EaseUtil.isNullOrBlank(dlApplication.getApplicant()
41  							.getIdCard()
42  							.getExistingIncompleteApplicationReasonCode())
43  					&& EaseUtil.isNotNull(dlApplication.getApplicant()
44  							.getIdCard().getIssuedDate())) {
45  				isNewInventoryRequired = false;
46  			}
47  		}
48  		//If ID Card is Applied then check if any Issued DL Exists
49  		else if (IndicatorsConstants.TTC_IDA.equalsIgnoreCase(ttc)) {
50  			if (dlApplication.getApplicant().getDriverLicense().getIssued()
51  					&& EaseUtil.isNotNull(dlApplication.getApplicant()
52  							.getDriverLicense().getEffectiveDate())) {
53  				isNewInventoryRequired = false;
54  			}
55  		}
56  		return isNewInventoryRequired;
57  	}
58  }

30  	/* (non-Javadoc)

33  	@Override
34  	protected void invokeService(ProcessContext context) {
35  		CounselorInquiryProcessContext counselorInquiryProcessContext = (CounselorInquiryProcessContext) context;
36  		CounselorInquiryForDisplayRequestTs counselorInquiryForDisplayRequestTs = new CounselorInquiryForDisplayRequestTs(
37  				counselorInquiryProcessContext.getUserContext(),
38  				counselorInquiryProcessContext.getCounselorInquiryApplication()
39  						.getApplicant().getDriverLicense().getLicenseNumber());
40  		CounselorInquiryForDisplayResponseTs counselorInquiryForDisplayResponseTs = counselorInquiryTaskService
41  				.execute(counselorInquiryForDisplayRequestTs);
42  		if (counselorInquiryForDisplayResponseTs.hasErrors()) {
43  			this.populateErrorMessages(counselorInquiryProcessContext,
44  					counselorInquiryForDisplayResponseTs.getErrorCollector());
45  		}
46  		else {
47  			counselorInquiryProcessContext
48  					.setInquiryReturnData(counselorInquiryForDisplayResponseTs
49  							.getInquiryReturnData());
50  			counselorInquiryProcessContext
51  					.setDisplayStatusInfoCode(counselorInquiryForDisplayResponseTs
52  							.getDisplayStatusInfoCode());
53  		}
54  	}
55

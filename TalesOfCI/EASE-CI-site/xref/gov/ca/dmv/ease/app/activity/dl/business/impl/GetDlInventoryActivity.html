32  public class GetDlInventoryActivity extends BusinessActivity {
33  	/*** The Constant serialVersionUID. */
34  	private static final long serialVersionUID = 7395638466054958188L;
35  	private InventoryTaskService inventoryTaskService;
36  
37  	/*
38  	 * (non-Javadoc)
39  	 * 
40  	 * @see
41  	 * gov.ca.dmv.ease.activity.BusinessActivity#invokeService(gov.ca.dmv.ease
42  	 * .app.context.ProcessContext)
43  	 */
44  	/***
45  	 * This method creates GetDlInventoryRequestTs request class with required
46  	 * business objects from the process context using factory and calls DL Task
47  	 * Service to get inventory for DL. Once the response is received from the
48  	 * Task Service it maps the business objects to the process context.
49  	 * 
50  	 */
51  	@Override
52  	public void invokeService(ProcessContext processContext) {
53  		HqInquiryProcessContext hqInquiryProcessContext = (HqInquiryProcessContext) processContext;
54  		DlTaskServiceRequestFactory dlTaskReqFactory = new DlTaskServiceRequestFactory();
55  		boolean isDl = processContext.getUserContext().getTtc().startsWith("D") ? Boolean.TRUE
56  				: Boolean.FALSE;
57  		GetDlInventoryRequestTs request = dlTaskReqFactory
58  				.createGetDlInventoryRequest(isDl, processContext
59  						.getUserContext());
60  		GetDlInventoryResponseTs response = inventoryTaskService
61  				.execute(request);
62  		if (response.hasErrors()) {
63  			populateErrorMessages(hqInquiryProcessContext, response
64  					.getErrorCollector());
65  		}
66  		else {
67  			InventoryItem inventoryItem = response.getInventoryItem();
68  			if (hqInquiryProcessContext.getUserContext().getTtc()
69  					.equalsIgnoreCase(ITtcConstants.DLA)) {
70  				if (hqInquiryProcessContext.getDlApplicant() != null
71  						&& hqInquiryProcessContext.getDlApplicant()
72  								.getDriverLicense() != null) {
73  					hqInquiryProcessContext.getDlApplicant().getDriverLicense()
74  							.setIssued(true);
75  					hqInquiryProcessContext.getDlApplicant().getDriverLicense()
76  							.setLicenseNumber(
77  									((DriverLicense) inventoryItem)
78  											.getLicenseNumber());
79  				}
80  			}
81  			else if (hqInquiryProcessContext.getUserContext().getTtc()
82  					.equalsIgnoreCase(ITtcConstants.IDA)) {
83  				if (hqInquiryProcessContext.getDlApplicant() != null
84  						&& hqInquiryProcessContext.getDlApplicant().getIdCard() != null) {
85  					hqInquiryProcessContext.getDlApplicant().getDriverLicense()
86  							.setIssued(true);
87  					hqInquiryProcessContext
88  							.getDlApplicant()
89  							.getIdCard()
90  							.setIdNumber(
91  									((IdCard) inventoryItem).getLicenseNumber());
92  				}
93  			}
94  			hqInquiryProcessContext.setInventoryRequested(true);
95  		}
96  	}
97

99  	 * @param inventoryTaskService the inventoryTaskService to set
100 	 */
101 	public void setInventoryTaskService(
102 			InventoryTaskService inventoryTaskService) {
103 		this.inventoryTaskService = inventoryTaskService;
104 	}

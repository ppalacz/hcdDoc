37  public class ValidateAutomatedReceipt extends BusinessActivity {
38  	/*** The Constant serialVersionUID. */
39  	private static final long serialVersionUID = -8489154466528344108L;
40  	/*** The issue dl task service. */
41  	private IssueDlTaskService issueDlTaskService;
42  
43  	/***
44  	 * Sets the issue dl task service.
45  	 * 
46  	 * @param issueDlTaskService the issueDlTaskService to set
47  	 */
48  	public void setIssueDlTaskService(IssueDlTaskService issueDlTaskService) {
49  		this.issueDlTaskService = issueDlTaskService;
50  	}
51  
52  	/*
53  	 * (non-Javadoc)
54  	 * 
55  	 * @see
56  	 * gov.ca.dmv.ease.app.activity.BusinessActivity#invokeService(gov.ca.dmv
57  	 * .ease.app.context.ProcessContext)
58  	 */
59  	@Override
60  	protected void invokeService(ProcessContext processContext) {
61  		DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) processContext;
62  		List <Voucher> vouchers = dlFeeProcessContext.getInvoice()
63  				.getVoucherPayments();
64  		TaskServiceRequestFactoryImpl tsRequestFactory = new TaskServiceRequestFactoryImpl();
65  		// Creating request
66  		ValidateAutomatedReceiptRequestTs request = tsRequestFactory
67  				.createValidateAutomatedReceiptRequest(vouchers, null, processContext
68  						.getUserContext());
69  		// Invoking task service
70  		ValidateAutomatedReceiptResponseTs response = issueDlTaskService
71  				.execute(request);
72  		if (response.getErrorCollector().hasErrors()) {
73  			processContext.setValidationMessageObject(response
74  					.getValidationMessageObject());
75  		}
76  		else {
77  			List <Payment> payments = new ArrayList <Payment>();
78  			for (Payment payment : dlFeeProcessContext.getInvoice()
79  					.getPayments()) {
80  				//TODO this calls for having a selector method on Payment subclasses that return a boolean
81  				//what is the property that below types of payments have?
82  				if (payment instanceof CashPayment
83  						|| payment instanceof CheckPayment
84  						|| payment instanceof DebitPayment
85  						|| payment instanceof CreditPayment) {
86  					payments.add(payment);
87  				}
88  			}
89  			List <VoucherReturnCode> voucherReturnCodes = response
90  					.getVoucherReturnCodes();
91  			for (int i = 0; i < voucherReturnCodes.size(); i++) {
92  				voucherReturnCodes.get(i).getValidatedVoucher().setOffice(
93  						voucherReturnCodes.get(i).getInquiredVoucher()
94  								.getOffice());
95  				voucherReturnCodes.get(i).getValidatedVoucher().setEmployee(
96  						voucherReturnCodes.get(i).getInquiredVoucher()
97  								.getEmployee());
98  				voucherReturnCodes.get(i).getValidatedVoucher()
99  						.setSequenceNumber(
100 								voucherReturnCodes.get(i).getInquiredVoucher()
101 										.getSequenceNumber());
102 				payments.add(voucherReturnCodes.get(i).getValidatedVoucher());
103 			}
104 			//	for (VoucherReturnCode voucherReturnCode : voucherReturnCodes) {
105 			//		/		payments.add(voucherReturnCode.getVoucher());
106 			//	}
107 			dlFeeProcessContext.getInvoice().setPayments(payments);
108 		}
109 		/*
110 		 * Not sure how the transaction is accessed. Transaction transaction =
111 		 * response.getTransaction();
112 		 */
113 	}

60  	/*** The issue dl task service. */

63  	/***
64  	 * This method creates CancelDlRequestTs request class with
65  	 * required business objects from the process context using factory and
66  	 * calls DL Task Service to search person by license number.
67  	 * 
68  	 * Once the response is received from the Task Service it maps the business
69  	 * objects to the process context.
70  	 * 
71  	 */
72  	@Override
73  	protected void invokeService(ProcessContext context) {
74  		CdlProcessContext cdlProcessContext = (CdlProcessContext) context;
75  		Person person = cdlProcessContext.getPerson();
76  		UserContext userContext = (UserContext) EaseProxyUtils
77  				.getProxyObject(context.getRootContext().getUserContext());
78  		//Create request
79  		CancelDlRequestTs request = new CancelDlRequestTs(person,
80  				cdlProcessContext.getTransactionSubCode(), userContext);
81  		
82  		//Get the response
83  		CancelDlResponseTs response = miscellaneousTaskService.execute(request);
84  		if (response.hasErrors()) {
85  			// TODO 'UP','PE','01','03','06','07'
86  			this.populateErrorMessages(cdlProcessContext, response.getErrorCollector());
87  		}
88  		else {
89  			if (TRANSACTION_SUBCODE_INQUIRY.equalsIgnoreCase(cdlProcessContext.getTransactionSubCode())) {
90  				if(response.getResponseInfoCode() != null) {
91  					CodeSetElement responseInfoCode = response.getResponseInfoCode();
92  					IErrorCollector errorCollector = new ErrorCollector();
93  					
94  					if(!isNullorEmpty(responseInfoCode) && !isNullorEmpty(errorCollector)) {
95  						if("".equalsIgnoreCase(responseInfoCode.getCode()))
96  							errorCollector.register(new ApplicationException("Q050"));	//INVALID HQTRS FILE ERROR RETURN
97  						if("0E".equalsIgnoreCase(responseInfoCode.getCode()))
98  							errorCollector.register(new ApplicationException("Q051"));	//TRAINING RECORD NOT FOUND
99  						if("AA".equalsIgnoreCase(responseInfoCode.getCode()))
100 							errorCollector.register(new ApplicationException("Q621"));	//CONTROLLED RECORD - CONTACT CONFIDENTIAL RECORDS UNIT
101 						if("AF".equalsIgnoreCase(responseInfoCode.getCode()))
102 							errorCollector.register(new ApplicationException("Q631"));	//CONTROLLED RECORD - CONTACT DLFAU
103 						if("1".equalsIgnoreCase(responseInfoCode.getCode()) ||
104 								"11".equalsIgnoreCase(responseInfoCode.getCode()) ||
105 								"17".equalsIgnoreCase(responseInfoCode.getCode()) ||
106 								"19".equalsIgnoreCase(responseInfoCode.getCode()) ||
107 								"27".equalsIgnoreCase(responseInfoCode.getCode()) ||
108 								"99".equalsIgnoreCase(responseInfoCode.getCode()))
109 							errorCollector.register(new ApplicationException("Q022"));	//CNA - LONG TERM
110 						if("2".equalsIgnoreCase(responseInfoCode.getCode()) ||
111 								"3".equalsIgnoreCase(responseInfoCode.getCode()) ||
112 								"-21".equalsIgnoreCase(responseInfoCode.getCode()))
113 							errorCollector.register(new ApplicationException("Q023"));	//CNA - SHORT TERM
114 						if("4".equalsIgnoreCase(responseInfoCode.getCode()))
115 							errorCollector.register(new ApplicationException("Q501"));	//NO HIT ON DL DATA BASE
116 						if("7".equalsIgnoreCase(responseInfoCode.getCode()))
117 							errorCollector.register(new ApplicationException("Q552"));	//DRIVER RECORD NOT AVAILABLE
118 						if("8".equalsIgnoreCase(responseInfoCode.getCode()))
119 							errorCollector.register(new ApplicationException("Q605"));	//CONTROLLED RECORD--CALL RSIU
120 						if("10".equalsIgnoreCase(responseInfoCode.getCode()))
121 							errorCollector.register(new ApplicationException("Q076"));	//NO MATCH ON NAME IN DL RECORD
122 						if("12".equalsIgnoreCase(responseInfoCode.getCode()))
123 							errorCollector.register(new ApplicationException("Q574"));	//NO MATCH ON NAME AND/OR BIRTHDATE IN DL RECORD
124 						if("14".equalsIgnoreCase(responseInfoCode.getCode()))
125 							errorCollector.register(new ApplicationException("Q539"));	//NO ID CARD ON RECORD
126 						if("15".equalsIgnoreCase(responseInfoCode.getCode()))
127 							errorCollector.register(new ApplicationException("Q540"));	//PREVIOUSLY CANCELLED
128 						if("16".equalsIgnoreCase(responseInfoCode.getCode()))
129 							errorCollector.register(new ApplicationException("Q559"));	//ALREADY EXPIRED - CANCELLATION INVALID
130 						if("31".equalsIgnoreCase(responseInfoCode.getCode()))
131 							errorCollector.register(new ApplicationException("Q586"));	//EFFECTIVE DATE MUST BE LATER THAN ISSUE DATE
132 						if("32".equalsIgnoreCase(responseInfoCode.getCode()))
133 							errorCollector.register(new ApplicationException("Q554"));	//ORIG ACTION EFF DATE OR AUTH SECTION DOES NOT MATCH DRIVER RECORD
134 						if("46".equalsIgnoreCase(responseInfoCode.getCode()))
135 							errorCollector.register(new ApplicationException("Q584"));	//DL NOT VALID OR NEVER ISSUED	
136 					}
137 					this.populateErrorMessages(cdlProcessContext, errorCollector);
138 				}
139 				if (response.getPersonName() != null
140 						&& response.getResponseInfoCode() != null
141 						&& "00".equalsIgnoreCase(response.getResponseInfoCode().getCode())) {
142 					// SINGLE HIT; need to add the Document Generation/Print code;
143 					//Create list of personNames and add the person 
144 					//List <PersonName> personNames = new ArrayList <PersonName>();
145 					//personNames.add(response.getPersonName());
146 					//Set person to the cdlProcessCtx
147 					person.setCurrentName(response.getPersonName());
148 					cdlProcessContext.setPerson(person);
149 					printReceipt(cdlProcessContext);
150 				}
151 				else {
152 					// TODO ' ','CE','NH','ER'
153 					// no place holders for the above error codes
154 				}
155 			}
156 		}
157 	}

159 	private void printReceipt(CdlProcessContext cdlCtx) {
160 
161 		IUserContext userCtx = cdlCtx.getUserContext();
162 		//create TS request for printing the cancel DL receipt
163 		DlVoluntaryCancellationRequestTus printRequestTs = new DlVoluntaryCancellationRequestTus(
164 				userCtx, cdlCtx.getPerson(), new DlVoluntaryCancellationConverter());
165 		//Get the response
166 		PrintResponseTs printResponseTs = miscellaneousTaskService
167 				.execute(printRequestTs);
168 		if (printResponseTs.hasErrors()) {
169 			populateErrorMessages(cdlCtx, printResponseTs.getErrorCollector());
170 			// TODO 'UP','PE','01','03','06','07'
171 		}
172 	}

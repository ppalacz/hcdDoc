103 	public CdlisPdpsProcessContext() {
104 		dlApplication = new DlApplication();
105 		dlApplication.setNewLicenseNumberRequired(false);
106 		Person person = new Person();
107 		person.setCurrentName(new PersonName());
108 		List <PersonName> personNames = new ArrayList <PersonName>();
109 		personNames.add(new PersonName());
110 		person.setPersonNames(personNames);
111 		person.setDriverLicense(new DriverLicense());
112 		person.getDriverLicense().setAppliedLicenseClasses(
113 				new ArrayList <CodeSetElement>());
114 		dlApplication.setApplicant(person);
115 		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
116 		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
117 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
118 		legalPresenceDocument.setVerificationCode(new CodeSetElement());
119 		List <Document> list = new ArrayList <Document>();
120 		list.add(ssnDocument);
121 		list.add(legalPresenceDocument);
122 		person.setDocuments(list);
123 		Address mailingAddress = new Address(AddressType.MAILING);
124 		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
125 		person.addAddress(mailingAddress);
126 		person.addAddress(residenceAddress);
127 		Invoice invoice = new Invoice();
128 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
129 		invoice.setInvoiceItems(items);
130 		List <Payment> payments = new ArrayList <Payment>();
131 		invoice.setPayments(payments);
132 		dlApplication.setInvoice(invoice);
133 	}

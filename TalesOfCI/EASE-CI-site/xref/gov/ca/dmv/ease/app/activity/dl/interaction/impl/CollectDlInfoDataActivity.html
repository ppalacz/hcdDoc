26  public class CollectDlInfoDataActivity extends FallbackableInteractionActivity {
27  	/*** The Constant serialVersionUID. */
28  	private static final long serialVersionUID = 1817609475068358233L;
29  	/*** The collectDlInfoDataRules for validation */
30  	private CollectDlInfoDataRules collectDlInfoDataRules;
31  
32  	/* (non-Javadoc)
33  	 * @see gov.ca.dmv.ease.app.activity.impl.Activity#preCondition(gov.ca.dmv.ease.app.context.impl.ProcessContext)
34  	 */
35  	@Override
36  	protected void postCondition(ProcessContext context) {
37  		if (isNotByPassValidations(context)) {
38  			collectDlInfoDataRules.validate(context);
39  			populateErrorMessages(context, collectDlInfoDataRules
40  					.getErrorCollector());
41  			if (EaseUtil.isNullOrBlank(collectDlInfoDataRules
42  					.getErrorCollector().getEntries())) {
43  				collectDlInfoDataRules.militaryExtensionRule(context);
44  			}
45  		}
46  	}
47  
48  	/***
49  	 * Set collectDlInfoDataRules.
50  	 * 
51  	 * @param collectDlInfoDataRules the collectDlInfoDataRules to set
52  	 */
53  	public void setCollectDlInfoDataRules(
54  			CollectDlInfoDataRules aCollectDlInfoDataRules) {
55  		collectDlInfoDataRules = aCollectDlInfoDataRules;
56  	}
57  }

21  public class RouteCodeIndicators implements IndicatorsConstants {
22  	/***
23  	 * This method computes Route Code Indicator.
24  	 * 
25  	 * @param currentTtc the Current TTC
26  	 * @param routeCode the Route Code
27  	 * @param isRouteCodeProcessed indicator for Route Code Processing
28  	 * 
29  	 * @return the Route Code Indicator
30  	 */
31  	public String computeRouteCodeIndicator(String currentTtc, int routeCode,
32  			boolean isRouteCodeProcessed) {
33  		String routeCodeIndicator = null;
34  		//Check for TTC - DLA, DLC, DLD, DLE, DLP, DRT, IDA, IDS, IDC, IRP, SPO, SPC
35  		if (TTC_DLA.equalsIgnoreCase(currentTtc)
36  				|| TTC_DLA.equalsIgnoreCase(currentTtc)
37  				|| TTC_DLC.equalsIgnoreCase(currentTtc)
38  				|| TTC_DLD.equalsIgnoreCase(currentTtc)
39  				|| TTC_DLE.equalsIgnoreCase(currentTtc)
40  				|| TTC_DLP.equalsIgnoreCase(currentTtc)
41  				|| TTC_DRT.equalsIgnoreCase(currentTtc)
42  				|| TTC_IDA.equalsIgnoreCase(currentTtc)
43  				|| TTC_IDS.equalsIgnoreCase(currentTtc)
44  				|| TTC_IDC.equalsIgnoreCase(currentTtc)
45  				|| TTC_IRP.equalsIgnoreCase(currentTtc)
46  				|| TTC_SPO.equalsIgnoreCase(currentTtc)
47  				|| TTC_SPC.equalsIgnoreCase(currentTtc)) {
48  			if (ROUTE_CODE_3503 == routeCode) {
49  				if (isRouteCodeProcessed) {
50  					routeCodeIndicator = ROUTE_RECORD_CODE_B;
51  				}
52  				else {
53  					routeCodeIndicator = ROUTE_RECORD_CODE_M;
54  				}
55  			}
56  			if (ROUTE_CODE_MIN >= routeCode || routeCode <= ROUTE_CODE_MAX) {
57  				if (isRouteCodeProcessed) {
58  					routeCodeIndicator = ROUTE_RECORD_CODE_B;
59  				}
60  				else {
61  					routeCodeIndicator = ROUTE_RECORD_CODE_M;
62  				}
63  			}
64  		}
65  		return routeCodeIndicator;
66  	}
67  }

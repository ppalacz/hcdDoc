31  public class DeleteCreditedVoucher extends BusinessActivity {
32  	/*** The Constant serialVersionUID. */
33  	private static final long serialVersionUID = -2464130413861316088L;
34  	/*** The inventory task service. */
35  	private TaskService issueDlTaskService;

37  	/***
38  	 * @param issueDlTaskService the issueDlTaskService to set
39  	 */
40  	public void setIssueDlTaskService(TaskService issueDlTaskService) {
41  		this.issueDlTaskService = issueDlTaskService;
42  	}
43  
44  	/*
45  	 * (non-Javadoc)
46  	 * 
47  	 * @see
48  	 * gov.ca.dmv.ease.app.activity.BusinessActivity#invokeService(gov.ca.dmv
49  	 * .ease.app.context.ProcessContext)
50  	 */
51  	@Override
52  	protected void invokeService(ProcessContext context) {
53  		DlProcessContext dlProcessContext = (DlProcessContext) context;
54  		DlTaskServiceRequestFactory dlTaskReqFactory = new DlTaskServiceRequestFactory();
55  		List <Voucher> vouchers = null;
56  		Invoice invoice = dlProcessContext.getInvoice();
57  		if (invoice != null) {
58  			vouchers = invoice.getVoucherPayments();
59  			if (vouchers != null) {
60  				for (Voucher voucher : vouchers) {
61  					if (voucher.getAmount() != null
62  							&& ((voucher.getStatusCode().getCode().equals("C")))) {
63  						DeleteMasterReceiptFileRequestTs request = dlTaskReqFactory
64  								.createDeleteMasterReceiptFile(voucher, context
65  										.getUserContext());
66  						DeleteMasterReceiptFileResponseTs response = issueDlTaskService
67  								.execute(request);
68  						if (response.getErrorCollector().hasErrors()) {
69  							populateErrorMessages(context, response
70  									.getErrorCollector());
71  						}
72  					}
73  				}
74  			}
75  		}
76  	}
77  }

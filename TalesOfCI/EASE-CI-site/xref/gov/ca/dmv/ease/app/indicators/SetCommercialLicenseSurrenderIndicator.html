25  public class SetCommercialLicenseSurrenderIndicator {
26  	private static final String CDL_DESIGNATION_P = "P";
27  	private static final String CDL_DESIGNATION_Q = "Q";
28  	private static final String CDL_DESIGNATION_S = "S";
29  	private static final String ERROR_CODE_D903 = "D903";
30  
31  	public List <CommercialLicenseSurrenderIndicatorsEnum> execute(
32  			IErrorCollector errorCollector, UserContext userContext,
33  			List <OutOfStateLicense> previousLics, CdlisRecord cdlisRecord) {
34  		boolean isD903ErrorExist = false;
35  		if (!errorCollector.isEmpty()) {
36  			List <IErrorCollectorEntry> errors = errorCollector.getEntries();
37  			for (IErrorCollectorEntry error : errors) {
38  				String errorCode = error.getExceptionMessage();
39  				if (isNullOrBlank(errorCode)
40  						&& errorCode.equals(ERROR_CODE_D903)) {
41  					isD903ErrorExist = true;
42  					break;
43  				}
44  			}
45  		}
46  		boolean osLicEnteredOnCdl1 = false;
47  		Boolean isOsLicComml = null;
48  		if (isNotNull(previousLics) && (previousLics.size() > 0)) {
49  			osLicEnteredOnCdl1 = true;
50  			isOsLicComml = previousLics.get(0).isCommercial();
51  		}
52  		if (osLicEnteredOnCdl1 && isNotNull(isOsLicComml) && isOsLicComml) {
53  			if (isD903ErrorExist) {
54  				CommercialLicenseSurrenderIndicatorsEnum
55  						.setIarcCCCommlCompareSetInvalidCdlisInd(true);
56  				CommercialLicenseSurrenderIndicatorsEnum
57  						.setNoCdlDocsAllowedNonCdlOkToIssueInd(true);
58  				CommercialLicenseSurrenderIndicatorsEnum
59  						.setFirstTimeCalifWithPriorCdlInd(false);
60  			}
61  		}
62  		else {
63  			if (isNotNull(cdlisRecord)) {
64  				CodeSetElement designateActionCode = cdlisRecord
65  						.getDesignateAction();
66  				if (!isNullOrBlank(designateActionCode)) {
67  					String designateAction = designateActionCode.getCode();
68  					if (designateAction.equals(CDL_DESIGNATION_S)
69  							|| designateAction.equals(CDL_DESIGNATION_P)) {
70  						if (isD903ErrorExist) {
71  							CommercialLicenseSurrenderIndicatorsEnum
72  									.setIarcCCCommlCompareSetInvalidCdlisInd(true);
73  							CommercialLicenseSurrenderIndicatorsEnum
74  									.setNoCdlDocsAllowedNonCdlOkToIssueInd(true);
75  							CommercialLicenseSurrenderIndicatorsEnum
76  									.setFirstTimeCalifWithPriorCdlInd(false);
77  						}
78  					}
79  					else {
80  						if (osLicEnteredOnCdl1 && isOsLicComml) {
81  							if (designateAction.equals(CDL_DESIGNATION_Q)) {
82  								if (isD903ErrorExist) {
83  									CommercialLicenseSurrenderIndicatorsEnum
84  											.setIarcCCCommlCompareSetInvalidCdlisInd(true);
85  									CommercialLicenseSurrenderIndicatorsEnum
86  											.setFirstTimeCalifWithPriorCdlInd(false);
87  									previousLics.get(0).setCommercial(null); // TODO: should this exist here ???
88  								}
89  							}
90  						}
91  					}
92  				}
93  			}
94  		}
95  		List <CommercialLicenseSurrenderIndicatorsEnum> commercialLicenseSurrenderIndicators = new ArrayList <CommercialLicenseSurrenderIndicatorsEnum>();
96  		commercialLicenseSurrenderIndicators
97  				.add(CommercialLicenseSurrenderIndicatorsEnum.IARC_CC_COMML_COMPARE_SET_INVALID_CDLIS_IND);
98  		commercialLicenseSurrenderIndicators
99  				.add(CommercialLicenseSurrenderIndicatorsEnum.NO_CDL_DOCS_ALLOWED_NON_CDL_OK_TO_ISSUE_IND);
100 		commercialLicenseSurrenderIndicators
101 				.add(CommercialLicenseSurrenderIndicatorsEnum.FIRST_TIME_CALIF_WITH_PRIOR_CDL_IND);
102 		return commercialLicenseSurrenderIndicators;
103 	}
104 }

26  public class IsLegalPresenceCheckRequiredDecision extends DecisionActivity
27  		implements IDlaDecisionActivityConstantsForBpDefinition {
28  	private static final long serialVersionUID = -5420199914113352264L;
29  
30  	/***
31  	 * This method evaluates if legal presence check is required or not. 
32  	 * If required it returns true, if not returns false.
33  	 */
34  	@Override
35  	protected String evaluate(ProcessContext processContext) {
36  		CustomerInfoProcessContext customerInfoProcessContext = (CustomerInfoProcessContext) processContext;
37  		if (EaseUtil.isNullOrBlank(customerInfoProcessContext.getDlApplicant()
38  				.getLegalPresenceDocument().getDocumentId())) {
39  			return CHECK_NOT_REQUIRED;
40  		}
41  		return CHECK_REQUIRED;
42  	}
43  }

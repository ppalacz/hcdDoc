56  	/***
57  	 * Assign daf number.
58  	 * 
59  	 * @param context the context
60  	 */
61  	private void assignDafNumber(ProcessContext context) {
62  		DlProcessContext dlProcessContext = (DlProcessContext) context;
63  		Integer transactionIdentifier = new Integer("000");
64  		//Getting the Max DAF number in the Database
65  		IPersistenceServiceRequest request = persistenceServiceRequestFactory
66  				.createRetrieveCriteriaObjectRequest(context.getUserContext(),
67  						Transaction.class);
68  		RetrieveCriteriaObjectResponse response = (RetrieveCriteriaObjectResponse) request
69  				.execute();
70  		if (!response.hasErrors()) {
71  			if (response.getCriteriaInstance().list().size() > 0) {
72  				Criteria criteria = response.getCriteriaInstance();
73  				criteria.add(Restrictions.like("officeId", context
74  						.getUserContext().getOfficeId()));
75  				ProjectionList projList = Projections.projectionList();
76  				projList.add(Projections.max("transactionIdentifier"));
77  				criteria.setProjection(projList);
78  				List listOfBusinessObjects = criteria.list();
79  				for (Object obj : listOfBusinessObjects) {
80  					if (obj != null) {
81  						transactionIdentifier = (Integer) obj;
82  					}
83  				}
84  				transactionIdentifier++;
85  				dlProcessContext.setDafRecordNumber(transactionIdentifier);
86  			}
87  			else {
88  				transactionIdentifier++;
89  				dlProcessContext.setDafRecordNumber(transactionIdentifier);
90  			}
91  		}
92  		else if (response.hasErrors()) {
93  			List <IErrorCollectorEntry> errors = response.getErrorCollector()
94  					.getEntries();
95  			for (IErrorCollectorEntry errorCollectorEntry : errors) {
96  				LOGGER.error(errorCollectorEntry);
97  			}
98  		}
99  		else {
100 			LOGGER.error("Unable to assign DAF number to DLARecord");
101 		}
102 	}

104 	/***
105 	 * Assign sequence number.
106 	 * 
107 	 * @param context the context
108 	 */
109 	private void assignSequenceNumber(ProcessContext context) {
110 		DlProcessContext dlProcessContext = (DlProcessContext) context;
111 		String processId = null;
112 		GenerateSequenceNoRequestTs tsRequest = null;
113 		GenerateSequenceNoResponseTs tsResponse = null;
114 		ITaskServiceRequestFactory requestFactory = null;
115 		TaskService taskService = null;
116 		taskService = new IssueDlTaskService();
117 		processId = dlProcessContext.getProcessId();
118 		requestFactory = new TaskServiceRequestFactoryImpl();
119 		tsRequest = requestFactory.createGenerateSequenceNumberRequest(
120 				processId, dlProcessContext.getUserContext());
121 		tsResponse = taskService.execute(tsRequest);
122 		Integer sequenceNumber = tsResponse.getSequenceNumber();
123 		//Adding from Hidden EDL Logic Document 
124 		//If ControlCashierSeq# > 7000 Then 
125 		//then Reduce ControlCashierSeq# by 2000
126 		if (sequenceNumber > 7000) {
127 			sequenceNumber = sequenceNumber - 2000;
128 		}
129 		dlProcessContext.setCashierTransactionSequence(sequenceNumber);
130 		((UserContext) dlProcessContext.getUserContext())
131 				.setControlCashierSeqNumber(sequenceNumber);
132 	}

134 	/***
135 	 * Creates the process history.
136 	 * 
137 	 * @param context the context
138 	 */
139 	private void createProcessHistory(ProcessContext context) {
140 		DlProcessContext dlProcessContext = (DlProcessContext) context;
141 		ProcessHistory processHistory = new ProcessHistory(
142 				((UserContext) ((UserContext) context.getUserContext()).clone()),
143 				null, null, null, new Date(), new Integer(dlProcessContext
144 						.getDafRecordNumber()), ""
145 						+ dlProcessContext.getCashierTransactionSequence());
146 		Date applicationDate = null;
147 		if (context instanceof SpProcessContext) {
148 			SpProcessContext spProcessContext = (SpProcessContext) context;
149 			SalesPersonApplication salesPersonApplication = spProcessContext
150 					.getSalesPersonApplication();
151 			applicationDate = salesPersonApplication.getApplicationDate();
152 			salesPersonApplication.addProcessHistory(processHistory);
153 		}
154 		else if (context instanceof DlProcessContext) {
155 			applicationDate = dlProcessContext.getDlApplication()
156 					.getApplicationDate();
157 			dlProcessContext.getDlApplication().addProcessHistory(
158 					processHistory);
159 		}
160 		processHistory.setApplicationDate(applicationDate);
161 	}

74  	public IdProcessContext() {
75  		setDlApplication(new DlApplication());
76  		getDlApplication().setNewLicenseNumberRequired(false);
77  		aniResults = new ArrayList <AniRecord>();
78  		Person person = new Person();
79  		person.setCurrentName(new PersonName());
80  		List <PersonName> personNames = new ArrayList <PersonName>();
81  		person.setPersonNames(personNames);
82  		person.setIdCard(new IdCard());
83  		getDlApplication().setApplicant(person);
84  		AniRecord aniRecord = new AniRecord();
85  		aniResults.add(aniRecord);
86  		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
87  		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
88  		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
89  		legalPresenceDocument.setVerificationCode(new CodeSetElement());
90  		List <Document> list = new ArrayList <Document>();
91  		Passport passport = new Passport();
92  		list.add(ssnDocument);
93  		list.add(legalPresenceDocument);
94  		list.add(passport);
95  		person.setDocuments(list);
96  		getDlApplication().setApplicationTypeCode(
97  				new CodeSetElement("", "", ""));
98  		Address mailingAddress = new Address(AddressType.MAILING);
99  		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
100 		person.addAddress(mailingAddress);
101 		person.addAddress(residenceAddress);
102 		Invoice invoice = new Invoice();
103 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
104 		invoice.setInvoiceItems(items);
105 		List <Payment> payments = new ArrayList <Payment>();
106 		invoice.setPayments(payments);
107 		CashPayment cashPayment = new CashPayment();
108 		payments.add(cashPayment);
109 		CheckPayment checkPayment = new CheckPayment();
110 		payments.add(checkPayment);
111 		DebitPayment debitPayment = new DebitPayment();
112 		payments.add(debitPayment);
113 		List <IProcessHistory> processHistory = new ArrayList <IProcessHistory>();
114 		getDlApplication().setProcessHistory(processHistory);
115 		getDlApplication().setInvoice(invoice);
116 		setChangedName(new PersonName());
117 	}

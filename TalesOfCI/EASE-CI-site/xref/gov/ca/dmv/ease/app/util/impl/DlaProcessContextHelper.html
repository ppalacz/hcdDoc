37  	/***
38  	 * This method is for copying values from EliRecord to DlaProcessContext.
39  	 * Not all the values of EliRecord are copied only the person object from 
40  	 * EliRecord is copied to DlaProcessContext. Before copying we are setting 
41  	 * the current/new license type, certificates etc.. to the EliPerson
42  	 * 
43  	 * @param processContext
44  	 * @param eliRecord
45  	 */
46  	public static void copyEliRecord(ProcessContext processContext,
47  			EliRecord eliRecord) {
48  		if (processContext instanceof HqInquiryProcessContext) {
49  			HqInquiryProcessContext hqInquiryProcessContext = (HqInquiryProcessContext) processContext;
50  			if (eliRecord.getPerson() != null) {
51  				if (eliRecord.getPerson().getDriverLicense() != null) {
52  					eliRecord.getPerson().getDriverLicense()
53  							.setAppliedLicenseClasses(
54  									hqInquiryProcessContext.getDriverLicense()
55  											.getAppliedLicenseClasses());
56  					eliRecord.getPerson().getDriverLicense().setCertificates(
57  							hqInquiryProcessContext.getDriverLicense()
58  									.getCertificates());
59  					eliRecord.getPerson().getDriverLicense()
60  							.setNewEndorsements(
61  									hqInquiryProcessContext.getDriverLicense()
62  											.getNewEndorsements());
63  					eliRecord.getPerson().getDriverLicense().setNewAttachments(
64  							hqInquiryProcessContext.getDriverLicense()
65  									.getNewAttachments());
66  				}
67  				eliRecord.getPerson().setPreviousLicenses(
68  						hqInquiryProcessContext.getDlApplication()
69  								.getApplicant().getPreviousLicenses());
70  			}
71  			if (eliRecord.getPerson().getSocialSecurityDocument().getSsn() == null) {
72  				eliRecord.getPerson().getSocialSecurityDocument().setSsn(
73  						hqInquiryProcessContext.getDlApplication()
74  								.getApplicant().getSocialSecurityDocument()
75  								.getSsn());
76  			}
77  			if (eliRecord.getUserVerifiedSsnCode() != null) {
78  				hqInquiryProcessContext.getDlApplication()
79  						.setNewUserVerifiedSsnCode(
80  								eliRecord.getUserVerifiedSsnCode());
81  			}
82  			if (eliRecord.isAbsentParent() != null) {
83  				hqInquiryProcessContext.getDlApplication().setAbsentParent(
84  						eliRecord.isAbsentParent());
85  			}
86  			if (eliRecord.getApplicationPendingIndicator() != null) {
87  				hqInquiryProcessContext.getDlApplication()
88  						.setApplicationPendingIndicatorCode(
89  								eliRecord.getApplicationPendingIndicator());
90  			}
91  			Person applicant = PersonUtil.copyDemographicInfo(eliRecord
92  					.getPerson());
93  			hqInquiryProcessContext.getDlApplication().setApplicant(applicant);
94  			hqInquiryProcessContext.getEliRecord().setHitCounter(
95  					eliRecord.getHitCounter());
96  		}
97  	}

99  	/***
100 	 * This method returns a ANI record based on the license number
101 	 * @param aniRecords ANI Records
102 	 * @param licenseNumber license number
103 	 * @return AniRecord instance
104 	 */
105 	public static AniRecord getAniRecordWithLicenseNumber(
106 			List <AniRecord> aniRecords, String licenseNumber) {
107 		Iterator <AniRecord> iterator = aniRecords.iterator();
108 		while (iterator.hasNext()) {
109 			AniRecord aniRecord = iterator.next();
110 			String aniLicenseNumber = aniRecord.getDriverLicenseNumber();
111 			if (aniLicenseNumber.equals(licenseNumber)) {
112 				return aniRecord;
113 			}
114 		}
115 		return null;
116 	}

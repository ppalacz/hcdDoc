38  public class SearchRecordInLocalRepositoryActivity extends BusinessActivity {
39  	/*** The Constant serialVersionUID. */
40  	private static final long serialVersionUID = 47498410242653528L;
41  	private CdaTaskService cdaTaskService;
42  
43  	/*
44  	 * (non-Javadoc)
45  	 * 
46  	 * @see
47  	 * gov.ca.dmv.ease.app.activity.impl.BusinessActivity#invokeService(gov.
48  	 * ca.dmv.ease.app.context.impl.ProcessContext)
49  	 */
50  	@Override
51  	protected void invokeService(ProcessContext processContext) {
52  		HqInquiryProcessContext hqInquiryProcessContext = (HqInquiryProcessContext) processContext;
53  		License license = hqInquiryProcessContext.getDriverLicense();
54  		DafRecordInquiryRequestTs tsRequest = null;
55  		DafRecordInquiryResponseTs tsResponse = null;
56  		IErrorCollector errorCollector = new ErrorCollector();
57  		try {
58  			tsRequest = new DafRecordInquiryRequestTs(hqInquiryProcessContext
59  					.getUserContext(), license.getLicenseNumber());
60  			if (license instanceof DriverLicense) {
61  				tsRequest.setDafInquiryTypes(DafInquiryTypes.Driver_License);
62  			}
63  			else {
64  				tsRequest.setDafInquiryTypes(DafInquiryTypes.ID_Card);
65  			}
66  			tsResponse = cdaTaskService.execute(tsRequest);
67  			if (tsResponse.hasErrors()) {
68  				//this.populateErrorMessages(hqInquiryProcessContext, tsResponse.getErrorCollector());
69  			}
70  			else {
71  				ITransaction transaction = tsResponse.getTransaction();
72  				String licenseNumber = tsResponse.getTransaction()
73  						.getLicenseNumber();
74  				//Q556
75  				if (!((transaction == null) || StringUtil
76  						.isEmpty(licenseNumber))) {
77  					// This means there exists a record for the ID provided.
78  					// Add the error message
79  					errorCollector
80  							.register(new EaseValidationException(
81  									IInquiryErrorMessageConstants.DL_DAF_RECORD_ALREADY_CREATED_FOR_THIS_DL_NUMBER));
82  					populateErrorMessages(processContext, errorCollector);
83  				}
84  			}
85  		}
86  		catch (Exception e) {
87  			errorCollector.register(e);
88  			populateErrorMessages(processContext, errorCollector);
89  		}
90  	}
91

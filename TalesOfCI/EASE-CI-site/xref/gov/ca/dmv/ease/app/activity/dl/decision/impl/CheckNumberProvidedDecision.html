26  public class CheckNumberProvidedDecision extends DecisionActivity implements
27  		IDlaDecisionActivityConstantsForBpDefinition {
28  	/*** The Constant serialVersionUID. */
29  	private static final long serialVersionUID = 1L;
30  
31  	/* (non-Javadoc)
32  	 * @see gov.ca.dmv.ease.activity.DecisionActivity#evaluate(gov.ca.dmv.ease.app.context.ProcessContext)
33  	 */
34  	/***
35  	 * This method evaluates if license number is provided or not. If provided it returns true, if not returns false.
36  	 */
37  	@Override
38  	protected String evaluate(ProcessContext processContext) {
39  		CustomerInfoProcessContext customerInfoProcessContext = (CustomerInfoProcessContext) processContext;
40  		DriverLicense drivingLicense = customerInfoProcessContext
41  				.getDriverLicense();
42  		IdCard idCard = customerInfoProcessContext.getDlApplication()
43  				.getIdCard();
44  		String strClass = null;
45  		String strExpirationDate = null;
46  		if (drivingLicense != null && drivingLicense.getLicenseNumber() != null
47  				&& drivingLicense.getLicenseNumber().trim().length() > 0) {
48  			strClass = drivingLicense.getAppliedLicenseClasses().get(0)
49  					.getCode();
50  			if (strClass != null)
51  				return DL_NUMBER_PROVIDED;
52  		}
53  		else if (idCard != null && idCard.getIdNumber() != null
54  				&& idCard.getIdNumber().length() > 0) {
55  			strExpirationDate = idCard.getExpirationYear();
56  			if (strExpirationDate != null)
57  				return ID_NUMBER_PROVIDED;
58  		}
59  		else {
60  			customerInfoProcessContext.setDlaStartedwith("SearchPersonByName");
61  		}
62  		return NUMBER_NOT_PROVIDED;
63  	}
64  }

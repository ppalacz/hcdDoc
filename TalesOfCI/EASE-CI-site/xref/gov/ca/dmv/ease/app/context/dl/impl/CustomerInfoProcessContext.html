131 	public CustomerInfoProcessContext() {
132 		this(new DlApplication());
133 		getDlApplication().setNewLicenseNumberRequired(false);
134 		aniResults = new ArrayList <AniRecord>();
135 		Person person = new Person();
136 		person.setCurrentName(new PersonName());
137 		List <PersonName> personNames = new ArrayList <PersonName>();
138 		person.setPersonNames(personNames);
139 		person.setDriverLicense(new DriverLicense());
140 		person.getDriverLicense().setAppliedLicenseClasses(
141 				new ArrayList <CodeSetElement>());
142 		OutOfStateLicense outOfStateLicense = new OutOfStateLicense();
143 		outOfStateLicense.setName(new PersonName());
144 		person.setPreviousLicenses(new ArrayList <OutOfStateLicense>());
145 		person.addPreviousLicense(outOfStateLicense);
146 		getDlApplication().setApplicant(person);
147 		AniRecord aniRecord = new AniRecord();
148 		aniResults.add(aniRecord);
149 		lpvRecord = new LpvRecord();
150 		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
151 		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
152 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
153 		legalPresenceDocument.setVerificationCode(new CodeSetElement());
154 		List <Document> list = new ArrayList <Document>();
155 		Passport passport = new Passport();
156 		list.add(ssnDocument);
157 		list.add(legalPresenceDocument);
158 		list.add(passport);
159 		person.setDocuments(list);
160 		getDlApplication().setApplicationTypeCode(
161 				new CodeSetElement("", "", ""));
162 		Address mailingAddress = new Address(AddressType.MAILING);
163 		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
164 		person.addAddress(mailingAddress);
165 		person.addAddress(residenceAddress);
166 		Invoice invoice = new Invoice();
167 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
168 		invoice.setInvoiceItems(items);
169 		List <Payment> payments = new ArrayList <Payment>();
170 		invoice.setPayments(payments);
171 		CashPayment cashPayment = new CashPayment();
172 		payments.add(cashPayment);
173 		CheckPayment checkPayment = new CheckPayment();
174 		payments.add(checkPayment);
175 		DebitPayment debitPayment = new DebitPayment();
176 		payments.add(debitPayment);
177 		List <IProcessHistory> processHistory = new ArrayList <IProcessHistory>();
178 		getDlApplication().setProcessHistory(processHistory);
179 		getDlApplication().setInvoice(invoice);
180 		setChangedName(new PersonName());
181 	}

513 	/***
514 	 * This method returns Person object of the primary record from ANI response.
515 	 * 
516 	 * @return the primary ani record
517 	 */
518 	public AniRecord getPrimaryAniRecord() {
519 		Iterator <AniRecord> iterator = getAniRecords().iterator();
520 		while (iterator.hasNext()) {
521 			AniRecord aniRecord = iterator.next();
522 			if (aniRecord.isPrimary()) {
523 				return aniRecord;
524 			}
525 		}
526 		return null;
527 	}

43  	 * This method processes School Bus Restriction Code.
44  	 * Refer - DLA7_Restrictions.doc
45  	 * Invoke this after Collection Data Screen
46  	 * 
47  	 * Note - Existing Application Restrictions will be modified by this method.
48  	 * 
49  	 * @param dlApplication the Input DL Application
50  	 * @param schoolBusRestriction29 the School Bus Restriction
51  	 * 
52  	 * @return Updated List of Existing Applied Restrictions
53  	 * 
54  	 */
55  	public List <DrivingRestriction> processSchoolBusRestrictionCode(
56  			DlApplication dlApplication,
57  			DrivingRestriction schoolBusRestriction29) {
58  		List <CodeSetElement> applicationEndrosements = dlApplication
59  				.getLicense().getNewEndorsements();
60  		boolean isSchoolBusEndorsementExist = false;
61  		boolean isSchoolBusRestrictionExist = false;
62  		List <DrivingRestriction> applicationRestrictions = dlApplication
63  				.getLicense().getNewRestrictions();
64  		if (!EaseUtil.isNullOrBlank(applicationEndrosements)) {
65  			//Iterate through Application Endorsements
66  			for (CodeSetElement endorsement : applicationEndrosements) {
67  				if (IDlEndorsementCategories.SCHOOL_BUS
68  						.equalsIgnoreCase(endorsement.getCode())) {
69  					isSchoolBusEndorsementExist = true;
70  					break;
71  				}
72  			}
73  		}
74  		//If School Bus Endorsement is Present
75  		//Step 1 - If Restriction Code 29 not Present then Add Restriction Code 29
76  		if (!EaseUtil.isNullOrBlank(applicationRestrictions)) {
77  			for (DrivingRestriction drivingRestriction : applicationRestrictions) {
78  				if (IRestrictionCodesConstants.NO_STUDENTS_WITHOUT_CERT
79  						.equalsIgnoreCase(drivingRestriction
80  								.getRestrictionTypeCode().getCode())) {
81  					isSchoolBusRestrictionExist = true;
82  					break;
83  				}
84  			}
85  		}
86  		if (isSchoolBusEndorsementExist) {
87  			if (!isSchoolBusRestrictionExist
88  					&& EaseUtil.isNotNull(applicationRestrictions)) {
89  				//Add Restriction Code 29 - If 5 Restrictions exist then add in 5th Position
90  				if (applicationRestrictions.size() < IndicatorsConstants.RESTRICTIONS_MAX) {
91  					applicationRestrictions.add(schoolBusRestriction29);
92  				}
93  				else if (applicationRestrictions.size() >= 5) {
94  					applicationRestrictions.add(4, schoolBusRestriction29);
95  				}
96  			}
97  		}
98  		else if (isSchoolBusRestrictionExist) {
99  			//Remove Restriction Code 29
100 			applicationRestrictions.remove(schoolBusRestriction29);
101 		}
102 		return applicationRestrictions;
103 	}

261 	/***
262 	 * This method Validates Categories and Restrictions.
263 	 * If Restrictions that does not apply for Categories exist,
264 	 * then it removes it and returns updated list of Restrictions.
265 	 * Table 4 - Determine if there are restrictions that should be cleared.
266 	 * @param ttc the TTC
267 	 * @param applicationRestrictions the Application Restrictions
268 	 * @param applicationCategories the Application Categories
269 	 * 
270 	 * @return the List of updated Application Restrictions
271 	 */
272 	public List <DrivingRestriction> checkForValidRestrictions(String ttc,
273 			List <DrivingRestriction> applicationRestrictions,
274 			List <CodeSetElement> applicationCategories) {
275 		boolean isNonCommercialCategories = false;
276 		boolean isRestrictionInRangeExist = true;
277 		int replacementIndex = -1;
278 		if (IndicatorsConstants.TTC_DLA.equalsIgnoreCase(ttc)
279 				|| IndicatorsConstants.TTC_DLC.equalsIgnoreCase(ttc)) {
280 			//Check for any Non Commercial Restrictions
281 			if (!EaseUtil.isNullOrBlank(applicationCategories)) {
282 				for (CodeSetElement category : applicationCategories) {
283 					if (IndicatorsConstants.CATEGORY_1
284 							.equalsIgnoreCase(category.getCode())
285 							|| IndicatorsConstants.CATEGORY_2
286 									.equalsIgnoreCase(category.getCode())
287 							|| IndicatorsConstants.CATEGORY_3
288 									.equalsIgnoreCase(category.getCode())) {
289 						//Non Commercial Category Exist
290 						isNonCommercialCategories = true;
291 						break;
292 					}
293 				}
294 			}
295 			//Is there an Application Restriction = 71-73 or 80-87
296 			while (isRestrictionInRangeExist && isNonCommercialCategories) {
297 				replacementIndex = restrictionInRangeIndex(applicationRestrictions);
298 				isRestrictionInRangeExist = replacementIndex == -1 ? false
299 						: true;
300 				//Remove the Restriction
301 				if (isRestrictionInRangeExist) {
302 					applicationRestrictions.remove(replacementIndex);
303 				}
304 			}
305 		}
306 		return applicationRestrictions;
307 	}

310 	 * This method checks Application Restrictions and computes Previous/Existing Categories.
311 	 * Table 5 - Based upon the Restriction Codes of Record set the Previous Non-Commercial Categories 
312 	 * Perform tests in order and exit after first match.  
313 	 * These variables are used in document production, test results, etc. 
314 	 * 
315 	 * @param applicationRestrictions the Application Restrictions
316 	 * 
317 	 * @return the List of Previous/Existing Categories
318 	 */
319 	public List <CodeSetElement> computeExistingCategories(
320 			List <DrivingRestriction> applicationRestrictions) {
321 		List <CodeSetElement> existingCategories = new ArrayList <CodeSetElement>();
322 		CodeSetElement category1 = new CodeSetElement(
323 				IndicatorsConstants.CATEGORY_1, IndicatorsConstants.CATEGORY_1,
324 				IndicatorsConstants.CATEGORY_1);
325 		CodeSetElement category2 = new CodeSetElement(
326 				IndicatorsConstants.CATEGORY_2, IndicatorsConstants.CATEGORY_2,
327 				IndicatorsConstants.CATEGORY_2);
328 		CodeSetElement category3 = new CodeSetElement(
329 				IndicatorsConstants.CATEGORY_3, IndicatorsConstants.CATEGORY_3,
330 				IndicatorsConstants.CATEGORY_3);
331 		//If Restriction is either 71,72 or 73 - Add Category 3
332 		if (checkIfRestrictionExist(applicationRestrictions,
333 				fetchDrivingRestriction(71))
334 				|| checkIfRestrictionExist(applicationRestrictions,
335 						fetchDrivingRestriction(72))
336 				|| checkIfRestrictionExist(applicationRestrictions,
337 						fetchDrivingRestriction(73))) {
338 			existingCategories.add(category3);
339 		}
340 		//If Restriction is either 80, 81 or 82 - Add Category 2
341 		else if (checkIfRestrictionExist(applicationRestrictions,
342 				fetchDrivingRestriction(80))
343 				|| checkIfRestrictionExist(applicationRestrictions,
344 						fetchDrivingRestriction(81))
345 				|| checkIfRestrictionExist(applicationRestrictions,
346 						fetchDrivingRestriction(82))) {
347 			existingCategories.add(category2);
348 		}
349 		//If Restriction is either 83 or 84 - Add Category 1
350 		else if (checkIfRestrictionExist(applicationRestrictions,
351 				fetchDrivingRestriction(83))
352 				|| checkIfRestrictionExist(applicationRestrictions,
353 						fetchDrivingRestriction(84))) {
354 			existingCategories.add(category1);
355 		}
356 		//If Restriction is either 85, 86 or 87 - Add Category 1 & 3
357 		else if (checkIfRestrictionExist(applicationRestrictions,
358 				fetchDrivingRestriction(85))
359 				|| checkIfRestrictionExist(applicationRestrictions,
360 						fetchDrivingRestriction(86))
361 				|| checkIfRestrictionExist(applicationRestrictions,
362 						fetchDrivingRestriction(87))) {
363 			existingCategories.add(category1);
364 			existingCategories.add(category3);
365 		}
366 		return existingCategories;
367 	}

404 	 * @param toBeCheckedRestriction the Input Restriction to Check

408 	private boolean checkIfRestrictionExist(
409 			List <DrivingRestriction> applicationRestrictions,
410 			DrivingRestriction toBeCheckedRestriction) {
411 		boolean isRestrictionExist = false;
412 		//Null Check
413 		if (!EaseUtil.isNullOrBlank(applicationRestrictions)
414 				&& EaseUtil.isNotNull(toBeCheckedRestriction)
415 				&& !EaseUtil.isNullOrBlank(toBeCheckedRestriction
416 						.getRestrictionTypeCode())) {
417 			//Iterate Driving Restrictions
418 			for (DrivingRestriction drivingRestriction : applicationRestrictions) {
419 				if (!EaseUtil.isNullOrBlank(drivingRestriction
420 						.getRestrictionTypeCode())) {
421 					if (drivingRestriction
422 							.getRestrictionTypeCode()
423 							.getCode()
424 							.equalsIgnoreCase(
425 									toBeCheckedRestriction
426 											.getRestrictionTypeCode().getCode())) {
427 						isRestrictionExist = true;
428 						break;
429 					}
430 				}
431 			}
432 		}
433 		return isRestrictionExist;
434 	}

436 	/***
437 	 * This method will Check and then Append Driving Restrictions.
438 	 * 
439 	 * @param applicationRestrictions the Application Restrictions
440 	 * @param toBeAddedRestriction the Driving Restriction to be Appended
441 	 * 
442 	 * @return Updated List of Existing Applied Restrictions
443 	 */
444 	private List <DrivingRestriction> checkAndAppendRestriction(
445 			List <DrivingRestriction> applicationRestrictions,
446 			DrivingRestriction toBeAddedRestriction) {
447 		boolean isRestrictionInRangeExist = false;
448 		int replacementIndex = -1;
449 		if (EaseUtil.isNotNull(applicationRestrictions)) {
450 			//Step 1 - If Restrictions between 71-73 && 80-87 already exists then replace with new Restriction
451 			replacementIndex = restrictionInRangeIndex(applicationRestrictions);
452 			isRestrictionInRangeExist = replacementIndex == -1 ? false : true;
453 			if (isRestrictionInRangeExist) {
454 				//Replace the Restriction
455 				applicationRestrictions.add(replacementIndex,
456 						toBeAddedRestriction);
457 			}
458 			else if (applicationRestrictions.size() < IndicatorsConstants.RESTRICTIONS_MAX) {
459 				//Add to the Restrictions
460 				applicationRestrictions.add(toBeAddedRestriction);
461 			}
462 			else if (applicationRestrictions.size() == IndicatorsConstants.RESTRICTIONS_MAX) {
463 				//Add to the Restrictions at 5th Position
464 				applicationRestrictions.add(
465 						IndicatorsConstants.RESTRICTIONS_MAX - 1,
466 						toBeAddedRestriction);
467 			}
468 		}
469 		return applicationRestrictions;
470 	}

472 	/***
473 	 * This method checks and returns the index of Restriction 71-73 Or 80-87.
474 	 * 
475 	 * @param applicationRestrictions the Application Restrictions
476 	 * 
477 	 * @return the index of the matching Restriction
478 	 */
479 	private int restrictionInRangeIndex(
480 			List <DrivingRestriction> applicationRestrictions) {
481 		int restrictionToReplaceIndex = -1;
482 		String restrictionCode = null;
483 		DrivingRestriction drivingRestriction = null;
484 		boolean isMatchFound = false;
485 		if (!EaseUtil.isNullOrBlank(applicationRestrictions)) {
486 			//Iterate Restrictions
487 			Iterator <DrivingRestriction> iterator = applicationRestrictions
488 					.iterator();
489 			while (iterator.hasNext()) {
490 				restrictionToReplaceIndex++;
491 				drivingRestriction = iterator.next();
492 				if (!EaseUtil.isNullOrBlank(drivingRestriction
493 						.getRestrictionTypeCode())) {
494 					restrictionCode = drivingRestriction
495 							.getRestrictionTypeCode().getCode();
496 					//If between 71-73 Or 80-87
497 					if (CLASS_A_SINGLE_TRAILER_NO_PAY
498 							.equalsIgnoreCase(restrictionCode)
499 							|| CLASS_A_SINGLE_TRAILER_PAY_ONLY
500 									.equalsIgnoreCase(restrictionCode)
501 							|| CLASS_AC_SINGLE_TRAILER_NO_PAY
502 									.equalsIgnoreCase(restrictionCode)
503 							|| CLASS_B_FIREFIGHTING_ONLY
504 									.equalsIgnoreCase(restrictionCode)
505 							|| CLASS_A_FIREFIGHTING_ONLY
506 									.equalsIgnoreCase(restrictionCode)
507 							|| CLASS_BC_FIREFIGHTING_ONLY
508 									.equalsIgnoreCase(restrictionCode)
509 							|| CLASS_AC_FIREFIGHTING_ONLY
510 									.equalsIgnoreCase(restrictionCode)
511 							|| CLASS_AB_FIREFIGHTING_ONLY
512 									.equalsIgnoreCase(restrictionCode)
513 							|| CLASS_A_FIREFIGHTING_OR_TOW_SINGLE
514 									.equalsIgnoreCase(restrictionCode)
515 							|| CLASS_AB_FIREFIGHTING_OR_TOW_SINGLE
516 									.equalsIgnoreCase(restrictionCode)
517 							|| CLASS_AC_FIREFIGHTING_OR_TOW_SINGLE
518 									.equalsIgnoreCase(restrictionCode)) {
519 						isMatchFound = true;
520 						break;
521 					}
522 				}
523 			}
524 		}
525 		if (!isMatchFound) {
526 			restrictionToReplaceIndex = -1;
527 		}
528 		return restrictionToReplaceIndex;
529 	}

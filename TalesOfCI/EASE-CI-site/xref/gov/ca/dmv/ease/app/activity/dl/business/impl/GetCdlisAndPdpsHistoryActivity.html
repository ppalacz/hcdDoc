35  	/***
36  	 * This method creates the GetCdlisAndPdpsResponseRequestTs request class
37  	 * with required data from the process context using factory and calls DL
38  	 * Task Service to get the Cdlis and Pdps history. Once the
39  	 * response is received from the Task Service it maps it back to the process
40  	 * context.
41  	 * 
42  	 */
43  	@Override
44  	protected void invokeService(ProcessContext context) {
45  		CdlisPdpsProcessContext cdlisPdpsProcessContext = (CdlisPdpsProcessContext) context;
46  		// Sending Asynchronous Response Promise to get CDLIS/PDPS inquiry
47  		// response
48  		IDlTaskServiceRequestFactory reqFactory = new DlTaskServiceRequestFactory();
49  		SendCdlisAndPdpsRequestTs requestTs = reqFactory
50  				.createSendPdpsAndCdlisInquiryRequest(cdlisPdpsProcessContext
51  						.getDriverLicense(), cdlisPdpsProcessContext
52  						.getDlApplicant(), cdlisPdpsProcessContext
53  						.getDlApplication().getSocialSecurityDocument(),
54  						cdlisPdpsProcessContext.getUserContext());
55  		CdlisAndPdpsHistoryResponseTs responseTs = issueDlTaskService
56  				.getCdlisAndPdpsResponse(requestTs);
57  		if (responseTs.getErrorCollector() != null
58  				&& responseTs.getErrorCollector().hasErrors()) {
59  			//TODO Now what? 
60  		}
61  		else {
62  			if (responseTs.getCdlisRecordHits().size() > 1) {
63  				cdlisPdpsProcessContext.setCdlisResultHits(responseTs
64  						.getCdlisRecordHits());
65  				cdlisPdpsProcessContext.setPdpsResultHits(responseTs
66  						.getPdpsRecordHits());
67  			}
68  			else {
69  				cdlisPdpsProcessContext.setCdlisResultHitsCdu(responseTs
70  						.getCdlisRecordHits());
71  				cdlisPdpsProcessContext.setPdpsResultHitsCdu(responseTs
72  						.getPdpsRecordHits());
73  			}
74  		}
75  	}

29  public class DlPrintRequestFallback extends DecisionActivity {
30  	/*** The Constant serialVersionUID. */
31  	private static final long serialVersionUID = -4120078087961900186L;
32  
33  	/*
34  	 * (non-Javadoc)
35  	 * 
36  	 * @see
37  	 * gov.ca.dmv.ease.app.activity.impl.DecisionActivity#evaluate(gov.ca.dmv
38  	 * .ease.app.context.impl.ProcessContext)
39  	 * 
40  	 * @return String, if return "license_type_A-E" fallback to test result 2
41  	 * screen, if return "license_type_FandM" fallback to test result 1 screen.
42  	 * license_type_A-E license_type_FandM
43  	 */
44  	@Override
45  	protected String evaluate(ProcessContext context) {
46  		DlProcessContext processContext = (DlProcessContext) context;
47  		String licenseClass = ICdaADecisionActivityConstants.NON_COMMERCIAL_LICENSE_TYPE;
48  		if (processContext.getDlApplication().getApplicant() != null) {
49  			List <CodeSetElement> licenseClasses = processContext
50  					.getDlApplication().getApplicant().getDriverLicense()
51  					.getAppliedLicenseClasses();
52  			if (DlApplicationUtil.isCommercialLicense(licenseClasses)) {
53  				licenseClass = ICdaADecisionActivityConstants.COMMERCIAL_LICENSE_TYPE;
54  			}
55  		}
56  		return licenseClass;
57  	}
58  }

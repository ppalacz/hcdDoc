41  		CdaProcessContext cdaContext = (CdaProcessContext) context;
42  		PrintRequestTs tsRequest = null;
43  		//FIXME - hardcoding of the activity class name is not acceptable - use polymorphism instead
44  		if (context.getCurrentInteractionActivity().getClass().getSimpleName()
45  				.equalsIgnoreCase("CollectTestResults1Activity")) {
46  			tsRequest = new PrintRequestTs(cdaContext.getUserContext(),
47  					cdaContext.getDlApplication());
48  			Set <PrintRequestTypes> printRequestTypes = new HashSet <PrintRequestTypes>();
49  			printRequestTypes.add(PrintRequestTypes.TEST_RESULTS_1_LDD0631);
50  			tsRequest.setPrintRequestTypes(printRequestTypes);
51  		}
52  		//FIXME - hardcoding of the activity class name is not acceptable - use polymorphism instead
53  		else if (context.getCurrentInteractionActivity().getClass()
54  				.getSimpleName()
55  				.equalsIgnoreCase("CollectTestResults2Activity")) {
56  			tsRequest = new PrintRequestTs(cdaContext.getUserContext(),
57  					cdaContext.getDlApplication());
58  			Set <PrintRequestTypes> printRequestTypes = new HashSet <PrintRequestTypes>();
59  			printRequestTypes.add(PrintRequestTypes.TEST_RESULTS_2_LDD0632);
60  			tsRequest.setPrintRequestTypes(printRequestTypes);
61  		}
62  		PrintResponseTs tsResponse = cdaTaskService.execute(tsRequest);
63  		if (tsResponse.hasErrors()) {
64  			populateErrorMessages(cdaContext, tsResponse.getErrorCollector());
65  		}
66  	}

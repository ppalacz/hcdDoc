145 	public HqInquiryProcessContext() {
146 		dlApplication = new DlApplication();
147 		dlApplication.setNewLicenseNumberRequired(false);
148 		Person person = new Person();
149 		IdCard idCard = new IdCard();
150 		person.setIdCard(idCard);
151 		person.setCurrentName(new PersonName());
152 		List <PersonName> personNames = new ArrayList <PersonName>();
153 		personNames.add(new PersonName());
154 		person.setPersonNames(personNames);
155 		person.setDriverLicense(new DriverLicense());
156 		person.getDriverLicense().setAppliedLicenseClasses(
157 				new ArrayList <CodeSetElement>());
158 		dlApplication.setApplicant(person);
159 		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
160 		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
161 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
162 		legalPresenceDocument.setVerificationCode(new CodeSetElement());
163 		List <Document> list = new ArrayList <Document>();
164 		list.add(ssnDocument);
165 		list.add(legalPresenceDocument);
166 		person.setDocuments(list);
167 		Address mailingAddress = new Address(AddressType.MAILING);
168 		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
169 		person.addAddress(mailingAddress);
170 		person.addAddress(residenceAddress);
171 		Invoice invoice = new Invoice();
172 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
173 		invoice.setInvoiceItems(items);
174 		List <Payment> payments = new ArrayList <Payment>();
175 		invoice.setPayments(payments);
176 		aniResults = new ArrayList <AniRecord>();
177 		setAniResults(aniResults);
178 		dlApplication.setInvoice(invoice);
179 		SalesPersonApplication salesPersonApplication = new SalesPersonApplication();
180 		VehicleSalesPerson salesPerson = new VehicleSalesPerson();
181 		salesPerson.setCurrentName(new PersonName());
182 		List <PersonName> spPersonNames = new ArrayList <PersonName>();
183 		spPersonNames.add(new PersonName());
184 		salesPerson.setPersonNames(spPersonNames);
185 		salesPerson.setDriverLicense(new DriverLicense());
186 		salesPerson.getDriverLicense().setAppliedLicenseClasses(
187 				new ArrayList <CodeSetElement>());
188 		Address spMailingAddress = new Address(AddressType.MAILING);
189 		Address spResidenceAddress = new Address(AddressType.RESIDENTIAL);
190 		salesPerson.addAddress(spMailingAddress);
191 		salesPerson.addAddress(spResidenceAddress);
192 		salesPersonApplication.setApplicant(salesPerson);
193 		setSalesPersonApplication(salesPersonApplication);
194 	}

364 	 * This method returns Person object of the primary record from ANI response.
365 	 * 
366 	 * @return the primary ani record
367 	 */
368 	public AniRecord getPrimaryAniRecord() {
369 		Iterator <AniRecord> iterator = getAniRecords().iterator();
370 		while (iterator.hasNext()) {
371 			AniRecord aniRecord = iterator.next();
372 			if (aniRecord.isPrimary()) {
373 				return aniRecord;
374 			}
375 		}
376 		return null;
377 	}

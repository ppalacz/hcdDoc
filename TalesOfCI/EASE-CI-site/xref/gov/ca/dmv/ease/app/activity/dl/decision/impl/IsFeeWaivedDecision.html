31  	/***
32  	 * This method evaluates if fee has been waived. 
33  	 * If the fee is waived it returns true, if not returns false.
34  	 */
35  	@Override
36  	protected String evaluate(ProcessContext context) {
37  		Boolean isFeeRequired = null;
38  		if (context instanceof DlFeeProcessContext) {
39  			DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) context;
40  			SalesPersonApplication salesPersonApplication = dlFeeProcessContext
41  					.getSalesPersonApplication();
42  			DlApplication dlApplication = dlFeeProcessContext
43  					.getDlApplication();
44  			if (salesPersonApplication != null) {
45  				isFeeRequired = salesPersonApplication.isFeeRequired();
46  			}
47  			else if (dlApplication != null) {
48  				isFeeRequired = dlApplication.isFeeRequired();
49  			}
50  		}
51  		/*	if(context instanceof SpProcessContext) {
52  				SpProcessContext spProcessContext  = (SpProcessContext) context;
53  				isFeeRequired = spProcessContext.getSalesPersonApplication().isFeeRequired();			
54  			} else if (context instanceof DlaProcessContext) {
55  				DlaProcessContext dlaProcessContext = (DlaProcessContext) context;
56  				isFeeRequired = dlaProcessContext.getDlApplication().isFeeRequired();			
57  			}*/
58  		if (isFeeRequired != null && isFeeRequired) {
59  			return FEE_NOT_WAIVED;
60  		}
61  		return FEE_WAIVED;
62  	}
63  }

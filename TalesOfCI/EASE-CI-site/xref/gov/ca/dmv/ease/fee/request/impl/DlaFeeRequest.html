77  		StringBuilder strBuilder = new StringBuilder();
78  		String ruleLicense1Type = CLASSIFIERTYPES.CLASS.type();
79  		String ruleLicense1 = licenseClass1.type();
80  		String ruleLicense2Type = CLASSIFIERTYPES.CLASS.type();
81  		String ruleLicense2 = licenseClass2.type();
82  		String ruleOriginalRenewalType = CLASSIFIERTYPES.ORIGINALRENEWAL.type();
83  		String ruleOriginalRenewal = orginalRenewal.type();
84  		String ruleCategoryType = CLASSIFIERTYPES.CATEGORY.type();
85  		String ruleCategory = category.type();
86  		strBuilder.append(ruleLicense1Type).append(EQUALS).append(ruleLicense1)
87  				.append(COLONDELIMITER);
88  		strBuilder.append(ruleLicense2Type).append(EQUALS).append(ruleLicense2)
89  				.append(COLONDELIMITER);
90  		strBuilder.append(ruleOriginalRenewalType).append(EQUALS).append(
91  				ruleOriginalRenewal).append(COLONDELIMITER);
92  		strBuilder.append(ruleCategoryType).append(EQUALS).append(ruleCategory);
93  		String keyClassifier = strBuilder.toString();
94  		return keyClassifier;
95  	}

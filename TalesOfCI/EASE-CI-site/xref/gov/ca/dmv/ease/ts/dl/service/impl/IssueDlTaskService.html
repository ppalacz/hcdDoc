183 	 * Execute.
184 	 * 
185 	 * @param tsRequest
186 	 *            the request
187 	 * 
188 	 * @return the calculate dl fee response
189 	 */
190 	public CalculateDlFeeResponseTs execute(CalculateDlFeeRequestTs tsRequest) {
191 		CalculateDlFeeResponseTs tsResponse = null;
192 		CalculateDlFeeRequestBus busRequest = null;
193 		CalculateDlFeeResponseBus busResponse = null;
194 		IBusDlRequestFactory requestFactory = new BusDlRequestFactory();
195 		IDlTaskServiceResponseFactory responseFactory = new DlTaskServiceResponseFactory();
196 		IErrorCollector collector = new ErrorCollector();
197 		try {
198 			//collector = tsRequest.validate();
199 			if (collector.hasErrors()) {
200 				tsResponse = responseFactory
201 						.createCalculateDlFeeResponseForTs(collector);
202 			}
203 			else {
204 				busRequest = requestFactory.createCalculateDlFeeRequestForBus(
205 						tsRequest.getDlApplication(), tsRequest
206 								.getUserContext());
207 				busResponse = getFeeCalculationService().execute(busRequest);
208 				if (busResponse.hasErrors()) {
209 					tsResponse = responseFactory
210 							.createCalculateDlFeeResponseForTs(busResponse
211 									.getErrorCollector());
212 				}
213 				else {
214 					tsResponse = responseFactory
215 							.createCalculateDlFeeResponseForTs(busResponse
216 									.getFee());
217 				}
218 			}
219 		}
220 		catch (Exception exception) {
221 			// Register exception in error collector and assign it to response
222 			// object.
223 			collector.register(exception);
224 			tsResponse = responseFactory
225 					.createCalculateDlFeeResponseForTs(collector);
226 			return tsResponse;
227 		}
228 		return tsResponse;
229 	}

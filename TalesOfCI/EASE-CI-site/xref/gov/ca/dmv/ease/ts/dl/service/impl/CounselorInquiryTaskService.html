44  	/***
45  	 * @param tsRequest
46  	 * @return
47  	 */
48  	public CounselorInquiryForDisplayResponseTs execute(
49  			CounselorInquiryForDisplayRequestTs tsRequest) {
50  		CounselorInquiryForDisplayResponseTs tsResponse = null;
51  		CounselorInquiryForDisplayRequestBus busRequest = null;
52  		CounselorInquiryForDisplayResponseBus busResponse = null;
53  		IErrorCollector collector = new ErrorCollector();
54  		try {
55  			//collector = tsRequest.validate();
56  			if (collector.hasErrors()) {
57  				tsResponse = new CounselorInquiryForDisplayResponseTs(collector);
58  			}
59  			else {
60  				busRequest = new CounselorInquiryForDisplayRequestBus(tsRequest
61  						.getUserContext(), tsRequest.getDriverLicense());
62  				busResponse = counselorInquiryService.execute(busRequest);
63  				if (busResponse.hasErrors()) {
64  					tsResponse = new CounselorInquiryForDisplayResponseTs(
65  							busResponse.getErrorCollector());
66  				}
67  				else {
68  					tsResponse = new CounselorInquiryForDisplayResponseTs(
69  							busResponse.getInquiryReturnData(), busResponse
70  									.getDisplayStatusInfoCode());
71  				}
72  			}
73  		}
74  		catch (Exception e) {
75  			// Register exception in error collector and assign it to response
76  			// object.
77  			collector.register(e);
78  			tsResponse = new CounselorInquiryForDisplayResponseTs(collector);
79  		}
80  		return tsResponse;
81  	}

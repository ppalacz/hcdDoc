54  	 * Verifies the validness of the given driver license/id number
55  	 * 
56  	 * @param tsRequest
57  	 * @param tsResponse
58  	 */
59  	public AddressChangeDlIdInfoResponseTs execute(
60  			AddressChangeDlIdInfoRequestTs tsRequest) {
61  		AddressChangeDlIdInfoResponseTs tsResponse = null;
62  		AddressChangeDlIdInfoRequestBus busRequest = null;
63  		AddressChangeDlIdInfoResponseBus busResponse = null;
64  		IErrorCollector collector = new ErrorCollector();
65  		try {
66  			//collector = tsRequest.validate();
67  			if (collector.hasErrors()) {
68  				tsResponse = new AddressChangeDlIdInfoResponseTs(collector);
69  			}
70  			else {
71  				Transaction transaction = new Transaction();
72  				transaction.setOfficeId(tsRequest.getUserContext()
73  						.getOfficeId());
74  				busRequest = new AddressChangeDlIdInfoRequestBus(tsRequest
75  						.getUserContext(), tsRequest.getPerson());
76  				busResponse = miscellaneousBusinessUtilityService
77  						.execute(busRequest);
78  				if (busResponse.hasErrors()) {
79  					tsResponse = new AddressChangeDlIdInfoResponseTs(
80  							busResponse.getErrorCollector());
81  				}
82  				else {
83  					tsResponse = new AddressChangeDlIdInfoResponseTs(
84  							busResponse.getDlmRecord());
85  				}
86  			}
87  		}
88  		catch (Exception e) {
89  			// Register exception in error collector and assign it to response
90  			// object.
91  			collector.register(e);
92  			tsResponse = new AddressChangeDlIdInfoResponseTs(collector);
93  		}
94  		return tsResponse;
95  	}

185 		CancelDlResponseBus busResponse = null;

195 			else {

197 						cancelDlRequestTs.getPerson(), cancelDlRequestTs
198 								.getSubCode(), cancelDlRequestTs
199 								.getUserContext());
200 				busResponse = miscellaneousBusinessUtilityService
201 						.execute(requestsForBus);
202 				if (busResponse.hasErrors()) {
203 					tsResponse = new CancelDlResponseTs(busResponse
204 							.getErrorCollector());
205 				}
206 				else {
207 					//tsResponse = new CancelDlResponseTs(busResponse.getEliRecord());
208 					tsResponse = new CancelDlResponseTs(busResponse
209 							.getPersonName(), busResponse.getResponseInfoCode());
210 				}
211 			}
212 		}
213 		catch (PersonInquiryRequestValidationException exception) {
214 			// Register exception in error collector and assign it to response

216 			collector.register(exception);

38  public abstract class AbstractDlInformationSearchConverter extends
39  		AbstractEcsMessageConverter {
40  	/*** The variable TCode */

45  	/***
46  	 * Gets the VR Info Code
47  	 * @return vr info code
48  	 */
49  	private static String getVrInfoCode(
50  			VrInformationRequest vrInformationRequest) {
51  		if (!EaseUtil.isNullOrBlank(vrInformationRequest.getInfoCode())) {
52  			return vrInformationRequest.getInfoCode().getCode();
53  		}
54  		else {
55  			return IEcsConverterConstants.INFO_CODE_4;
56  		}
57  	}

59  	/***
60  	 * Gets the Vehicle File Code
61  	 * @return the vehicle file code
62  	 */
63  	private static String getVehicleFileCode(VehicleRecord vehicleRecord) {
64  		if (!EaseUtil.isNullOrBlank(vehicleRecord.getVehicleFileCode())) {
65  			return vehicleRecord.getVehicleFileCode().getCode();
66  		}
67  		else
68  			return BLANK;
69  	}
70  
71  	/***
72  	 * Gets the get vehicle vin if present
73  	 * @return
74  	 */
75  	private static String getVehicleVin(VehicleRecord vehicleRecord) {
76  		if (!EaseUtil.isNullOrBlank(vehicleRecord.getVehicleVin())) {
77  			return vehicleRecord.getVehicleVin();
78  		}
79  		else {
80  			return BLANK;
81  		}
82  	}

87  	@Override
88  	public String createMessage(IEcsRequest request) {
89  		AbstractDlInformationSearchRequestEcs requestFromClient = (AbstractDlInformationSearchRequestEcs) request;
90  		VrInformationRequest vrInformationRequest = requestFromClient
91  				.getVrInformationRequest();
92  		VehicleRecord vehicleRecord = vrInformationRequest.getVehicleRecord();
93  		populateMandatorySegment(requestFromClient);
94  		if (TCODE.equalsIgnoreCase(IEcsConverterConstants.TCODE_R62)) {
95  			transformRequestForR62(vrInformationRequest, vehicleRecord);
96  		}
97  		else if (TCODE.equalsIgnoreCase(IEcsConverterConstants.TCODE_RHC)
98  				|| TCODE.equalsIgnoreCase(IEcsConverterConstants.TCODE_RHI)) {
99  			transformRequestForRHCAndRHI(vrInformationRequest, vehicleRecord);
100 		}
101 		Object[] createdMessage = new Object[] { message };
102 		return createMessageFormat(createdMessage);
103 	}

105 	/***
106 	 * transforms the request for TCode, R62.
107 	 */
108 	private void transformRequestForR62(
109 			VrInformationRequest vrInformationRequest,
110 			VehicleRecord vehicleRecord) {
111 		if (!EaseUtil.isNullOrBlank(vehicleRecord
112 				.getVehicleLicensePlateNumber())) {
113 			populateSegment2(vrInformationRequest, vehicleRecord);
114 		}
115 		else if (!EaseUtil.isNullOrBlank(vehicleRecord.getVehicleVin())) {
116 			populateSegment3(vrInformationRequest, vehicleRecord);
117 		}
118 	}

120 	/***
121 	 * populates the mandatory initial segment for TCodes R62, RHC and RHI
122 	 * @param requestFromClient
123 	 */
124 	private void populateMandatorySegment(
125 			AbstractDlInformationSearchRequestEcs requestFromClient) {
126 		message.append(TCODE);
127 		message.append(IEcsConverterConstants.CONSTANT_99933);
128 		message.append(filler(requestFromClient.getUserContext().getOfficeId(),
129 				3));
130 	}
131 
132 	/*** 
133 	 * populates the R62 segment 2 for vehicle license plate #
134 	 */
135 	private void populateSegment2(VrInformationRequest vrInformationRequest,
136 			VehicleRecord vehicleRecord) {
137 		message.append(filler(getVrInfoCode(vrInformationRequest), 1));
138 		message.append(filler(getVehicleFileCode(vehicleRecord), 1));
139 		message.append(filler(vehicleRecord.getVehicleLicensePlateNumber(), 7));
140 		message.append(convertDateToString(vehicleRecord.getVehicleAsOfDate()));
141 	}
142 
143 	/***
144 	 * populates the R62 segment 3 for VIN
145 	 */
146 	private void populateSegment3(VrInformationRequest vrInformationRequest,
147 			VehicleRecord vehicleRecord) {
148 		message.append(filler(getVrInfoCode(vrInformationRequest), 1));
149 		message.append(filler(IEcsConverterConstants.CONSTANT_V, 1));
150 		message.append(getVehicleVin(vehicleRecord));
151 		message.append(convertDateToString(vehicleRecord.getVehicleAsOfDate()));
152 	}
153 
154 	/***
155 	 * transforms the request for TCodes RHC and RHI.
156 	 */
157 	private void transformRequestForRHCAndRHI(
158 			VrInformationRequest vrInformationRequest,
159 			VehicleRecord vehicleRecord) {
160 		message.append(filler(getVrInfoCode(vrInformationRequest), 1));
161 		message.append(filler(getVehicleFileCode(vehicleRecord), 1));
162 		message.append(filler(vehicleRecord.getVehicleLicensePlateNumber(), 7));
163 		if (TCODE.equalsIgnoreCase(IEcsConverterConstants.TCODE_RHI)) {
164 			message.append(filler(convertDateToString(vehicleRecord
165 					.getVehicleAsOfDate()), 6));
166 		}
167 		message.append(filler(getVehicleVin(vehicleRecord), 3));
168 	}

170 	/* (non-Javadoc)
171 	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createResponse(java.lang.String)
172 	 */
173 	@Override
174 	public IEcsResponse createResponseFromMultiplePayloads(
175 			List <String> inboundMessagePayloads) {
176 		StringBuilder strBuilder = new StringBuilder();
177 		for (String payload : inboundMessagePayloads) {
178 			String ebcidic = null;
179 			String screen = null;
180 			String formattedScreen = null;
181 			try {
182 				ebcidic = new String(payload.getBytes("Cp1047"));
183 				screen = Decoder3270.decode(ebcidic);
184 				formattedScreen = Decoder3270.formatScreen(screen);
185 				strBuilder.append(formattedScreen).append(LINE_SEPARATOR);
186 			}
187 			catch (UnsupportedEncodingException e) {
188 				throw new MessageConversionException(e);
189 			}
190 		}
191 		DataStreamResponseEcs response = new DataStreamResponseEcs();
192 		String formatResponse = StringUtils
193 				.trimTrailingWhitespace(removeDuplicateHeader(strBuilder));
194 		response.setOutputText(formatResponse);
195 		return response;
196 	}

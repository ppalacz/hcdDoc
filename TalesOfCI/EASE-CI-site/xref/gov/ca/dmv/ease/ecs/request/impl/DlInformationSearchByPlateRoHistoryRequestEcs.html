56  	/* (non-Javadoc)
57  	 * @see gov.ca.dmv.ease.ecs.request.impl.SynchronousEcsRequest#execute()
58  	 */
59  	@Override
60  	public DataStreamResponseEcs execute() throws EcsServiceException {
61  		DataStreamResponseEcs dataStreamResponseEcs = null;
62  		IErrorCollector errorCollector = getValidator().validate(this);
63  		if (errorCollector.hasErrors()) {
64  			dataStreamResponseEcs = new DataStreamResponseEcs(errorCollector);
65  		}
66  		else {
67  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
68  			//In the catch block we need to register the error to the error collector.
69  			try {
70  				dataStreamResponseEcs = (DataStreamResponseEcs) super.execute();
71  			}
72  			catch (EcsResponseIsNullException responseIsNullException) {
73  				errorCollector = new ErrorCollector();
74  				errorCollector.register(new EaseValidationException(
75  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
76  				dataStreamResponseEcs = new DataStreamResponseEcs(
77  						errorCollector);
78  			}
79  			catch (SystemException sysEx) {
80  				errorCollector = new ErrorCollector();
81  				errorCollector.register(new EaseValidationException(
82  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
83  				dataStreamResponseEcs = new DataStreamResponseEcs(
84  						errorCollector);
85  			}
86  		}
87  		return dataStreamResponseEcs;
88  	}

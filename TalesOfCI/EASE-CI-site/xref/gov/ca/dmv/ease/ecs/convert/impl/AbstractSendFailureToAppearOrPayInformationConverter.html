36  public abstract class AbstractSendFailureToAppearOrPayInformationConverter
37  		extends AbstractEcsMessageConverter {
38  	/*** The converter TCode. */
39  	protected String tCode = BLANK;
40  
41  	/* (non-Javadoc)
42  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
43  	 */
44  	@Override
45  	public String createMessage(IEcsRequest requestFromClient) {
46  		SendFailureToAppearOrPayInformationRequestEcs request = (SendFailureToAppearOrPayInformationRequestEcs) requestFromClient;
47  		IUserContext currentUserContext = request.getUserContext();
48  		Person person = request.getPerson();
49  		CitationDetail citationDetail = request.getCitationDetail();
50  		String controlCashierSequenceNo = request.getControlCashierSequenceNo();
51  		Object[] message = new Object[] {
52  				tCode,
53  				currentUserContext.getTtc(),
54  				currentUserContext.getOfficeId(),
55  				currentUserContext.getTechId(),
56  				filler(person.getDriverLicense().getLicenseNumber(), 8),
57  				ConverterHelper.getFirstThreeCharsOfLastName(person),
58  				filler(convertDateToString(person.getBirthDate()), 6),
59  				filler(citationDetail.getDocketNumber(), 17),
60  				filler(convertDateToString(citationDetail.getViolationDate()),
61  						6),
62  				filler(getSectionsViolated(citationDetail), 49),
63  				filler(getConvictionDate(currentUserContext, citationDetail), 6),
64  				filler(getCourtCode(citationDetail), 5),
65  				getWorkDate(currentUserContext),
66  				getWorkDate(currentUserContext),
67  				filler(controlCashierSequenceNo, 4),
68  				getFineAmount(citationDetail.getFineAmount()),
69  				IEcsConverterConstants.CONSTANT_SERVICE_FEE };
70  		return createMessageFormat(message);
71  	}
72  
73  	/***
74  	 * Gets the statute code and section violation number from sections violated
75  	 * @return
76  	 */
77  	private String getSectionsViolated(CitationDetail citationDetail) {
78  		List <SectionViolated> sectionsViolated = citationDetail
79  				.getSectionsViolated();
80  		StringBuilder statuteCodeWithViolation = new StringBuilder();
81  		for (SectionViolated sectionViolated : sectionsViolated) {
82  			if (!EaseUtil.isNullOrBlank(sectionViolated.getStatuteCode())) {
83  				statuteCodeWithViolation.append(sectionViolated
84  						.getStatuteCode().getCode());
85  			}
86  			else {
87  				statuteCodeWithViolation.append(filler(BLANK, 1));
88  			}
89  			statuteCodeWithViolation.append(filler(sectionViolated
90  					.getSectionViolatedNumber(), 6));
91  		}
92  		return statuteCodeWithViolation.toString();
93  	}
94  
95  	/***
96  	 * Gets the court code
97  	 * @return
98  	 */
99  	private String getCourtCode(CitationDetail citationDetail) {
100 		if (!EaseUtil.isNullOrBlank(citationDetail.getCourtCode())) {
101 			return citationDetail.getCourtCode().getCode();
102 		}
103 		else {
104 			return BLANK;
105 		}
106 	}
107 
108 	/***
109 	 * Gets the conviction date based on TCode
110 	 * @return
111 	 */
112 	private String getConvictionDate(IUserContext currentUserContext,
113 			CitationDetail citationDetail) {
114 		String convictionDate = null;
115 		if (tCode.equalsIgnoreCase(IEcsConverterConstants.TCODE_DFA)) {
116 			convictionDate = getWorkDate(currentUserContext);
117 		}
118 		else {
119 			convictionDate = convertDateToString(citationDetail
120 					.getConvictionDate());
121 		}
122 		return convictionDate;
123 	}
124 
125 	/***
126 	 * Gets the fine amount with padded 0's on the left.
127 	 * @param fineAmount
128 	 * @return
129 	 */
130 	private String getFineAmount(BigDecimal fineAmount) {
131 		String strFineAmt = BLANK;
132 		if (!EaseUtil.isNullOrBlank(fineAmount)) {
133 			strFineAmt = filler(fineAmount.multiply(new BigDecimal(100))
134 					+ BLANK, 6, '0', EcsConverterHelper.PAD_LEFT);
135 		}
136 		else {
137 			strFineAmt = filler(BLANK, 6);
138 		}
139 		return strFineAmt;
140 	}

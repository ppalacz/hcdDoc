50  	@Override
51  	public String createMessage(IEcsRequest requestFromClient) {
52  		ValidateAutomatedReceiptRequestEcs request = (ValidateAutomatedReceiptRequestEcs) requestFromClient;
53  		Object[] message = new Object[] {
54  				IEcsConverterConstants.TCODE_CIA, // TRANSACTION CODE (HQ)
55  				getOfficeId(request.getUserContext()), // OFFICE ID
56  				getWorkDate(request.getUserContext()), // WORKDATE (YYMMDD)
57  				getEmployeeID(request.getUserContext()),// EMPLOYEE ID
58  				filler(convertDateToString(new Date(),
59  						EcsConverterHelper.FORMATTER_HH_MM), 4, '0',
60  						EcsConverterHelper.PAD_LEFT), // SEQUENCE NUMBER
61  				parseVouchers(request.getVouchers()) }; // MASTER RECEIPTS
62  		return createMessageFormat(message);
63  	}

66  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createResponse(java.lang.String)
67  	 */
68  	@Override
69  	public IEcsResponse createResponse(String ciaMessage) {
70  		if (isNullOrBlank(ciaMessage)
71  				|| ciaMessage.length() < IEcsConverterConstants.SIZE_OF_INITIAL_PART_OF_INQUIRY) { // First level validation of response size 
72  			throw new MessageConversionException(
73  					IEcsErrorMessageConstants.INVALID_NULL_RESPONSE);
74  		}
75  		ValidateAutomatedReceiptResponseEcs validateAutomatedReceiptResponseEcs = new ValidateAutomatedReceiptResponseEcs();
76  		List <VoucherReturnCode> voucherReturnCodes = new ArrayList <VoucherReturnCode>();
77  		int totalErrorResponses = 0;
78  		int totalGoodResponses = 0;
79  		int totalVouchers = 0;
80  		// Request structure is also a part of response structure. Set initial part (first 18 characters) of this request structure.
81  		setInitialPartOfInquiry(ciaMessage, validateAutomatedReceiptResponseEcs);
82  		String messageExcludingInitialPart = ciaMessage
83  				.substring(IEcsConverterConstants.SIZE_OF_INITIAL_PART_OF_INQUIRY);// Response message after first 18 characters
84  		if (messageExcludingInitialPart.length() == 0) {// Second level validation of response size. If validation fails then it means that it doesn't have any voucher information.
85  			throw new MessageConversionException(
86  					IEcsErrorMessageConstants.INVALID_RESPONSE_LENGTH);
87  		}
88  		int count = 0; // count increases if there is an error response or good response
89  		int startPos = 0;
90  		while (startPos < messageExcludingInitialPart.length()) {
91  			if (messageExcludingInitialPart.charAt(startPos) == 'E') {// Return code for error response starts with 'E'
92  				totalErrorResponses++;
93  				VoucherReturnCode voucherReturnCode = voucherReturnCodes
94  						.get(count);
95  				validateResponseSize(ciaMessage, voucherReturnCodes,
96  						totalVouchers, totalErrorResponses, totalGoodResponses);// Third level validation of response size. If validation fails then it means that it doesn't have proper error information. 
97  				String errorResponse = messageExcludingInitialPart
98  						.substring(
99  								startPos,
100 								startPos += IEcsConverterConstants.SIZE_OF_ERROR_RESPONSE);
101 				String retCodeStr = errorResponse.substring(0, 3).trim();// e.g. E02
102 				String errorCodeStr = errorResponse.substring(3, 5).trim(); // Valid error codes = 73,74,75,TR
103 				// set return code and error code for voucher
104 				Voucher voucher = new Voucher();
105 				voucherReturnCode.setValidatedVoucher(voucher);
106 				setReturnCodeForVoucher(retCodeStr, voucherReturnCode);
107 				setErrorCodeForVoucher(errorCodeStr, voucherReturnCode);
108 				setCommnetAndStatusForVoucher(errorCodeStr, voucherReturnCode
109 						.getValidatedVoucher());
110 				//voucherReturnCodes.add(voucherReturnCode);
111 				count++;
112 			}
113 			else if (messageExcludingInitialPart.charAt(startPos) == 'M') {// Return code for error response starts with 'M'
114 				totalGoodResponses++;
115 				VoucherReturnCode voucherReturnCode = voucherReturnCodes
116 						.get(count);
117 				validateResponseSize(ciaMessage, voucherReturnCodes,
118 						totalVouchers, totalErrorResponses, totalGoodResponses);// Third level validation of response size. If validation fails then it means that it doesn't have proper information for good response. 
119 				String goodResponse = messageExcludingInitialPart
120 						.substring(
121 								startPos,
122 								startPos += IEcsConverterConstants.SIZE_OF_GOOD_RESPONSE);
123 				// set return code and error code for voucher				
124 				String retCodeStr = goodResponse.substring(0, 3).trim();// e.g. M02
125 				//VoucherReturnCode voucherReturnCode = new VoucherReturnCode();
126 				Voucher voucher = new Voucher();
127 				voucherReturnCode.setValidatedVoucher(voucher);
128 				setReturnCodeForVoucher(retCodeStr, voucherReturnCode);
129 				// set good response for validated voucher 
130 				addGoodResponseForVoucher(goodResponse, voucherReturnCode
131 						.getValidatedVoucher());
132 				//voucherReturnCodes.add(voucherReturnCode);
133 				count++;
134 			}
135 			else {// If not 'E' or 'M' then it's an inquired voucher information of size 15		
136 				totalVouchers++;
137 				VoucherReturnCode voucherReturnCode = new VoucherReturnCode();
138 				validateResponseSize(ciaMessage, voucherReturnCodes,
139 						totalVouchers, totalErrorResponses, totalGoodResponses);// Third level validation of response size 
140 				String voucherStr = messageExcludingInitialPart
141 						.substring(
142 								startPos,
143 								startPos
144 										+ IEcsConverterConstants.SIZE_OF_ONE_INQUIRED_VOUCHER);
145 				startPos += IEcsConverterConstants.SIZE_OF_ONE_INQUIRED_VOUCHER;
146 				addVoucherReturnCode(voucherReturnCodes, voucherReturnCode,
147 						voucherStr);
148 			}
149 		}
150 		validateAutomatedReceiptResponseEcs
151 				.setVoucherReturnCodes(voucherReturnCodes);
152 		return validateAutomatedReceiptResponseEcs;
153 	}

155 	/***
156 	 * Add the good response for voucher.
157 	 * 
158 	 * @param goodResponse the good response
159 	 * @param voucher the voucher
160 	 */
161 	private void addGoodResponseForVoucher(String goodResponse, Voucher voucher) {
162 		int start = 3;
163 		String tCode = goodResponse.substring(start, start += 3).trim(); // TCODE - No need to set in any domain object
164 		String receiptPhaseNumber = tCode.equalsIgnoreCase("CIB") ? "2" : "1"; // Receipt Phase Number
165 		goodResponse.substring(start, start += 3).trim(); // Filler - No need to set in any domain object
166 		String dollarAmount = goodResponse.substring(start, start += 7).trim(); // DOLLAR AMOUNT
167 		String voucherDate = goodResponse.substring(start, start += 6).trim(); // DATE PAID
168 		String licenseNumber = goodResponse.substring(start, start += 7).trim();// LICENSE NUMBER
169 		String bundleId = goodResponse.substring(start, start += 5).trim();// BUNDLE ID
170 		String last3Vin = goodResponse.substring(start, start += 3).trim();// 3 POSITION VIN NUMBER
171 		String centsAmount = goodResponse.substring(start, start += 2).trim();// CENTS AMOUNT
172 		goodResponse.substring(start, start += 36).trim();// Filler - No need to set in any domain object
173 		goodResponse.substring(start, start += 3).trim(); // TYPE TRANSACTION CODE - No need to set in any domain object		
174 		String comment = goodResponse.substring(start, start += 17).trim(); // CIACOMM
175 		voucher.setReceiptPhaseNumber(receiptPhaseNumber);
176 		voucher.setAmount(getAmount(dollarAmount, centsAmount, tCode));
177 		voucher.setDate(voucherDate.matches("[0]*") ? null
178 				: convertStringToDate(voucherDate,
179 						EcsConverterHelper.FORMATTER_YY_MM_DD)); // if date paid = 000000 then it is same as null
180 		voucher.setLicenseNumber(licenseNumber);
181 		voucher.setNameOrBundleId(bundleId);
182 		voucher.setLast3Vin(isNullOrBlank(last3Vin) ? null : last3Vin);
183 		voucher.setComment(comment); // not used
184 	}

186 	/***
187 	 * Add the voucher return code.
188 	 * 
189 	 * @param validatedVouchers the validated vouchers
190 	 * @param voucherStr the voucher str
191 	 */
192 	private void addVoucherReturnCode(
193 			List <VoucherReturnCode> validatedVouchers,
194 			VoucherReturnCode voucherReturnCode, String voucherStr) {
195 		//VoucherReturnCode voucherReturnCode = new VoucherReturnCode();
196 		Voucher inquiredVoucher = new Voucher();
197 		// set office id
198 		Office office = new Office();
199 		office.setOfficeId(voucherStr.substring(0, 3).trim());
200 		inquiredVoucher.setOffice(office);
201 		// set work date
202 		Employee employee = new Employee();
203 		EmployeeWorkdateControl employeeWorkdateControl = new EmployeeWorkdateControl();
204 		OfficeWorkdate officeWorkDate = new OfficeWorkdate();
205 		employeeWorkdateControl.setOfficeWorkdate(officeWorkDate);
206 		employeeWorkdateControl.getOfficeWorkdate().setAuthorizedWorkDate(
207 				convertStringToDate(voucherStr.substring(3, 9),
208 						EcsConverterHelper.FORMATTER_YY_MM_DD));
209 		employee.setEmployeeWorkdateControl(employeeWorkdateControl);
210 		// set employee tech id
211 		employee.setEmployeeTechId(voucherStr.substring(9, 11).trim());
212 		inquiredVoucher.setEmployee(employee);
213 		// set sequence number
214 		inquiredVoucher.setSequenceNumber(convertStringToInteger(voucherStr
215 				.substring(11, 15).trim()));
216 		voucherReturnCode.setInquiredVoucher(inquiredVoucher);
217 		validatedVouchers.add(voucherReturnCode);
218 	}

220 	/***
221 	 * Get the amount.
222 	 * 
223 	 * @param dollarAmount the dollar amount
224 	 * @param centsAmount the cents amount
225 	 * @param tCode the t code
226 	 * 
227 	 * @return the amount
228 	 */
229 	private BigDecimal getAmount(String dollarAmount, String centsAmount,
230 			String tCode) {
231 		if (tCode.equalsIgnoreCase("CIB")) { // Phase code 2 means "CIB". If CIB is there, then it means that with dollars, pennies are also available in the response. 
232 			return new BigDecimal(dollarAmount + "." + centsAmount);
233 		}
234 		else { // Phase code 1 means tcode other than "CIB". It means that, pennies are not available.
235 			return new BigDecimal(dollarAmount);
236 		}
237 	}

239 	/***
240 	 * Get the error message.
241 	 * 
242 	 * @param errorCode the error code
243 	 * 
244 	 * @return the error message
245 	 */
246 	private String getErrorMessage(String errorCode) {
247 		String errorMsg = BLANK;
248 		if (errorCode.equals("73") || errorCode.equals("74")) {
249 			errorMsg = IEcsConverterConstants.ERRORMSG_FOR_ERRORCODE_73_AND_74;
250 		}
251 		else if (errorCode.equals("75")) {
252 			errorMsg = IEcsConverterConstants.ERRORMSG_FOR_ERRORCODE_75;
253 		}
254 		else if (errorCode.equals("TR")) {
255 			errorMsg = IEcsConverterConstants.ERRORMSG_FOR_ERRORCODE_TR;
256 		}
257 		return errorMsg;
258 	}

260 	/***
261 	 * Parse the vouchers.
262 	 * 
263 	 * @param vouchers the vouchers
264 	 * 
265 	 * @return the string
266 	 */
267 	private String parseVouchers(List <Voucher> vouchers) {
268 		StringBuilder parsedVouchers = new StringBuilder();
269 		for (Voucher voucher : vouchers) {
270 			Office office = voucher.getOffice();
271 			Employee employee = voucher.getEmployee();
272 			EmployeeWorkdateControl employeeWorkdateControl = employee
273 					.getEmployeeWorkdateControl();
274 			parsedVouchers.append(filler(office.getOfficeId(), 3));// M/R KEY OFFICE
275 			parsedVouchers.append(convertDateToString(employeeWorkdateControl
276 					.getOfficeWorkdate().getAuthorizedWorkDate(),
277 					EcsConverterHelper.FORMATTER_YY_MM_DD));// M/R KEY WORKDATE (YYMMDD)
278 			parsedVouchers.append(filler(employee.getEmployeeTechId(), 2));// REG TECH
279 			parsedVouchers
280 					.append(filler(convertIntegerToString(voucher
281 							.getSequenceNumber()), 4, '0',
282 							EcsConverterHelper.PAD_LEFT));// M/R KEY SEQUENCE #
283 		}
284 		String returnString = parsedVouchers.toString();
285 		return returnString;
286 	}

289 	 * Set the status for voucher.
290 	 * 
291 	 * @param errorCode the error code
292 	 * @param voucher the voucher
293 	 */
294 	private void setCommnetAndStatusForVoucher(String errorCode, Voucher voucher) {
295 		String voucherStatus = BLANK;
296 		/*UserContext userContext = (UserContext) EaseApplicationContext
297 				.getApplicationContext().getBean("userContext");*/
298 		String functionalMode = getUserContext().getFunctionalMode().getCode();
299 		if (errorCode.equals("73") || errorCode.equals("74")) {
300 			voucher.setComment(IEcsConverterConstants.MISC_RECEIPT_COMMENT);
301 			if (functionalMode.equals("D") || functionalMode.equals("V")) {
302 				voucherStatus = "I";
303 			}
304 			else {
305 				voucherStatus = "S";
306 			}
307 			voucher
308 					.setStatusCode(new CodeSetElement(null, voucherStatus, null));
309 		}
310 	}

358 	/***
359 	 * Validate response size.
360 	 * 
361 	 * @param ciaMessage the cia message
362 	 * @param voucherReturnCodes the voucher return codes
363 	 * @param totalVouchers the total vouchers
364 	 * @param totalErrorResponses the total error responses
365 	 * @param totalGoodResponses the total good responses
366 	 */
367 	private void validateResponseSize(String ciaMessage,
368 			List <VoucherReturnCode> voucherReturnCodes, int totalVouchers,
369 			int totalErrorResponses, int totalGoodResponses) {
370 		int expectedSizeOfVouchers = totalVouchers
371 				* IEcsConverterConstants.SIZE_OF_ONE_INQUIRED_VOUCHER;
372 		int expectedSizeOfrrorResponses = totalErrorResponses
373 				* IEcsConverterConstants.SIZE_OF_ERROR_RESPONSE;
374 		int expectedSizeOfGoodResponses = totalGoodResponses
375 				* IEcsConverterConstants.SIZE_OF_GOOD_RESPONSE;
376 		int expectedSizeOfCiaMessage = IEcsConverterConstants.SIZE_OF_INITIAL_PART_OF_INQUIRY
377 				+ expectedSizeOfVouchers
378 				+ expectedSizeOfrrorResponses
379 				+ expectedSizeOfGoodResponses;
380 		if (ciaMessage.length() < expectedSizeOfCiaMessage) {
381 			throw new MessageConversionException(
382 					"Invalid Response: Message size is not proper");
383 		}
384 	}

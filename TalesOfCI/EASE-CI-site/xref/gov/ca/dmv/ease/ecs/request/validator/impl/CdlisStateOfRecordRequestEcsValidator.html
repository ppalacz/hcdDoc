25  public class CdlisStateOfRecordRequestEcsValidator extends EcsRequestValidator {
26  	/*** The Converter. */
27  	private static final String CONVERTER = "CdlisStateOfRecordConverter";
28  	/*** The Request. */
29  	private static final String REQUEST = "CdlisStateOfRecordRequestEcs";
30  	/*** The Constant serialVersionUID. */
31  	private static final long serialVersionUID = -8221831000103969671L;
32  
33  	/* (non-Javadoc)
34  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
35  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
36  	 */
37  	@Override
38  	protected void validateRequest(IEcsRequest request,
39  			IErrorCollector aCollector) {
40  		if (!(request instanceof CdlisStateOfRecordRequestEcs)) {
41  			addValidationExceptionInErrorCollector(
42  					REQUEST_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
43  					new String[] { REQUEST }, aCollector);
44  		}
45  		else {
46  			if (!(request.getMessageConverter() instanceof CdlisStateOfRecordConverter)) {
47  				addValidationExceptionInErrorCollector(
48  						CONVERTER_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
49  						new String[] { CONVERTER }, aCollector);
50  			}
51  			CdlisStateOfRecordRequestEcs cdlisStateOfRecordRequestEcs = (CdlisStateOfRecordRequestEcs) request;
52  			if (EaseUtil.isNullOrBlank(cdlisStateOfRecordRequestEcs
53  					.getCdlisRecord())) {
54  				addValidationExceptionInErrorCollector(
55  						CDLIS_RECORD_IS_EMPTY_ERROR_MESSAGE, null, aCollector);
56  			}
57  			if (EaseUtil.isNullOrBlank(cdlisStateOfRecordRequestEcs
58  					.getCdlisRecord().getCurrentName())) {
59  				addValidationExceptionInErrorCollector(
60  						CDLIS_RECORD_NAME_IS_EMPTY_ERROR_MESSAGE, null,
61  						aCollector);
62  			}
63  			if (EaseUtil.isNullOrBlank(cdlisStateOfRecordRequestEcs
64  					.getCdlisRecord().getCurrentBirthDate())) {
65  				addValidationExceptionInErrorCollector(
66  						CDLIS_RECORD_BIRTHDATE_IS_EMPTY_ERROR_MESSAGE, null,
67  						aCollector);
68  			}
69  			if (EaseUtil.isNullOrBlank(cdlisStateOfRecordRequestEcs
70  					.getCdlisRecord().getCurrentPrimaryDlNumberState())) {
71  				addValidationExceptionInErrorCollector(
72  						CDLIS_RECORD_LICENSE_STATE_IS_EMPTY_ERROR_MESSAGE,
73  						null, aCollector);
74  			}
75  		}
76  	}
77  }

91  	/***
92  	 * This method computes the License Term Years.
93  	 * 
94  	 * @param license the License
95  	 * @param invoice the Invoice
96  	 * @param processHistory the Process History
97  	 * @param processPendingSegment indicates if its pending segment
98  	 * @return the Limited Term Years
99  	 */
100 	private String getLicenseTerm(License license) {
101 		int termOfYears = 0;
102 		boolean processPendingSegment = false;
103 		if (license instanceof IdCard) {
104 			termOfYears = 0;
105 			if (EaseUtil.isNotNull(previousProcessHistory)
106 					&& previousProcessHistory.getUserContext().getTtc()
107 							.equalsIgnoreCase(CONST_IDA_TTC)) {
108 				termOfYears = 6;
109 			}
110 		}
111 		else if (license instanceof DriverLicense
112 				&& EaseUtil.isNotNull(license)) {
113 			Integer limitedLicenseTermYears = ((DriverLicense) license)
114 					.getLimitedLicenseTermYears();
115 			if (!EaseUtil.isNullOrBlank(dlApplication.getProcessHistory())
116 					&& !dlApplication.getProcessHistory().isEmpty()) {
117 				if (dlApplication.getProcessHistory().size() == 1) {
118 					processPendingSegment = false;
119 				}
120 				else if (dlApplication.getProcessHistory().size() == 2) {
121 					processPendingSegment = true;
122 				}
123 			}
124 			Integer renewalLicenseTermYears = UpdateHeadquartersSystemHelper
125 					.calculateRenewalLicenseTermYears((DriverLicense) license,
126 							invoice, previousProcessHistory,
127 							processPendingSegment);
128 			if (!EaseUtil.isNullOrBlank(limitedLicenseTermYears)) {
129 				termOfYears = limitedLicenseTermYears % 10;
130 			}
131 			else if (!EaseUtil.isNullOrBlank(renewalLicenseTermYears)) {
132 				termOfYears = renewalLicenseTermYears % 10;
133 			}
134 			else
135 				termOfYears = 5;
136 		}
137 		return filler(BLANK + (new Integer(termOfYears)), 1);
138 	}

321 	 * Gets the drive test waiver class.
322 	 * @param driver license, invoice, return type
323 	 * @return the drive test waiver class
324 	 */
325 	private String getDriveTestWaiverClassOrDate(String returnType) {
326 		String existingLicenseClass1 = BLANK;
327 		String appliedLicenseClass1 = BLANK;
328 		String driveTestWaiverClass = null;
329 		Date driveTestWaiverDate = null;
330 		if (EaseUtil.isNotNull(driverLicense)) {
331 			List <CodeSetElement> existingLicenseClasses = driverLicense
332 					.getExistingLicenseClasses();
333 			List <CodeSetElement> appliedLicenseClasses = driverLicense
334 					.getAppliedLicenseClasses();
335 			Date dateFeeReceived = null;
336 			if (EaseUtil.isNotNull(invoice)
337 					&& EaseUtil.isNotNull(invoice.getPayments())
338 					&& !invoice.getPayments().isEmpty()) {
339 				dateFeeReceived = invoice.getPayments().get(0).getDate();
340 			}
341 			Date dlExpirationDate = driverLicense.getExpirationDate();
342 			if (EaseUtil.isNotNull(existingLicenseClasses)
343 					&& !existingLicenseClasses.isEmpty()
344 					&& EaseUtil.isNotNull(existingLicenseClasses.get(0)))
345 				existingLicenseClass1 = existingLicenseClasses.get(0).getCode();
346 			if (EaseUtil.isNotNull(appliedLicenseClasses)
347 					&& !appliedLicenseClasses.isEmpty()
348 					&& EaseUtil.isNotNull(appliedLicenseClasses.get(0)))
349 				appliedLicenseClass1 = appliedLicenseClasses.get(0).getCode();
350 			if ((existingLicenseClass1.compareTo(appliedLicenseClass1) < 0 && ("A"
351 					.equals(existingLicenseClass1)
352 					|| "B".equals(existingLicenseClass1)
353 					|| "C".equals(existingLicenseClass1)
354 					|| "D".equals(existingLicenseClass1) || "E"
355 					.equals(existingLicenseClass1)))) {
356 				if (EaseUtil.isNotNull(dateFeeReceived)
357 						&& dateFeeReceived.before(addAnYear(dlExpirationDate))) {
358 					if (isClassUpgrade(existingLicenseClass1,
359 							appliedLicenseClass1)) {
360 						driveTestWaiverDate = addAnYear(dlExpirationDate);
361 						driveTestWaiverClass = existingLicenseClass1;
362 					}
363 				}
364 			}
365 		}
366 		if ("WAIVER_CLASS".equals(returnType)) {
367 			return driveTestWaiverClass;
368 		}
369 		else
370 			return convertDateToString(driveTestWaiverDate);
371 	}

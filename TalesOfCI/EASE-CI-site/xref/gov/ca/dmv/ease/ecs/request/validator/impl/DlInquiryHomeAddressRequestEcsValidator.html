27  public class DlInquiryHomeAddressRequestEcsValidator extends
28  		EcsRequestValidator {
29  	/*** The Constant serialVersionUID. */
30  	private static final long serialVersionUID = 8547457919995397309L;
31  	/*** The DCF TCODE. */
32  	private static final String C52_TCODE = "C52";
33  	/*** The Request. */
34  	private static final String REQUEST = "DlInquiryHomeAddressRequestEcs";
35  	/*** The Converter. */
36  	private static final String CONVERTER = "DlInquiryHomeAddressConverter";
37  	/*** The TTC. */
38  	private static final String TTC = "37U";
39  
40  	/***
41  	 * Constructor.
42  	 */
43  	public DlInquiryHomeAddressRequestEcsValidator() {
44  		super();
45  	}
46  
47  	/* (non-Javadoc)
48  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
49  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
50  	 */
51  	@Override
52  	protected void validateRequest(IEcsRequest request,
53  			IErrorCollector aCollector) {
54  		if (request == null) {
55  			this.addValidationExceptionInErrorCollector(NULL_REQUEST,
56  					new String[] { C52_TCODE }, aCollector);
57  		}
58  		else {
59  			if (!(request instanceof DlInquiryHomeAddressRequestEcs)) {
60  				this.addValidationExceptionInErrorCollector(
61  						IMPROPER_REQUEST_TYPE, new String[] { REQUEST },
62  						aCollector);
63  			}
64  			else {
65  				if (!(request.getMessageConverter() instanceof DlInquiryHomeAddressConverter)) {
66  					this.addValidationExceptionInErrorCollector(
67  							IMPROPER_MESSAGE_CONVERTER_ERROR_MESSAGE,
68  							new String[] { CONVERTER }, aCollector);
69  				}
70  				if (!(StringUtils
71  						.equals(request.getUserContext().getTtc(), TTC))) {
72  					this.addValidationExceptionInErrorCollector(
73  							INVALID_TTC_ERROR_MESSAGE, new String[] { TTC },
74  							aCollector);
75  				}
76  				DlInquiryHomeAddressRequestEcs dlInquiryRequest = (DlInquiryHomeAddressRequestEcs) request;
77  				String fileCode = dlInquiryRequest.getFileCode();
78  				String vehicleLicenseNumber = dlInquiryRequest
79  						.getVehicleLicenseNumber();
80  				//File Code
81  				if (EaseUtil.isNullOrBlank(fileCode)) {
82  					this.addValidationExceptionInErrorCollector(
83  							FILE_CODE_IS_EMPTY_ERROR_MESSAGE, aCollector);
84  				}
85  				//Vehicle License Number
86  				if (EaseUtil.isNullOrBlank(vehicleLicenseNumber)) {
87  					this.addValidationExceptionInErrorCollector(
88  							VEHICLE_LICENSE_NUMBER_IS_EMPTY_ERROR_MESSAGE,
89  							aCollector);
90  				}
91  			}//End of Check for Request
92  		}//End of Check for Request Type
93  	}
94  }

84  	/***
85  	 * Get the driver license or id number.
86  	 * 
87  	 * @param person the person
88  	 * 
89  	 * @return the driver license or id number
90  	 */
91  	private String getDlorIdNumber(Person person) {
92  		String dlOrIdNumber = BLANK;
93  		DriverLicense driverLicense = person.getDriverLicense();
94  		IdCard idCard = person.getIdCard();
95  		if (isNotNull(driverLicense)
96  				&& isNotNull(driverLicense.getLicenseNumber())) {
97  			dlOrIdNumber = driverLicense.getLicenseNumber();
98  		}
99  		else if (isNotNull(idCard) && isNullOrBlank(idCard.getIdNumber())) {
100 			dlOrIdNumber = idCard.getIdNumber();
101 		}
102 		return dlOrIdNumber;
103 	}

105 	/***
106 	 * Get the reversed name.
107 	 * 
108 	 * @param person the person
109 	 * 
110 	 * @return the reversed name
111 	 */
112 	private String getReversedTrueName(Person person) {
113 		PersonName personName = null;
114 		StringBuilder reversedName = new StringBuilder();
115 		// Assumption : validation for "getCurrentName() should return PersonName" will be a part of respective Validator
116 		if (isNotNull(person.getCurrentName())) {
117 			personName = (PersonName) person.getCurrentName();
118 		}
119 		if (isNotNull(personName)) {
120 			reversedName.append(isNull(personName.getLastName()) ? BLANK
121 					: personName.getLastName());
122 			reversedName.append(EcsConverterHelper.EMPTY_FILLER);
123 			reversedName.append(isNull(personName.getFirstName()) ? BLANK
124 					: personName.getFirstName());
125 			reversedName.append(EcsConverterHelper.EMPTY_FILLER);
126 			reversedName.append(isNull(personName.getMiddleName()) ? BLANK
127 					: personName.getMiddleName());
128 			reversedName.append(EcsConverterHelper.EMPTY_FILLER);
129 			reversedName.append(isNull(personName.getSuffix()) ? BLANK
130 					: personName.getSuffix());
131 		}
132 		return reversedName.toString().trim();
133 	}

56  	@Override
57  	public String createMessage(IEcsRequest recordCashierTransactionRequestEcs) {
58  		StringBuilder messageBuffer = new StringBuilder();
59  		// Get the DlApplication from the Request object.
60  		DlApplication dlApplication = (DlApplication) ((RecordCashierTransactionRequestEcs) recordCashierTransactionRequestEcs)
61  				.getApplication();
62  		if (dlApplication != null) {
63  			// 1. Build the Control Segment
64  			RecordCashierTransactionControlSegmentConverter controlSegmentConverter = new RecordCashierTransactionControlSegmentConverter();
65  			// Add Control Segment Data
66  			String controlSegment = controlSegmentConverter.createMessage(
67  					recordCashierTransactionRequestEcs.getUserContext(),
68  					dlApplication);
69  			messageBuffer.append(controlSegment);
70  			LOGGER.info("Control Message is " + controlSegment);
71  			// 2. Build the Fee Segment
72  			RecordCashierTransactionFeeSegmentConverter feeSegmentConverter = new RecordCashierTransactionFeeSegmentConverter();
73  			// Add Control Segment Data
74  			String feeSegment = feeSegmentConverter
75  					.createMessage(dlApplication);
76  			messageBuffer.append(feeSegment);
77  			LOGGER.info("Fee Segment is " + feeSegment);
78  			// 3. Build the Payment Segment - TBD: In Process
79  			RecordCashierTransactionPaymentSegmentConverter paymentSegmentConverter = new RecordCashierTransactionPaymentSegmentConverter();
80  			// Add Payment Segment Data
81  			String paymentSegment = paymentSegmentConverter
82  					.createMessage(dlApplication);
83  			messageBuffer.append(paymentSegment);
84  			LOGGER.info("Payment Message is " + paymentSegment);
85  			// 4. Build the Refund Segment(s) - TBD: In Process
86  			RecordCashierTransactionRefundSegmentConverter refundSegmentConverter = new RecordCashierTransactionRefundSegmentConverter();
87  			// Add Refund Segment Data
88  			String refundSegment = refundSegmentConverter
89  					.createMessage(dlApplication);
90  			messageBuffer.append(refundSegment);
91  			LOGGER.info("Refund Message is " + refundSegment);
92  			// 5. Build the Inventory Segment(s) - TBD: In Process
93  			RecordCashierTransactionInventorySegmentConverter inventorySegmentConverter = new RecordCashierTransactionInventorySegmentConverter();
94  			// Add Payment Segment Data
95  			String inventorySegment = inventorySegmentConverter
96  					.createMessage(dlApplication);
97  			messageBuffer.append(inventorySegment);
98  			LOGGER.info("Inventory Message is " + inventorySegment);
99  			// 6. Build the Transaction Segment(s) - TBD: In Process
100 			RecordCashierTransactionTransactionSegmentConverter transactionSegmentConverter = new RecordCashierTransactionTransactionSegmentConverter();
101 			// Add Payment Segment Data
102 			String transactionSegment = transactionSegmentConverter
103 					.createMessage(dlApplication);
104 			messageBuffer.append(transactionSegment);
105 			LOGGER.info("transaction Message is " + transactionSegment);
106 		}
107 		return messageBuffer.toString();
108 	}

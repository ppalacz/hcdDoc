79  		SearchPersonByNameRequestEcs request = (SearchPersonByNameRequestEcs) aRequest;
80  		Person aPerson = request.getPerson();
81  		if (EaseUtil.isNullOrBlank(aPerson)) {
82  			addValidationExceptionInErrorCollector(
83  					IEcsErrorMessageConstants.ANI_PERSON_NULL_ERROR, null,
84  					aCollector);
85  		}
86  		else {
87  			PersonName personName = aPerson.getCurrentName();
88  			if (personName == null) {
89  				addValidationExceptionInErrorCollector(
90  						IEcsErrorMessageConstants.ANI_PERSON_NAME_NOT_AVAILABLE,
91  						null, aCollector);
92  			}
93  			if (aPerson.getBirthDate() == null) {
94  				addValidationExceptionInErrorCollector(
95  						IEcsErrorMessageConstants.ANI_PERSON_BIRTHDATE_NOT_AVAILABLE,
96  						null, aCollector);
97  			}
98  			String lastName = personName.getLastName();
99  			if (isNullOrBlank(lastName)) {
100 				addValidationExceptionInErrorCollector(
101 						IEcsErrorMessageConstants.ANI_PERSON_LASTNAME_NOT_AVAILABLE,
102 						null, aCollector);
103 			}
104 			String firstName = personName.getFirstName();
105 			if (isNullOrBlank(firstName)) {
106 				addValidationExceptionInErrorCollector(
107 						IEcsErrorMessageConstants.ANI_PERSON_FIRSTNAME_NOT_AVAILABLE,
108 						null, aCollector);
109 			}
110 			//Do not validate class for ID
111 			if (aRequest.getUserContext().getTtc().equalsIgnoreCase(
112 					ITtcConstants.DLA)) {
113 				DriverLicense driverLicense = aPerson.getDriverLicense();
114 				if (EaseUtil.isNullOrBlank(driverLicense)) {
115 					addValidationExceptionInErrorCollector(
116 							IEcsErrorMessageConstants.ANI_PERSON_DRIVER_LICENSE_NOT_AVAILABLE,
117 							null, aCollector);
118 				}
119 				else {
120 					validateClassOfInquiry(driverLicense, 1, 2, aCollector);
121 				}
122 			}

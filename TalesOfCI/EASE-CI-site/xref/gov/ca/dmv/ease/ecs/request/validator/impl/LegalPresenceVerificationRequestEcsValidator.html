33  	/*** The Constant DOCUMENT_TYPE A */
34  	private static final String DOCUMENT_TYPE_A = "A";
35  	/*** The Constant DOCUMENT_TYPE E */
36  	private static final String DOCUMENT_TYPE_I = "I";
37  	/*** The Constant INVALID_DOCUMENT_ID. */
38  	private static final String INVALID_DOCUMENT_ID = "Document Id is invalid";
39  	/*** The Constant MANDATORY_CITIZEN_I94_CONDITIONAL. */
40  	private static final String MANDATORY_CITIZEN_I94_CONDITIONAL = "Citizen I94 is null";
41  	/*** The Constant MANDATORY_DOCUMENT_EXPIRATION_DATE_CONDITIONAL. */
42  	private static final String MANDATORY_DOCUMENT_EXPIRATION_DATE_CONDITIONAL = "Document expiration date is null";
43  	/*** The Constant NULL_DOCUMENT. */
44  	private static final String NULL_DOCUMENT = "Document is null";
45  
46  	/***
47  	 * Validates citizen i94 number.
48  	 * 
49  	 * @param citizenI94Number the citizen i94 number
50  	 * @param documentId the document id
51  	 * @param aCollector the a collector
52  	 */
53  	private void validateCitizenI94Number(String documentTypeCode,
54  			String citizenI94Number, IErrorCollector aCollector) {
55  		if (DOCUMENT_TYPE_I.equals(documentTypeCode)
56  				|| DOCUMENT_TYPE_A.equals(documentTypeCode)) {
57  			if (EaseUtil.isNullOrBlank(citizenI94Number)) {
58  				aCollector.register(new EcsValidationException(
59  						MANDATORY_CITIZEN_I94_CONDITIONAL));
60  				return;
61  			}
62  		}
63  	}
64  
65  	/***
66  	 * Validates document expiration date.
67  	 * 
68  	 * @param documentId the document id
69  	 * @param expirationDate the expiration date
70  	 * @param aCollector the a collector
71  	 */
72  	private void validateDocumentExpirationDate(String documentTypeCode,
73  			Date expirationDate, IErrorCollector aCollector) {
74  		if (DOCUMENT_TYPE_I.equals(documentTypeCode)
75  				|| DOCUMENT_TYPE_A.equals(documentTypeCode)) {
76  			if (expirationDate == null) {
77  				addValidationExceptionInErrorCollector(
78  						MANDATORY_DOCUMENT_EXPIRATION_DATE_CONDITIONAL,
79  						aCollector);
80  			}
81  			return;
82  		}
83  	}
84  
85  	/***
86  	 * Validates document type code.
87  	 * 
88  	 * @param documentId the document id
89  	 * @param aCollector the a collector
90  	 */
91  	private void validateDocumentId(String documentId,
92  			IErrorCollector aCollector) {
93  		String docIdRegex = "[//w]*";
94  		if (EaseUtil.isNullOrBlank(documentId)) {
95  			addValidationExceptionInErrorCollector(INVALID_DOCUMENT_ID,
96  					aCollector);
97  			return;
98  		}
99  		else {
100 			if (!checkValueAsPerRegex(documentId, docIdRegex)) {
101 				addValidationExceptionInErrorCollector(INVALID_DOCUMENT_ID,
102 						aCollector);
103 				return;
104 			}
105 		}
106 	}

110 	 */
111 	@Override
112 	protected void validateRequest(IEcsRequest aRequest,
113 			IErrorCollector aCollector) {
114 		LegalPresenceVerificationRequestEcs legalPresenceVerificationRequestEcs = (LegalPresenceVerificationRequestEcs) aRequest;
115 		LegalPresenceDocument legalPresenceDocument = legalPresenceVerificationRequestEcs
116 				.getDlApplication().getLegalPresenceDocument();
117 		if (legalPresenceDocument.getDocumentId() != null) {
118 			validateDocumentId(legalPresenceDocument.getDocumentId(),
119 					aCollector);
120 			validateCitizenI94Number(legalPresenceDocument
121 					.getLegalPresenceDocumentType().getCode(),
122 					legalPresenceDocument.getDocumentId(), aCollector);
123 			validateDocumentExpirationDate(legalPresenceDocument
124 					.getDocumentId(),
125 					legalPresenceDocument.getExpirationDate(), aCollector);
126 		}
127 		else {
128 			addValidationExceptionInErrorCollector(NULL_DOCUMENT, aCollector);
129 		}
130 	}

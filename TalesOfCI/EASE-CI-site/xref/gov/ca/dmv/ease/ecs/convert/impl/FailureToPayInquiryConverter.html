64  		Date birthDate = failureToPayInquiryRequestEcs.getBirthDate();
65  		String inquiryDocketNumber = failureToPayInquiryRequestEcs
66  				.getInquiryDocketNumber();
67  		Object[] values = new Object[7];
68  		values[0] = TCODE; //MDLTCODE	3	HQ TRANSACTION CODE        
69  		values[1] = getTtc(userContext); //MDLTTC  	3	TYPE TRANS CODE
70  		values[2] = filler(licenseNumber, 8); //MDLDLID#	8	DL NUMBER             
71  		values[3] = lastName.substring(0, 3); //MDL3LNME	3	3 POS LAST NAME
72  		values[4] = filler(convertDateToString(birthDate), 6); //MDLBRTHD	6	BIRTHDATE             
73  		values[5] = filler(inquiryDocketNumber, 7); //MDLIDKT#	7	INQURIY DOCKET #
74  		values[6] = filler(convertDateToString(new Date()), 6); //MDLTRNDT	6	TRANSACTION DATE
75  		return createMessageFormat(values);
76  	}

86  		}
87  		FailureToPayInquiryResponseEcs response = null;
88  		//Check if Valid Response

90  			//Process Response

101 	/***
102 	 * This method processes the Response Message.
103 	 * 
104 	 * @param responseMessage the Response Message
105 	 * 
106 	 * @return the Failure To Pay Inquiry Response
107 	 */
108 	private FailureToPayInquiryResponseEcs processResponse(
109 			String responseMessage) {
110 		FailureToPayInquiryResponseEcs response = new FailureToPayInquiryResponseEcs(
111 				responseMessage.substring(36, 38));
112 		//Process the Response
113 		Person dlApplicant = new Person();
114 		dlApplicant.setCurrentName(ConverterResponseHelper
115 				.buildPersonNameFromString(responseMessage.substring(46, 81),
116 						DELIMITER));
117 		List <PersonName> personNames = new ArrayList <PersonName>();
118 		personNames.add(ConverterResponseHelper.buildPersonNameFromString(
119 				responseMessage.substring(81, 116), DELIMITER));
120 		dlApplicant.setPersonNames(personNames);
121 		Address mailingAddress = new Address(AddressType.MAILING);
122 		mailingAddress.setLine1(responseMessage.substring(116, 151).trim());
123 		mailingAddress.setCity(responseMessage.substring(151, 162).trim());
124 		String stateCode = responseMessage.substring(162, 164);
125 		mailingAddress.setStateCode(new CodeSetElement(stateCode, stateCode,
126 				stateCode));
127 		mailingAddress.setZip(responseMessage.substring(164, 169));
128 		dlApplicant.addAddress(mailingAddress);
129 		int numberOfRecords = Integer.parseInt(responseMessage.substring(341,
130 				343));
131 		response.setDlApplicant(dlApplicant);
132 		//Strip of 343 Bytes
133 		responseMessage = responseMessage.substring(343);
134 		List <CitationDetail> ftaCitationDetails = new ArrayList <CitationDetail>();
135 		List <CitationDetail> ftpCitationDetails = new ArrayList <CitationDetail>();
136 		String processRecord = null;
137 		for (int i = 0; i < numberOfRecords; i++) {
138 			processRecord = responseMessage.substring(0, 84);
139 			//Iterate through the Records, its either FTA/FTP
140 			if (isFTARecord(processRecord.substring(74, 75))) {
141 				//Add to FTA
142 				ftaCitationDetails.add(this
143 						.processCitationRecord(processRecord));
144 			}
145 			else {
146 				//Add to FTP
147 				ftpCitationDetails.add(this
148 						.processCitationRecord(processRecord));
149 			}
150 		}
151 		response.setFtaCitationDetails(ftaCitationDetails);
152 		response.setFtpCitationDetails(ftpCitationDetails);
153 		return response;
154 	}

166 		citationDetail.setViolationDate(convertStringToDate(processRecord

178 	/***
179 	 * This method processes Sections Violated.
180 	 * 
181 	 * @param sections the Sections
182 	 * 
183 	 * @return List of Sections Violated
184 	 */
185 	private List <SectionViolated> processSectionsViolated(String sections) {
186 		List <SectionViolated> sectionsViolated = new ArrayList <SectionViolated>();
187 		if (!EaseUtil.isNullOrBlank(sections)) {
188 			String statuteCode = null;
189 			String sectionViolatedNumber = null;
190 			String section = null;
191 			for (int i = 0; i < 7; i++) {
192 				SectionViolated sectionViolated = new SectionViolated();
193 				section = sections.substring(0 + i * 7, 7 + i * 7);
194 				statuteCode = section.substring(0 , 1);
195 				sectionViolatedNumber = section.substring(1, 7);
196 				if (!EaseUtil.isNullOrBlank(statuteCode)) {
197 					sectionViolated.setStatuteCode(new CodeSetElement(
198 							statuteCode, statuteCode, statuteCode));
199 				}
200 				if (!EaseUtil.isNullOrBlank(sectionViolatedNumber)) {
201 					sectionViolated
202 							.setSectionViolatedNumber(sectionViolatedNumber.trim());
203 				}
204 				if(EaseUtil.isNullOrBlank(statuteCode) && EaseUtil.isNullOrBlank(sectionViolatedNumber))
205 				{
206 					break;
207 				}
208 				sectionsViolated.add(sectionViolated);
209 			}
210 		}
211 		return sectionsViolated;
212 	}

226 	 * This method checks the Response is success.
227 	 * 
228 	 * @param true if Response is success
229 	 */
230 	private boolean isValidResponse(String responseMessage) {
231 		boolean isValidResponse = false;
232 		String responseCode = responseMessage.substring(36, 38);
233 		//Only 00, R0, RO - Are Good Return Codes
234 		if (responseCode.equals(RESPONSE_CODE_00)
235 				|| responseCode.equals(RESPONSE_CODE_R0)
236 				|| responseCode.equals(RESPONSE_CODE_RO)) {
237 			isValidResponse = true;
238 		}
239 		return isValidResponse;

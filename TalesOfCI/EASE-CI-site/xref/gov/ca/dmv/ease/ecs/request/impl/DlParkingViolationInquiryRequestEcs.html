59  	@Override
60  	public DlParkingViolationInquiryResponseEcs execute() {
61  		DlParkingViolationInquiryResponseEcs response = null;
62  		IErrorCollector errorCollector = getValidator().validate(this);
63  		if (errorCollector.hasErrors()) {
64  			response = new DlParkingViolationInquiryResponseEcs(errorCollector);
65  		}
66  		else {
67  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
68  			//In the catch block we need to register the error to the error collector.
69  			try {
70  				response = (DlParkingViolationInquiryResponseEcs) super
71  						.execute();
72  			}
73  			catch (EcsResponseIsNullException responseIsNullException) {
74  				errorCollector = new ErrorCollector();
75  				errorCollector.register(new EaseValidationException(
76  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
77  				response = new DlParkingViolationInquiryResponseEcs(
78  						errorCollector);
79  			}
80  			catch (SystemException sysEx) {
81  				errorCollector = new ErrorCollector();
82  				errorCollector.register(new EaseValidationException(
83  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
84  				response = new DlParkingViolationInquiryResponseEcs(
85  						errorCollector);
86  			}
87  		}
88  		return response;
89  	}

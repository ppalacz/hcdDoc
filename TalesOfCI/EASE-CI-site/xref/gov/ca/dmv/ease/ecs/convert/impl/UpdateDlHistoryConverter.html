42  	@Override
43  	public String createMessage(IEcsRequest requestFromClient) {
44  		UpdateDlHistoryRequestEcs request = (UpdateDlHistoryRequestEcs) requestFromClient;
45  		// Assumption : validation for null check of person object will be done by the respective validator class.
46  		List <CdlisRecord> cdlisRecords = request.getCdlisRecords();
47  		List <String> cdlisStates = new ArrayList <String>();
48  		int noCdlisRecords = 0;
49  		if (isNotNull(cdlisRecords)) {
50  			noCdlisRecords = cdlisRecords.size();
51  		}
52  		if (noCdlisRecords > 0) {
53  			for (CdlisRecord cdlisRecord : cdlisRecords) {
54  				String cdlisSate = cdlisRecord.getCurrentPrimaryDlNumberState()
55  						.getCode(); //
56  				cdlisStates.add(cdlisSate);
57  			}
58  		}
59  		Person person = request.getPerson();
60  		return createMessageFormat(new Object[] {
61  				// Segment A - Initial information 
62  				getInitialInformation(person, noCdlisRecords),
63  				// CDLIS Hit data + CA license info + Out of state license info
64  				filler(getCdlisHitData(cdlisRecords, noCdlisRecords)
65  						+ getcaliforniaData(person)
66  						+ getOutOfStateInformation(person, cdlisStates), 1200),
67  				filler(BLANK, 23) // Segment D - Filler
68  		});
69  	}

72  	 * Gets the segment a.
73  	 * 
74  	 * @param person the person
75  	 * 
76  	 * @return the segment a
77  	 */
78  	private String getInitialInformation(Person person, int noCdlisRecords) {
79  		String dlOrIdNumber = ConverterHelper.getDlorIdNumber(person); // Driver/ID number
80  		// Number of records in this transaction (no of cdlis records + out of state licences + CA license)
81  		String numberOfRecords = BLANK
82  				+ (noCdlisRecords + person.getPreviousLicenses().size() + 1);
83  		return ((new StringBuilder()).append(IEcsConverterConstants.TCODE_DPR)
84  				.append(filler(dlOrIdNumber, 8)).append(filler(numberOfRecords,
85  				2, '0', EcsConverterHelper.PAD_LEFT))).toString();
86  	}

88  	/***
89  	 * Get CdlisHitData.
90  	 * 
91  	 * @param cdlisRecords
92  	 * @return CdlisHitData
93  	 */
94  	private String getCdlisHitData(List <CdlisRecord> cdlisRecords,
95  			int noCdlisRecords) {
96  		String cdlisData = BLANK;
97  		if (noCdlisRecords > 0) {
98  			for (CdlisRecord cdlisRecord : cdlisRecords) {
99  				String stateCode = cdlisRecord.getCurrentPrimaryDlNumberState()
100 						.getCode(); // State Code "CA"
101 				String dlOrIdNumber = cdlisRecord.getLicenseNumber(); // DRIVER LIC OR ID NUMBER
102 				String personName = getReversedName(cdlisRecord
103 						.getCurrentName()); // DRIVER NAME
104 				String birthDate = convertDateToString(cdlisRecord
105 						.getCurrentBirthDate(),
106 						EcsConverterHelper.FORMATTER_MM_DD_YYCC); // BIRTHDATE (MMDDYYCC)
107 				CodeSetElement genderCode = cdlisRecord.getGenderCode();
108 				String gender = BLANK; // GENDER
109 				if (!isNull(genderCode) && !isNull(genderCode.getCode())) {
110 					gender = genderCode.getCode().trim(); // M or F
111 				}
112 				// SSN
113 				String ssn = cdlisRecord.getCurrentPrimarySsn();
114 				cdlisData += ((new StringBuilder()).append(stateCode).append(
115 						filler(dlOrIdNumber, 25))
116 						.append(filler(personName, 35)).append(
117 								filler(birthDate, 8)).append(filler(gender, 1))
118 						.append(filler(ssn, 9))).toString();
119 			}
120 		}
121 		return cdlisData;
122 	}

131 	private String getcaliforniaData(Person person) {
132 		// Assumption : All the elements of Segment B are mandatory. In person's name, only last name is mandatory. Necessary validations will be written in respective validator class.
133 		String stateCode = IEcsConverterConstants.STATE_CA; // State Code "CA"
134 		String dlOrIdNumber = ConverterHelper.getDlorIdNumber(person); // DRIVER LIC OR ID NUMBER
135 		String personName = getReversedName(person.getCurrentName()); // DRIVER NAME
136 		String birthDate = convertDateToString(person.getBirthDate(),
137 				EcsConverterHelper.FORMATTER_MM_DD_YYCC); // BIRTHDATE (MMDDYYCC)
138 		CodeSetElement genderCode = person.getGenderCode();
139 		String gender = BLANK; // GENDER
140 		if (!isNull(genderCode) && !isNull(genderCode.getCode())) {
141 			gender = genderCode.getCode().trim(); // M or F
142 		}
143 		String ssn = BLANK; // SSN
144 		List <Document> documents = person.getDocuments();
145 		if (documents.size() > 0) {
146 			SocialSecurityDocument ssnDocument = (SocialSecurityDocument) documents
147 					.get(0); // Assumption: first element in the list of documents is social security document. Necessary validations will be written in respective validator class.
148 			if (!isNull(ssnDocument) && !isNull(ssnDocument.getSsn())) {
149 				ssn = ssnDocument.getSsn().trim();
150 			}
151 		}
152 		return ((new StringBuilder()).append(stateCode).append(
153 				filler(dlOrIdNumber, 25)).append(filler(personName, 35))
154 				.append(filler(birthDate, 8)).append(filler(gender, 1))
155 				.append(filler(ssn, 9))).toString();
156 	}

161 	 * @param person the person
162 	 * 
163 	 * @return the segment c

165 	private String getOutOfStateInformation(Person person,
166 			List <String> cdlisStates) {
167 		int SEGMENT_C_SIZE = 1120;
168 		StringBuilder segmentC = new StringBuilder();
169 		List <OutOfStateLicense> outOfStateLicenses = person
170 				.getPreviousLicenses();
171 		// Assumption : First element of outOfStateLicenses list is mandatory. Necessary validations will be written in respective validator class.
172 		// Assumption : Max size of outOfStateLicenses list is 14. Necessary validations will be written in respective validator class.
173 		// Assumption : If outOfStateLicense is not null, then all the information in outOfStateLicense can not be null. Necessary validations will be written in respective validator class.
174 		for (OutOfStateLicense outOfStateLicense : outOfStateLicenses) {
175 			String osCode = BLANK; // O/S CODE
176 			String osDlNumber = BLANK; // O/S DRIVER LICENSE NUMBER
177 			String osDriverName = BLANK; // O/S DRIVER NAME
178 			String osBirthDate = BLANK; // O/S BIRTHDATE (MMDDYYCC)
179 			String osGender = BLANK; // O/S GENDER
180 			String osSsn = BLANK; // O/S SSN
181 			if (cdlisStates.indexOf(outOfStateLicense.getIssuedByState()
182 					.getCode()) < 0) {
183 				if (!isNull(outOfStateLicense)) {
184 					CodeSetElement issuedByState = outOfStateLicense
185 							.getIssuedByState();
186 					if (!isNull(issuedByState)
187 							&& !isNull(issuedByState.getCode())) {
188 						osCode = issuedByState.getCode().trim();
189 					}
190 					//				osDlNumber = !isNull(outOfStateLicense.getLicenseNumber()) ? outOfStateLicense
191 					//						.getLicenseNumber().trim()
192 					//						: BLANK;
193 					osDlNumber = outOfStateLicense.getLicenseNumber();
194 					//				osDriverName = !isNull(outOfStateLicense.getCurrentName()) ? getReversedName(outOfStateLicense
195 					//						.getCurrentName())
196 					//						: BLANK;
197 					osDriverName = getReversedName(outOfStateLicense.getName());
198 					osBirthDate = convertDateToString(outOfStateLicense
199 							.getBirthDate(),
200 							EcsConverterHelper.FORMATTER_MM_DD_YYCC);
201 					CodeSetElement genderCode = outOfStateLicense
202 							.getGenderCode();
203 					if (!isNull(genderCode) && !isNull(genderCode.getCode())) {
204 						osGender = genderCode.getCode().trim();
205 					}
206 					osSsn = outOfStateLicense.getSsn();
207 				}
208 				segmentC.append(filler(osCode, 2)).append(
209 						filler(osDlNumber, 25))
210 						.append(filler(osDriverName, 35)).append(
211 								filler(osBirthDate, 8)).append(
212 								filler(osGender, 1)).append(filler(osSsn, 9));
213 			}
214 		}
215 		return filler(segmentC.toString(), SEGMENT_C_SIZE);
216 	}

220 	 * 
221 	 * @param personName the person name
222 	 * 
223 	 * @return the reversed name
224 	 */
225 	private String getReversedName(PersonName personName) {
226 		//final int MAX_NAME_SIZE = 35;
227 		StringBuilder reversedName = new StringBuilder();
228 		if (isNotNull(personName)) {
229 			// Assumption : last name can never be empty.  Necessary validations will be written in respective validator class.
230 			String lastName = isNull(personName.getLastName()) ? BLANK
231 					: personName.getLastName().trim();
232 			String firstName = isNull(personName.getFirstName()) ? BLANK
233 					: personName.getFirstName().trim();
234 			String middleName = isNull(personName.getMiddleName()) ? BLANK
235 					: personName.getMiddleName().trim();
236 			String suffix = isNull(personName.getSuffix()) ? BLANK : personName
237 					.getSuffix().trim();
238 			int lastNameLength = lastName.length();
239 			int firstNameLength = firstName.length();
240 			int middleNameLength = middleName.length();
241 			int suffixLength = suffix.length();
242 			// Assumption : Last name is mandatory
243 			if (lastNameLength > 0 && firstNameLength > 0
244 					&& middleNameLength > 0 && suffixLength > 0) { // if all are available				
245 				reversedName.append(lastName).append(
246 						IEcsConverterConstants.SYM_AT).append(firstName)
247 						.append(IEcsConverterConstants.SYM_AT).append(
248 								middleName).append(
249 								IEcsConverterConstants.SYM_AT).append(suffix);
250 			}
251 			else if (lastNameLength > 0 && firstNameLength == 0
252 					&& middleNameLength == 0 && suffixLength == 0) { // if only last name is available				
253 				reversedName.append(lastName).append(
254 						IEcsConverterConstants.SYM_AT).append(lastName); // reversedName = last name + @ + last name			 
255 			}
256 			else if (lastNameLength == 0 && firstNameLength > 0
257 					&& middleNameLength == 0 && suffixLength == 0) { // if only first name is available				
258 				reversedName.append(firstName).append(
259 						IEcsConverterConstants.SYM_AT).append(firstName); // reversedName = firstName + @ + firstName			 
260 			}
261 			else if (lastNameLength > 0 && firstNameLength > 0
262 					&& middleNameLength > 0 && suffixLength == 0) { // if only last name, first name and middle name are available
263 				reversedName.append(lastName).append(
264 						IEcsConverterConstants.SYM_AT).append(firstName)
265 						.append(IEcsConverterConstants.SYM_AT).append(
266 								middleName);
267 			}
268 			else if (lastNameLength > 0 && firstNameLength > 0
269 					&& middleNameLength == 0 && suffixLength == 0) { // if only last name and first name are available				
270 				reversedName.append(lastName).append(
271 						IEcsConverterConstants.SYM_AT).append(firstName);
272 			}
273 			else if (lastNameLength > 0 && firstNameLength > 0
274 					&& middleNameLength == 0 && suffixLength > 0) { // if only last name, first name and suffix are available			
275 				reversedName.append(lastName).append(
276 						IEcsConverterConstants.SYM_AT).append(firstName)
277 						.append(IEcsConverterConstants.SYM_AT).append(suffix);
278 			}
279 			else if (lastNameLength > 0 && firstNameLength == 0
280 					&& middleNameLength > 0 && suffixLength == 0) { // if only last name and middle name are available				
281 				reversedName.append(lastName).append(
282 						IEcsConverterConstants.SYM_AT).append(middleName);
283 			}
284 			else if (lastNameLength > 0 && firstNameLength == 0
285 					&& middleNameLength == 0 && suffixLength > 0) { // if only last name and suffix are available
286 				String twiceLastNameAndSuffix = lastName
287 						+ IEcsConverterConstants.SYM_AT + suffix
288 						+ IEcsConverterConstants.SYM_AT + lastName
289 						+ IEcsConverterConstants.SYM_AT + suffix;
290 				//if (twiceLastNameAndSuffix.length() <= MAX_NAME_SIZE) { // if size of (last name + @ + suffix + @ + last name + @ + suffix) <= 35					
291 				reversedName.append(twiceLastNameAndSuffix); // reversedName = last name + @ + suffix + @ + last name + @ + suffix					
292 				//}
293 				//				else { // if size of (last name + @ + suffix + @ + last name + @ + suffix) > 35				
294 				//					reversedName.append(lastName).append("@").append(suffix)
295 				//							.append("@").append(lastName); // reversedName = last name + @ + suffix + @ + last name 
296 				//				}
297 			}
298 		}
299 		// If @ is the last character of reversed name then remove it	
300 		//		String finalReversedName = reversedName.toString();
301 		//		if(finalReversedName.length() > MAX_NAME_SIZE) {
302 		//			finalReversedName = finalReversedName.substring(0, MAX_NAME_SIZE);
303 		//		}		
304 		//		if(finalReversedName.charAt(finalReversedName.length()-1) == '@') {
305 		//			finalReversedName = finalReversedName.substring(0, finalReversedName.length() - 1);
306 		//		}
307 		//		return finalReversedName;		
308 		return reversedName.toString();

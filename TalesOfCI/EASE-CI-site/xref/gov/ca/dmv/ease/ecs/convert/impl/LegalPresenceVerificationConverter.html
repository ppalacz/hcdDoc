42  	/* (non-Javadoc)
43  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
44  	 */
45  	@Override
46  	public String createMessage(IEcsRequest requestFronmClient) {
47  		LegalPresenceVerificationRequestEcs legalPresenceVerificationRequest = (LegalPresenceVerificationRequestEcs) requestFronmClient;
48  		DlApplication dlApplication = legalPresenceVerificationRequest
49  				.getDlApplication();
50  		LegalPresenceDocument legalPresenceDocument = dlApplication
51  				.getLegalPresenceDocument();
52  		Person person = dlApplication.getApplicant();
53  		PersonName personName = person.getCurrentName();
54  		String documentId = legalPresenceDocument.getDocumentId();
55  		String documentType = null;
56  		if (isNotNull(legalPresenceDocument.getLegalPresenceDocumentType())) {
57  			documentType = legalPresenceDocument.getLegalPresenceDocumentType()
58  					.getCode();
59  		}
60  		String bdlpVerificationCode = null;
61  		if (isNotNull(dlApplication.getNewUserVerifiedBdlpCode())) {
62  			bdlpVerificationCode = dlApplication.getNewUserVerifiedBdlpCode()
63  					.getCode();
64  		}
65  		IUserContext userContext = requestFronmClient.getUserContext();
66  		Object[] message = {
67  				filler(IEcsConverterConstants.TCODE_LPV, 3),
68  				filler(IEcsConverterConstants.VERSION_NUMBER, 30),// Version Number.
69  				filler(getAlienIdNumber(documentId, documentType), 11),// Alien #.
70  				filler(getCitizenI94Number(documentId, documentType), 11),// I94 #.
71  				filler(BLANK, 10),// SEVIS Id - Not Referenced.
72  				filler(getPassportNumber(legalPresenceDocument,
73  						bdlpVerificationCode), 12),// Passport # .
74  				filler(BLANK, 8),// Visa # - Not Referenced.
75  				filler(getReceiptNumber(legalPresenceDocument,
76  						bdlpVerificationCode), 13),// Receipt # .
77  				filler(getNaturalizationNumber(legalPresenceDocument,
78  						bdlpVerificationCode), 12),// Naturalization #.
79  				filler(getCitizenshipNumber(legalPresenceDocument,
80  						bdlpVerificationCode), 12),// Citizenship #.
81  				filler(getUserField(person, userContext), 40),// User field (DL/ID #, Office ID, Tech ID).
82  				filler(getBenefitCode(legalPresenceVerificationRequest
83  						.getUserContext().getTtc()), 2),// Benefit Code.						
84  				filler(personName.getLastName(), 40),// Last Name. 
85  				filler(personName.getFirstName(), 25),// First Name.
86  				filler(personName.getMiddleName(), 25),// Middle Name.
87  				filler(convertDateToString(person.getBirthDate(),
88  						EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY),
89  						10),// Date Of Birth.
90  				filler(getDocumentId(bdlpVerificationCode), 2),// Document Id.
91  				filler(getDocumentDescription(bdlpVerificationCode), 35),// Document Description.
92  				filler(convertDateToString(legalPresenceDocument
93  						.getExpirationDate(),
94  						EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY),
95  						10),// Document Expiration Date. 
96  				filler(BLANK, 9) };// blank
97  		return createMessageFormat(message);
98  	}

101 	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createResponse(java.lang.String)

103 	@Override
104 	public IEcsResponse createResponse(String anInboundMessagePayload) {
105 		if (isNullOrBlank(anInboundMessagePayload)) {
106 			throw new MessageConversionException(
107 					IEcsErrorMessageConstants.INVALID_NULL_RESPONSE);
108 		}
109 		int aMessageLength = anInboundMessagePayload.length();
110 		if (aMessageLength != IEcsConverterConstants.LPV_RESPONSE_LENGTH) {
111 			throw new MessageConversionException(
112 					IEcsErrorMessageConstants.INVALID_RESPONSE_LENGTH);
113 		}
114 		LegalPresenceVerificationResponseEcs legalPresenceVerificationResponseEcs = new LegalPresenceVerificationResponseEcs();
115 		LpvRecord lpvRecord = new LpvRecord();
116 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
117 		PersonName personName = new PersonName();
118 		// Set a case number.
119 		String caseNumber = anInboundMessagePayload.substring(0, 15).trim();
120 		legalPresenceDocument.setVerificationNumber(caseNumber);
121 		// Set a last name.
122 		String lastName = anInboundMessagePayload.substring(15, 55).trim();
123 		personName.setLastName(lastName);
124 		// Set a first name.
125 		String firstName = anInboundMessagePayload.substring(55, 80).trim();
126 		personName.setFirstName(firstName);
127 		// Set a middle name.
128 		String middleName = anInboundMessagePayload.substring(80, 105).trim();
129 		personName.setMiddleName(middleName);
130 		//Set a birth date.
131 		Date birthDate = convertStringToDate(anInboundMessagePayload.substring(
132 				105, 115).trim(),
133 				EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY);
134 		legalPresenceDocument.setBirthDate(birthDate);
135 		// Set a entry date.
136 		Date entryDate = convertStringToDate(anInboundMessagePayload.substring(
137 				115, 125).trim(),
138 				EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY);
139 		legalPresenceDocument.setEntryDate(entryDate);
140 		// Set a country of citizenship.
141 		String countryOfCitizenship = anInboundMessagePayload.substring(150,
142 				155).trim();
143 		legalPresenceDocument.setCountry(countryOfCitizenship);
144 		// Set the expiration date.
145 		Date expirationDate = convertStringToDate(anInboundMessagePayload
146 				.substring(158, 168).trim(),
147 				EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY);
148 		legalPresenceDocument.setExpirationDate(expirationDate);
149 		String eligibilityCodeCondition = anInboundMessagePayload.substring(
150 				168, 170).trim();
151 		// Set the eligibility statement code.
152 		if (eligibilityCodeCondition
153 				.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_1)
154 				|| eligibilityCodeCondition
155 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_4)
156 				|| eligibilityCodeCondition
157 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_10)
158 				|| eligibilityCodeCondition
159 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_18)
160 				|| eligibilityCodeCondition
161 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_20)
162 				|| eligibilityCodeCondition
163 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_22)
164 				|| eligibilityCodeCondition
165 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_24)) {
166 			legalPresenceDocument
167 					.setEligibilityCode(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_E);
168 		}
169 		else if (eligibilityCodeCondition
170 				.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_CONDITION_13)) {
171 			legalPresenceDocument
172 					.setEligibilityCode(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_I);
173 		}
174 		else {
175 			legalPresenceDocument
176 					.setEligibilityCode(IEcsConverterConstants.ELIGIBILITY_STATEMENT_CODE_O);
177 		}
178 		int executionStatusCode = Integer.parseInt(anInboundMessagePayload
179 				.substring(389, 393).trim());
180 		// Set the execution status message.
181 		CodeSetElement err = new CodeSetElement();
182 		if (executionStatusCode == IEcsConverterConstants.EXECUTION_STATUS_CODE_0) {
183 			err.setCode(Integer
184 					.toString(IEcsConverterConstants.EXECUTION_STATUS_CODE_0));
185 			err
186 					.setDescription(IEcsConverterConstants.EXECUTION_STATUS_MESSAGE_0);
187 			lpvRecord.setExecutionStatus("0");
188 			legalPresenceVerificationResponseEcs.setErrorCode(err);
189 		}
190 		else if (executionStatusCode == IEcsConverterConstants.EXECUTION_STATUS_CODE_1) {
191 			err.setCode(Integer
192 					.toString(IEcsConverterConstants.EXECUTION_STATUS_CODE_1));
193 			err
194 					.setDescription(IEcsConverterConstants.EXECUTION_STATUS_MESSAGE_1);
195 			lpvRecord.setExecutionStatus("1");
196 			legalPresenceVerificationResponseEcs.setErrorCode(err);
197 		}
198 		else {
199 			err
200 					.setDescription(IEcsConverterConstants.EXECUTION_STATUS_MESSAGE_LOG_ERROR);
201 			lpvRecord.setExecutionStatus("1");
202 			legalPresenceVerificationResponseEcs.setErrorCode(err);
203 		}
204 		lpvRecord.setPersonName(personName);
205 		lpvRecord.setLegalPresenceDocument(legalPresenceDocument);
206 		// Set the LpvRecord in response.
207 		legalPresenceVerificationResponseEcs.setLpvRecord(lpvRecord);
208 		return legalPresenceVerificationResponseEcs;
209 	}

240 	 * @param legalPresenceDocument the legal presence document
241 	 * @param bdlpVerifCode the BDLP verification code
242 	 * 
243 	 * @return the passport number
244 	 */
245 	private String getPassportNumber(
246 			LegalPresenceDocument legalPresenceDocument, String bdlpVerifCode) {
247 		// Assumption : validation will be written for checking the valid instance of legalPresenceDocument. 
248 		// If Document type="O" then legalPresenceDocument will be an instance of Passport.
249 		String passportNumber = BLANK;
250 		//BDLP CODE 'O' - Valid (unexpired) Canadian passport with I-94, No Form# associated.
251 		if (IEcsConverterConstants.BDLP_CODE_FOR_VALID_CANADIAN_PASSPORT_WITH_I94
252 				.equals(bdlpVerifCode)) {
253 			if (legalPresenceDocument instanceof Passport) {
254 				passportNumber = ((Passport) legalPresenceDocument)
255 						.getPassportNumber();
256 			}
257 		}
258 		return passportNumber;
259 	}

306 						.equals(bdlpVerifCode)

308 				|| IEcsConverterConstants.BDLP_CODE_FOR_EMPLOYMENT_AUTH_DOC
309 						.equals(bdlpVerifCode)) {

350 	private String getUserField(Person applicant, IUserContext userContext) {
351 		String dlOrIdNumber = BLANK;
352 		IdCard idCard = applicant.getIdCard();
353 		DriverLicense driverLicense = applicant.getDriverLicense();
354 		if (isNotNull(idCard) && isNotNull(idCard.getIdNumber())) {
355 			dlOrIdNumber = idCard.getIdNumber();
356 		}
357 		else if (isNotNull(driverLicense)
358 				&& isNotNull(driverLicense.getLicenseNumber())) {
359 			dlOrIdNumber = driverLicense.getLicenseNumber();
360 		}
361 		return filler(dlOrIdNumber, 8) + getOfficeId(userContext)
362 				+ getEmployeeID(userContext);
363 	}

374 			ttc = ttc.trim();
375 			if (ConverterHelper.isDriverLicenseApplicationTtc(ttc)) {
376 				// BDLP CODE '5' - Refugee Travel Document, Form# I-571
377 				return IEcsConverterConstants.BDLP_CODE_FOR_REFUGEE_TRAVEL_DOCUMENT;
378 			}
379 			else if (ConverterHelper.isIdentificationCardApplicationTtc(ttc)) {
380 				return IEcsConverterConstants.BENEFIT_CODE_60;
381 			}
382 		}
383 		return BLANK;
384 	}

440 			}
441 			else if (IEcsConverterConstants.BDLP_CODE_FOR_REFUGEE_TRAVEL_DOCUMENT
442 					.equals(bdlpVerifCode)) {
443 				return IEcsConverterConstants.DOCUMENT_ID_5;

450 					.equals(bdlpVerifCode)) {
451 				return IEcsConverterConstants.DOCUMENT_ID_8;

455 				return IEcsConverterConstants.DOCUMENT_ID_10;

457 			// BDLP CODE 'M' - Temporary Resident Identification Card, Form# I-688

459 					.equals(bdlpVerifCode)) {
460 				return IEcsConverterConstants.DOCUMENT_ID_6;
461 			}

464 	}

473 	private String getDocumentDescription(String bdlpVerifCode) {
474 		if (IEcsConverterConstants.BDLP_CODE_FOR_CITIZEN_IDENTIFICATION_CARD
475 				.equals(bdlpVerifCode)) {
476 			return IEcsConverterConstants.DOC_DESC_FOR_BDLP_VERIF_CODE_K;

479 				.equals(bdlpVerifCode)) {
480 			return IEcsConverterConstants.DOC_DESC_FOR_BDLP_VERIF_CODE_L;
481 		}
482 		else if (IEcsConverterConstants.BDLP_CODE_FOR_NON_RESIDENT_ALIEN_CANADIAN_BORDER_CROSSING_CARD
483 				.equals(bdlpVerifCode)) {

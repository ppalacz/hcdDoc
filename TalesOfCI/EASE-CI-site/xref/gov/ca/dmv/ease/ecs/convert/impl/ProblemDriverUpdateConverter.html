37  	@Override
38  	public String createMessage(IEcsRequest requestFromClient) {
39  		ProblemDriverUpdateRequestEcs request = (ProblemDriverUpdateRequestEcs) requestFromClient;
40  		Object[] message = new Object[6];
41  		// Segment A
42  		message[0] = IEcsConverterConstants.TCODE_DPS;
43  		DlApplication dlApplication = request.getDlApplication();
44  		Person applicant = dlApplication.getApplicant();
45  		DriverLicense driverLicense = applicant.getDriverLicense();
46  		message[1] = filler(driverLicense.getLicenseNumber(), 8); // DL number
47  		message[2] = filler(dlApplication
48  				.getNewIncompleteApplicationReasonCode().getCode(), 2); // Incomplete Application Reason Code
49  		message[3] = getDlIssueDate(request); // DL Issue Date --- Assumption: format of issued date is MMDDYY
50  		message[4] = filler(BLANK, 17); // 17 Characters of Message
51  		message[5] = getSegmentB(request);
52  		return createMessageFormat(message);
53  	}

72  	private String getDlIssueDate(ProblemDriverUpdateRequestEcs request) {
73  		String issueDate = null;
74  		String incompleteAppReasonCode = request.getDlApplication()
75  				.getNewIncompleteApplicationReasonCode().getCode();
76  		if (incompleteAppReasonCode == null
77  				|| incompleteAppReasonCode.trim().length() == 0) {
78  			issueDate = convertDateToString(request.getDlApplication()
79  					.getApplicant().getDriverLicense().getEffectiveDate(),
80  					EcsConverterHelper.FORMATTER_MM_DD_YY);
81  		}
82  		else {
83  			issueDate = filler(issueDate, 6);
84  		}
85  		return issueDate;

88  	/***
89  	 * Get the segment B appended by last segment C.
90  	 * 
91  	 * @param request the request
92  	 * 
93  	 * @return the segment B appended by last segment C.
94  	 */
95  	private String getSegmentB(ProblemDriverUpdateRequestEcs request) {
96  		String segmentBString = BLANK;
97  		String osCode = null;
98  		String eligibilityIndicator = BLANK;
99  		String questionableIndicator = BLANK;
100 		String typeIndicator = BLANK;
101 		String osDlNumber = BLANK;
102 		String osDriverName = BLANK;
103 		String osSsn = BLANK;
104 		String osDlNumberLastFiveNumber = BLANK;
105 		List <PdpsRecord> pdpsRecords = request.getDlApplication()
106 				.getPdpsRecords();
107 		int pdpsRecordSize = pdpsRecords.size();
108 		int noOfEmptySegments = 16 - (pdpsRecordSize + 1);
109 		int i = 0;
110 		String officeId = request.getUserContext().getOfficeId();
111 		while (i < pdpsRecordSize) {
112 			PdpsRecord pdpsRecord = pdpsRecords.get(i);
113 			if (isNotNull(pdpsRecord)) {
114 				osCode = filler(pdpsRecord.getCurrentPrimaryDlNumberState()
115 						.getCode(), 2); // O/S CODE
116 				String designateAction = pdpsRecord.getDesignateAction()
117 						.getCode().trim();
118 				if ((officeId
119 						.equalsIgnoreCase(IEcsConverterConstants.ISSUANCE_OFFICE_212) || officeId
120 						.equalsIgnoreCase(IEcsConverterConstants.ISSUANCE_OFFICE_231))
121 						&& (request.getDlApplication().isClearedByIssuance())) {
122 					eligibilityIndicator = filler(
123 							IEcsConverterConstants.ELIGIBILITY_INDICATOR_E, 1);
124 				}
125 				else if ((designateAction
126 						.equalsIgnoreCase(IEcsConverterConstants.DESIGNATE_ACTION_Q) || designateAction
127 						.equalsIgnoreCase(IEcsConverterConstants.DESIGNATE_ACTION_S))
128 						&& (pdpsRecord.getPassword() == null || pdpsRecord
129 								.getPassword().trim().length() == 0)) {
130 					eligibilityIndicator = filler(
131 							IEcsConverterConstants.ELIGIBILITY_INDICATOR_N, 1);
132 				}
133 				else if (!eligibilityIndicator
134 						.equalsIgnoreCase(IEcsConverterConstants.ELIGIBILITY_INDICATOR_N)
135 						&& !request.getDlApplication().isCNACondition()) {
136 					eligibilityIndicator = filler(
137 							IEcsConverterConstants.ELIGIBILITY_INDICATOR_E, 1);
138 				}
139 				questionableIndicator = filler(pdpsRecord.getDesignateAction()
140 						.getCode(), 1);
141 				if (pdpsRecord.getRecordTypeOrStateOfOrigin().trim()
142 						.equalsIgnoreCase(
143 								IEcsConverterConstants.STATE_OF_ORIGIN)) {
144 					typeIndicator = filler(
145 							IEcsConverterConstants.TYPE_INDICATOR_N, 1);
146 				}
147 				else {
148 					typeIndicator = filler(
149 							IEcsConverterConstants.TYPE_INDICATOR_P, 1);
150 				}
151 				String licenseNumber = pdpsRecord.getLicenseNumber().trim();
152 				osDlNumber = filler(licenseNumber, 20); // O/S DRIVER LICENSE NUMBER
153 				PersonName personName = pdpsRecord.getCurrentName();
154 				osDriverName = filler(getReversedName(personName), 35);
155 				osSsn = filler(pdpsRecord.getCurrentPrimarySsn(), 9);
156 				if (osDlNumber.length() >= 5) {
157 					osDlNumberLastFiveNumber = osDlNumber.substring(osDlNumber
158 							.length() - 5, osDlNumber.length()); // O/S DRIVER LICENSE NUMBER LAST 5 DIGITS
159 				}
160 				else {
161 					osDlNumberLastFiveNumber = filler(licenseNumber, 5);
162 				}
163 				segmentBString += osCode + eligibilityIndicator
164 						+ questionableIndicator + typeIndicator + osDlNumber
165 						+ osDriverName + osSsn + osDlNumberLastFiveNumber
166 						+ EcsConverterHelper.EMPTY_FILLER;
167 				i++;
168 			}
169 		}
170 		segmentBString += getSegmentC(request);
171 		for (int j = 0; j < noOfEmptySegments; j++) {
172 			segmentBString += filler(BLANK, 75);
173 		}
174 		return segmentBString;
175 	}

177 	/***
178 	 * Get segment C.
179 	 * 
180 	 * @param request a ProblemDriverUpdateRequestEcs
181 	 * @return segmentCString a segment C
182 	 */
183 	private String getSegmentC(ProblemDriverUpdateRequestEcs request) {
184 		String segmentCString = BLANK;
185 		String osCode = BLANK;
186 		String eligibilityIndicator = BLANK;
187 		String questionableIndicator = BLANK;
188 		String typeIndicator = BLANK;
189 		String osDlNumber = BLANK;
190 		String osDriverName = BLANK;
191 		String osSsn = BLANK;
192 		String osDlNumberLastFiveNumber = BLANK;
193 		DlApplication dlApplication = request.getDlApplication();
194 		Person applicant = dlApplication.getApplicant();
195 		List <OutOfStateLicense> outOfStateLicenses = (List <OutOfStateLicense>) applicant
196 				.getPreviousLicenses();
197 		int outOfStateLicensesSize = outOfStateLicenses.size();
198 		if (outOfStateLicensesSize > 0) {
199 			for (OutOfStateLicense outOfStateLicense : outOfStateLicenses) {
200 				if (isNotNull(outOfStateLicense)) { // If 16th element exist
201 					osCode = outOfStateLicense.getIssuedByState().getCode()
202 							.trim(); // O/S CODE
203 					if ((osCode.trim().length() != 0)
204 							&& (!osCode
205 									.equals(IEcsConverterConstants.OTHER_OSCODE))) {
206 						if (dlApplication
207 								.getNewIncompleteApplicationReasonCode() == null
208 								|| dlApplication
209 										.getNewIncompleteApplicationReasonCode()
210 										.getCode().trim().length() == 0) {
211 							eligibilityIndicator = IEcsConverterConstants.ELIGIBILITY_INDICATOR_E; // ELIGIBILITY_INDICATOR
212 							questionableIndicator = IEcsConverterConstants.QUESTIONABLE_INDICATOR_S; // SAME AS/QUESTIONABLE INDICATOR
213 							typeIndicator = IEcsConverterConstants.TYPE_INDICATOR_S;// TYPE_INDICATOR
214 							String licenseNumber = outOfStateLicense
215 									.getLicenseNumber().trim();
216 							osDlNumber = filler(licenseNumber, 20); // O/S DRIVER LICENSE NUMBER
217 							osDriverName = filler(
218 									getReversedName(outOfStateLicense.getName()),
219 									35); // O/S DRIVER NAME
220 							osSsn = filler(outOfStateLicense.getSsn().trim(), 9);
221 							osDlNumberLastFiveNumber = osDlNumber.substring(
222 									osDlNumber.length() - 5, osDlNumber
223 											.length()); // O/S DRIVER LICENSE NUMBER LAST 5 DIGITS
224 							segmentCString += osCode + eligibilityIndicator
225 									+ questionableIndicator + typeIndicator
226 									+ osDlNumber + osDriverName + osSsn
227 									+ osDlNumberLastFiveNumber
228 									+ EcsConverterHelper.EMPTY_FILLER;
229 						}
230 					}
231 					else {
232 						segmentCString = filler(BLANK, 75);
233 					}
234 				}
235 			}
236 		}
237 		else {
238 			segmentCString = filler(BLANK, 75);
239 		}
240 		return segmentCString;
241 	}

244 	 * Get the reversed name.
245 	 * 
246 	 * @param name the name
247 	 * 
248 	 * @return the reversed name
249 	 */
250 	private String getReversedName(PersonName name) {
251 		PersonName personName = null;
252 		// Assumption : validation for "getCurrentName() should return PersonName" will be a part of respective Validator
253 		if (isNotNull(name)) {
254 			personName = (PersonName) name;
255 		}
256 		return (isNull(personName.getLastName()) ? BLANK : personName
257 				.getLastName())
258 				+ EcsConverterHelper.EMPTY_FILLER
259 				+ (isNull(personName.getFirstName()) ? BLANK : personName
260 						.getFirstName())
261 				+ EcsConverterHelper.EMPTY_FILLER
262 				+ (isNull(personName.getMiddleName()) ? BLANK : personName
263 						.getMiddleName())
264 				+ EcsConverterHelper.EMPTY_FILLER
265 				+ (isNull(personName.getSuffix()) ? BLANK : personName
266 						.getSuffix());
267 	}

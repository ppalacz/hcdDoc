37  	private static final String NON_KEY_BOARD_CHAR_REGEX = "[^//x20-//x7E]";
38  	private static final String TCODE = "DLX";
39  	private static final String TYPE_TRANS_CODE = "43U";
40  	private static final String TRANSACTION_SUB_CODE = "I";
41  	private static final String REQUESTER_CODE_PREFIX = "80";

43  	private static final String RESPONSE_CODE = " ";

45  	/*
46  	 * (non-Javadoc)
47  	 * 
48  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter
49  	 * #createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
50  	 */
51  	@Override
52  	public String createMessage(IEcsRequest request) {
53  		DriverLicenseTerminationUpdateRequestEcs aRequestEcs = (DriverLicenseTerminationUpdateRequestEcs) request;
54  		List <Object> objectList = new ArrayList <Object>(0);
55  		/* TCODE */
56  		objectList.add(filler(TCODE, 3));
57  		/* TYPE TRANS CODE */
58  		objectList.add(filler(TYPE_TRANS_CODE, 3));
59  		/* TRANSACTION SUB-CODE */
60  		objectList.add(filler(TRANSACTION_SUB_CODE, 1));
61  		/* REQUESTER CODE PREFIX */
62  		objectList.add(filler(REQUESTER_CODE_PREFIX, 2));
63  		/* OFFICE ID (REQUESTER CODE SUFFIX) */
64  		objectList.add(filler(super.getOfficeId(request.getUserContext()), 3));
65  		/* EMPLOYEE ID */
66  		objectList
67  				.add(filler(super.getEmployeeID(request.getUserContext()), 2));
68  		/* DL/ID NUMBER */
69  		objectList.add(filler(aRequestEcs.getDlOrIdCardNumber(), 8));
70  		/* FIRST 3 POSITIONS LAST NAME */
71  		objectList.add(filler(aRequestEcs.getLastName(), 3));
72  		/*TERMINATION AUTHORITY */
73  		objectList.add(filler(aRequestEcs.getTerminationAuthorityCode(), 7));
74  		/*TERMINATION DATE*/
75  		objectList.add(filler(convertDateToString(aRequestEcs
76  				.getLicenseSurrenderedProcessDate(),
77  				EcsConverterHelper.FORMATTER_MM_DD_YY), 6));
78  		/*ORIGINAL ACTION AUTHORITY*/
79  		objectList.add(filler(aRequestEcs.getOriginalAuthoritySelection(), 7));
80  		/*ORIGINAL ACTION EFFECTIVE DATE*/
81  		objectList.add(filler(convertDateToString(aRequestEcs
82  				.getOriginalActionEffectiveDate(),
83  				EcsConverterHelper.FORMATTER_MM_DD_YY), 6));
84  		/*LICENSE LOCATION */
85  		objectList.add(filler(aRequestEcs.getLicenseLocation(), 2));
86  		/*LICENSE LOCATION CONSTANT*/
87  		objectList.add(filler(LICENSE_LOCATION_CONSTANT, 1));
88  		/*RESPONSE CODE*/
89  		objectList.add(filler(RESPONSE_CODE, 2));
90  		Object[] tcodeRequestElementsArray = objectList.toArray();
91  		return createMessageFormat(tcodeRequestElementsArray);
92  		//return "DLX43UI80130M2I0004841TES13362  11151113362  021006600";
93  	}

112 	/* (non-Javadoc)
113 	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createResponse(java.lang.String)
114 	 */
115 	public IEcsResponse createResponseFromMultiplePayloads(
116 			List <String> inboundMessagePayloads) {
117 		StringBuilder strBuilder = new StringBuilder();
118 		for (String payload : inboundMessagePayloads) {
119 			String ebcidic = null;
120 			String screen = null;
121 			String formattedScreen = null;
122 			try {
123 				ebcidic = new String(payload.getBytes("Cp1047"));
124 				screen = Decoder3270.decode(ebcidic);
125 				formattedScreen = Decoder3270.formatScreen(screen);
126 				strBuilder.append(formattedScreen).append(LINE_SEPARATOR);
127 			}
128 			catch (UnsupportedEncodingException e) {
129 				throw new MessageConversionException(e);
130 			}
131 		}
132 		String formatResponse = StringUtils
133 				.trimTrailingWhitespace(removeDuplicateHeader(strBuilder));
134 		return new DriverLicenseTerminationUpdateResponseEcs(formatResponse);
135 	}

137 	/***
138 	 * Format the response by replacing the non keyboard chars with spaces and
139 	 * replace *1 with \n\n to match the exact AS-IS format.
140 	 * 
141 	 * @param ltrMessage
142 	 * @return
143 	*/
144 	private String formatReponse(String ltrMessage) {
145 		// If the response contains "*1" then replace with "\n\n" to display in
146 		// the next line.
147 		// Replace all non-keyboard characters with spcae.
148 		return ltrMessage.replaceAll(NON_KEY_BOARD_CHAR_REGEX, "").replace(
149 				"*1", "\n\n");
150 	}

83  	/***
84  	 * RecordDlReviewConverter.java
85  	 * This method creates the message text for a BDRV DL Review bridge message.
86  	 * 
87  	 * @param recordDReviewRequestEcs
88  	 * 		An IEcsRequest RecordDLReviewRequestEcs object that
89  	 *      contains an embedded DlApplication object that contains the
90  	 *      information to be transformed to the desired message text.
91  	 * 
92  	 * @return String The String value of the message text for the fee segment.  
93  	 * 		If the bridge message cannot be created successfully a null object is
94  	 * 		returned - for example if the DlApplication embedded within the incoming
95  	 * 		IEcsRequest object is null
96  	 * 
97  	 * @see gov.ca.dmv.ease.ecs.request.impl.RecordDlReviewRequestEcs
98  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
99  	 */
100 	@Override
101 	public String createMessage(IEcsRequest recordDlReviewRequestEcs) {
102 		// Get the Current User Context
103 		IUserContext currentUserContext = recordDlReviewRequestEcs
104 				.getUserContext();
105 		if (!(((RecordDlReviewRequestEcs) recordDlReviewRequestEcs)
106 				.getApplication() instanceof DlApplication)) {
107 			return null;
108 		}
109 		// Get the DlApplication from the Request object.
110 		DlApplication dlApplication = (DlApplication) ((RecordDlReviewRequestEcs) recordDlReviewRequestEcs)
111 				.getApplication();
112 		String messageText = null;
113 		if (dlApplication != null) {
114 			StringBuilder messageBuffer = null;
115 			try {
116 				// 1. Build the basic Dl Review Message Template
117 				messageBuffer = buildDlReviewMessageTemplate(
118 						currentUserContext, dlApplication);
119 				// 2. Populate all the fee review action(s) for this transaction (if
120 				// any)
121 				populateFeeWaivedActionMessageText(messageBuffer,
122 						currentUserContext, dlApplication);
123 				// 3. Populate the photo review action for this message, if any.
124 				populateDuplicatePhotoActionMessageText(messageBuffer,
125 						currentUserContext, dlApplication);
126 				// Update count of actual number of review records in message
127 				StringBuilder tempStringBuffer = new StringBuilder();
128 				this.addNumberToMessageText(tempStringBuffer,
129 						numberOfReviewActions, NBR_OF_REVIEW_ACTIONS_LENGTH);
130 				// StringBuffer replace end value must be one greater than the 
131 				// ending index (begin with IX=zero)
132 				messageBuffer.replace(START_NBR_OF_REVIEW_ACTIONS_IX,
133 						(END_NBR_OF_REVIEW_ACTIONS_IX + 1), tempStringBuffer
134 								.toString());
135 				messageText = messageBuffer.toString();
136 			}
137 			catch (EaseBridgeMessageException ebme) {
138 				LOGGER.error(
139 						"Unable to build BDRV RecordDlReview bridge message. ",
140 						ebme);
141 				messageText = null;
142 			}
143 		}
144 		LOGGER.info("Message:[" + messageText + "]");
145 		return messageText;
146 	}

148 	/***
149 	 * Build the basic DL Review message template which is common for all the
150 	 * Review Actions. The messageText will subsequently be editted to populate
151 	 * the review action specific data values, as needed.
152 	 * 
153 	 * Method visibility was initially set to private, but was revised to public in order to unit test.
154 	 * 
155 	 * @param currentUserContext The current IUserContext which contains basic operator 
156 	 * 		information
157 	 * @param dlApplication The DlApplication which contains the details of the current DL
158 	 * 		transaction.
159 	 * @return StringBuilder A StringBuilder that contains the message text that has been 
160 	 * 		prepared from the input data.
161 	 * @throws EaseBridgeMessageException The exception that is thrown if the data in 
162 	 * 		the DlApplication cannot be mapped successfully  - for example if the DlApplication is null
163 	 * 
164 	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
165 	 */
166 	public StringBuilder buildDlReviewMessageTemplate(
167 			IUserContext currentUserContext, DlApplication dlApplication)
168 			throws EaseBridgeMessageException {
169 		if (dlApplication == null) {
170 			throw new EaseBridgeMessageException(
171 					"Unpopulated DlApplication object");
172 		}
173 		StringBuilder messageBuffer = new StringBuilder();
174 		// Build the DL REVIEW Message - TCODE
175 		this.addStringToMessageText(messageBuffer, DL_REVIEW_RECORD_TCODE,
176 				DL_REVIEW_RECORD_TCODE_LENGTH);
177 		// Add Process Date
178 		this.addStringToMessageText(messageBuffer, WORKDATE_FORMAT
179 				.format(currentUserContext.getWorkDate()), PROCESS_DATE_LENGTH);
180 		// Add Office ID Number
181 		this.addStringToMessageText(messageBuffer, currentUserContext
182 				.getOfficeId(), OFFICE_ID_LENGTH);
183 		// Add Keyer's Id Code
184 		this.addStringToMessageText(messageBuffer, currentUserContext
185 				.getTechId(), KEYER_S_ID_LENGTH);
186 		// Add Type Transaction Code
187 		this.addStringToMessageText(messageBuffer, currentUserContext.getTtc(),
188 				TYPE_TRANSACTION_CODE_LENGTH);
189 		// Add Driver License Number
190 		this.addStringToMessageText(messageBuffer, BLANK,
191 				DRIVER_LICENSE_NUMBER_LENGTH);
192 		// Add First 3 Positions of Last Name
193 		this.addStringToMessageText(messageBuffer, BLANK,
194 				FIRST_3_OF_LAST_NAME_LENGTH);
195 		// Add initial number of records to message
196 		this.addNumberToMessageText(messageBuffer, 0,
197 				NBR_OF_REVIEW_ACTIONS_LENGTH);
198 		return messageBuffer;
199 	}

213 	/***
214 	 * Populate the Review Action message data for a fee waived review action.
215 	 * 
216 	 * Method visibility was initially set to private, but was revised to public in order to unit test.
217 	 * 
218 	 * @param currentUserContext The current IUserContext which contains basic operator 
219 	 * 		information
220 	 * @param dlApplication The DlApplication which contains the details of the current DL
221 	 * 		transaction.
222 	 * @param messageBuffer A StringBuilder that contains the message text that has been 
223 	 * 		prepared from the input data.  Fee Waived information will be appended to this 
224 	 * 		messageBuffer.
225 	 * 
226 	 * @throws EaseBridgeMessageException The exception that is thrown if the data in 
227 	 * 		the DlApplication cannot be mapped successfully - for example if the DlApplication is null
228 	 * 
229 	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
230 	 */
231 	private void populateFeeWaivedActionMessageText(
232 			StringBuilder messageBuffer, IUserContext currentUserContext,
233 			DlApplication dlApplication) throws EaseBridgeMessageException {
234 		if (dlApplication == null) {
235 			throw new EaseBridgeMessageException(
236 					"Unpopulated DlApplication object");
237 		}
238 		for (InvoiceItem invoiceItem : dlApplication.getInvoice()
239 				.getInvoiceItems()) {
240 			if (invoiceItem.getFeeWaivedReasonCode() != null) {
241 				numberOfReviewActions++;
242 				// Add Action Code Type
243 				this.addStringToMessageText(messageBuffer,
244 						FEE_WAIVED_ACTION_CODE, TYPE_OF_ACTION_CODE_LENGTH);
245 				// Add Docket/Citation Number
246 				this.addStringToMessageText(messageBuffer, BLANK,
247 						DOCKET_OR_CITATION_NUMBER_LENGTH);
248 				// Add Disposition Codes
249 				this.addStringToMessageText(messageBuffer, BLANK,
250 						DISPOSITION_CODES_LENGTH);
251 				// Add Type of Fee Waived Code
252 				this
253 						.addStringToMessageText(messageBuffer, invoiceItem
254 								.getFeeWaivedReasonCode().getCode(),
255 								CODE_WAIVED_LENGTH);
256 				// Add Violation Date
257 				this.addStringToMessageText(messageBuffer, BLANK,
258 						VIOLATION_DATE_LENGTH);
259 				// Add Conviction/Effective Date
260 				this.addStringToMessageText(messageBuffer, BLANK,
261 						CONVICTION_OR_EFFECTIVE_DATE_LENGTH);
262 				// Add Court Code
263 				this.addStringToMessageText(messageBuffer, BLANK,
264 						COURT_CODE_LENGTH);
265 				// Add Authority Section
266 				this.addStringToMessageText(messageBuffer, BLANK,
267 						AUTHORITY_SECTION_LENGTH);
268 				// Add Manager ID
269 				this.addStringToMessageText(messageBuffer, BLANK,
270 						MANAGER_ID_LENGTH);
271 				// Add Action Type
272 				this.addStringToMessageText(messageBuffer, BLANK,
273 						ACTION_TYPE_LENGTH);
274 				// Add Pad
275 				this.addStringToMessageText(messageBuffer, BLANK, PAD_LENGTH);
276 			}
277 		}
278 		return;
279 	}

99  	@Override
100 	public IEcsResponse createResponse(String anInboundMessagePayload) {
101 		int resultSize = 0;
102 		if (isNullOrBlank(anInboundMessagePayload)) {
103 			throw new MessageConversionException(
104 					IEcsErrorMessageConstants.ADQ_INVALID_MESSAGE);
105 		}
106 		AutomatedNameIndexDlInquiryResponseEcs response = new AutomatedNameIndexDlInquiryResponseEcs();
107 		String returnCode = anInboundMessagePayload.substring(80, 82);
108 		if (IEcsConverterConstants.GOOD_RETURN_CODE
109 				.equalsIgnoreCase(returnCode)) {
110 			resultSize = getResultSize(anInboundMessagePayload);
111 		}
112 		response.setStatusInfoCode(getCodeSetElement(
113 				IEcsConverterConstants.ADQ_CODESET_NAME, returnCode));
114 		if (anInboundMessagePayload.length() != (IEcsConverterConstants.ADQ_REQUEST_PART_OF_RESPONSE_LENGTH
115 				+ IEcsConverterConstants.ADQ_ERRORCODE_OF_RESPONSE_LENGTH + (resultSize * IEcsConverterConstants.ADQ_ANIRECORD_OF_RESPONSE_LENGTH))) {
116 			throw new MessageConversionException(
117 					IEcsErrorMessageConstants.INVALID_RESPONSE_LENGTH);
118 		}
119 		List <AniRecord> aniRecords = new ArrayList <AniRecord>();
120 		// Remove request from the result string
121 		String aniActualResponse = anInboundMessagePayload.substring(82);
122 		for (int i = 0; i < resultSize; i++) {
123 			AniRecord aniRecord = new AniRecord();
124 			String driverLicenseNumber = aniActualResponse.substring(0, 8)
125 					.trim();
126 			if (driverLicenseNumber.length() > 0) {
127 				aniRecord.setDriverLicenseNumber(driverLicenseNumber);
128 			}
129 			String names = aniActualResponse.substring(8, 43);
130 			aniRecord.setPersonName(createPersonName(names));
131 			Date birthDate = (convertStringToDate(aniActualResponse.substring(
132 					43, 49).trim(), EcsConverterHelper.FORMATTER_MM_DD_YY));
133 			aniRecord.setBirthDate(birthDate);
134 			aniRecord.setCityName(aniActualResponse.substring(49, 62).trim());
135 			aniRecords.add(aniRecord);
136 			aniActualResponse = aniActualResponse.substring(62);
137 		}
138 		response.setMatches(aniRecords);
139 		return response;
140 	}

142 	/***
143 	 * Parse the result.
144 	 * 
145 	 * @param names the names
146 	 * 
147 	 * @return personName the PersonName
148 	 */
149 	private PersonName createPersonName(String names) {
150 		StringTokenizer token = new StringTokenizer(names);
151 		PersonName personName = new PersonName();
152 		int count = 0;
153 		while (token.hasMoreElements()) {
154 			String str = ((String) token.nextElement()).trim();
155 			if (count == 0) {
156 				personName.setLastName(str);
157 			}
158 			else if (count == 1) {
159 				personName.setFirstName(str);
160 			}
161 			else if (count == 2) {
162 				if ((IEcsConverterConstants.SUFFIX_JR.equalsIgnoreCase(str))
163 						|| (IEcsConverterConstants.SUFFIX_SR
164 								.equalsIgnoreCase(str)
165 								|| (IEcsConverterConstants.SUFFIX_II
166 										.equalsIgnoreCase(str))
167 								|| (IEcsConverterConstants.SUFFIX_III
168 										.equalsIgnoreCase(str)) || (IEcsConverterConstants.SUFFIX_IV
169 								.equalsIgnoreCase(str)))) {
170 					personName.setSuffix(str);
171 				}
172 				else {
173 					personName.setMiddleName(str);
174 				}
175 			}
176 			else if (count == 3) {
177 				personName.setSuffix(str);
178 			}
179 			count++;
180 		}
181 		return personName;
182 	}

185 	 * Gets the city based on the state of the person.
186 	 * 
187 	 * @param person the person
188 	 * 
189 	 * @return the city of the person
190 	 */
191 	private String getCity(Person person) {
192 		Address address = null;
193 		//The list of locators has only 1 address.
194 		if (isNotNull(person.getLocators())) {
195 			if (person.getLocators().size() > 0) {
196 				if (isNotNull(person.getLocators().get(0))) {
197 					address = (Address) person.getLocators().get(0);
198 					if (isNotNull(address)) {
199 						CodeSetElement state = address.getStateCode();
200 						String city = address.getCity();
201 						if (isNotNull(city) && isNotNull(state)) {
202 							if (state.getCode().equalsIgnoreCase(
203 									IEcsConverterConstants.STATE_CA)
204 									|| state.getCode().equalsIgnoreCase(
205 											IEcsConverterConstants.SPACE)) {
206 								return filler(city, 13);
207 							}
208 							else {
209 								return filler(IEcsConverterConstants.STAR
210 										+ city, 12)
211 										+ IEcsConverterConstants.SPACE;
212 							}
213 						}
214 					}
215 				}
216 			}
217 		}
218 		return BLANK;
219 	}

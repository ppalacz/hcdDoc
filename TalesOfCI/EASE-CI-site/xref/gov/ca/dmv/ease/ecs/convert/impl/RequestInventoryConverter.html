49  		// NOTE : This Tcode is used in both DL and VR. For now, it is implemented only for DL.

51  		RequestInventoryRequestEcs request = (RequestInventoryRequestEcs) requestFromClient;
52  		Object[] message = new Object[] {
53  				TCODE, //TCODE
54  				getOfficeId(request.getUserContext()), // OFFICE_ID
55  				getWorkDate(request.getUserContext(),
56  						EcsConverterHelper.FORMATTER_MM_DD_YY), // WORKDATE
57  				getEmployeeID(request.getUserContext()),// TECHNICIAN ID		
58  				// As per the confirmation from requirement team, SEQUENCE# is a current system date in MMDD format				
59  				convertDateToString(new Date(),
60  						EcsConverterHelper.FORMATTER_HH_MM), // SEQUENCE #		
61  				INV_ITEM_CODE_FOR_DL_AND_ID, // INVENTORY ITEM CODE
62  				filler(BLANK, 8), // CENTRAL ISSUED LICENSE #
63  				// NOTE : For DL/ID, TYPE LICENSE is blank. For VR, it is not blank.
64  				filler(BLANK, 2),// TYPE LICENSE
65  				// NOTE : For DL/ID, VIN is blank. For VR, it is not blank.
66  				filler(BLANK, 30), // VIN
67  				// NOTE : For DL/ID, FIRST 3 POS MAKE is blank, For VR, it is not blank.
68  				filler(BLANK, 3),// FIRST 3 POS MAKE
69  				getTypeTransactionCode(request.getUserContext()), // TYPE TRANS CODE (DMVA)
70  				filler(BLANK, 7) // FILLER
71  		};
72  		return createMessageFormat(message);
73  	}

99  	public IEcsResponse createResponse(String c77MessagePayload) {
100 		RequestInventoryResponseEcs response = new RequestInventoryResponseEcs();
101 		// NOTE : This Tcode is used in both DL and VR. For now, it is implemented only for DL. In future, it has to be modified to accept VR related information also.
102 		final int C77_MESSAGE_LENGTH = 74;
103 		if (isNullOrBlank(c77MessagePayload)) {
104 			throw new MessageConversionException(
105 					"Invalid Response: Unable to process on null message");
106 		}
107 		int aMessageLength = c77MessagePayload.length();
108 		if (aMessageLength != C77_MESSAGE_LENGTH) {
109 			throw new MessageConversionException(
110 					"Invalid Response: Message size is: " + aMessageLength
111 							+ " expected: " + C77_MESSAGE_LENGTH);
112 		}
113 		//String tCode = anInboundMessagePayload.substring(0, 3); // TCODE
114 		//String officeId = anInboundMessagePayload.substring(3, 6); // OFFICE ID --- Not referenced
115 		//String workDate = anInboundMessagePayload.substring(6, 12); // WORK DATE --- Not referenced
116 		//String techId = anInboundMessagePayload.substring(12, 14); // TECHNICIAN ID --- Not referenced
117 		//String techSequenceNumber =  anInboundMessagePayload.substring(14, 18); // TECH SEQUENCE NUMBER --- Not referenced
118 		String ttc = c77MessagePayload.substring(64, 67); // TYPE TRANS CODE (DMVA) --- Not referenced
119 		String inventoryItemCode = c77MessagePayload.substring(18, 21).trim();// INVENTORY ITEM CODE	--- It is always 400 for both DL and ID
120 		setIssuedLicenseOrIdNumber(c77MessagePayload.substring(21, 29).trim(),
121 				ttc, inventoryItemCode, response); // CENTRAL ISSUED LICENSE/ID #		
122 		// String typeLicense = anInboundMessagePayload.substring(29, 31); // TYPE LICENSE --- Not referenced
123 		// String vin = anInboundMessagePayload.substring(31, 61); // VIN --- Not referenced
124 		// String first3PosMake = anInboundMessagePayload.substring(61, 64); // FIRST 3 POS MAKE --- Not referenced
125 		// For good response, error code is blank. For error code = 12, error message = Out of Central Inventory). For all other error codes, error message = "CNA"
126 		String errorCode = c77MessagePayload.substring(67).trim();// ERROR CODE
127 		if (!isBlank(errorCode)) {
128 			if (errorCode.equals(ERROR_CODE_12)) {
129 				response.setErrorMessage(MESSAGE_FOR_ERROR_CODE_12);
130 			}
131 			else {
132 				response.setErrorMessage(MESSAGE_FOR_OTHER_ERROR_CODE);
133 			}
134 		}
135 		return response;
136 	}

138 	/***
139 	 * Sets the issued license or id number.
140 	 * 
141 	 * @param issuedLicenseOrIdNumber the issued license or id number
142 	 * @param ttc the ttc
143 	 * @param inventoryItemCode the inventory item code
144 	 * @param response the response
145 	 */
146 	private void setIssuedLicenseOrIdNumber(String issuedLicenseOrIdNumber,
147 			String ttc, String inventoryItemCode,
148 			RequestInventoryResponseEcs response) {
149 		if (ttc.startsWith("D")) { // For all DL related TCodes, issued inventory is Driver License
150 			DriverLicense driverLicense = new DriverLicense();
151 			driverLicense.setItemCode(inventoryItemCode);
152 			driverLicense.setLicenseNumber(issuedLicenseOrIdNumber);
153 			response.setLicense(driverLicense);
154 		}
155 		else if (ttc.startsWith("I")) { // For all ID related TCodes, issued inventory is Id Card 
156 			IdCard idCard = new IdCard();
157 			idCard.setItemCode(inventoryItemCode);
158 			idCard.setLicenseNumber(issuedLicenseOrIdNumber);
159 			response.setLicense(idCard);
160 		}
161 		else {
162 			response.setLicense(null);
163 		}
164 	}

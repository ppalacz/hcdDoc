218 	@Override
219 	public IEcsResponse createResponse(String anInboundMessagePayload) {
220 		CdlisStateOfRecordResponseEcs response = new CdlisStateOfRecordResponseEcs();
221 		Map <String, String> tcodeResponseElementsMap = this
222 				.buildTcodeResponseElementsMap(anInboundMessagePayload);
223 		this.errorProcessing(tcodeResponseElementsMap, response);
224 		if (!response.hasErrors()) {
225 			this.createResponse(tcodeResponseElementsMap, response);
226 		}
227 		return response;
228 	}

230 	/***
231 	 * The purpose of this method is to process the errors.
232 	 * 
233 	 * @param tcodeResponseElementsMap elements map for CDS response
234 	 * @param response response object
235 	 */
236 	private void errorProcessing(Map <String, String> tcodeResponseElementsMap,
237 			CdlisStateOfRecordResponseEcs response) {
238 		String messageType = tcodeResponseElementsMap.get("MESSAGE_TYPE");
239 		String processingStatus = tcodeResponseElementsMap
240 				.get("PROCESSING_STATUS");
241 		String errorIndicator = tcodeResponseElementsMap.get("ERROR_INDICATOR");
242 		String matchIndicator = tcodeResponseElementsMap.get("MATCH_INDICATOR");
243 		String soiHeader = tcodeResponseElementsMap.get("SOI_HEADER_FIELD");
244 		String firstErrorCode = tcodeResponseElementsMap
245 				.get("1ST_ERROR_CODE_1");
246 		String firstErrorCodeDescription = tcodeResponseElementsMap
247 				.get("1ST_ERROR_DESCRIPTION_1");
248 		if (StringUtils.equals(messageType, "HG")) {
249 			if (StringUtils.equals(matchIndicator, "Y")
250 					&& StringUtils.equals(errorIndicator, "N")) {
251 				return;
252 			}
253 			if (StringUtils.equals(processingStatus, "01")
254 					|| StringUtils.isBlank(processingStatus)) {
255 				ConverterResponseHelper.addErrorToResponse(response, "Q618",
256 						"CDLIS/STATE OF RECORD DISCREPANCY");
257 			}
258 			else if (StringUtils.equals(processingStatus, "02")) {
259 				ConverterResponseHelper.addErrorToResponse(response, "Q617",
260 						"STATE OF RECORD CNA");
261 			}
262 			else if (StringUtils.equals(processingStatus, "05")) {
263 				ConverterResponseHelper.addErrorToResponse(response, "Q624",
264 						"RECORD UNAVAILABLE - RETRY IN 4 DAYS");
265 			}
266 			else {
267 				ConverterResponseHelper.addErrorToResponse(response, "0569",
268 						"NETWORK TRANSMISSION ERROR");
269 			}
270 		}
271 		if (StringUtils.equals(errorIndicator, "D")) {
272 			ConverterResponseHelper.addErrorToResponse(response, "0616",
273 					"AAMVA NETWORK CNA");
274 		}
275 		else if (StringUtils.equals(errorIndicator, "U")) {
276 			ConverterResponseHelper.addErrorToResponse(response, "0617",
277 					"STATE OF RECORD CNA");
278 		}
279 		else if (StringUtils.equals(errorIndicator, "Y")) {
280 			if (StringUtils.equals(processingStatus, "03")) {
281 				if (StringUtils.isNotBlank(firstErrorCode)) {
282 					ConverterResponseHelper.addErrorToResponse(response,
283 							firstErrorCode, firstErrorCodeDescription);
284 				}
285 				else {
286 					ConverterResponseHelper.addErrorToResponse(response, "",
287 							"EDIT ERROR BLOCK MISSING");
288 				}
289 			}
290 			else {
291 				ConverterResponseHelper.addErrorToResponse(response, "0569",
292 						"NETWORK TRANSMISSION ERROR");
293 			}
294 		}
295 		else if (StringUtils.equals(soiHeader, "CNA")
296 				|| StringUtils.equals(soiHeader, "TIM")) {
297 			ConverterResponseHelper.addErrorToResponse(response, "Q616",
298 					"AAMVA NETWORK CNA");
299 		}
300 		else if (StringUtils.equals(soiHeader, "PAR")) {
301 			ConverterResponseHelper.addErrorToResponse(response, "0569",
302 					"NETWORK TRANSMISSION ERROR");
303 		}
304 	}

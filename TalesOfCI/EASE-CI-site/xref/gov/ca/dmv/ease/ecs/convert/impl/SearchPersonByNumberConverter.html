200

202 			List <CodeSetElement> appliedLicenseClasses = ((DriverLicense) license)

260 	@Override
261 	public IEcsResponse createResponse(String anInboundMessagePayload) {
262 		final int SIZE_OF_SEGMENT_0 = 327;
263 		LOGGER.info("anInboundMessagePayload : [" + anInboundMessagePayload
264 				+ "]");
265 		SearchPersonByNumberResponseEcs response = initializeBusinessObjects();
266 		final String ELI_TCODE = getMessageFromResourceBundle("TCODE_ELI");
267 		// Validating and populating Echoed Request	
268 		validateMessageSize(anInboundMessagePayload, MIN_RESPONSE_SIZE);
269 		populateEchoedRequest(anInboundMessagePayload, 0, response);
270 		// Set Error Code and Error Message
271 		setReturnCode(anInboundMessagePayload);// ERROR CODE		
272 		if (anInboundMessagePayload.length() == MIN_RESPONSE_SIZE) {// If there is a error in the response, then response length will be 23
273 			return response;
274 		}
275 		// Validating and populating Segment 0
276 		validateMessageSize(anInboundMessagePayload, MIN_RESPONSE_SIZE
277 				+ SIZE_OF_SEGMENT_0);
278 		addPopulatingSegmentIndicator(SEGMENT_0, ELI_TCODE,
279 				anInboundMessagePayload);
280 		populateSegment0(anInboundMessagePayload, MIN_RESPONSE_SIZE);
281 		// Validating and populating other segments
282 		if (anInboundMessagePayload.length() > MIN_RESPONSE_SIZE
283 				+ SIZE_OF_SEGMENT_0) {// Along with Segment 0, if more segments exists
284 			// Decide the start positions of different segments depending on the availability of segments A,A2 and B
285 			int nextSegmentStartPos = -1; // next segment can be C, E, H or I
286 			final int SIZE_OF_SEGMENTAORB_EXISTANCE_INDICATOR = 1;
287 			final int SIZE_OF_SEGMENT_A = 50;
288 			final int SIZE_OF_SEGMENT_A2 = 195;
289 			final int SIZE_OF_SEGMENT_B = 55;
290 			final int SIZE_OF_SEGMENT_C = 60;
291 			final int SIZE_OF_SEGMENT_E = 60;
292 			final int SIZE_OF_SEGMENT_H = 120;
293 			final int SIZE_OF_SEGMENT_I = 120;
294 			String segmentIndicator = getSubString(anInboundMessagePayload,
295 					MIN_RESPONSE_SIZE + SIZE_OF_SEGMENT_0, MIN_RESPONSE_SIZE
296 							+ SIZE_OF_SEGMENT_0 + 1); // DL/ID PENDING SUBRECORD SEGMENT
297 			LOGGER.info("segmentIndicator:" + segmentIndicator);
298 			if (segmentIndicator.equals(SEGMENT_A)) {// If A then A and A2 exist. Validate and populate Segment A and A2.
299 				int segmentAStartPos = MIN_RESPONSE_SIZE + SIZE_OF_SEGMENT_0
300 						+ 1;
301 				int segmentA2StartPos = segmentAStartPos + SIZE_OF_SEGMENT_A;
302 				LOGGER.info("Segment A and A2 exist");
303 				validateMessageSize(anInboundMessagePayload, MIN_RESPONSE_SIZE
304 						+ SIZE_OF_SEGMENT_0
305 						+ SIZE_OF_SEGMENTAORB_EXISTANCE_INDICATOR
306 						+ SIZE_OF_SEGMENT_A);
307 				addPopulatingSegmentIndicator(SEGMENT_A, ELI_TCODE,
308 						anInboundMessagePayload);
309 				String birthDate = populateSegmentA(anInboundMessagePayload,
310 						segmentAStartPos);
311 				validateMessageSize(anInboundMessagePayload, MIN_RESPONSE_SIZE
312 						+ SIZE_OF_SEGMENT_0
313 						+ SIZE_OF_SEGMENTAORB_EXISTANCE_INDICATOR
314 						+ SIZE_OF_SEGMENT_A + SIZE_OF_SEGMENT_A2);
315 				addPopulatingSegmentIndicator(SEGMENT_A2, ELI_TCODE,
316 						anInboundMessagePayload);
317 				populateSegmentA2(anInboundMessagePayload, segmentA2StartPos,
318 						birthDate);
319 				nextSegmentStartPos = segmentA2StartPos + SIZE_OF_SEGMENT_A2;
320 			}
321 			else if (segmentIndicator.equals(SEGMENT_B)) {// If B then A and B exist. Validate and populate Segment A and B.
322 				int segmentAStartPos = MIN_RESPONSE_SIZE + SIZE_OF_SEGMENT_0
323 						+ 1;
324 				int segmentBStartPos = segmentAStartPos + SIZE_OF_SEGMENT_A;
325 				LOGGER.info("Segment A and B exist");
326 				validateMessageSize(anInboundMessagePayload, MIN_RESPONSE_SIZE
327 						+ SIZE_OF_SEGMENT_0
328 						+ SIZE_OF_SEGMENTAORB_EXISTANCE_INDICATOR
329 						+ SIZE_OF_SEGMENT_A);
330 				addPopulatingSegmentIndicator(SEGMENT_A, ELI_TCODE,
331 						anInboundMessagePayload);
332 				String birthDate = populateSegmentA(anInboundMessagePayload,
333 						segmentAStartPos);
334 				validateMessageSize(anInboundMessagePayload, MIN_RESPONSE_SIZE
335 						+ SIZE_OF_SEGMENT_0
336 						+ SIZE_OF_SEGMENTAORB_EXISTANCE_INDICATOR
337 						+ SIZE_OF_SEGMENT_A + SIZE_OF_SEGMENT_B);
338 				addPopulatingSegmentIndicator(SEGMENT_B, ELI_TCODE,
339 						anInboundMessagePayload);
340 				populateSegmentB(anInboundMessagePayload, segmentBStartPos,
341 						birthDate);
342 				nextSegmentStartPos = segmentBStartPos + SIZE_OF_SEGMENT_B;
343 			}
344 			else { // If A, A2 and B don't exist, then start from next segment(C/E/H/I)
345 				LOGGER.info("Segment A, A2 and B does not exist");
346 				nextSegmentStartPos = MIN_RESPONSE_SIZE + SIZE_OF_SEGMENT_0;
347 			}
348 			// Validate and populate other segments
349 			int originalSizeofInputMsg = anInboundMessagePayload.length();
350 			while (nextSegmentStartPos < originalSizeofInputMsg) {
351 				boolean validSegmentFound = false;
352 				boolean isSegmentH = false;
353 				String inboundMessageForNextSegment = anInboundMessagePayload
354 						.substring(nextSegmentStartPos);
355 				segmentIndicator = inboundMessageForNextSegment.substring(0, 1);
356 				if (segmentIndicator.equals(SEGMENT_C)) {// Validate and populate Segment C
357 					validSegmentFound = true;
358 					validateMessageSize(inboundMessageForNextSegment,
359 							SIZE_OF_SEGMENT_C);
360 					addPopulatingSegmentIndicator(SEGMENT_C, ELI_TCODE,
361 							inboundMessageForNextSegment);
362 					populateSegmentC(inboundMessageForNextSegment, 0);
363 					nextSegmentStartPos = nextSegmentStartPos
364 							+ SIZE_OF_SEGMENT_C;
365 				}
366 				else if (segmentIndicator.equals(SEGMENT_E)) {// Validate and populate Segment E
367 					validSegmentFound = true;
368 					validateMessageSize(inboundMessageForNextSegment,
369 							SIZE_OF_SEGMENT_E);
370 					addPopulatingSegmentIndicator(SEGMENT_E, ELI_TCODE,
371 							inboundMessageForNextSegment);
372 					populateSegmentE(inboundMessageForNextSegment, 0);
373 					nextSegmentStartPos = nextSegmentStartPos
374 							+ SIZE_OF_SEGMENT_E;
375 				}
376 				else if (segmentIndicator.equals(SEGMENT_H)) {// Validate and populate Segment H
377 					isSegmentH = true;
378 					validSegmentFound = true;
379 					validateMessageSize(inboundMessageForNextSegment,
380 							SIZE_OF_SEGMENT_H);
381 					addPopulatingSegmentIndicator(SEGMENT_H, ELI_TCODE,
382 							inboundMessageForNextSegment);
383 					populateSegmentH(inboundMessageForNextSegment, 0);
384 					nextSegmentStartPos = nextSegmentStartPos
385 							+ SIZE_OF_SEGMENT_H;
386 				}
387 				else if (segmentIndicator.equals(SEGMENT_I)) {// Validate and populate Segment I
388 					validSegmentFound = true;
389 					validateMessageSize(inboundMessageForNextSegment,
390 							SIZE_OF_SEGMENT_I);
391 					addPopulatingSegmentIndicator(SEGMENT_I, ELI_TCODE,
392 							inboundMessageForNextSegment);
393 					populateSegmentI(inboundMessageForNextSegment, 0);
394 					nextSegmentStartPos = nextSegmentStartPos
395 							+ SIZE_OF_SEGMENT_I;
396 				}
397 				setBackgroundIndicatorAndExpirationDate(
398 						getEchoedDriverLicense(), getEchoedTCode(), isSegmentH);
399 				validateSegmentIndicator(validSegmentFound, nextSegmentStartPos);
400 			}
401 		}
402 		return response;
403 	}

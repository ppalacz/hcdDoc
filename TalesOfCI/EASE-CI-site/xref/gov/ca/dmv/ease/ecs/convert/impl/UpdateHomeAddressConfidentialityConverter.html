30  public class UpdateHomeAddressConfidentialityConverter extends
31  		AbstractEcsMessageConverter {
32  	/* (non-Javadoc)
33  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
34  	 */
35  	@Override
36  	public String createMessage(IEcsRequest requestFromClient) {
37  		UpdateHomeAddressConfidentialityRequestEcs aRequest = 
38  			(UpdateHomeAddressConfidentialityRequestEcs) requestFromClient;
39  		IUserContext userContext = aRequest.getUserContext();
40  		// Assumption : validation for null check of all request parameters will 
41  		//be done by the respective validator class.
42  		Object[] values = new Object[9];
43  		values[0] = IEcsConverterConstants.TCODE_RP1; //TRANSACTION CODE
44  		values[1] = getOfficeId(userContext); //OFFICE ID
45  		values[2] = getWorkDate(userContext); // WORK DATE
46  		values[3] = getEmployeeID(userContext); //EMPLOYEE ID
47  		values[4] = aRequest.getVehicleFileCode(); //FILE CODE
48  		values[5] = filler(aRequest.getVehicleLicenseNumber(), 7,
49  				EcsConverterHelper.EMPTY_FILLER, EcsConverterHelper.PAD_LEFT); // LICENSE NUMBER
50  		values[6] = filler(aRequest.getRegisteredOwnerName(), 3); //BASIC R/O NAME
51  		values[7] = filler(this.getPendingRegisteredOwnerName(
52  				aRequest.getPendingRegisteredOwnerName(), aRequest.getUpdateFor()), 3); // PENDING R/O NAME
53  		values[8] = filler(this.getEmployeeAgencyName(aRequest
54  				.getEmployingAgencyName(), aRequest.getRequestedFor()), 28); // PEACE OFFICER AGENCY
55  		MessageFormat messageFormat = new MessageFormat(
56  				"{0}{1}{2}{3}{4}{5}{6}{7}{8}");
57  		String returnString = messageFormat.format(values);
58  		return returnString;
59  	}
60  
61  	/***
62  	 * Gets the pending registered owner name.
63  	 * 
64  	 * @param pendingRegisteredOwnerName the pending registered owner name
65  	 * @param updateFor the update for
66  	 * 
67  	 * @return the pending registered owner name
68  	 */
69  	private String getPendingRegisteredOwnerName(
70  			String pendingRegisteredOwnerName, String updateFor) {
71  		String requiredName = BLANK;
72  		if(!EaseUtil.isNullOrBlank(pendingRegisteredOwnerName) 
73  				&& !EaseUtil.isNullOrBlank(updateFor)) {
74  			if(updateFor.equalsIgnoreCase(
75  					IEcsConverterConstants.PEACE_OFFICE_REQUEST_UPDATE_FOR_REGISTERED_OWNER)) {
76  				//If update is for only 'Registered Owner' then pending name will be blank.
77  				return BLANK;
78  			}
79  			else {
80  				return pendingRegisteredOwnerName;
81  			}
82  		}
83  		return requiredName;
84  	}
85  	/***
86  	 * Gets the employee agency name.
87  	 * 
88  	 * @param agencyName the agency name
89  	 * @param updateFor the update for
90  	 * 
91  	 * @return the employee agency name
92  	 */
93  	private String getEmployeeAgencyName(
94  			String agencyName, String requestedFor) {
95  		String requiredName = BLANK;
96  		if (!EaseUtil.isNullOrBlank(agencyName) && !EaseUtil.isNullOrBlank(requestedFor)) {
97  			if(requestedFor.equalsIgnoreCase(
98  					IEcsConverterConstants.PEASE_OFFICER_REUQEST_FOR_EMPLOYEE)) {
99  				//If requested code for Employee
100 				requiredName = agencyName;
101 			}
102 			else {
103 				//If requested code for Spouse or Child append first character to the return string
104 				requiredName = agencyName + EMPTY_SPACE + requestedFor.charAt(0);
105 			}
106 		}
107 		return requiredName;
108 	}

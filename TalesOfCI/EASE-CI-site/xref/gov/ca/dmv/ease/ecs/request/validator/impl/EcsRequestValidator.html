75  	@Override
76  	public void validateValidatable(IValidatable validatable,
77  			IErrorCollector collector) {
78  		if (validatable == null) {
79  			collector.register(new EaseValidationException(
80  					NULL_REQUEST_ERROR_MESSAGE));
81  			return;
82  		}
83  		else {
84  			if (validatable instanceof IEcsRequest) {
85  				//Validate User Context
86  				validateUserContext(((IEcsRequest) validatable)
87  						.getUserContext(), collector);
88  				//Validate Request
89  				validateRequest((IEcsRequest) validatable, collector);
90  			}
91  			else {
92  				collector.register(new EaseValidationException(
93  						WRONG_REQUEST_TYPE_ERROR_MESSAGE
94  								+ validatable.getClass().getName()));
95  			}
96  		}
97  	}

100 	 * Validates user context.
101 	 *

103 	 * @param aCollector the a collector

105 	 * @throws ValidationException the validation exception
106 	 * @throws EaseValidationException the ease validation exception
107 	 */
108 	protected static void validateUserContext(IUserContext userContext,
109 			IErrorCollector aCollector) throws EaseValidationException {
110 		if (null != userContext) {
111 			String strEmpId = userContext.getTechId();
112 			String strOfficeId = userContext.getOfficeId();
113 			String strTtc = userContext.getTtc();
114 			Date workDate = userContext.getWorkDate();
115 			String strRacfId = userContext.getRacfId();
116 			//Validating Employee Id
117 			if (isNullOrBlank(strEmpId)) {
118 				aCollector.register(new EaseValidationException(
119 						"Invalid Employee Id"));
120 			}
121 			//Validating Office Id
122 			if (isNullOrBlank(strOfficeId)) {
123 				aCollector.register(new EaseValidationException(
124 						"Invalid Office Id"));
125 			}
126 			//Validating TTC
127 			if (isNullOrBlank(strTtc)) {
128 				aCollector.register(new EaseValidationException("Invalid TTC"));
129 			}
130 			//Validating Work date
131 			if (workDate == null) {
132 				aCollector.register(new EaseValidationException(
133 						"Invalid Work date"));
134 			}
135 			//Validating RACF Id
136 			if (isNullOrBlank(strRacfId)) {
137 				aCollector.register(new EaseValidationException(
138 						"Invalid RACF Id"));
139 			}
140 		}
141 		else {
142 			aCollector.register(new EaseValidationException(
143 					"Invalid UserContext : NULL"));
144 		}

172 	/***
173 	 * Validates person.
174 	 * 
175 	 * @param aPerson the a person
176 	 * @param aCollector the a collector
177 	 */
178 	protected void validatePerson(Person aPerson, IErrorCollector aCollector) {
179 		if (aPerson == null) {
180 			aCollector.register(new EaseValidationException("Person is null"));
181 			return;
182 		}
183 		PersonName personName = aPerson.getCurrentName();
184 		if (personName == null) {
185 			aCollector.register(new EaseValidationException(
186 					"Person name is null"));
187 		}
188 		if (aPerson.getBirthDate() == null) {
189 			aCollector.register(new EaseValidationException(
190 					"Person DOB is null"));
191 		}
192 		String lastName = personName.getLastName();
193 		if (isNullOrBlank(lastName)) {
194 			aCollector.register(new EaseValidationException(
195 					"Person's last name is not set"));
196 		}
197 		String firstName = personName.getFirstName();
198 		if (isNullOrBlank(firstName)) {
199 			aCollector.register(new EaseValidationException(
200 					"Person's first name is not set"));
201 		}
202 		Address address = aPerson.getAddress(AddressType.MAILING);
203 		if (address == null) {
204 			aCollector.register(new EaseValidationException("Address is null"));
205 		}
206 		String city = address.getCity();
207 		if (isNullOrBlank(city)) {
208 			aCollector.register(new EaseValidationException(
209 					"City name is not set"));
210 		}
211 	}

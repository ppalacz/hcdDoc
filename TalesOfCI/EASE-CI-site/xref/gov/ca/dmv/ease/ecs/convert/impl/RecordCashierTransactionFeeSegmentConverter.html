52  	/***
53  	 * This method creates the message text for the fee segment within a BCTR
54  	 * Cashier Transaction bridge message.
55  	 * 
56  	 * @param dlApplication
57  	 *            A DlApplication object that contains the information to be
58  	 *            transformed to the desired message text.
59  	 * 
60  	 * @return String The String value of the message text for the fee segment.
61  	 * 
62  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage
63  	 *      (gov.ca.dmv.ease.ecs.request.IEcsRequest)
64  	 */
65  	public String createMessage(DlApplication dlApplication) {
66  		//The StringBuilder
67  		StringBuilder messageBuffer = new StringBuilder();
68  		//Add constant "F"
69  		addStringToMessageText(messageBuffer, "F", 1);
70  		Invoice invoice = dlApplication.getInvoice();
71  		List <InvoiceItem> invoiceItems = null;
72  		if (EaseUtil.isNotNull(invoice))
73  			invoiceItems = invoice.getInvoiceItems();
74  		else {
75  			addNumberToMessageText(messageBuffer, 0, 1); //Add Number of Fee Entries
76  			addNumberToMessageText(messageBuffer, 0, 1); //Add Next Location Number - CONSTANT, always 0
77  			return messageBuffer.toString();
78  		}
79  		int numberOfFees = 0; // number of fees
80  		// Get the number of fees. Is it correct to get only the 1st invoice item?		
81  		if ((invoiceItems.get(0).getDriverLicenceFee() != null)
82  				&& (invoiceItems.get(0).getDriverLicenceFee()
83  						.getDriverLicenseRateStatuses() != null)) {
84  			numberOfFees = invoiceItems.get(0).getDriverLicenceFee()
85  					.getDriverLicenseRateStatuses().size();
86  			addNumberToMessageText(messageBuffer, numberOfFees, 2); // Add number of Fees Entries
87  			addNumberToMessageText(messageBuffer, 0, 1); // Add Next Location Number
88  			// Add fee info for each fee
89  			for (InvoiceItem invoiceItem : invoiceItems) {
90  				Fee fee = invoiceItem.getFee();
91  				if (EaseUtil.isNotNull(fee)) {
92  					numberOfFees++;
93  					// Add Fee Code
94  					/***
95  					 * TODO: validate w/ BO that it is the first DriverLicenseRateStatuses in the list what we need to use;
96  					 */
97  					List <DriverLicenseRateStatus> dlRateStatusList = ((DriverLicenseFee) fee)
98  							.getDriverLicenseRateStatuses();
99  					if ((EaseUtil.isNotNull(dlRateStatusList)
100 							&& (dlRateStatusList.size() > 0)
101 							&& EaseUtil.isNotNull(dlRateStatusList.get(0)) && !EaseUtil
102 							.isNullOrBlank(dlRateStatusList.get(0)
103 									.getFeeRatecode())))
104 						addStringToMessageText(messageBuffer, dlRateStatusList
105 								.get(0).getFeeRatecode().getCode(), 3);
106 					else
107 						addStringToMessageText(messageBuffer, BLANK, 3);
108 					// Add Waiver Reason Code
109 					if (!EaseUtil.isNullOrBlank(invoiceItem
110 							.getFeeWaivedReasonCode()))
111 						addStringToMessageText(messageBuffer, invoiceItem
112 								.getFeeWaivedReasonCode().getCode(), 1);
113 					else
114 						this.addStringToMessageText(messageBuffer, BLANK, 1);
115 					// Add Fee Amount 
116 					if (EaseUtil.isNotNull(fee)
117 							&& EaseUtil.isNotNull(fee.getAmount()))
118 						addBigDecimalToMessageText(messageBuffer, fee
119 								.getAmount(), 9);
120 					else
121 						addBigDecimalToMessageText(messageBuffer,
122 								BigDecimal.ZERO, 9);
123 					// Add Fee Waived Amount
124 					if (EaseUtil.isNotNull(invoiceItem.getWaivedAmount()))
125 						addBigDecimalToMessageText(messageBuffer, invoiceItem
126 								.getWaivedAmount(), 9);
127 					else
128 						addBigDecimalToMessageText(messageBuffer,
129 								BigDecimal.ZERO, 9);
130 					// Add Approving Manager ID TODO: I think we need to add field to
131 					// data object Invoice or InvoiceItem for Manager
132 					addStringToMessageText(messageBuffer, BLANK, 2);
133 				}
134 				else
135 					addStringToMessageText(messageBuffer, BLANK, 24);
136 			}
137 			StringBuilder numberOfFeesBuffer = new StringBuilder();
138 			//		this.addNumberToMessageText(numberOfFeesBuffer, numberOfFees,
139 			//				NUMBER_OF_FEE_ENTRIES_LENGTH);
140 			//		messageBuffer.replace(START_NUMBER_OF_FEE_ENTRIES_IX,
141 			//				END_NUMBER_OF_FEE_ENTRIES_IX, numberOfFeesBuffer.toString());
142 			return messageBuffer.toString();
143 		}
144 		else {
145 			addNumberToMessageText(messageBuffer, 0, 1); //Add Number of Fee Entries
146 			addNumberToMessageText(messageBuffer, 0, 1); //Add Next Location Number - CONSTANT, always 0
147 			return messageBuffer.toString();
148 		}
149 	}

64  	 * @return CDLIS/PDPS header record
65  	 */
66  	private CdlisPdpsHeaderInfo buildCdlisHeaderInfo(
67  			Map <String, String> tcodeResponseElementsMap) {
68  		CdlisPdpsHeaderInfo cdlisPdpsHeaderInfo = new CdlisPdpsHeaderInfo();
69  		/* MESSAGE_TYPE */
70  		String messageType = tcodeResponseElementsMap.get(MESSAGE_TYPE_KEY);
71  		cdlisPdpsHeaderInfo.setMessageType(messageType);
72  		/* ERROR_INDICATOR */
73  		String errorIndicator = tcodeResponseElementsMap
74  				.get(ERROR_INDICATOR_KEY);
75  		cdlisPdpsHeaderInfo.setErrorIndicator(errorIndicator);
76  		/* MATCH_COUNT */
77  		String matchCount = tcodeResponseElementsMap.get(MATCH_COUNT_KEY);
78  		cdlisPdpsHeaderInfo.setMatchCount(Integer.parseInt(matchCount));
79  		/* MATCH_INDICATOR */
80  		String matchIndicator = tcodeResponseElementsMap
81  				.get(MATCH_INDICATOR_KEY);
82  		cdlisPdpsHeaderInfo.setMatchIndicator(matchIndicator);
83  		/* AKA_DL_NUMBER_COUNT */
84  		String akaDlNumberCount = tcodeResponseElementsMap
85  				.get(AKA_DL_NUMBER_COUNT_KEY);
86  		if (StringUtils.isNotBlank(akaDlNumberCount)) {
87  			cdlisPdpsHeaderInfo.setAkaDlNumberCount(Integer
88  					.parseInt(akaDlNumberCount));
89  		}
90  		/* AKA_SSN_NUMBER_COUNT */
91  		String akaSsNumberCount = tcodeResponseElementsMap
92  				.get(AKA_SSN_NUMBER_COUNT_KEY);
93  		if (StringUtils.isNotBlank(akaSsNumberCount)) {
94  			cdlisPdpsHeaderInfo.setAkaSsNumberCount(Integer
95  					.parseInt(akaSsNumberCount));
96  		}
97  		/* MATCH_LIMIT_INDICATOR */
98  		String matchLimitIndicator = tcodeResponseElementsMap
99  				.get(MATCH_LIMIT_INDICATOR_KEY);
100 		cdlisPdpsHeaderInfo.setMatchLimitIndicator(matchLimitIndicator);
101 		/* MATCH_SEQUENCE_NUMBER_ID */
102 		String recordNumber = tcodeResponseElementsMap
103 				.get(MATCH_SEQUENCE_NUMBER_ID_KEY);
104 		if (StringUtils.isNotBlank(recordNumber)) {
105 			cdlisPdpsHeaderInfo.setMatchSequenceNumberId(Integer
106 					.parseInt(recordNumber));
107 		}
108 		/* STATE_OF_RECORD_CODE */
109 		cdlisPdpsHeaderInfo.setStateOfRecordCode(tcodeResponseElementsMap
110 				.get(STATE_OF_RECORD_CODE_KEY));
111 		/* APPLICATION_ID */
112 		String applicationId = tcodeResponseElementsMap.get(APPLICATION_ID_KEY);
113 		if (StringUtils.isNotBlank(applicationId)) {
114 			cdlisPdpsHeaderInfo.setApplicationId(Integer
115 					.parseInt(applicationId));
116 		}
117 		return cdlisPdpsHeaderInfo;
118 	}

120 	/***
121 	 * The purpose of this method is to build the information for CDLIS.
122 	 * 
123 	 * @param tcodeResponseElementsMap the map of tcode CDU response elements
124 	 * 
125 	 * @return CDLIS/PDPS record
126 	 */
127 	private CduResponseInfo buildCdlisInfo(
128 			Map <String, String> tcodeResponseElementsMap) {
129 		CduResponseInfo cduResponseInfo = new CduResponseInfo();
130 		String strMessageType = tcodeResponseElementsMap.get(MESSAGE_TYPE_KEY);
131 		if (StringUtils.isNotBlank(strMessageType)
132 				&& StringUtils.equals(strMessageType.trim(), CONST_RC)) {
133 			//Header Info
134 			CdlisPdpsHeaderInfo cdlisPdpsHeaderInfo = this
135 					.buildCdlisHeaderInfo(tcodeResponseElementsMap);
136 			cduResponseInfo.setCdlisPdpsHeaderInfo(cdlisPdpsHeaderInfo);
137 			CduRequestInfo cduRequestInfo = this
138 					.buildCduRequestInfo(tcodeResponseElementsMap);
139 			//Copy over Request Information
140 			CduRequestInfo cduRequestCopyInfo = new CduRequestInfo();
141 			cduRequestCopyInfo.setRrDateOfBirth(cduRequestInfo
142 					.getRrDateOfBirth());
143 			cduRequestCopyInfo.setRrDriverLicenseNumber(cduRequestInfo
144 					.getRrDriverLicenseNumber());
145 			cduRequestCopyInfo.setRrName(cduRequestInfo.getRrName());
146 			cduRequestCopyInfo.setRrSocialSecurityNumber(cduRequestInfo
147 					.getRrSocialSecurityNumber());
148 			cduResponseInfo.setCduRequestInfo(cduRequestCopyInfo);
149 		}
150 		else if (StringUtils.isNotBlank(strMessageType)
151 				&& StringUtils.equals(strMessageType, CONST_RD)) {
152 			//CDLIS Info
153 			DrivingRecord drivingRecord = this
154 					.buildPdpsCdlisDataForHit(tcodeResponseElementsMap);
155 			cduResponseInfo.setDrivingRecord(drivingRecord);
156 		}
157 		else if (StringUtils.isNotBlank(strMessageType)
158 				&& StringUtils.equals(strMessageType, CONST_HC)) {
159 			//Driver History Info
160 			CduDriverHistoryInfo cduDriverHistoryInfo = this
161 					.buildDriverHistoryInfo(tcodeResponseElementsMap);
162 			cduResponseInfo.setCduDriverHistoryInfo(cduDriverHistoryInfo);
163 		}
164 		return cduResponseInfo;
165 	}

367 				0);
368 		/* AKA1_NAME */
369 		if (StringUtils.isNotBlank(tcodeResponseElementsMap.get(AKA1_NAME_KEY))) {

382 			akaInformation1.setAkaDlNumber(firstAkaDlNumber);
383 			akaInformationList.add(akaInformation1);

385 		if (StringUtils.isNotBlank(tcodeResponseElementsMap.get(AKA2_NAME_KEY))) {
386 			/* AKA2_NAME */
387 			PersonName aka2PersonName = ConverterResponseHelper
388 					.buildPersonNameFromString(tcodeResponseElementsMap
389 							.get(AKA2_NAME_KEY));
390 			AkaInformation akaInformation2 = new AkaInformation();
391 			akaInformation2.setAkaName(aka2PersonName);
392 			akaInformationList.add(akaInformation2);

400 			akaInformation3.setAkaName(aka3PersonName);
401 			akaInformationList.add(akaInformation3);
402 		}
403 		drivingRecord.setAkaInformation(akaInformationList);
404 		/* CURRENT_DATE_OF_BIRTH */
405 		Date dateOfBirth = convertStringToDate(tcodeResponseElementsMap
406 				.get(CURRENT_DATE_OF_BIRTH_KEY),
407 				EcsConverterHelper.FORMATTER_CCYY_MM_DD);
408 		drivingRecord.setCurrentBirthDate(dateOfBirth);
409 		/* CURRENT_SEX_CODE */
410 		String currentSexCode = tcodeResponseElementsMap
411 				.get(CURRENT_SEX_CODE_KEY);
412 		if (StringUtils.equals(currentSexCode, MALE_GENDER_CODE)) {
413 			drivingRecord.setGenderCode(createCodeSetElement(
414 					ICodeSetNameConstants.SEX_CODE, MALE_GENDER_CODE_VALUE));
415 		}
416 		else if (StringUtils.equals(currentSexCode, FEMALE_GENDER_CODE)) {
417 			drivingRecord.setGenderCode(createCodeSetElement(
418 					ICodeSetNameConstants.SEX_CODE, FEMALE_GENDER_CODE_VALUE));
419 		}
420 		/* CURRENT_HEIGHT */
421 		String currentHeight = tcodeResponseElementsMap.get(CURRENT_HEIGHT_KEY);
422 		if (StringUtils.isNotBlank(currentHeight)) {
423 			String heightInFeet = currentHeight.substring(0, 1);
424 			String heightInInches = currentHeight.substring(1);
425 			drivingRecord.setHeightInFeet(Integer.parseInt(heightInFeet));
426 			drivingRecord.setHeightInInch(Integer.parseInt(heightInInches));
427 		}
428 		/* CURRENT_WEIGHT */
429 		String currentWeight = tcodeResponseElementsMap.get(CURRENT_WEIGHT_KEY);
430 		drivingRecord.setWeight(Integer.parseInt(currentWeight));
431 		/* CURRENT_EYE_COLOR */
432 		String currentEyeColorCode = tcodeResponseElementsMap
433 				.get(CURRENT_EYE_COLOR_KEY);
434 		if (StringUtils.equals(currentEyeColorCode, BROWN_COLOR_CODE)) {
435 			drivingRecord.setEyeColorCode(createCodeSetElement(
436 					ICodeSetNameConstants.EYE_COLOR_CODE,
437 					BROWN_COLOR_CODE_VALUE));
438 		}
439 		else if (StringUtils.equals(currentEyeColorCode, HAZEL_COLOR_CODE)) {
440 			drivingRecord.setEyeColorCode(createCodeSetElement(
441 					ICodeSetNameConstants.EYE_COLOR_CODE,
442 					HAZEL_COLOR_CODE_VALUE));
443 		}
444 		else if (StringUtils.equals(currentEyeColorCode, PINK_COLOR_CODE)
445 				|| StringUtils.equals(currentEyeColorCode, DIC_COLOR_CODE)
446 				|| StringUtils.equals(currentEyeColorCode, UNK_COLOR_CODE)) {
447 			drivingRecord.setEyeColorCode(createCodeSetElement(BLANK));
448 		}

450 			drivingRecord
451 					.setEyeColorCode(createCodeSetElement(currentEyeColorCode));
452 		}

454 		//FIXME Use CodeSetService framework
455 		/*		drivingRecord.setCurrentPrimaryDlNumberState(getCodeSetElement(
456 						"StateCode", tcodeResponseElementsMap
457 								.get("CURRENT_PRIMARY_DL_NUMBER_STATE")));*/
458 		drivingRecord.setCurrentPrimaryDlNumberState(createCodeSetElement(
459 				ICodeSetNameConstants.STATE_CODE, tcodeResponseElementsMap
460 						.get(CURRENT_PRIMARY_DL_NUMBER_STATE_KEY)));
461 		/* CURRENT_PRIMARY_DL_NUMBER */
462 		drivingRecord.setLicenseNumber(tcodeResponseElementsMap
463 				.get(CURRENT_PRIMARY_DL_NUMBER_KEY));

1795 	 * The purpose of this method is to process the CDU response. 
1796 	 * The CDU response will be having chunks of data of length 874.
1797 	 * Every chunk of data will be identified by message type and message sequence number id. 
1798 	 * This method will process the entire CDU response
1799 	 * and then creates CDLIS and PDPS records by merging the chunks of data 
1800 	 * based on message type and message sequence number id.
1801 	 * 
1802 	 * @param anInboundMessagePayloadBody the cdu response with out header
1803 	 * @param recordResponseCount CDU response records count
1804 	 * 
1805 	 * @return the response of CDLIS/PDPS inquiry to the ECS
1806 	 */

1808 			String anInboundMessagePayloadBody, int recordResponseCount) {
1809 		List <CduResponseInfo> cduResponseInfoList = new ArrayList <CduResponseInfo>(
1810 				0);
1811 		for (int responseCount = 0; responseCount < recordResponseCount; responseCount++) {
1812 			int startAt = responseCount * MESSAGE_PAYLOAD_CDLIS_RECORD_LENGTH;
1813 			int endAt = startAt + MESSAGE_PAYLOAD_CDLIS_RECORD_LENGTH;
1814 			String anInboundMessagePayloadResponse = anInboundMessagePayloadBody
1815 					.substring(startAt, endAt);
1816 			Map <String, String> tcodeResponseElementsMap = this
1817 					.buildTcodeResponseElementsMap(anInboundMessagePayloadResponse);
1818 			CduResponseInfo cduResponseInfo = null;
1819 			/* The state of origin field determines if the record is for CDLIS or PDPS. */
1820 			String strRecordType = tcodeResponseElementsMap.get(
1821 					STATE_OF_ORIGIN_KEY).trim();
1822 			if (StringUtils.isNotBlank(strRecordType)
1823 					&& StringUtils.equals(strRecordType, CONST_XX)) {
1824 				cduResponseInfo = buildCdlisInfo(tcodeResponseElementsMap);
1825 			}
1826 			else {
1827 				cduResponseInfo = buildPdpsInfo(tcodeResponseElementsMap);
1828 			}
1829 			cduResponseInfo.setOriginType(strRecordType);
1830 			String strMessageType = tcodeResponseElementsMap.get(
1831 					MESSAGE_TYPE_KEY).trim();
1832 			cduResponseInfo.setMessageType(strMessageType);
1833 			cduResponseInfoList.add(cduResponseInfo);
1834 		}
1835 		CdlisAndPdpsResponseEcs response = this
1836 				.processCduResponse(cduResponseInfoList);
1837 		return response;
1838 	}

1874 				ConverterResponseHelper.addErrorToResponse(response, CONST_C,

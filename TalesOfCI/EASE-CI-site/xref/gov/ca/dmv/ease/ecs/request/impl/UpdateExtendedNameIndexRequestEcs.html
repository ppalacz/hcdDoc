59  		FireAndForgetReceipt updateExtendedNameIndexResponseEcs = null;
60  		IErrorCollector errorCollector = getValidator().validate(this);
61  		if (errorCollector.hasErrors()) {
62  			updateExtendedNameIndexResponseEcs = new FireAndForgetReceipt(
63  					errorCollector);
64  		}
65  		else {
66  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
67  			//In the catch block we need to register the error to the error collector.
68  			try {
69  				updateExtendedNameIndexResponseEcs = (FireAndForgetReceipt) super
70  						.execute();
71  			}
72  			catch (EcsResponseIsNullException responseIsNullException) {
73  				errorCollector = new ErrorCollector();
74  				errorCollector.register(new EaseValidationException(
75  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
76  				updateExtendedNameIndexResponseEcs = new FireAndForgetReceipt(
77  						errorCollector);
78  			}

81  				errorCollector.register(new EaseValidationException(
82  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
83  				updateExtendedNameIndexResponseEcs = new FireAndForgetReceipt(
84  						errorCollector);
85  			}

88  	}

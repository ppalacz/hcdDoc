28  public class DeleteMasterReceiptFileRequestEcsValidator extends
29  		EcsRequestValidator implements IRequestValidator {
30  	/*** The Constant serialVersionUID. */
31  	private static final long serialVersionUID = -4987082337848015464L;
32  	/*** The Voucher. */
33  	private static final String VOUCHER = "Voucher";
34  	/*** The Request. */
35  	private static final String REQUEST = "DeleteMasterReceiptFileRequestEcs";
36  
37  	/* (non-Javadoc)
38  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
39  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
40  	 */
41  	@Override
42  	protected void validateRequest(IEcsRequest request,
43  			IErrorCollector aCollector) {
44  		if (!(request instanceof DeleteMasterReceiptFileRequestEcs)) {
45  			addValidationExceptionInErrorCollector(
46  					REQUEST_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
47  					new String[] { REQUEST }, aCollector);
48  		}
49  		else {
50  			DeleteMasterReceiptFileRequestEcs requestEcs = (DeleteMasterReceiptFileRequestEcs) request;
51  			Voucher voucher = requestEcs.getVoucher();
52  			validateVoucher(voucher, aCollector);
53  		}
54  	}
55  
56  	/***
57  	 * This method validates Voucher.
58  	 * 
59  	 * @param voucher the Voucher to validate
60  	 * @param aCollector the Error Collector
61  	 */
62  	private void validateVoucher(Voucher voucher, IErrorCollector aCollector) {
63  		if (isNullOrBlank(voucher)) {
64  			addValidationExceptionInErrorCollector(
65  					OBJECT_IS_NULL_OR_BLANK_ERROR_MESSAGE,
66  					new String[] { VOUCHER }, aCollector);
67  		}
68  		else {
69  			Office office = voucher.getOffice();
70  			Employee employee = voucher.getEmployee();
71  			if (isNullOrBlank(voucher.getWorkDate())) {
72  				addValidationExceptionInErrorCollector(
73  						VOUCHER_WORK_DATE_IS_EMPTY_ERROR_MESSAGE, null,
74  						aCollector);
75  			}
76  			if (isNullOrBlank(office) || isNullOrBlank(office.getOfficeId())) {
77  				addValidationExceptionInErrorCollector(
78  						VOUCHER_OFFICE_ID_IS_EMPTY_ERROR_MESSAGE, null,
79  						aCollector);
80  			}
81  			if (isNullOrBlank(employee)
82  					|| isNullOrBlank(employee.getEmployeeTechId())) {
83  				addValidationExceptionInErrorCollector(
84  						VOUCHER_EMPLOYEE_ID_IS_EMPTY_ERROR_MESSAGE, null,
85  						aCollector);
86  			}
87  			if (isNullOrBlank(voucher.getSequenceNumber())) {
88  				addValidationExceptionInErrorCollector(
89  						VOUCHER_SEQUENCE_NUMBER_IS_EMPTY_ERROR_MESSAGE, null,
90  						aCollector);
91  			}
92  			if (isNullOrBlank(voucher.getAmount())) {
93  				addValidationExceptionInErrorCollector(
94  						VOUCHER_AMOUNT_IS_EMPTY_ERROR_MESSAGE, null, aCollector);
95  			}
96  			//TODO: cashier sequence number's BO mapping is not available. Once it is available, write a validation for that.
97  		}
98  	}
99  }

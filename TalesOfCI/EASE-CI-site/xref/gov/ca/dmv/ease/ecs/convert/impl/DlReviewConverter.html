40  		String messageText = null;
41  		int numberOfRecords = 0;
42  		List <Object> objectDlReviewBodyList = buildReviewRecords((RecordDlReviewRequestEcs) request);
43  		numberOfRecords = findNumberOfRecords(objectDlReviewBodyList);
44  		if (numberOfRecords != 0) {
45  			List <Object> objectHeaderList = buildHeader(request,
46  					numberOfRecords);
47  			objectHeaderList.addAll(objectDlReviewBodyList);
48  			messageText = StringUtils.join(objectHeaderList.toArray());
49  		}
50  		return messageText;
51  	}

68  	/***
69  	 * Builds the header.
70  	 * 
71  	 * @param objectList the object list
72  	 * @param request the request
73  	 * @param numberOfRecords the number of records
74  	 */
75  	private List <Object> buildHeader(IEcsRequest request, int numberOfRecords) {
76  		List <Object> objectList = new ArrayList <Object>(0);
77  		RecordDlReviewRequestEcs recordDlReviewRequestEcs = (RecordDlReviewRequestEcs) request;
78  		// Get the Current User Context
79  		IUserContext currentUserContext = recordDlReviewRequestEcs
80  				.getUserContext();
81  		// Get the DlApplication from the Request object.
82  		DlApplication dlApplication = (recordDlReviewRequestEcs)
83  				.getDlProcessContext().getDlApplication();
84  		/* TCODE */
85  		objectList.add(filler("BDRV", 4));
86  		/* PROCESS DATE */
87  		objectList.add(filler(convertDateToString(currentUserContext
88  				.getWorkDate(), EcsConverterHelper.FORMATTER_MM_DD_YYCC), 8));
89  		/* OFFICE ID */
90  		objectList.add(filler(currentUserContext.getOfficeId(), 3));
91  		/* KEYER'S ID */
92  		objectList.add(filler(currentUserContext.getTechId(), 2));
93  		/* TYPE TRANSACTION CODE */
94  		objectList.add(filler(currentUserContext.getTtc(), 3));
95  		/* DRIVER LICENSE NUMBER/SalesPerson ID */
96  		String licenseNumber = "";
97  		if (dlApplication.getApplicant() != null) {
98  			licenseNumber = dlApplication.getApplicant().getDriverLicense().getLicenseNumber();
99  		}
100 		objectList.add(filler(licenseNumber, 8));
101 		/* FIRST 3 POSTIONS OF LAST NAME */
102 		String lastName = "";
103 		if (dlApplication.getApplicant() != null) {
104 			lastName = dlApplication.getApplicant().getCurrentName().getLastName();
105 		}
106 		objectList.add(filler(lastName, 3));
107 		/* PROCESSOR MNEMONIC */
108 		objectList.add(filler("", 2));
109 		/* NUMBER OF RECORDS */
110 		objectList.add(filler(String.valueOf(numberOfRecords), 3));
111 		return objectList;
112 	}

46  	/***
47  	 * Copy current name to aka names.
48  	 */
49  	public void copyCurrentNameToAkaNames() {
50  		List <PersonName> listAkaNames = new ArrayList <PersonName>();
51  		PersonName existingPersonName = new PersonName();
52  		existingPersonName.setFirstName((getDlApplicant().getCurrentName())
53  				.getFirstName());
54  		existingPersonName.setLastName((getDlApplicant().getCurrentName())
55  				.getLastName());
56  		existingPersonName.setMiddleName((getDlApplicant().getCurrentName())
57  				.getMiddleName());
58  		PersonName newPersonName = new PersonName();
59  		newPersonName.setFirstName(this.firstName);
60  		newPersonName.setMiddleName(this.middleName);
61  		newPersonName.setLastName(this.lastName);
62  		newPersonName.setSuffix(this.suffix);
63  		listAkaNames.add(existingPersonName);
64  		listAkaNames.add(newPersonName);
65  		getDlaProcessContext().getDlApplicant().setPersonNames(listAkaNames);
66  		getDlaProcessContext().setChangedName(existingPersonName);
67  	}

69  	/***
70  	 * Fill previous values.
71  	 */
72  	public void fillPreviousValues() {
73  		List <PersonName> existingPersonNamesList = getDlaProcessContext()
74  				.getDlApplication().getApplicant().getPersonNames();
75  		if (existingPersonNamesList != null
76  				&& existingPersonNamesList.size() > 0) {
77  			PersonName personName = existingPersonNamesList.get(0); // storing aka names at index 1 and at index 0 previous current names will be stored.
78  			firstName = personName.getFirstName();
79  			middleName = personName.getMiddleName();
80  			lastName = personName.getLastName();
81  			suffix = personName.getSuffix();
82  		}
83  		else {
84  			PersonName personName = getDlaProcessContext().getDlApplication()
85  					.getApplicant().getCurrentName();
86  			firstName = personName.getFirstName();
87  			middleName = personName.getMiddleName();
88  			lastName = personName.getLastName();
89  			suffix = personName.getSuffix();
90  		}
91  	}

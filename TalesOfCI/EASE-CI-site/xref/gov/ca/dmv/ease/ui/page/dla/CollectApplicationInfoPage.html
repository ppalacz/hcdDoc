291 	public String getLicenseNumber() {
292 		if (getCustomerInfoProcessContext().getDlApplicant().getDriverLicense() != null
293 				&& getCustomerInfoProcessContext().getDlApplication()
294 						.getApplicant().getDriverLicense()
295 						.getExistingLicenseClasses() != null
296 				&& getCustomerInfoProcessContext().getDlApplication()
297 						.getApplicant().getDriverLicense()
298 						.getExistingLicenseClasses().size() > 0) {
299 			this.licenseNumber = getDriverLicense().getLicenseNumber();
300 		}
301 		else if (getCustomerInfoProcessContext().getDlApplication()
302 				.getApplicant().getIdCard() != null
303 				&& getCustomerInfoProcessContext().getDlApplication()
304 						.getApplicant().getIdCard().getIdNumber() != null) {
305 			licenseNumber = (getCustomerInfoProcessContext().getDlApplication()
306 					.getIdCard().getIdNumber());
307 		}
308 		return licenseNumber;
309 	}

311 	/***
312 	 * Gets the Expiration Date from DL process context.
313 	 * 
314 	 * @return the Date
315 	 */
316 	public String getDlExpirationDate() {
317 		try {
318 			if (getCustomerInfoProcessContext().getEliRecord() != null) {
319 				if (getCustomerInfoProcessContext().getEliRecord()
320 						.getDriverLicense().getExpirationDate() != null) {
321 					SimpleDateFormat sdf = new SimpleDateFormat("MMddyy");
322 					dlExpirationDate = sdf
323 							.format(getCustomerInfoProcessContext()
324 									.getEliRecord().getDriverLicense()
325 									.getExpirationDate());
326 				}
327 				else {
328 					dlExpirationDate = (getCustomerInfoProcessContext()
329 							.getEliRecord().getDriverLicense()
330 							.getLicenseStatusCode() != null ? getCustomerInfoProcessContext()
331 							.getEliRecord().getDriverLicense()
332 							.getLicenseStatusCode().getDescription()
333 							: "NONE");
334 				}
335 			}
336 			else {
337 				dlExpirationDate = "NONE";
338 			}
339 		}
340 		catch (Exception e) {
341 			LOGGER.error("Binding Error: inquiryResponsePage dlExpirationDate "
342 					+ e.getMessage());
343 		}
344 		return dlExpirationDate;
345 	}

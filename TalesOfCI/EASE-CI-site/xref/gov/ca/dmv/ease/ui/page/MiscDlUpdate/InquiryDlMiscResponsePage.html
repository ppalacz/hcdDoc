146 	public String getExpirationDate() {
147 		try {
148 			if (getMiscDlProcessContext().getDlmRecord() != null) {
149 				if (getMiscDlProcessContext().getDlmRecord().getPerson()
150 						.getDriverLicense().getExpirationDate() != null) {
151 					SimpleDateFormat sdf = new SimpleDateFormat("MMddyy");
152 					expirationDate = sdf.format(getMiscDlProcessContext()
153 							.getDlmRecord().getPerson().getDriverLicense()
154 							.getExpirationDate());
155 				}
156 				else {
157 					expirationDate = (getMiscDlProcessContext().getDlmRecord()
158 							.getPerson().getDriverLicense()
159 							.getLicenseStatusCode() != null ? getMiscDlProcessContext()
160 							.getDriverLicense().getLicenseStatusCode()
161 							.getDescription()
162 							: "NONE");
163 				}
164 			}
165 			else {
166 				expirationDate = "NONE";
167 			}
168 		}
169 		catch (Exception e) {
170 			LOGGER.error("Binding Error: inquiryDlMiscResponse expirationDate "
171 					+ e.getMessage());
172 		}
173 		return expirationDate;
174 	}

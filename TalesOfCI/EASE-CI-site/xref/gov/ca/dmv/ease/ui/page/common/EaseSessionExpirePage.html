37  	/*** The Constant serialVersionUID. */
38  	private static final long serialVersionUID = -8291294772888457478L;
39  	/*** Logger for this class */
40  	private static final Log logger = LogFactory
41  			.getLog(EaseSessionExpirePage.class);
42  	/*** The session data. */
43  	private SessionData sessionData;
44  	/*** The message. */
45  	private String message = "Your EASE session has timed out. Would you like to restore the session?";

53  		logger.info("Cancel the restore functionality");
54  		return "CollectModeSelectionActivity";
55  	}
56  
57  	/***
58  	 * The purpose of this class is to process the restore functionality.
59  	 * 
60  	 * @return the string
61  	 */
62  	public String processRestore() {
63  		logger.info("Process the restore functionality");
64  		logger.info("Last Interaction Activity Name: "
65  				+ this.sessionData.getLastInteractionActivityName());
66  		FacesContext facesContext = FacesContext.getCurrentInstance();
67  		ServletContext servletContext = (ServletContext) facesContext
68  				.getExternalContext().getContext();
69  		ApplicationContext applicationContext = WebApplicationContextUtils
70  			.getRequiredWebApplicationContext(servletContext);
71  		SessionData sessionDataBean = (SessionData) EaseProxyUtils
72  				.getProxyObject(applicationContext.getBean("sessionData"));
73  		HttpServletRequest httpServletRequest = (HttpServletRequest)facesContext
74  				.getExternalContext().getRequest();
75  		this.restoreSession(httpServletRequest, applicationContext,
76  				sessionDataBean);
77  		return sessionDataBean.getLastInteractionActivityName();
78  	}
79  	
80  	/***
81  	 * The purpose of this method is to restore the session.
82  	 * 
83  	 * @param httpServletRequest the http servlet request
84  	 * @param applicationContext the application context
85  	 */
86  	private void restoreSession(HttpServletRequest httpServletRequest,
87  			ApplicationContext applicationContext, SessionData sessionDataBean) {
88  		HttpSession httpSession = httpServletRequest.getSession(true);
89  		String sessionId = httpSession.getId();
90  		ISessionService sessionService = (SessionService) applicationContext
91  				.getBean("sessionService");
92  		SessionData sessionData = sessionService.restoreState(sessionId);
93  		if (sessionData != null) {
94  			/* Restore state to bean */
95  			sessionDataBean.copy(sessionData);
96  			sessionService.storeState(sessionId, sessionData);
97  			SessionContext sessionContextBean = (SessionContext) EaseProxyUtils
98  					.getProxyObject(applicationContext.getBean("sessionContext"));
99  			if ((sessionData.getCurrentProcessContext() != null) && (sessionContextBean != null)) sessionContextBean
100 					.setCurrentProcessContext((ProcessContext)sessionData.getCurrentProcessContext());
101 		}
102 	}

147 	/***
148 	 * Gets the allowed ttc.
149 	 * 
150 	 * @return the allowed ttc
151 	 */
152 	@SuppressWarnings(value = { "unchecked" })
153 	public List <subList> getAllowedTTC() {
154 		Map <String, String> roleTtcMap = authAndAuthService
155 				.getAuthorizedTtcsMap(getContext().getRoles());
156 		String mode = (getContext().getFunctionalMode() != null) ? getContext().getFunctionalMode().getCode() : "";
157 		if (roleTtcMap.containsKey("DAF-TTC")
158 				&& (mode.equalsIgnoreCase("M") || mode.equalsIgnoreCase("P"))) {
159 			roleTtcMap.remove("DAF-TTC");
160 			getContext().getAllowedTtc().remove("DAF");
161 		}
162 		List <subList> sl = new ArrayList <subList>();
163 		Iterator it = roleTtcMap.entrySet().iterator();
164 		while (it.hasNext()) {
165 			Map.Entry pairs1 = (Map.Entry) it.next();
166 			subList s = new subList();
167 			s.setName(pairs1.getKey().toString());
168 			String[] m1 = pairs1.getValue().toString().split(",");
169 			s.setSubListName(m1);
170 			sl.add(s);
171 		}
172 		return sl;
173 	}

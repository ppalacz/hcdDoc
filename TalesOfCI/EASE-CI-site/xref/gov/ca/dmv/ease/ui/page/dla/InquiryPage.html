82  	/***
83  
84  	* Convert the list of applied license classes into
85  
86  	* applied License class and motorcycle .
87  
88  	*/
89  	private void convertGetAppliedLicenseClass() {
90  		try {
91  			List <CodeSetElement> licenseClassCodes = getHqInquiryProcessContext()
92  					.getDlApplication().getApplicant().getDriverLicense()
93  					.getAppliedLicenseClasses();
94  			if (licenseClassCodes.size() > 0) {
95  				appliedLicenseClass = licenseClassCodes.get(0);
96  			}
97  			else {
98  				appliedLicenseClass = null;
99  			}
100 			if (licenseClassCodes.size() > 1
101 					&& licenseClassCodes.get(1) != null) {
102 				motorCycle = true;
103 			}
104 			else {
105 				motorCycle = false;
106 			}
107 		}
108 		catch (Exception e) {
109 			LOGGER
110 					.error("Converter Error:  InquiryPage convertGetAppliedLicenseClass"
111 							+ e.getMessage());
112 		}
113 	}
114 
115 	/***
116 	 * Convert the list of applied license classes and sets
117 	 * applied License class and motorcycle .
118 	 */
119 	private void convertSetAppliedLicenseClass() {
120 		try {
121 			DriverLicense license = getHqInquiryProcessContext().getDlApplication()
122 					.getApplicant().getDriverLicense();
123 			List <CodeSetElement> licenseClasses = new ArrayList <CodeSetElement>();
124 			licenseClasses.add(appliedLicenseClass == null ? null
125 					: appliedLicenseClass);
126 			if (motorCycle) {
127 				licenseClasses.add(new CodeSetElement("M", "M", "Motor Cycle"));
128 			}
129 			license.setAppliedLicenseClasses(licenseClasses);
130 		}
131 		catch (Exception e) {
132 			LOGGER
133 					.error("Converter Error: InquiryPage convertSetAppliedLicenseClass"
134 							+ e.getMessage());
135 		}
136 	}

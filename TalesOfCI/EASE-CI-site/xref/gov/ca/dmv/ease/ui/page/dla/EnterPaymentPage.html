135 	/***
136 	 * Adds the receipts.
137 	 * @return the string
138 	 */
139 	public String addReceipts() {
140 		try {
141 			String officeId = "";
142 			Date workDate = null;
143 			String employeeTechId = "";
144 			Integer sequenceNumber = 0;
145 			if (this.voucherPayments.size() > 0) {
146 				Voucher voucher = this.voucherPayments.get(this.voucherPayments
147 						.size() - 1);
148 				officeId = voucher.getOffice().getOfficeId();
149 				workDate = voucher.getEmployee().getEmployeeWorkdateControl().getOfficeWorkdate()
150 						.getAuthorizedWorkDate();
151 				employeeTechId = voucher.getEmployee().getEmployeeTechId();
152 				sequenceNumber = voucher.getSequenceNumber();
153 				if ((officeId != null) && (workDate != null)
154 						&& (employeeTechId != null) && (sequenceNumber != null)) {
155 					Voucher voucherPayment = new Voucher();
156 					Office office = new Office();
157 					Employee employee = new Employee();
158 					EmployeeWorkdateControl employeeWorkdateControl = new EmployeeWorkdateControl();
159 					OfficeWorkdate officeWorkdate = new OfficeWorkdate();
160 					employeeWorkdateControl.setOfficeWorkdate(officeWorkdate);
161 					employee
162 							.setEmployeeWorkdateControl(employeeWorkdateControl);
163 					voucherPayment.setOffice(office);
164 					voucherPayment.setEmployee(employee);
165 					getDlFeeProcessContext().getInvoice().addPayment(
166 							voucherPayment);
167 				}
168 			}
169 		} catch (Exception e) {
170 			FacesUtils
171 					.addErrorMessage("There was a problem executing the submitPreProcessing in addReceipts() "
172 							+ e.getMessage());
173 			LOGGER.error(e);
174 			e.printStackTrace();
175 		}
176 		return "";
177 	}

179 	/***
180 	 * Clear the empty payment objects and set it to process context.
181 	 */
182 	public void clearEmptyPaymentObjects() {
183 		List <Payment> paymentList = new ArrayList <Payment>();
184 		List <Voucher> voucherPaymentList = getDlFeeProcessContext().getInvoice()
185 				.getVoucherPayments();
186 		for (int i = 0; i < voucherPaymentList.size(); i++) {
187 			Voucher voucher = voucherPaymentList.get(i);
188 			if (!isVoucherEmpty(voucher)) {
189 				paymentList.add(voucher);
190 			}
191 		}
192 		List <DebitPayment> debitPaymentList = getDlFeeProcessContext()
193 				.getInvoice().getDebitPayments();
194 		for (int i = 0; i < debitPaymentList.size(); i++) {
195 			DebitPayment debitPayment = debitPaymentList.get(i);
196 			if (debitPayment.getAmount() != null) {
197 				paymentList.add(debitPayment);
198 			}
199 		}
200 		List <CheckPayment> checkPaymentList = getDlFeeProcessContext()
201 				.getInvoice().getCheckPayments();
202 		for (int i = 0; i < checkPaymentList.size(); i++) {
203 			CheckPayment checkPayment = checkPaymentList.get(i);
204 			if (checkPayment.getAmount() != null) {
205 				paymentList.add(checkPayment);
206 			}
207 		}
208 		if (getDlFeeProcessContext().getInvoice().getCashPayment() != null
209 				&& getDlFeeProcessContext().getInvoice().getCashPayment()
210 						.getAmount() != null
211 				&& getDlFeeProcessContext().getInvoice().getCashPayment()
212 						.getAmount().doubleValue() > 0) {
213 			paymentList.add(getDlFeeProcessContext().getInvoice()
214 					.getCashPayment());
215 		}
216 		getDlFeeProcessContext().getInvoice().setPayments(paymentList);
217 	}

396 	 * 
397 	 * @param voucher the voucher
398 	 * 
399 	 * @return the boolean
400 	 */
401 	private Boolean isVoucherEmpty(Voucher voucher) {
402 		Boolean hasOfficeId = voucher.getOffice().getOfficeId() == null;
403 		Boolean hasWorkdate = voucher.getEmployee()
404 				.getEmployeeWorkdateControl().getOfficeWorkdate().getAuthorizedWorkDate() == null;
405 		Boolean hasEmpId = voucher.getEmployee().getEmployeeTechId() == null;
406 		Boolean hasSequence = voucher.getSequenceNumber() == null;
407 		return (hasOfficeId && hasWorkdate && hasEmpId && hasSequence);
408 	}

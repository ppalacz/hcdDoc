30  public class ProcessNetValidator extends AbstractDlValidator {
31  	/*** The Constant serialVersionUID. */
32  	private static final long serialVersionUID = 7771157977363980969L;
33  
34  	/***
35  	 * Checks if is cash refund eligible.
36  	 * 
37  	 * @param mode
38  	 *            the mode
39  	 * @param invoiceTotal
40  	 *            the invoice total
41  	 * @param amount
42  	 *            the amount
43  	 * @return the boolean
44  	 */
45  	protected Boolean isCashRefundEligible(String mode, Float invoiceTotal,
46  			Float amount) {
47  		if (amount > invoiceTotal && !mode.equals(IEaseModes.COUNTER)
48  				|| !IEaseModes.MANUAL.equals(mode)
49  				|| !IEaseModes.TRAVEL_CREW.equals(mode)) {
50  			return Boolean.FALSE;
51  		}
52  		return Boolean.TRUE;
53  	}
54  
55  	/***
56  	 * Checks if is password needed for cash refund.
57  	 * 
58  	 * @param invoiceTotal
59  	 *            the invoice total
60  	 * @param amount
61  	 *            the amount
62  	 * @param password
63  	 *            the password
64  	 * 
65  	 * @return the boolean
66  	 */
67  	protected Boolean isPasswordNeededForCashRefund(Float invoiceTotal,
68  			Float amount, String password) {
69  		Float cashRefund = invoiceTotal - amount;
70  		if (cashRefund >= 20 && cashRefund < 100 && isNullorEmpty(password)) {
71  			return Boolean.FALSE;
72  		}
73  		return Boolean.TRUE;
74  	}
75  
76  	@Override
77  	public void validate(List <IValidatable> validatables,
78  			IErrorCollector collector) {
79  		DlApplication dlApplication = (DlApplication) validatables.get(0);
80  		IUserContext userContext = (IUserContext) validatables.get(1);
81  		BigDecimal invoiceTotal = dlApplication.getInvoice().getInvoiceTotal();
82  		BigDecimal amount = dlApplication.getInvoice().getCashPayment()
83  				.getAmount();
84  		Float invoiceTotalFloatValue = invoiceTotal.floatValue();
85  		Float amountFloatValue = amount.floatValue();
86  		String feeActionCode = dlApplication.getInvoice()
87  				.getFeeRefundActionCode().getCode();
88  		String mode = userContext.getOperationalMode().getCode();
89  		String password = dlApplication.getAuthorizingPassword();
90  		String string = null;
91  		if ((string = validateActionCode(feeActionCode, invoiceTotalFloatValue,
92  				amountFloatValue, password)) != null) {
93  			addErrorMessage(collector, string);
94  		}
95  		if (!isCashRefundEligible(mode, invoiceTotalFloatValue,
96  				amountFloatValue)) {
97  			addErrorMessage(collector,
98  					CAN_ONLY_ISSUE_CHANGE_IN_COUNTER_MANUAL_OR_TRAVEL_CREW_MODE);
99  		}
100 		if (isPasswordNeededForCashRefund(invoiceTotalFloatValue,
101 				amountFloatValue, password)) {
102 			addErrorMessage(collector,
103 					REQUIRES_APPROVAL_AUTHORITY_PASSWORD_FOR_CHANGE_OVER);
104 		}
105 	}
106 
107 	/***
108 	 * Validate action code.
109 	 * 
110 	 * @param feeActionCode
111 	 *            the fee action code
112 	 * @param invoiceTotal
113 	 *            the invoice total
114 	 * @param amount
115 	 *            the amount
116 	 * @param password
117 	 *            the password
118 	 * 
119 	 * @return the string
120 	 */
121 	protected String validateActionCode(String feeActionCode,
122 			Float invoiceTotal, Float amount, String password) {
123 		Float cashRefund = invoiceTotal - amount;
124 		if (CASH.equals(feeActionCode) && amount > invoiceTotal) {
125 			return CAN_ONLY_ISSUE_CHANGE_ON_OVERPAYMENT;
126 		}
127 		if (cashRefund > 100 && !HEAD_QUARTERS_REFUND.equals(feeActionCode)) {
128 			return CANNOT_SSUE_CHANGE_REQUEST_HEADQUARTERS_REFUND;
129 		}
130 		if (cashRefund > 100 && HEAD_QUARTERS_REFUND.equals(feeActionCode)
131 				&& isNullorEmpty(password)) {
132 			return CANNOT_SSUE_CHANGE_REQUEST_HEADQUARTERS_REFUND;
133 		}
134 		return null;
135 	};
136 }

35  public class CdaD846Rule {
36  	/***
37  	 * Validate if the photo is not required and test results are entered.
38  	 * 
39  	 * @param commercialTestResultGroup
40  	 * @param dlApplication
41  	 * @param driverLicense
42  	 * @param errorCollector
43  	 */
44  	public void validate(List <CommercialTestResult> testResultList,
45  			CodeSetElement photoRequiredCodeSet,
46  			PhotoInformation photoInformation, IErrorCollector errorCollector) {
47  		String photoRequiredCode = CodeUtil
48  				.getCodeFromCodeSetElement(photoRequiredCodeSet);
49  		if (testResultList != null && testResultList.size() > 0) {
50  			if (IPhotoRequiredCodeConstants.REQUIRED
51  					.equalsIgnoreCase(photoRequiredCode)
52  					&& photoInformation == null) {
53  				errorCollector.register(new EaseValidationException(
54  						IDataCollectionErrorMessageConstants.PHOTO_REQUIRED));
55  			}
56  		}
57  	}
58  
59  	/* (non-Javadoc)

63  			CommercialTestResultGroup commercialTestResultGroup,

42  	 * Retrieve daf from database.
43  	 * 
44  	 * @param dafProcessContext the daf process context
45  	 * @param selectedDafRecordNumber the selected daf record number
46  	 * @param errorCollector the error collector
47  	 * @param cdaTaskService the cda task service
48  	 * 
49  	 * @return the transaction
50  	 */
51  	protected Transaction retrieveDafFromDatabase(
52  			DafProcessContext dafProcessContext,
53  			Integer selectedDafRecordNumber, IErrorCollector errorCollector,
54  			CdaTaskService cdaTaskService) {
55  		Transaction tempDafRecord = null;
56  		DafRecordInquiryRequestTs tsRequest = new DafRecordInquiryRequestTs(
57  				dafProcessContext.getUserContext(), selectedDafRecordNumber
58  						.intValue());
59  		tsRequest.setDafInquiryTypes(DafInquiryTypes.DAF);
60  		DafRecordInquiryResponseTs tsResponse = cdaTaskService
61  				.execute(tsRequest);
62  		if (tsResponse.hasErrors()) {
63  			populateErrorMessages(tsResponse.getErrorCollector(),
64  					errorCollector);
65  		}
66  		else {
67  			tempDafRecord = (Transaction) tsResponse.getTransaction();
68  		}
69  		return tempDafRecord;
70  	}

162 	 * 
163 	 * @param persistenceServiceErrors the persistence service errors
164 	 * @param errorCollector the error collector
165 	 */
166 	protected void populateErrorMessages(
167 			IErrorCollector persistenceServiceErrors,
168 			IErrorCollector errorCollector) {
169 		if (persistenceServiceErrors != null
170 				&& !(persistenceServiceErrors.isEmpty())) {
171 			for (IErrorCollectorEntry persistenceServiceError : persistenceServiceErrors
172 					.getEntries()) {
173 				errorCollector.getEntries().add(persistenceServiceError);
174 			}
175 		}
176 	}

329 						.getPreviousProcessHistory();
330 				if (previousProcessHistory != null) {

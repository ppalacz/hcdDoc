28  public class SpD718Rule extends AbstractSpRule {
29  	public void validate(ProcessContext processContext) {
30  		String ttc = getTtcCode(processContext);
31  		String applicableTtcs = ITtcConstants.SPO + "," + ITtcConstants.SPC
32  				+ "," + ITtcConstants.SPR + "," + ITtcConstants.SRX + ","
33  				+ ITtcConstants.SPD;
34  		if (isEnterAction(processContext) && applicableTtcs.contains(ttc)
35  				&& getApplicantIdCard(processContext) != null) {
36  			if (!isIdCardExpired(processContext) && hasChanges(processContext)) {
37  				getErrorCollector()
38  						.register(
39  								new ApplicationException(
40  										IDataCollectionErrorMessageConstants.CHANGES_MUST_FIRST_BE_MADE_VIA_AN_ID_APP));
41  			}
42  		}
43  	}

45  	/***
46  	 * Checks for changes to the name, gender or birth date. 
47  	 *
48  	 * @param context the context
49  	 * @return true, if successful
50  	 */
51  	private boolean hasChanges(ProcessContext context) {
52  		// has bdate changed?
53  		SpProcessContext spContext = (SpProcessContext) context;
54  		if (spContext.getChangedBirthDate() != null){
55  			if(getDateDifferenceInDays(spContext.getChangedBirthDate(),
56  						getApplicant(context).getBirthDate()) != 0){
57  				return true;
58  			}
59  		}
60  		// name change? 
61  		// TODO: should this compare to person.getPersonNames? Is getPreviousName used? 
62  		if (!getApplicant(context).getCurrentName().equals(
63  				getApplicant(context).getPreviousName()))
64  			return true;
65  		// Gender changed? 
66  		if (getApplicant(context).isGenderChanged())
67  			return true;
68  		return false;
69  	}

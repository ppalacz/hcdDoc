31  public class Daf0001Rule extends AbstractDafRule {
32  	public void validate(ProcessContext processContext) {
33  		DafProcessContext dafProcessContext = (DafProcessContext) processContext;
34  		IErrorCollector errorCollector = new ErrorCollector();
35  		ValidateHelper helper = new ValidateHelper();
36  		String activityName = getCurrentActivityName(dafProcessContext);
37  		String dafRecordNumber = getDafNumberInString(dafProcessContext);
38  		String dlDafNumber = getDafDlNumber(dafProcessContext);
39  		String idDafNumber = getDafIdNumber(dafProcessContext);
40  		String spDafNumber = getDafSalesPersonNumber(dafProcessContext);
41  		if (activityName != null
42  				&& activityName.equalsIgnoreCase("DafMaintenanceActivity")) {
43  			if (compareSelectedAction(
44  					new Action[] { getAction(IActionsRegistry.PURGE_ACTION) },
45  					dafProcessContext)) {
46  				ArrayList <String> list = new ArrayList <String>();
47  				list.add(dafRecordNumber);
48  				list.add(dlDafNumber);
49  				list.add(idDafNumber);
50  				list.add(spDafNumber);
51  				if (helper.itemCounter(list) == 0) {
52  					errorCollector.register(new ApplicationException("0001"));
53  				}
54  			}
55  		}
56  		else if (activityName != null
57  				&& activityName.equalsIgnoreCase("BrowseDailyAppFileActivity")) {
58  			if (compareSelectedAction(new Action[] {
59  					getAction(IActionsRegistry.DISPLAY_ACTION),
60  					getAction(IActionsRegistry.PURGE_ACTION) },
61  					dafProcessContext)) {
62  				ArrayList <String> list = new ArrayList <String>();
63  				list.add(dafRecordNumber);
64  				list.add(dlDafNumber);
65  				list.add(idDafNumber);
66  				list.add(spDafNumber);
67  				if (helper.itemCounter(list) == 0) {
68  					errorCollector.register(new ApplicationException("0001"));
69  				}
70  			}
71  		}
72  		else if (activityName != null
73  				&& activityName
74  						.equalsIgnoreCase("DisplayOfIdApplicationDataActivity")) {// TODO: Not tested because it is related to IdCard and out of scope for now.
75  			if (compareSelectedAction(
76  					new Action[] { getAction(IActionsRegistry.UPDATE_PENDING_ACTION) },
77  					dafProcessContext)) {
78  				Transaction tempDafRecord = dafProcessContext.getTransaction();
79  				if (tempDafRecord != null) {
80  					if (tempDafRecord.getLicenseType().equals(
81  							LicenseType.ID_CARD)) {
82  						String newIncompleteApplicationReasonCode = getNewIncompleteApplicationReasonCode(dafProcessContext
83  								.getDlApplication());
84  						if (newIncompleteApplicationReasonCode == null) {
85  							errorCollector.register(new ApplicationException(
86  									"0557"));
87  						}
88  					}
89  				}
90  			}
91  		}
92  		setErrorCollector(errorCollector);
93  	}
94  }

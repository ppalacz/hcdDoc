39  public class EnterPaymentRule extends AbstractValidateRule {
40  	EnterPaymentValidator enterPaymentValidator = new EnterPaymentValidator();
41  
42  	/* (non-Javadoc)
43  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
44  	 */
45  	public void validate(ProcessContext processContext) {
46  		IErrorCollector errorCollector = new ErrorCollector();
47  		IUserContext userContext = processContext.getUserContext();
48  		DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) processContext;
49  		Action selectedAction = dlFeeProcessContext.getSelectedAction();
50  		if (selectedAction != null
51  				&& selectedAction == getActionsRegistry().getAction(
52  						IActionsRegistry.ENTER_ACTION)) {
53  			DlApplication dlApplication = dlFeeProcessContext
54  					.getDlApplication();
55  			Invoice invoice = dlApplication.getInvoice();
56  			List <Voucher> vouchers = invoice.getVoucherPayments();
57  			CreditPayment creditPayment = invoice.getCreditPayment();
58  			List <CheckPayment> checkPayments = invoice.getCheckPayments();
59  			List <DebitPayment> manualPayments = invoice.getDebitPayments();
60  			if (!enterPaymentValidator.isCheckAndManualCreditFieldsConsecutive(// TODO: not tested because of check and manual cr field issues in UI.
61  					checkPayments, manualPayments)) {
62  				//F072
63  				errorCollector
64  						.register(new ApplicationException(
65  								IFeeAndPaymentErrorMessageConstants.CHECK_MANUAL_CREDIT_FIELDS_MUST_BE_CONSECUTIVE));
66  			}
67  			if (enterPaymentValidator
68  					.isSequenceNumberMissingFromAutomatedReceipts(vouchers)) {
69  				//F017
70  				errorCollector.register(new ApplicationException(
71  						IFeeAndPaymentErrorMessageConstants.SEQUENCE_NUMBER_MISSING));
72  			}
73  			if (enterPaymentValidator
74  					.isOfficeIdMissingFromAutomatedReceipts(vouchers)) {
75  				//F014
76  				errorCollector.register(new ApplicationException(
77  						IFeeAndPaymentErrorMessageConstants.OFFICE_ID_MISSING));
78  			}
79  			if (enterPaymentValidator
80  					.isEmployeeIdMissingFromAutomatedReceipts(vouchers)) {
81  				//F016
82  				errorCollector.register(new ApplicationException(
83  						IFeeAndPaymentErrorMessageConstants.EMPLOYEE_ID_MISSING));
84  			}
85  			if (enterPaymentValidator
86  					.isWorkDateMissingFromAutomatedReceipts(vouchers)) {
87  				//F015
88  				errorCollector.register(new ApplicationException(
89  						IFeeAndPaymentErrorMessageConstants.WORK_DATE_MISSING));
90  			}
91  			if (enterPaymentValidator.isCreditCardAmountEntryExist(
92  					creditPayment, userContext)) {// TODO: not tested because credit card fields are not available on screen
93  				//F102
94  				errorCollector
95  						.register(new ApplicationException(
96  								IFeeAndPaymentErrorMessageConstants.CREDIT_CARD_ENTRY_NOT_ALLOWED_FOR_THIS_OFFICE));
97  			}
98  			if (enterPaymentValidator.isCreditCardNumberAndExpiryDateExist(
99  					creditPayment, userContext)) {// TODO: not tested because credit card fields are not available on screen
100 				// 0153
101 				errorCollector.register(new ApplicationException("0153"));
102 			}
103 			if (enterPaymentValidator.isAutomatedReceiptsDuplicated(vouchers)) {
104 				// F047
105 				errorCollector.register(new ApplicationException(
106 						IFeeAndPaymentErrorMessageConstants.DUPLICATED_AUTOMATED_RECEIPTS));
107 			}
108 			//TODO: If automated receipt is entered and cash amount is not entered then below validation is not allowing to move further.
109 			// Ideally, in case automated receipt entry, it should not ask to enter cash amount. So, below validation is commented for now.
110 			/*if (!enterPaymentValidator.isPaymentMadeInFull(invoice)) {
111 				// F509
112 				errorCollector.register(new ApplicationException(IFeeAndPaymentMessage.PAYMENT_IN_FULL_REQUIRED));
113 			}*/
114 			if (enterPaymentValidator.isCnaCondition(dlApplication,
115 					CnaConditionType.CNA_VOUCHER_INQUIRY)) {// TODO: not tested
116 				// F010
117 				errorCollector
118 						.register(new ApplicationException(
119 								IFeeAndPaymentErrorMessageConstants.CNA_CANNOT_VERIFY_AUTOMATED_RECEIPTS));
120 			}
121 		}
122 		setErrorCollector(errorCollector);
123 	}
124 }

71  	/***
72  	 * when the transaction is not issuance office transaction then the CDL
73  	 * designation may not be equal to N
74  	 * 
75  	 * @param dlaProcessContext
76  	 * @param errorCollector
77  	 * 
78  	 */
79  	public void cdlis0013(DlApplication dlApplication,
80  			List <DrivingRecord> cdlisRecords, IErrorCollector errorCollector) {
81  		for (DrivingRecord cdlisRecord : cdlisRecords) {
82  			if (EaseUtil.isNullOrBlank(dlApplication.isClearedByIssuance())
83  					|| !dlApplication.isClearedByIssuance()) {
84  				if (EaseUtil.isNullOrBlank(cdlisRecord.getDesignateAction())
85  						|| cdlisRecord.getDesignateAction().getCode() == "N") {
86  					errorCollector
87  							.register(new EaseValidationException(
88  									IGeneralErrorMessageConstants.DATA_DOES_NOT_MATCH_VALID_CODES));
89  					;
90  				}
91  			}
92  		}
93  	}

95  	/***
96  	 * when the current class is not equal to commercial class A, B, C then the
97  	 * Additional out of state driver license information can be equal to Y
98  	 * 
99  	 * Conversely when class = A, B, C if Additional out of state driver license
100 	 * = TRUE its an error
101 	 * 
102 	 * @param dlaProcessContext
103 	 * @param errorCollector
104 	 * 
105 	 */
106 	public void cdlis0153(Boolean isAdditionalOsDlInfo,
107 			List <DrivingRecord> cdlisRecords, IErrorCollector errorCollector) {
108 		for (DrivingRecord cdlisRecord : cdlisRecords) {
109 			if (!EaseUtil.isNullOrBlank(cdlisRecord
110 					.getDriverLicenseClassCodes())) {
111 				for (CodeSetElement classCode : cdlisRecord
112 						.getDriverLicenseClassCodes()) {
113 					if (classCode.getCode() == "A"
114 							|| classCode.getCode() == "B"
115 							|| classCode.getCode() == "C") {
116 						if (isAdditionalOsDlInfo) {
117 							errorCollector
118 									.register(new EaseValidationException(
119 											IGeneralErrorMessageConstants.INVALID_FIELD_PRESENT));
120 						}
121 					}
122 				}
123 			}
124 		}
125 	}

229 	/***
230 	 * Validate user name and password.
231 	 * 
232 	 * @param processContext
233 	 * @param errorCollector
234 	 */
235 	private void validateAuthorizingPassword(ProcessContext processContext,
236 			IErrorCollector errorCollector) {
237 		CdlisPdpsProcessContext cdlisPdpsProcessContext = (CdlisPdpsProcessContext) processContext;
238 		if (cdlisPdpsProcessContext.getAuthorizingUserId() != null
239 				&& cdlisPdpsProcessContext.getAuthorizingPassword() != null) {
240 			String userId = cdlisPdpsProcessContext.getAuthorizingUserId()
241 					.toLowerCase();
242 			String password = cdlisPdpsProcessContext.getAuthorizingPassword();
243 			int counter = 0;
244 			for (Iterator <DrivingRecord> iterator = cdlisPdpsProcessContext
245 					.getCdlisResultHits().iterator(); iterator.hasNext();) {
246 				DrivingRecord drivingRecord = (DrivingRecord) iterator.next();
247 				if (drivingRecord != null
248 						&& drivingRecord.getDesignateAction() != null
249 						&& (drivingRecord.getDesignateAction().getCode()
250 								.equalsIgnoreCase("S") || drivingRecord
251 								.getDesignateAction().getCode()
252 								.equalsIgnoreCase("Q"))) {
253 					counter++;
254 				}
255 			}
256 			for (Iterator <DrivingRecord> iterator = cdlisPdpsProcessContext
257 					.getPdpsResultHits().iterator(); iterator.hasNext();) {
258 				DrivingRecord drivingRecord = (DrivingRecord) iterator.next();
259 				if (drivingRecord != null
260 						&& drivingRecord.getDesignateAction() != null
261 						&& (drivingRecord.getDesignateAction().getCode()
262 								.equalsIgnoreCase("S") || drivingRecord
263 								.getDesignateAction().getCode()
264 								.equalsIgnoreCase("Q"))) {
265 					counter++;
266 				}
267 			}
268 			if (counter > 0
269 					&& (EaseUtil.isNullOrBlank(userId) || EaseUtil
270 							.isNullOrBlank(password))) {
271 				errorCollector
272 						.register(new EaseValidationException(
273 								IDataCollectionErrorMessageConstants.APPROVAL_AUTHORITY_PASSWORD_REQUIRED));
274 			}
275 			if (counter > 0 && !EaseUtil.isNullOrBlank(userId)
276 					&& !EaseUtil.isNullOrBlank(password)) {
277 				try {
278 					if (!getAuthAndAuthService().isAuthorizedApprover(userId,
279 							password))
280 						errorCollector
281 								.register(new EaseValidationException(
282 										IFeeAndPaymentErrorMessageConstants.APPROVAL_AUTHORITY_PASSWORD_INVALID));
283 				}
284 				catch (LoginException e) {
285 					errorCollector
286 							.register(new EaseValidationException(
287 									IFeeAndPaymentErrorMessageConstants.APPROVAL_AUTHORITY_PASSWORD_INVALID));
288 				}
289 			}
290 		}
291 	}

30  //TODO: Not tested because cashier sequence number is not available. At present, process history is not being populated.
31  public class DafQ520Rule extends AbstractDafRule {
32  	/*** The cda task service. */
33  	private CdaTaskService cdaTaskService;// Injected by spring
34  
35  	/***
36  	 * Sets the cda task service.
37  	 * 
38  	 * @param cdaTaskService the new cda task service
39  	 */
40  	public void setCdaTaskService(CdaTaskService cdaTaskService) {
41  		this.cdaTaskService = cdaTaskService;
42  	}
43  	/* (non-Javadoc)
44  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
45  	 */
46  	public void validate(ProcessContext processContext) {
47  		DafProcessContext dafProcessContext = (DafProcessContext) processContext;
48  		IErrorCollector errorCollector = new ErrorCollector();		
49  		Integer dafRecordNumber = getDafNumber(dafProcessContext);
50  		String dlDafNumber = getDafDlNumber(dafProcessContext);
51  		String idDafNumber = getDafIdNumber(dafProcessContext);
52  		String spDafNumber = getDafSalesPersonNumber(dafProcessContext);		
53  		String activityName = getCurrentActivityName(dafProcessContext);
54  		
55  		if (activityName != null && activityName.equalsIgnoreCase("DafMaintenanceActivity")) {			
56  			if (compareSelectedAction(new Action[] {getAction(IActionsRegistry.PURGE_ACTION)}, dafProcessContext)) {
57  				Transaction tempDafRecord = null;
58  				String selectedDafRecordNumber = null;
59  				if(dafRecordNumber != null) {
60  					tempDafRecord = retrieveDafFromDatabase(dafProcessContext, dafRecordNumber, errorCollector, cdaTaskService);
61  					selectedDafRecordNumber = dafRecordNumber.toString();
62  				} else if(dlDafNumber != null) {
63  					tempDafRecord = retrieveDlDafFromDatabase(dafProcessContext, dlDafNumber, errorCollector, cdaTaskService);
64  					selectedDafRecordNumber = dlDafNumber;
65  				} else if(idDafNumber != null) {// TODO: not tested because id card daf is out of scope for now.c
66  					tempDafRecord = retrieveIdDafFromDatabase(dafProcessContext, idDafNumber, errorCollector, cdaTaskService);
67  					selectedDafRecordNumber = idDafNumber;
68  				} else if(spDafNumber != null) {// TODO: not tested because sales person daf is out of scope for now.
69  					tempDafRecord = retrieveIdDafFromDatabase(dafProcessContext, spDafNumber, errorCollector, cdaTaskService);
70  					selectedDafRecordNumber = spDafNumber;
71  				}
72  				
73  				if(tempDafRecord != null) {
74  					Integer transactionIdentifier = tempDafRecord.getTransactionIdentifier();
75  					if(transactionIdentifier != null) {
76  						if(transactionIdentifier.intValue() == Integer.valueOf(selectedDafRecordNumber).intValue()) {// active daf record found							
77  							if(!EaseUtil.isNullOrBlank(getCashierSequenceNumber(tempDafRecord))) {
78  								errorCollector.register(new ApplicationException("Q520"));
79  							}
80  						}								
81  					}						
82  				}
83  			}
84  
85  		}		
86  		setErrorCollector(errorCollector);
87  	}
88  }

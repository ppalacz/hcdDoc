36  public class CdaD558Rule  extends AbstractValidateRule {
37  	private static final String CLASS_A = "A";
38  	private static final String CLASS_B = "B";
39  	private static final String CLASS_C = "C";
40  	private static final String CLASS_D = "D";
41  	private static final String CLASS_E = "E";
42  	private static final String CLASS_F = "F";
43  	private static final String CLASS_M = "M";
44  
45  	public void validate(ProcessContext processContext) {
46  	}
47  	/***
48  	 * D558 - If value entered but test is not required
49  	 * 
50  	 * @param driverLicense
51  	 * @param commercialTestResultGroup
52  	 */
53  	public void validate(CommercialTestResultGroup commercialTestResultGroup,
54  			DriverLicense driverLicense) {
55  		IErrorCollector errorCollector = new ErrorCollector();
56  		if (isUneededTest(driverLicense, commercialTestResultGroup)) {
57  			errorCollector
58  					.register(new EaseValidationException(
59  							IDataCollectionErrorMessageConstants.APP_CLASS_ENDORSEMENT_VS_LAW_TEST));
60  		}
61  	}
62  
63  	/***
64  	 * isUneededTest - return true if test not needed
65  	 * 
66  	 * @param driverLicense
67  	 * @param commercialTestResultGroup
68  	 */
69  	private boolean isUneededTest(DriverLicense driverLicense,
70  			CommercialTestResultGroup commercialTestResultGroup) {
71  		List <String> endorsementCodes = DlApplicationUtil
72  				.getAllEndorsementCodes(driverLicense);
73  		List <String> categoryCodes = DlApplicationUtil
74  				.getAllCategoryCodes(driverLicense);
75  		List <String> appliedLicenseClassCodes = DlApplicationUtil
76  				.getAppliedLicenseClassCodes(driverLicense);
77  		List <CommercialTestResult> generalTestResults = commercialTestResultGroup
78  				.getCommercialTestResults(CommercialTestResultType.GENERAL_TEST_RESULT);
79  		List <CommercialTestResult> classATestResults = commercialTestResultGroup
80  				.getCommercialTestResults(CommercialTestResultType.CLASSA_COMBINATION_TEST_RESULT);
81  		List <CommercialTestResult> trlCoachTestResults = commercialTestResultGroup
82  				.getCommercialTestResults(CommercialTestResultType.TRAILERCOACH_TEST_RESULT);
83  		List <CommercialTestResult> fireFighterTestResults = commercialTestResultGroup
84  				.getCommercialTestResults(CommercialTestResultType.FIREFIGHTER_TEST_RESULT);
85  		List <CommercialTestResult> fireFighterClassATestResults = commercialTestResultGroup
86  				.getCommercialTestResults(CommercialTestResultType.FIREFIGHTER_CLASSA_TEST_RESULT);
87  		List <CommercialTestResult> passengerTestResults = commercialTestResultGroup
88  				.getCommercialTestResults(CommercialTestResultType.PASSENGER_TEST_RESULT);
89  		List <CommercialTestResult> doubleOrTripleTestResults = commercialTestResultGroup
90  				.getCommercialTestResults(CommercialTestResultType.DOUBLES_OR_TRIPLES_TEST_RESULT);
91  		List <CommercialTestResult> tankTestResults = commercialTestResultGroup
92  				.getCommercialTestResults(CommercialTestResultType.TANK_TEST_RESULT);
93  		List <CommercialTestResult> hmTestResults = commercialTestResultGroup
94  				.getCommercialTestResults(CommercialTestResultType.HAZMAT_TEST_RESULT);
95  		if (ArrayUtils.contains(appliedLicenseClassCodes, CLASS_A, CLASS_B,
96  				CLASS_F)) {
97  			if (!EaseUtil.isNullOrBlank(fireFighterTestResults)) {
98  				return true;
99  			}
100 		}
101 		if (ArrayUtils.contains(appliedLicenseClassCodes, CLASS_A, CLASS_E,
102 				CLASS_F)) {
103 			if (!EaseUtil.isNullOrBlank(trlCoachTestResults)
104 					|| !EaseUtil.isNullOrBlank(fireFighterClassATestResults)) {
105 				return true;
106 			}
107 		}
108 		if (ArrayUtils.contains(appliedLicenseClassCodes, CLASS_C, CLASS_D,
109 				CLASS_E, CLASS_F)) {
110 			if (!EaseUtil.isNullOrBlank(generalTestResults)
111 					|| !EaseUtil.isNullOrBlank(classATestResults)
112 					|| !EaseUtil.isNullOrBlank(passengerTestResults)) {
113 				return true;
114 			}
115 		}
116 		if (ArrayUtils.contains(appliedLicenseClassCodes, CLASS_F, CLASS_F)) {
117 			if (!EaseUtil.isNullOrBlank(doubleOrTripleTestResults)) {
118 				return true;
119 			}
120 		}
121 		// TRL/COACH Cat
122 		if (!ArrayUtils.contains(categoryCodes,
123 				IDlClassTypeConstants.TRAILER_COACH)) {
124 			if (!EaseUtil.isNullOrBlank(trlCoachTestResults)) {
125 				return true;
126 			}
127 		}
128 		// FF Class A Cat
129 		if (!ArrayUtils.contains(categoryCodes,
130 				IDlClassTypeConstants.FIRE_FIGHTER_CLASSS_A)) {
131 			if (!EaseUtil.isNullOrBlank(fireFighterClassATestResults)) {
132 				return true;
133 			}
134 		}
135 		// FF Class B Cat
136 		if (!ArrayUtils.contains(categoryCodes,
137 				IDlClassTypeConstants.FIRE_FIGHTER_CLASSS_B)) {
138 			if (!EaseUtil.isNullOrBlank(fireFighterTestResults)) {
139 				return true;
140 			}
141 		}
142 		// DOUBLES/TRPLS End
143 		if (!ArrayUtils.contains(endorsementCodes,
144 				IDlEndorsementCategories.DOUBLE_TRIPLE)) {
145 			if (!EaseUtil.isNullOrBlank(doubleOrTripleTestResults)) {
146 				return true;
147 			}
148 		}
149 		// PASSNGER and BUS End
150 		if (!ArrayUtils.contains(endorsementCodes,
151 				IDlEndorsementCategories.PASSENGER_VEH)
152 				&& !ArrayUtils.contains(endorsementCodes,
153 						IDlEndorsementCategories.SCHOOL_BUS)) {
154 			if (!EaseUtil.isNullOrBlank(passengerTestResults)) {
155 				return true;
156 			}
157 		}
158 		// HAZ MAT End
159 		if (!ArrayUtils.contains(endorsementCodes,
160 				IDlEndorsementCategories.HAZARDOUS_MATERIAL)) {
161 			if (!EaseUtil.isNullOrBlank(hmTestResults)) {
162 				return true;
163 			}
164 		}
165 		// TANK End
166 		if (!ArrayUtils.contains(endorsementCodes,
167 				IDlEndorsementCategories.TANK_VEHICLE)) {
168 			if (!EaseUtil.isNullOrBlank(tankTestResults)) {
169 				return true;
170 			}
171 		}
172 		return false;
173 	}
174 }

19   * Description: A password may be keyed to bypass the usual vehicle salesperson 
20   * renewal transaction requirements for 14 days beyond the expiration date

31  public class SpD608Rule extends AbstractSpRule {
32  	/***
33  	 * 
34  	 */
35  	public void validate(ProcessContext processContext) {
36  		// get out of here if not enter pressed or we are creating a new sp license or we are reinstating 
37  		if (!isEnterAction(processContext)
38  				|| getTtcCode(processContext).equalsIgnoreCase(
39  						ITtcConstants.SPO)
40  				|| getTtcCode(processContext).equalsIgnoreCase(
41  						ITtcConstants.SRX)) {
42  			return;
43  		}
44  		SalesPersonApplication spApplication = getSalesPersonApplication(processContext);
45  		SpProcessContext context = (SpProcessContext)processContext; 
46  		boolean createError = false;
47  		License license = getSalesPersonLicense(processContext);
48  		// cant go on if we dont have the license to get the exp date in the first place
49  		if (license == null) {
50  			getErrorCollector().register(
51  					new ApplicationException(
52  							getMessageFromResourceBundle("NULL_LICENSE")));
53  			return;
54  		}
55  		//also can't compare if there is no expiration date associated with the license
56  		Date salesLicenseExpDate = license.getExpirationDate();
57  		if (salesLicenseExpDate == null) {
58  			getErrorCollector()
59  					.register(
60  							new ApplicationException(
61  									getMessageFromResourceBundle("NULL_OR_EMPTY_LICENSE_EXPIRATION_DATE")));
62  			return;
63  		}
64  		// expect positive number if not expired, negative number if expired
65  		int daysDifference = getDateDifferenceInDays(salesLicenseExpDate,
66  				getWorkDate(processContext));
67  		// first of all if we are expired and have no authentication?
68  		if (daysDifference < 0
69  				&& EaseUtil.isNullOrBlank(context.getAuthorizingPassword())) {
70  			createError = true;
71  		}
72  //		// you can renew with authorization and TTC SPR up to 14 days after the exp date
73  //		else if (daysDifference >= -14
74  //				&& daysDifference <= 0
75  //				&& getTtcCode(processContext).equalsIgnoreCase(
76  //						ITtcConstants.SPR)
77  //				&& EaseUtil.isNullOrBlank(spApplication
78  //						.getAuthorizingPassword())) {
79  //			createError = true;
80  //		}
81  		// if the expiration date has passed by 14 days and we are not using ttc srx then create error
82  		else if (daysDifference < -14
83  				&& !(getTtcCode(processContext)
84  						.equalsIgnoreCase(ITtcConstants.SRX))) {
85  			createError = true;
86  		}
87  		// if the ttc is SPC or SPD a password is required within 120 days of the exp date		
88  		else if ((getTtcCode(processContext)
89  				.equalsIgnoreCase(ITtcConstants.SPC) || getTtcCode(
90  				processContext).equalsIgnoreCase(ITtcConstants.SPD))
91  				&& daysDifference < 120
92  				&& EaseUtil.isNullOrBlank(context
93  						.getAuthorizingPassword())) {
94  			createError = true;
95  		}
96  		if (createError) {
97  			getErrorCollector()
98  					.register(
99  							new ApplicationException(
100 									IDataCollectionErrorMessageConstants.TTC_INVALID_FOR_EXP_DATE_UNLESS_PASSWORD_KEYED));
101 		}
102 	}
103 }

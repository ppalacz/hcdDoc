44  			Date processDate = getWorkDate(processContext);
45

47  				if(getDateDifferenceInDays(expirationDate, processDate) > 365*4){
48  					getErrorCollector().register(new ApplicationException(IDataCollectionErrorMessageConstants.CURRENT_EXP_YEAR_KEYED_MUST_BE_WITHIN_4_YEARS));
49  				}
50  			}
51  			else if(expirationDate == null){
52  				getErrorCollector().register(new ApplicationException(getMessageFromResourceBundle("NULL_OR_EMPTY_LICENSE_EXPIRATION_DATE")));
53  			}
54  			else if(processDate != null){				
55  				getErrorCollector().register(new ApplicationException(getMessageFromResourceBundle("NULL_WORK_DATE")));

29  public class SelectRecordMatchRules extends AbstractValidateRule {
30  	AniInquiryMultipleHitsValidator aniInquiryMultipleHitsValidator = null;
31  
32  	/* (non-Javadoc)
33  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
34  	 */
35  	public void validate(ProcessContext processContext) {
36  		HqInquiryProcessContext hqInquiryProcessContext = (HqInquiryProcessContext) processContext;
37  		DlApplication dlApplication = hqInquiryProcessContext
38  				.getDlApplication();
39  		DriverLicense driverLicense = dlApplication.getLicense();
40  		IErrorCollector errorCollector = new ErrorCollector();
41  		Action selectedAction = hqInquiryProcessContext.getSelectedAction();
42  		if (selectedAction.equals(getActionsRegistry().getAction(
43  				IActionsRegistry.ENTER_ACTION))) {
44  			aniInquiryMultipleHitsValidator.validate(driverLicense,
45  					hqInquiryProcessContext.getAniTryNumber(),
46  					hqInquiryProcessContext.getAniPrimaryCode(),
47  					hqInquiryProcessContext.getAniRecords(), errorCollector);
48  		}
49  		if (selectedAction.equals(getActionsRegistry().getAction(
50  				IActionsRegistry.NO_MORE_TRIES))) {
51  			aniInquiryMultipleHitsValidator.validateNoMoreTries(
52  					hqInquiryProcessContext.getAniTryNumber(),
53  					hqInquiryProcessContext.getAniRecords(),
54  					!hqInquiryProcessContext.isCamvAvailable(),
55  					hqInquiryProcessContext.getUserContext(), errorCollector);
56  		}
57  		if (selectedAction.equals(getActionsRegistry().getAction(
58  				IActionsRegistry.CNA_BYPASS))) {
59  			aniInquiryMultipleHitsValidator.validateCnaByPass(driverLicense,
60  					hqInquiryProcessContext.getAniTryNumber(),
61  					hqInquiryProcessContext.getAniRecords(),
62  					!hqInquiryProcessContext.isAniDatabaseAvailable(),
63  					errorCollector);
64  		}
65  		setErrorCollector(errorCollector);
66  	}
67  
68  	/***
69  	 * @param aniInquiryMultipleHitsValidator the aniInquiryMultipleHitsValidator to set
70  	 */
71  	public void setAniInquiryMultipleHitsValidator(
72  			AniInquiryMultipleHitsValidator aniInquiryMultipleHitsValidator) {
73  		this.aniInquiryMultipleHitsValidator = aniInquiryMultipleHitsValidator;
74  	}

36  	public void validate(DlApplication dlApplication,
37  			IErrorCollector errorCollector) {
38  		DriverLicense driverLicense = dlApplication.getLicense();
39  		if (!EaseUtil.isNullOrBlank(driverLicense)) {
40  			CommercialTestResultGroup commercialTestResultGroup = driverLicense
41  					.getCommercialTestResultGroup();
42  			if (!EaseUtil.isNullOrBlank(commercialTestResultGroup)) {
43  				validate(commercialTestResultGroup,
44  						CommercialTestResultType.GENERAL_TEST_RESULT,
45  						errorCollector);
46  				validate(
47  						commercialTestResultGroup,
48  						CommercialTestResultType.CLASSA_COMBINATION_TEST_RESULT,
49  						errorCollector);
50  				validate(commercialTestResultGroup,
51  						CommercialTestResultType.TRAILERCOACH_TEST_RESULT,
52  						errorCollector);
53  				validate(commercialTestResultGroup,
54  						CommercialTestResultType.FIREFIGHTER_TEST_RESULT,
55  						errorCollector);
56  				validate(
57  						commercialTestResultGroup,
58  						CommercialTestResultType.FIREFIGHTER_CLASSA_TEST_RESULT,
59  						errorCollector);
60  				validate(commercialTestResultGroup,
61  						CommercialTestResultType.PASSENGER_TEST_RESULT,
62  						errorCollector);
63  				validate(commercialTestResultGroup,
64  						CommercialTestResultType.PRETRIP_TEST_RESULT,
65  						errorCollector);
66  			}
67  		}
68  	}

70  	/***
71  	 * @param errorCollector
72  	 * @param commercialTestResultGroup
73  	 */
74  	private void validate(CommercialTestResultGroup commercialTestResultGroup,
75  			CommercialTestResultType commercialTestResultType,
76  			IErrorCollector errorCollector) {
77  		List <CommercialTestResult> commercialGeneralTestResults = commercialTestResultGroup
78  				.getCommercialTestResults(commercialTestResultType);
79  		if (EaseUtil.isNotNull(commercialGeneralTestResults)
80  				&& commercialGeneralTestResults.size() > 0) {
81  			for (CommercialTestResult commercialGeneralTestResult : commercialGeneralTestResults) {
82  				String testResultCode = commercialGeneralTestResult
83  						.getTestResultCode().getCode();
84  				if (IGeneralTestConstants.CORRECTION
85  						.equalsIgnoreCase(testResultCode)
86  						&& isDriveTestPresent(commercialTestResultGroup)) {
87  					errorCollector
88  							.register(new EaseValidationException(
89  									IDataCollectionErrorMessageConstants.CANNOT_CHANGE_TEST_RESULT));
90  				}
91  			}
92  		}
93  	}

95  	/***
96  	 * Return true if the drive test present.
97  	 * 
98  	 * @param cdaProcessContext
99  	 * @return a boolean
100 	 */
101 	private boolean isDriveTestPresent(
102 			CommercialTestResultGroup commercialTestResultGroup) {
103 		boolean isDriveTestPresent = false;
104 		List <CommercialTestResult> commercialPerformanceTestResults = commercialTestResultGroup
105 				.getCommercialTestResults(CommercialTestResultType.PERFORMANCE_TEST_RESULT);
106 		if (EaseUtil.isNotNull(commercialPerformanceTestResults)
107 				&& commercialPerformanceTestResults.size() > 0) {
108 			isDriveTestPresent = true;
109 		}
110 		return isDriveTestPresent;
111 	}

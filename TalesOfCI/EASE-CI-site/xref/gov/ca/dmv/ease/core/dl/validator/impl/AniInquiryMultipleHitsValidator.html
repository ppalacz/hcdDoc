40  	public void validate(DriverLicense driverLicense, Integer aniTryNumber,
41  			String primaryCode, List <AniRecord> aniRecords,
42  			IErrorCollector collector) {
43  		if (!EaseUtil.isNullOrBlank(primaryCode)) {
44  			collector.register(new EaseValidationException(
45  					IGeneralErrorMessageConstants.INVALID_FIELD_PRESENT));
46  		}
47  		//Q515
48  		String licenseNumber = driverLicense.getLicenseNumber();
49  		int primaryIndex = findPrimaryIndex(aniRecords);
50  		String aniDlNumber = null;
51  		if (primaryIndex >= 0) {
52  			aniDlNumber = ((AniRecord) aniRecords.get(primaryIndex))
53  					.getDriverLicenseNumber();
54  		}
55  		if (aniDlNumber != null && licenseNumber != null
56  				&& aniDlNumber.equals(licenseNumber)) {
57  			collector
58  					.register(new EaseValidationException(
59  							IInquiryErrorMessageConstants.TRY_NUMBER_MATCHES_DL_ID_IN_USE));
60  		}
61  		//0001
62  		if (EaseUtil.isNullOrBlank(aniTryNumber)) {
63  			collector.register(new EaseValidationException(
64  					IGeneralErrorMessageConstants.FIELD_REQUIRED));
65  		}
66  		//Q530
67  		else if (!EaseUtil.isNullOrBlank(aniTryNumber)
68  				&& aniTryNumber.intValue() > aniRecords.size()) {
69  			collector.register(new EaseValidationException(
70  					IInquiryErrorMessageConstants.INVALID_TRY_NUMBER));
71  		}
72  		//Q516
73  		else if (isPrimaryTryNumber(aniRecords, aniTryNumber)) {
74  			collector
75  					.register(new EaseValidationException(
76  							IInquiryErrorMessageConstants.INQUIRY_PREVIOUSLY_MADE_ON_THIS_RECORD));
77  		}
78  	}

80  	/***
81  	 * The multiple hits screen no more tries button validation, error code Q533, Q534
82  	 * @param enteredTryNumber
83  	 * @param aniRecords
84  	 * @param isCna
85  	 * @param collector
86  	 */
87  	public void validateCnaByPass(DriverLicense driverLicense,
88  			Integer aniTryNumber, List <AniRecord> aniRecords, Boolean isCna,
89  			IErrorCollector collector) {
90  		String licenseNumber = driverLicense.getLicenseNumber();
91  		if (aniRecords != null) {
92  			//Q533
93  			AniRecord aniRecord = aniRecords.get(aniRecords.size() - 1);
94  			if (!licenseNumber.equals(aniRecord.getDriverLicenseNumber())) {
95  				collector
96  						.register(new EaseValidationException(
97  								IInquiryErrorMessageConstants.INQUIRY_MUST_FIRST_BE_MADE_ON_TRY_NUMBER_PRESS_ENTER));
98  			}
99  		}
100 		//Q534
101 		boolean hasPreviousInquiry = false;
102 		int previousIndex = -1;
103 		for (int i = 0; i < aniRecords.size(); i++) {
104 			AniRecord aniRecord = aniRecords.get(i);
105 			if (licenseNumber.equals(aniRecord.getDriverLicenseNumber())) {
106 				hasPreviousInquiry = true;
107 				previousIndex = i;
108 				break;
109 			}
110 		}
111 		if (hasPreviousInquiry) {
112 			if (!EaseUtil.isNullOrBlank(aniTryNumber)
113 					&& (previousIndex != (aniTryNumber.intValue() - 1))) {
114 				collector
115 						.register(new EaseValidationException(
116 								IInquiryErrorMessageConstants.TRY_NUMBER_CANNOT_BE_CHANGED_IF_CNA_BYPASS_SELECTED));
117 			}
118 		}
119 	}

122 	 * The multiple hits screen no more tries button validation, error code 0153, Q549, Q561, Q564
123 	 * @param aniTryNumber
124 	 * @param aniRecords

126 	 * @param userContext

129 	public void validateNoMoreTries(Integer aniTryNumber,
130 			List <AniRecord> aniRecords, Boolean isCna,
131 			IUserContext userContext, IErrorCollector collector) {
132 		if (EaseUtil.isNotNull(aniTryNumber)) {
133 			collector.register(new EaseValidationException(
134 					IGeneralErrorMessageConstants.INVALID_FIELD_PRESENT));
135 		}
136 		//Q549
137 		if ((isCna != null) && !isCna) {
138 			if (!anyRecordNotProceed(aniRecords)) {
139 				collector
140 						.register(new EaseValidationException(
141 								IInquiryErrorMessageConstants.INQUIRE_AGAIN_ON_TRY_NUMBER_TO_RETRIEVE_RECORD));
142 			}
143 		}
144 		//Q561, need clarification
145 		//Q564
146 		String mode = userContext.getOperationalMode().getCode();
147 		if (IEaseModes.TRAVEL_CREW.equals(mode)
148 				|| IEaseModes.MANUAL.equals(mode)) {
149 			if (!aniRecordsHasPrimary(aniRecords)) {
150 				collector
151 						.register(new EaseValidationException(
152 								IInquiryErrorMessageConstants.RECORD_MUST_HAVE_BEEN_ACCEPTED_PRIMARY_OR_RESTART_WITH_DL_INQUIRY));
153 			}
154 		}
155 	}

161 	 * @param aniTryNumber the ani try number

165 	private boolean isPrimaryTryNumber(List <AniRecord> aniRecords,
166 			Integer aniTryNumber) {
167 		AniRecord aniRecord = aniRecords.get(aniTryNumber - 1);
168 		return aniRecord.isPrimary();

171 	/***
172 	 * @param aniRecords
173 	 * @return
174 	 */
175 	private int findPrimaryIndex(List <AniRecord> aniRecords) {
176 		int primaryIndex = -1;
177 		for (int i = 0; i < aniRecords.size(); i++) {
178 			AniRecord aniRecord = aniRecords.get(i);
179 			String actionCode = null;
180 			if (aniRecord.getActionCode() != null) {
181 				actionCode = aniRecord.getActionCode().getCode();
182 			}
183 			if (IDlMessageConstants.PRIMARY.equals(actionCode)) {
184 				primaryIndex = i;
185 				break;
186 			}
187 		}
188 		return primaryIndex;
189 	}

191 	/***
192 	 * @param aniRecords
193 	 * @return
194 	 */
195 	private boolean anyRecordNotProceed(List <AniRecord> aniRecords) {
196 		boolean isProceed = true;
197 		if (aniRecords != null) {
198 			for (AniRecord aniRecord : aniRecords) {
199 				if (EaseUtil.isNullOrBlank(aniRecord.getDriverLicenseNumber())) {
200 					isProceed = false;
201 				}
202 			}
203 		}
204 		return isProceed;
205 	}

207 	/***
208 	 * @param aniRecords
209 	 * @return
210 	 */
211 	private boolean aniRecordsHasPrimary(List <AniRecord> aniRecords) {
212 		boolean hasPrimary = false;
213 		if (aniRecords != null) {
214 			for (AniRecord aniRecord : aniRecords) {
215 				String actionCode = null;
216 				if (aniRecord.getActionCode() != null) {
217 					actionCode = aniRecord.getActionCode().getCode();
218 				}
219 				if (IDlMessageConstants.PRIMARY.equals(actionCode)) {
220 					hasPrimary = true;
221 				}
222 			}
223 		}
224 		return hasPrimary;
225 	}

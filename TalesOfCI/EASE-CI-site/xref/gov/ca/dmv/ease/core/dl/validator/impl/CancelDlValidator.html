31  public class CancelDlValidator extends AbstractValidateRule {
32  	
33  	/***
34  	 * Implementation of {@link gov.ca.dmv.ease.validator.IValidationRule#validate(ProcessContext)}
35  	 */
36  	public void validate(ProcessContext processContext) {
37  		CdlProcessContext cdlProcessContext = (CdlProcessContext) processContext;
38  		IErrorCollector errorCollector = new ErrorCollector();
39  
40  		Action selectedAction = cdlProcessContext.getSelectedAction();
41  		// Validations
42  		if (selectedAction != null
43  				&& selectedAction == getActionsRegistry().getAction(
44  						IActionsRegistry.ENTER_ACTION)) {
45  			String licenseNumber = null;
46  			Person applicant =cdlProcessContext.getDlApplicant();
47  			
48  			if(!isNullorEmpty(applicant)){
49  				if(!isNullorEmpty(applicant.getDriverLicense())) {
50  					licenseNumber = applicant.getDriverLicense().getLicenseNumber();
51  					if(!isLicenseNumberValid(licenseNumber))
52  						//ERROR CODE - Q505 - DL number prefix is NOT valid.
53  						errorCollector.register(new ApplicationException("Q505"));
54  				}
55  				
56  				if(!isNullorEmpty(applicant.getCurrentName()))
57  					if(!is3POSNameValid(applicant.getCurrentName().getLastName()))
58  						//ERROR CODE - D670 - The 3POS name entered is NOT valid.
59  						errorCollector.register(new ApplicationException("D670"));
60  			}
61  		}
62  		
63  		// Setting the error collector
64  		setErrorCollector(errorCollector);
65  	}
66  
67  	/***
68  	 * The System shall validate if the license number does NOT start with ["L", "O", "Q", "T"] (Q505)
69  	 * 
70  	 * @param String licenseNumber
71  	 * @return Boolean
72  	 */
73  	protected Boolean isLicenseNumberValid(String licenseNumber) {
74  		
75  		if(!isNullorEmpty(licenseNumber)) {
76  			String licensePrefix = licenseNumber.substring(0, 1);
77  			
78  			if("L".equals(licensePrefix) || "O".equals(licensePrefix) || "Q".equals(licensePrefix) || "T".equals(licensePrefix))
79  				return Boolean.FALSE;
80  			else
81  				return Boolean.TRUE;
82  		}
83  		return Boolean.FALSE;
84  	}
85  
86  	/***
87  	 * The System shall validate if the 3POS name is 2 or 3 characters but NOT 1 character (D670).  
88  	 * 
89  	 * @param String
90  	 * @return Boolean
91  	 */
92  	protected Boolean is3POSNameValid(String posName) {
93  
94  		if (isNullorEmpty(posName)
95  				|| posName.trim().length() == 1) 
96  			return Boolean.FALSE;
97  		else
98  			return Boolean.TRUE;
99  	}
100 }

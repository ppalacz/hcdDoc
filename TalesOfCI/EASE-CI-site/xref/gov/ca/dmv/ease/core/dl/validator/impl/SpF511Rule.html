27  	public void validate(ProcessContext processContext) {
28  		SalesPersonApplication application = getSalesPersonApplication(processContext);
29  		IUserContext userContext = processContext.getUserContext();
30  		CodeSetElement mode = userContext.getOperationalMode();
31  		String code = mode != null ? mode.getCode() : "";
32  		boolean hasNoFeeReasonCode = false;
33  		CodeSetElement feeRequiredReasonCode = application
34  				.getFeeRequiredReasonCode();
35  		
36  		String applicableCodes = ITtcConstants.SPO + "," + ITtcConstants.SPR
37  				+ "," + ITtcConstants.SPX + "," + ITtcConstants.SPD;//"SPO,SPR,SPX,SPD";
38  		
39  		if (isEnterAction(processContext)
40  				&& applicableCodes.contains(getTtcCode(processContext))
41  				&& feeRequiredReasonCode != null
42  				&& feeRequiredReasonCode.getCode().equalsIgnoreCase(
43  						IFeeResonCodeConstants.F_FEE_FOR_MANUAL_APPLICATION)) {
44  			hasNoFeeReasonCode = true;
45  		}
46  		/*
47  		 * if Fee action code is F-No Reason then make sure that it is only for
48  		 * manual or travel modes.
49  		 */
50  		if ((hasNoFeeReasonCode)
51  				&& !(code.equalsIgnoreCase(IEaseModes.TRAVEL_CREW) || code
52  						.equalsIgnoreCase(IEaseModes.MANUAL))) {
53  			getErrorCollector()
54  					.register(
55  							new EaseValidationException(
56  									IFeeAndPaymentErrorMessageConstants.NO_FEE_REASON_F_ALLOWED_IN_MANUAL_TRAVEL_CREW_MODES_ONLY));
57  		}
58  	}

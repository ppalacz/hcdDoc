35  // TODO: As this is an information message, it is not included in DafBusinessProcess.xml
36  public class DafQ508Rule extends AbstractDafRule {
37  	
38  	/*** The cda task service. */
39  	private CdaTaskService cdaTaskService;// Injected by spring
40  
41  	/***
42  	 * Sets the cda task service.
43  	 * 
44  	 * @param cdaTaskService the new cda task service
45  	 */
46  	public void setCdaTaskService(CdaTaskService cdaTaskService) {
47  		this.cdaTaskService = cdaTaskService;
48  	}
49  	/* (non-Javadoc)
50  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
51  	 */
52  	public void validate(ProcessContext processContext) {
53  		DafProcessContext dafProcessContext = (DafProcessContext) processContext;
54  		IErrorCollector errorCollector = new ErrorCollector();
55  		String activityName = getCurrentActivityName(dafProcessContext);				
56  		Integer selectedDafRecordNumber = getDafNumber(dafProcessContext);
57  		if (selectedDafRecordNumber != null) {
58  			if (activityName != null && (activityName.equalsIgnoreCase("DafMaintenanceActivity") || activityName.equalsIgnoreCase("BrowseDailyAppFileActivity"))) {
59  				if (compareSelectedAction(new Action[] { getAction(IActionsRegistry.DISPLAY_ACTION) }, dafProcessContext)) {
60  					// Try to retrieve entered daf number from database. If daf record exists and it has cnaStatus = 'Y' then set error message
61  					Transaction tempDafRecord = retrieveDafFromDatabase(dafProcessContext, selectedDafRecordNumber, errorCollector, cdaTaskService);
62  					if (tempDafRecord != null && tempDafRecord.getTransactionIdentifier().intValue() == selectedDafRecordNumber.intValue() && "Y".equalsIgnoreCase(tempDafRecord.getCnaStatus())) {
63  						errorCollector.register(new ApplicationException("Q508"));
64  					}
65  				}
66  			}		
67  			else if (activityName != null && activityName.equalsIgnoreCase("DisplayOfDlApplicationDataActivity")) {
68  				if(compareSelectedAction(new Action[] {	
69  						getAction(IActionsRegistry.ENTER_ACTION)},					
70  						dafProcessContext)) {
71  					Transaction tempDafRecord = dafProcessContext.getTransaction();
72  					if (tempDafRecord != null && tempDafRecord.getTransactionIdentifier().intValue() == selectedDafRecordNumber.intValue() && "Y".equalsIgnoreCase(tempDafRecord.getCnaStatus())) {
73  						errorCollector.register(new ApplicationException("Q508"));
74  					}
75  				}
76  			}
77  		}
78  		setErrorCollector(errorCollector);
79  	}
80  }

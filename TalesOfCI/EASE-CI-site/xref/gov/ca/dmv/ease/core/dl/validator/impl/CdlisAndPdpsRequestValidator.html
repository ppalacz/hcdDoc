32  public class CdlisAndPdpsRequestValidator extends AbstractBaseValidator {
33  	/*** The Constant serialVersionUID. */
34  	private static final long serialVersionUID = 192018429049904164L;

36  	/***
37  	 * This method validates Person<code>Person</code>.
38  	 * 
39  	 * @param validatable
40  	 *            The Validate
41  	 * @return <code>IErrorCollector</code> containing Validation Messages.
42  	 * @see IErrorCollector
43  	 * @see Person
44  	 */
45  	@Override
46  	public void validateValidatable(IValidatable validatable,
47  			IErrorCollector collector) {
48  		if (validatable instanceof Person) {
49  			Person person = (Person) validatable;
50  			if (EaseUtil.isNullOrBlank(person.getBirthDate())) {
51  				collector.register(new EaseValidationException(SpringUtils
52  						.getMessage(PERSON_BIRTHDATE_IS_NULL_ERROR_MESSAGE)));
53  			}
54  			if (person.getPersonNames() == null
55  					|| person.getPersonNames().size() == 0) {
56  				// Check if Last Name is Null or Empty
57  				collector
58  						.register(new EaseValidationException(
59  								SpringUtils
60  										.getMessage(PERSON_CURRENT_NAME_IS_EMPTY_ERROR_MESSAGE)));
61  			}
62  			if (EaseUtil.isNullOrBlank(person.getPersonNames().get(0)
63  					.getLastName())) {
64  				// Check if Last Name is Null or Empty
65  				collector.register(new EaseValidationException(SpringUtils
66  						.getMessage(PERSON_LAST_NAME_IS_EMPTY_ERROR_MESSAGE)));
67  			}
68  			if (EaseUtil.isNullOrBlank(person.getPersonNames().get(0)
69  					.getFirstName())) {
70  				// Check if First Name is Null or Empty
71  				collector.register(new EaseValidationException(SpringUtils
72  						.getMessage(PERSON_FIRST_NAME_IS_EMPTY_ERROR_MESSAGE)));
73  			}
74  		}
75  	}
76  }

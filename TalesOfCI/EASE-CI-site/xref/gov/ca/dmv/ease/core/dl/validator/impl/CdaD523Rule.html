87  	/***
88  	 * @param cdaProcessContext
89  	 * @param errorCollector
90  	 */
91  	private void validate(CommercialTestResultGroup commercialTestResultGroup,
92  			CommercialTestResultType commercialTestResultType,
93  			IErrorCollector errorCollector) {
94  		boolean testWaived = false;
95  		boolean testFailed = false;
96  		List <CommercialTestResult> commercialAirBreaksTestResults = commercialTestResultGroup
97  				.getCommercialTestResults(commercialTestResultType);
98  		if (commercialAirBreaksTestResults != null
99  				&& commercialAirBreaksTestResults.size() > 1) {
100 			for (CommercialTestResult commercialAirBreaksTestResult : commercialAirBreaksTestResults) {
101 				String testResultCode = commercialAirBreaksTestResult
102 						.getTestResultCode().getCode();
103 				if (IGeneralTestConstants.WAIVED
104 						.equalsIgnoreCase(testResultCode)) {
105 					testWaived = true;
106 				}
107 				if (commercialAirBreaksTestResult.getTestFailureDate() != null) {
108 					testFailed = true;
109 				}
110 			}
111 		}
112 		if (testWaived && testFailed) {
113 			errorCollector
114 					.register(new ApplicationException(
115 							IDataCollectionErrorMessageConstants.TEST_PREVIOUSLY_FAILED_CANNOT_WAIVE));
116 		}
117 	}

32  				&& (activityName
33  						.equalsIgnoreCase("CollectDafTestResults1Activity") || activityName
34  						.equalsIgnoreCase("CollectDafTestResults2Activity"))) {
35  			if (compareSelectedAction(
36  					new Action[] { getAction(IActionsRegistry.UPDATE_PENDING_ACTION) },
37  					dafProcessContext)) {
38  				DlApplication currentDlApplication = dafProcessContext
39  						.getDlApplication();
40  				if (currentDlApplication != null) {
41  					String previousIncompleteApplicationReasonCode = ""; // TODO: We don't have BO to get maintain previously entered incomplete applicaiton reason code.
42  					String newIncompleteApplicationReasonCode = getNewIncompleteApplicationReasonCode(dafProcessContext
43  							.getDlApplication());
44  					if (newIncompleteApplicationReasonCode != null
45  							&& previousIncompleteApplicationReasonCode != null) {
46  						Map <String, Integer> iaOrderMap = IncompleteApplicationReasonsCodesOrder.orders;
47  						Integer prevIncompAppReasonPriority = iaOrderMap
48  								.get(previousIncompleteApplicationReasonCode);
49  						Integer newIncompAppReasonPriority = iaOrderMap
50  								.get(newIncompleteApplicationReasonCode);
51  						if (prevIncompAppReasonPriority > newIncompAppReasonPriority) {
52  							errorCollector.register(new ApplicationException(

58  					boolean isIncompleteAppReasonCodeValid = true; // TODO: For now hard coded.

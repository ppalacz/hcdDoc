17  /***

27  //DOSC
28  public class OutOfStateDlCollectionValidator extends AbstractDlValidator {
29  	/*** The Constant serialVersionUID. */
30  	private static final long serialVersionUID = -1308864558019731824L;
31  
32  	/***
33  	 * Checks if is driver license and state code matched.
34  	 * 
35  	 * @param outOfStateLicenseNumeber the out of state license number
36  	 * @param outOfState the out of state
37  	 * 
38  	 * @return true, if successful
39  	 */
40  	protected boolean areDriverLicenseAndStateCodeMatched(
41  			String outOfStateLicenseNumeber, String outOfState) {
42  		if (isNullorEmpty(outOfStateLicenseNumeber)
43  				|| outOfStateLicenseNumeber.length() < 2) {
44  			return Boolean.FALSE;
45  		}
46  		String subOfLicense = outOfStateLicenseNumeber.substring(0, 2);
47  		if (subOfLicense.equals(outOfState)) {
48  			return Boolean.TRUE;
49  		}
50  		return Boolean.FALSE;
51  	}
52  
53  	/* (non-Javadoc)
54  	 * @see gov.ca.dmv.ease.fw.validate.impl.AbstractBaseValidator#validateValidatable(gov.ca.dmv.ease.fw.validate.IValidatable, gov.ca.dmv.ease.fw.error.IErrorCollector)
55  	 */
56  	@Override
57  	public void validateValidatable(IValidatable validatable,
58  			IErrorCollector collector) {
59  		DlApplication dlApplication = (DlApplication) validatable;
60  		List <OutOfStateLicense> outOfStateLicenses = dlApplication
61  				.getApplicant().getPreviousLicenses();
62  		for (OutOfStateLicense outOfStateLicense : outOfStateLicenses) {
63  			String outOfState = outOfStateLicense.getIssuedByState().getCode();
64  			String outOfStateLicenseNumeber = outOfStateLicense
65  					.getLicenseNumber();
66  			if (areDriverLicenseAndStateCodeMatched(outOfStateLicenseNumeber,
67  					outOfState)) {
68  				//ERROR CODE - D900
69  				addErrorMessage(collector, VERIFY_OUT_OF_STATE_DL_NUMBER);
70  			}
71  		}
72  	}
73  }

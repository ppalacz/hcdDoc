32  public class AutomatedReceiptValidator extends AbstractBusValidator {
33  	/*** The Constant serialVersionUID. */
34  	private static final long serialVersionUID = -2845278818605340592L;
35  
36  	/***
37  	 * This method validates List of vouchers <code>Voucher</code>.
38  	 * 
39  	 * @param validatables The List of Vouchers to validate
40  	 * @return <code>IErrorCollector</code> containing Validation Messages.
41  	 * @see IErrorCollector
42  	 * @see Voucher
43  	 */
44  	@Override
45  	public void validateValidatable(IValidatable validatables,
46  			IErrorCollector collector) {
47  		try {
48  			Voucher voucher = (Voucher) validatables;
49  			if (EaseUtil.isNullOrBlank(voucher.getSequenceNumber())) {
50  				collector
51  						.register(new EaseValidationException(
52  								SpringUtils
53  										.getMessage(VOUCHER_SEQUENCE_NUMBER_IS_NULL_ERROR_MESSAGE)));
54  			}
55  			else if (voucher.getSequenceNumber() < 1) {
56  				collector
57  						.register(new EaseValidationException(
58  								SpringUtils
59  										.getMessage(VOUCHER_SEQUENCE_NUMBER_IS_INVALID_ERROR_MESSAGE)));
60  			}
61  			else if (voucher.getSequenceNumber() > 9999) {
62  				collector
63  						.register(new EaseValidationException(
64  								SpringUtils
65  										.getMessage(VOUCHER_SEQUENCE_NUMBER_IS_INVALID_ERROR_MESSAGE)));
66  			}
67  		}
68  		catch (Exception e) {
69  			throw new AutomatedReceiptRequestValidationException(e);
70  		}
71  	}
72  }

29  public class SpD569Rule extends AbstractSpRule {
30  	public void validate(ProcessContext processContext) {
31  		String ttcList = ITtcConstants.SPD + "," + ITtcConstants.SPO + ","
32  				+ ITtcConstants.SPR + "," + ITtcConstants.SRX;
33  		// if we are ttc spd, spo, spr,  or srx  and we have indicated no fee then authorization is required
34  		if (isEnterAction(processContext)
35  				&& ttcList.contains(getTtcCode(processContext))
36  				&& !getSalesPersonApplication(processContext).isFeeRequired()) {
37  			// we've marked that no fee is required but did not indicate why
38  			if (getSalesPersonApplication(processContext)
39  					.getFeeRequiredReasonCode() == null) {
40  				getErrorCollector()
41  						.register(
42  								new EaseValidationException(
43  										IDataCollectionErrorMessageConstants.INVALID_FEE_INDICATOR));
44  			}
45  			else {
46  				// Cannot indicate a reason other than 'F' without also including an authorizing password (F only available for manual mode handled in other Rule.
47  				if ((!getSalesPersonApplication(processContext)
48  						.getFeeRequiredReasonCode()
49  						.getCode()
50  						.equalsIgnoreCase(
51  								IFeeReasonCodeConstants.F_FEE_FOR_MANUAL_APPLICATION))){
52  					// we provided a password, is it correct? 
53  					if(EaseUtil.isNotBlank(getAuthorizingPassword(processContext))){
54  						try{
55  							if(!getAuthAndAuthService().isAuthorizedApprover(
56  								getAuthorizingUserId(processContext), getAuthorizingPassword(processContext))){
57  								getErrorCollector()
58  								.register(new EaseValidationException(
59  										IFeeAndPaymentErrorMessageConstants.APPROVAL_AUTHORITY_PASSWORD_INVALID));
60  							}
61  						}
62  						catch(LoginException le){
63  							getErrorCollector()
64  							.register(new EaseValidationException(
65  									IFeeAndPaymentErrorMessageConstants.APPROVAL_AUTHORITY_PASSWORD_INVALID));
66  						}
67  						
68  					}
69  					else{// we did not provide a password: fail
70  						getErrorCollector()
71  						.register(new ApplicationException(
72  								IDataCollectionErrorMessageConstants.APPROVAL_AUTHORITY_PASSWORD_REQUIRED));	
73  					}
74  											
75  				}
76  				
77  			}
78  		}
79  	}
80  }

59  	 * Sets license expiration date based on military extension code
60  	 * 
61  	 * @param processContext
62  	 */
63  	public void militaryExtensionRule(ProcessContext processContext) {
64  		CustomerInfoProcessContext customerInfoProcessContext = (CustomerInfoProcessContext) processContext;
65  		DlApplication dlApplication = customerInfoProcessContext
66  				.getDlApplication();
67  		DriverLicense driverLicense = dlApplication.getLicense();
68  		String militaryExtensionCode = null;
69  		if (dlApplication.getMilitaryDutyCode() != null) {
70  			militaryExtensionCode = dlApplication.getMilitaryDutyCode()
71  					.getCode();
72  		}
73  		if (IMilitaryExtensionCodeConstants.END_OF_EXTENSION_E
74  				.equals(militaryExtensionCode)) {
75  			driverLicense.setExpirationDate(new Date());
76  		}
77  		else if (IMilitaryExtensionCodeConstants.RETENSION_OF_EXTENSION_OR_EXTENSION_Y
78  				.equals(militaryExtensionCode)) {
79  			driverLicense.setExpirationDate(DateUtil.addToCurrentDate(1, 0, 0));
80  		}
81  	}

144 	/***
145 	 * @param dlApplication
146 	 * @param errorCollector
147 	 */
148 	public void validateMotorcycle(DlApplication dlApplication,
149 			String selectedTtc, IErrorCollector errorCollector) {
150 		String motorCycleTraingCode = null;
151 		if (dlApplication.getMotorCycleTrainingCode() != null) {
152 			motorCycleTraingCode = dlApplication.getMotorCycleTrainingCode()
153 					.getCode();
154 		}
155 		String trainingWaiveCode = null;
156 		if (dlApplication.getOutOfStateDriverEducationAndTrainingWaiveCode() != null) {
157 			trainingWaiveCode = dlApplication
158 					.getOutOfStateDriverEducationAndTrainingWaiveCode()
159 					.getCode();
160 		}
161 		//D729
162 		if (IMotorcycleTrainingCodeConstants.WAIVED_W
163 				.equals(motorCycleTraingCode)) {
164 			if (!"R".equals(trainingWaiveCode)) {
165 				errorCollector
166 						.register(new EaseValidationException(
167 								IDataCollectionErrorMessageConstants.M_C_TRNG_W_REQUIRES_PREV_LIC));
168 			}
169 		}
170 		//0001
171 		Date birthdate = dlApplication.getApplicant().getBirthDate();
172 		if (birthdate != null) {
173 			if (DateUtil.isYoungerThan17AndHalf(birthdate)) {
174 				if (!"R".equals(trainingWaiveCode)) {
175 					if (trainingWaiveCode == null) {
176 						errorCollector.register(new EaseValidationException(
177 								IGeneralErrorMessageConstants.FIELD_REQUIRED));
178 					}
179 				}
180 			}
181 		}
182 		//D730
183 		if (IMotorcycleTrainingCodeConstants.PREVIOUS_CLASSM_OR_CNA_X
184 				.equals(motorCycleTraingCode)) {
185 			if (dlApplication.isCNACondition()) {
186 				if (ITtcConstants.DLA.equals(selectedTtc)) {
187 					errorCollector
188 							.register(new EaseValidationException(
189 									IDataCollectionErrorMessageConstants.M_C_TRNG_X_DURING_CNA_FOR_NON_ORIGINALS_ONLY));
190 				}
191 			}
192 		}
193 	}

202 	public void validateEducationAndTraining(IErrorCollector errorCollector,
203 			String number, String code, String trainingWaiveCode, Date birthDate) {
204 		if (birthDate != null) {
205 			if (DateUtil.isYoungerThan17AndHalf(birthDate)) {
206 				//0001
207 				if (code == null || number == null) {
208 					errorCollector.register(new EaseValidationException(
209 							IGeneralErrorMessageConstants.FIELD_REQUIRED));
210 				}
211 			}
212 			if ("R".equals(trainingWaiveCode)) {
213 				if (!EaseUtil.isNullOrBlank(code)
214 						|| !EaseUtil.isNullOrBlank(number)) {
215 					//0153
216 					errorCollector
217 							.register(new EaseValidationException(
218 									IGeneralErrorMessageConstants.INVALID_FIELD_PRESENT));
219 				}
220 			}
221 			else {
222 				if (DateUtil.isYoungerThan17AndHalf(birthDate)) {
223 					//0001
224 					if (code == null || number == null) {
225 						errorCollector.register(new EaseValidationException(
226 								IGeneralErrorMessageConstants.FIELD_REQUIRED));
227 					}
228 				}
229 			}
230 		}

233 	public void validateStatus(IErrorCollector errorCollector, String status,
234 			String barCodeNumber, String code, Date birthDate) {
235 		//D540
236 		if (DateUtil.isYoungerThan17AndHalf(birthDate)) {
237 			if (IDriverEducationConstants.NOT_ENROLLED.equals(status)) {
238 				errorCollector
239 						.register(new EaseValidationException(
240 								IDataCollectionErrorMessageConstants.DR_ED_DR_TRNG_ENROLLMENT_OR_COMPLETION_REQUIRED));
241 			}
242 		}
243 		//D539
244 		if (barCodeNumber != null && barCodeNumber.startsWith("Z")) {
245 			if (IDriverEducationConstants.SIMULTANEOUS_ENROLLMENT
246 					.equals(status)) {
247 				errorCollector
248 						.register(new EaseValidationException(
249 								IDataCollectionErrorMessageConstants.DR_ED_AND_DR_TRNG_COMPLETION_REQUIRED));
250 			}
251 		}
252 		//0001
253 		if (IDriverEducationConstants.COMPLETED_REQUIREMENTS.equals(status)) {
254 			if (barCodeNumber == null || code == null) {
255 				errorCollector.register(new EaseValidationException(
256 						IGeneralErrorMessageConstants.FIELD_REQUIRED));
257 			}
258 		}
259 	}

261 	/***
262 	 * @param errorCollector
263 	 * @param driverEducationStatusCode
264 	 * @param driverTrainingStatusCode
265 	 */
266 	public void validateWaivedStatus(DlApplication dlApplication,
267 			IErrorCollector errorCollector) {
268 		String driverEducationStatusCode = null;
269 		if (dlApplication.getDriverEducationStatusCode() != null) {
270 			driverEducationStatusCode = dlApplication
271 					.getDriverEducationStatusCode().getCode();
272 		}
273 		String driverTrainingStatusCode = null;
274 		if (dlApplication.getDriverTrainingStatusCode() != null) {
275 			driverTrainingStatusCode = dlApplication
276 					.getDriverTrainingStatusCode().getCode();
277 		}
278 		//D501
279 		if (IDriverEducationConstants.WAIVED_REQUIREMENTS
280 				.equals(driverEducationStatusCode)
281 				&& !IDriverEducationConstants.WAIVED_REQUIREMENTS
282 						.equals(driverTrainingStatusCode)) {
283 			errorCollector
284 					.register(new EaseValidationException(
285 							IDataCollectionErrorMessageConstants.W_INVALID_IF_ANY_OTHER_CODE_PRESENT));
286 		}
287 		else if (!IDriverEducationConstants.WAIVED_REQUIREMENTS
288 				.equals(driverEducationStatusCode)
289 				&& IDriverEducationConstants.WAIVED_REQUIREMENTS
290 						.equals(driverTrainingStatusCode)) {
291 			errorCollector
292 					.register(new EaseValidationException(
293 							IDataCollectionErrorMessageConstants.W_INVALID_IF_ANY_OTHER_CODE_PRESENT));
294 		}
295 		//D538
296 		String trainingWaiveCode = null;
297 		if (dlApplication.getOutOfStateDriverEducationAndTrainingWaiveCode() != null) {
298 			trainingWaiveCode = dlApplication
299 					.getOutOfStateDriverEducationAndTrainingWaiveCode()
300 					.getCode();
301 		}
302 		if (IDriverEducationConstants.WAIVED_REQUIREMENTS
303 				.equals(driverEducationStatusCode)
304 				|| IDriverEducationConstants.WAIVED_REQUIREMENTS
305 						.equals(driverTrainingStatusCode)) {
306 			if (!"R".equals(trainingWaiveCode)) {
307 				errorCollector
308 						.register(new EaseValidationException(
309 								IDataCollectionErrorMessageConstants.DR_ED_AND_DR_TRNG_VS_PREV_LIC));
310 			}
311 		}
312 	}

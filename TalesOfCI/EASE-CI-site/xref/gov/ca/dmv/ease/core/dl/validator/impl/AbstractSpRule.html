336 	/***
337 	 * if the applicant has an id card and it is expired return true. In
338 	 * all other cases return false.
339 	 *
340 	 * @param context the context
341 	 * @return true, if is id card expired
342 	 */
343 	protected boolean isIdCardExpired(ProcessContext context){
344 		
345 		Person applicant = getApplicant(context);
346 		if(applicant == null)
347 			return false; 
348 		
349 		if(applicant.getIdCard() == null)
350 			return false; 
351 		
352 		if(applicant.getIdCard().getExpirationDate() ==  null)
353 			return false; 
354 		
355 		return getDateDifferenceInDays(applicant.getIdCard().getExpirationDate(), getWorkDate(context)) < 0; 
356 	}

359 	 * return true if there is a drivers license and it has expired. 
360 	 * @param context
361 	 * @param ifNoDriversLicenseUseIdCard
362 	 * @return

364 	protected boolean isDriversLicenseExpired(ProcessContext context){
365 		Person applicant = getApplicant(context);
366 		if(applicant == null)
367 			return false; 
368 		
369 		if(applicant.getDriverLicense() == null)

376 	}

378 	/***
379 	 * Get the difference in days between two dates. If the difference is
380 	 * less than .5 days then return 0. If the firstDate is later than the second
381 	 * date then the return value will be positive. Otherwise the difference is
382 	 * negative or zero.
383 	 *
384 	 * @param firstDate the first date
385 	 * @param secondDate the second date
386 	 * @return the date difference in days
387 	 * @throws IllegalArgumentException the illegal argument exception
388 	 */
389 	protected int getDateDifferenceInDays(Date firstDate, Date secondDate) throws IllegalArgumentException{
390 		if(firstDate == null || secondDate == null){
391 			throw new IllegalArgumentException("A null date was passed, cannot compare.");
392 		}
393 
394 		Long diff = firstDate.getTime() - secondDate.getTime();
395 		
396 		Long diffInDays = new Long(Math.round(diff/(1000 * 60 * 60 * 24)));				
397 		
398 		return diffInDays.intValue(); 
399 	}

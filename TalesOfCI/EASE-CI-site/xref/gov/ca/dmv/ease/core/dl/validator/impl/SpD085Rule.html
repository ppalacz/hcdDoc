31  	public void validate(ProcessContext processContext) {
32  
33  		Person person = getApplicant(processContext); 
34  		if (isEnterAction(processContext) && person != null) {
35  			String mailingCityCode = ((SpProcessContext)processContext).getMailingCityCode();
36  			String mailingCityName = person.getAddress(
37  					AddressType.MAILING).getCity();
38  			String mailingStateName = person.getAddress(
39  					AddressType.MAILING).getState();
40  			String mailingStateZip = person.getAddress(
41  					AddressType.MAILING).getZip();
42  			validateUnableToGenerateZipOrCountyCode(mailingCityCode, mailingCityName, mailingStateName,
43  					mailingStateZip);
44  		}		
45  	}
46  
47  	/***
48  	 * @param errorCollector
49  	 * @param mailCityCode
50  	 * @param mailCityName
51  	 * @param mailState
52  	 * @param mailZipCode
53  	 */
54  	public void validateUnableToGenerateZipOrCountyCode(String mailCityCode,
55  			String mailCityName, String mailState, String mailZipCode) {
56  		if (!isNullOrBlank(mailCityName) && !isNullOrBlank(mailState)
57  				&& isNullOrBlank(mailCityCode)) {
58  			if (isNullOrBlank(mailZipCode) || !isZipValid(mailZipCode)) {
59  				getErrorCollector()
60  						.register(new EaseValidationException(
61  								IDataCollectionErrorMessageConstants.UNABLE_TO_GENERATE_ZIP_OR_COUNTY_CODE_MUST_BE_KEYED));
62  			}
63  		}
64  	}

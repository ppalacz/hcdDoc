30  public class SpD670Rule extends AbstractSpRule {
31  	/* (non-Javadoc)
32  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
33  	 */
34  	public void validate(ProcessContext processContext) {
35  		SpProcessContext spProcessContext = (SpProcessContext) processContext;
36  		SalesPersonApplication spApplication = spProcessContext
37  				.getSalesPersonApplication();
38  		Action selectedAction = spProcessContext.getSelectedAction();
39  		IErrorCollector errorCollector = new ErrorCollector();
40  		if (selectedAction != null
41  				&& selectedAction == getActionsRegistry().getAction(
42  						IActionsRegistry.ENTER_ACTION)) {
43  			   Person person = spApplication.getApplicant();
44  			   String lastName = person.getCurrentName().getLastName();
45  
46  				if (EaseUtil.isNullOrBlank(person) && EaseUtil.isNullOrBlank(lastName)
47  						&& lastName.trim().length() >= 1) {
48  					errorCollector
49  					.register(new EaseValidationException(
50  							IDataCollectionErrorMessageConstants.NAME_MUST_HAVE_MORE_THAN_ONE_LETTER));
51  				}
52  		}
53  		setErrorCollector(errorCollector);
54  	}

56  }

40  	public void validate(ProcessContext processContext) {
41  		String ttc = processContext.getUserContext().getTtc();
42  		if (ITtcConstants.CDA.equals(ttc)) {
43  			CdaProcessContext cdaProcessContext = (CdaProcessContext) processContext;
44  			Action selectedAction = cdaProcessContext.getSelectedAction();
45  			if (selectedAction != null) {
46  				IErrorCollector errorCollector = new ErrorCollector();
47  				if (selectedAction.equals(getActionsRegistry().getAction(
48  						IActionsRegistry.ENTER_ACTION))) {
49  					restrictionCodeValidator.validate(cdaProcessContext
50  							.getDlApplication(), cdaProcessContext
51  							.getDriveAndLawTestResultGroup(), errorCollector);
52  					testResults1MiscValidator.validate(cdaProcessContext
53  							.getDlApplication(), cdaProcessContext
54  							.getReadOnlyDlApplication(), cdaProcessContext
55  							.getDriveAndLawTestResultGroup(), cdaProcessContext
56  							.getUserContext().getTtc(), cdaProcessContext
57  							.getAuthorizingPassword(), cdaProcessContext
58  							.getAuthorizingUserId(), errorCollector);
59  					attachmentCodeValidator.validate(cdaProcessContext
60  							.getDlApplication(), cdaProcessContext
61  							.getDriveAndLawTestResultGroup(), cdaProcessContext
62  							.getCommercialTestResultGroup(), cdaProcessContext
63  							.getUserContext().getTtc(), errorCollector);
64  				}
65  				if (selectedAction.equals(getActionsRegistry().getAction(
66  						IActionsRegistry.UPDATE_PENDING_ACTION))) {
67  					cdaUpdatePendingRule.validate(cdaProcessContext
68  							.getDlApplication(), errorCollector);
69  				}
70  				if (selectedAction.equals(getActionsRegistry().getAction(
71  						IActionsRegistry.DAF_UPDATE_ACTION))) {
72  					cdaDafUpdateRule.validate(cdaProcessContext
73  							.getDlApplication(), errorCollector);
74  				}
75  				setErrorCollector(errorCollector);
76  			}
77  		}
78  		else if (ITtcConstants.DAF.equals(ttc)) {
79  			DafProcessContext dafProcessContext = (DafProcessContext) processContext;
80  			Action selectedAction = dafProcessContext.getSelectedAction();
81  			if (selectedAction != null) {
82  				IErrorCollector errorCollector = new ErrorCollector();
83  				if (selectedAction.equals(getActionsRegistry().getAction(
84  						IActionsRegistry.UPDATE_PENDING_ACTION))) {
85  					dafD551Rule.validate(dafProcessContext.getDlApplication(),
86  							dafProcessContext.getUserContext()
87  									.getOperationalMode().getCode(),
88  							errorCollector);
89  				}
90  				setErrorCollector(errorCollector);
91  			}
92  		}
93  	}

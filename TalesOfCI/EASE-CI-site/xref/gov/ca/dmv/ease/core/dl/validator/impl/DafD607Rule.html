30  public class DafD607Rule extends AbstractDafRule {
31  	/* (non-Javadoc)
32  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
33  	 */
34  	public void validate(ProcessContext processContext) {
35  		DafProcessContext dafProcessContext = (DafProcessContext) processContext;
36  		IErrorCollector errorCollector = new ErrorCollector();
37  		String activityName = getCurrentActivityName(dafProcessContext);
38  		if (activityName != null
39  				&& activityName
40  						.equalsIgnoreCase("DisplayOfIdApplicationDataActivity")) {
41  			if (compareSelectedAction(
42  					new Action[] { getAction(IActionsRegistry.UPDATE_PENDING_ACTION) },
43  					dafProcessContext)) {
44  				Transaction tempDafRecord = dafProcessContext.getTransaction();
45  				if (tempDafRecord != null) {
46  					if (tempDafRecord.getLicenseType().equals(
47  							LicenseType.ID_CARD)) {
48  						String newIncompleteApplicationReasonCode = getNewIncompleteApplicationReasonCode(dafProcessContext
49  								.getDlApplication());
50  						String existingIncompleteApplicationReasonCode = getExistingIdIncompleteApplicationReasonCode(dafProcessContext);
51  						if (newIncompleteApplicationReasonCode != null
52  								&& existingIncompleteApplicationReasonCode != null) {
53  							if (newIncompleteApplicationReasonCode
54  									.equals(existingIncompleteApplicationReasonCode)) {
55  								errorCollector
56  										.register(new ApplicationException(
57  												"D607"));
58  							}
59  						}
60  					}
61  				}
62  			}
63  		}
64  		else if (activityName != null
65  				&& (activityName
66  						.equalsIgnoreCase("CollectDafTestResults1Activity") || activityName
67  						.equalsIgnoreCase("CollectDafTestResults2Activity"))) {
68  			if (compareSelectedAction(
69  					new Action[] {// TODO: This validation is wrong and so it is not added in DafBusinessProcess.xml for now. As per requirement, we should be able to compare updated code with pre-populated code. If both are different then display an error message.	
70  					getAction(IActionsRegistry.ENTER_ACTION) },
71  					dafProcessContext)) {
72  				String newIncompleteApplicationReasonCode = getNewIncompleteApplicationReasonCode(dafProcessContext
73  						.getDlApplication());
74  				if (newIncompleteApplicationReasonCode != null) {
75  					errorCollector.register(new ApplicationException("D607"));
76  				}
77  			}
78  		}
79  		setErrorCollector(errorCollector);
80  	}
81  }

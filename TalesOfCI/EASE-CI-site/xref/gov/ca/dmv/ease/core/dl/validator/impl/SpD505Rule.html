23  public class SpD505Rule extends AbstractSpRule {
24  	public void validate(ProcessContext processContext) {
25  		String ttc = getTtcCode(processContext);
26  		String applicableTtcs = ITtcConstants.SPO + "," + ITtcConstants.SPC
27  				+ "," + ITtcConstants.SPR + "," + ITtcConstants.SRX + ","
28  				+ ITtcConstants.SPD;
29  		if (isEnterAction(processContext) && applicableTtcs.contains(ttc)
30  				&& getApplicantDriversLicense(processContext, false) != null) {
31  			if (!isDriversLicenseExpired(processContext) && hasChanges(processContext)) {
32  				getErrorCollector()
33  						.register(
34  								new ApplicationException(
35  										IDataCollectionErrorMessageConstants.CHANGES_MUST_FIRST_BE_MADE_VIA_A_DL_APP));
36  			}
37  		}
38  	}
39  
40  	/***
41  	 * Checks for changes to the name, gender or birth date. 
42  	 *
43  	 * @param context the context
44  	 * @return true, if successful
45  	 */
46  	private boolean hasChanges(ProcessContext context) {
47  		// has bdate changed?
48  		SpProcessContext spContext = (SpProcessContext) context;
49  		if (spContext.getChangedBirthDate() != null){
50  			if(getDateDifferenceInDays(spContext.getChangedBirthDate(),
51  						getApplicant(context).getBirthDate()) != 0){
52  				return true;
53  			}
54  		}
55  		// name change? 
56  		// TODO: should this compare to person.getPersonNames? Is getPreviousName used? 
57  		if (!getApplicant(context).getCurrentName().equals(
58  				getApplicant(context).getPreviousName()))
59  			return true;
60  		// Gender changed? 
61  		if (getApplicant(context).isGenderChanged())
62  			return true;
63  		return false;
64  	}

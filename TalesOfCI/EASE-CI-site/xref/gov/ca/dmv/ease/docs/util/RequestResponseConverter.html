25  public final class RequestResponseConverter {
26  	//Constants Declarations
27  	//Holds the value of Task Service Package 
28  	private static final String CONST_PACKAGE_TASK_SERVICE = "ts";
29  	//Holds the value of Business Utility Service Package 
30  	private static final String CONST_PACKAGE_BUSINESS_UTILITY_SERVICE = "bus";
31  	//Holds the value of Suffix for Task Service Files 
32  	private static final String CONST_SUFFIX_TASK_SERVICE = "Ts";
33  	//Holds the value of Suffix for Business Utility Service Files 
34  	private static final String CONST_SUFFIX_BUSINESS_UTILITY_SERVICE = "Bus";
35  
36  	/***
37  	 * This method provides utility to copy values from a Request/Response of one layer to
38  	 * the corresponding Request/Response of the next layer in Architecture.
39  	 * 
40  	 * @param fromObject Request/Response Object of Current Layer
41  	 * @param toObject Request/Response Object of Next Layer
42  	 * @param Object Request/Response Object of Next Layer populated with Current Layer Request/Response Values
43  	 */
44  	public static final Object convertObject(Object fromObject,
45  			Object toObject) {
46  		//Holds the Instance of Request/Response for Next Layer e.g. Business Service Request Object
47  		Object populatedObjectNextLayer = new Object();
48  		//Perform Null Check
49  		if (null == fromObject || null == toObject) {
50  			//If Either Current Layer or Next Layer Objects are Null
51  			//Do not Process and throw an Application Exception
52  			throw new ApplicationException(" Input Parameters cannot be Null ");
53  		}
54  		//Fetch the List of getter methods on the Current Layer Object e.g. Task Service Request
55  		try {
56  			//Fetch the Class from its Canonical Name
57  			Class classCurrentLayer = Class.forName(fromObject
58  					.getClass().getCanonicalName());
59  			//Fetch the Declared Methods
60  			Method methods[] = classCurrentLayer.getDeclaredMethods();
61  			//Holds the return objects for Each Method invoked i.e. GetXXX() only
62  			Object returnObjects[] = new Object[methods.length];
63  			//Iterate the List of Methods in the Class
64  			for (int i = 0; i < methods.length; i++) {
65  				//Invoke Each Method in the Request and store the return value in a List
66  				returnObjects[i] = methods[i].invoke(fromObject, null);
67  			}
68  			//Fetch Constructors of the Request/Response Object of Next Layer e.g. Business Service Request Object
69  			//TODO Validate Assumption - only 1 Constructor Exists
70  			//Fetch the Class from its Canonical Name
71  			Class classNextLayer = Class.forName(toObject.getClass()
72  					.getCanonicalName());
73  			//Fetch the Constructors of the Request Class
74  			Constructor constructorList[] = classNextLayer
75  					.getDeclaredConstructors();
76  			//Holds the Instance of a Constructor
77  			Constructor constructor = null;
78  			//Iterate the List of Constructors in the Class
79  			for (int i = 0; i < constructorList.length; i++) {
80  				constructor = constructorList[i];
81  				//Populate the Next Layer Object with Current Layer values
82  				populatedObjectNextLayer = constructor
83  						.newInstance(returnObjects);
84  			}
85  		}
86  		catch (Exception e) {
87  			throw new ApplicationException(
88  					" Unable to Return populated Next Layer Object "
89  							+ e.getMessage());
90  		}
91  		//Return the Instance of the Next Layer Object
92  		return populatedObjectNextLayer;
93  	}
94  
95  	/***
96  	 * This method provides utility to copy values from a Request/Response of one layer to
97  	 * the corresponding Request/Response of the next layer in Architecture.
98  	 * 
99  	 * @param objectCurrentLayer Request Object of Task Service Layer
100 	 * @return Object Request Object of Business Service Layer
101 	 * TODO - Need to make it Generic in Future with ENUMS for all Layers.
102 	 */
103 	public static final Object deepCopyUtility(Object objectCurrentLayer) {
104 		Object returnObject = new Object();
105 		//Fetch the List of getter methods on the Task Service Request
106 		try {
107 			//Fetch the Class from its Canonical Name
108 			String canonicalName = objectCurrentLayer.getClass()
109 					.getCanonicalName();
110 			//Replace the task folder with bus folder in package name
111 			String busCanonicalName = canonicalName.replaceAll(
112 					CONST_PACKAGE_TASK_SERVICE,
113 					CONST_PACKAGE_BUSINESS_UTILITY_SERVICE);
114 			//Replace ts with bus in the file name
115 			String busRequestCanonicalName = busCanonicalName.replace(
116 					CONST_SUFFIX_TASK_SERVICE,
117 					CONST_SUFFIX_BUSINESS_UTILITY_SERVICE);
118 			//Fetch the Class of the Return Object Type
119 			Class tsClass = Class.forName(busRequestCanonicalName);
120 			//Create the Instance of the Return Object
121 			returnObject = tsClass.newInstance();
122 		}
123 		catch (Exception e) {
124 			throw new ApplicationException(
125 					" Unable to Resolve the Return Object " + e.getMessage());
126 		}
127 		//Return the Instance of the Next Layer Object
128 		return returnObject;
129 	}
130 }

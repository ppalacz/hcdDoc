171 	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
172 		FtpFtaPaymentReceiptDocumentRequestTus request = (FtpFtaPaymentReceiptDocumentRequestTus) requestFromClient;
173 		IUserContext userContext = request.getUserContext();
174 		FtaFtpFinePaymentClearanceApplication application = request
175 				.getApplication();
176 		Invoice invoice = application.getInvoice();
177 		List <CitationDetail> citationDetails = application
178 				.getCitationDetails();
179 		Person person = application.getApplicant();
180 		Address address = person.getExistingAddress(AddressType.MAILING);
181 		List <String> elementsList = new ArrayList <String>();
182 		if (false) {//TODO:  Always File Copy?
183 			addToElementsList(elementsList, FILE_COPY_TEXT, FILE_COPY_ELEMENT);
184 		}
185 		addToElementsList(elementsList, getDlorIdNumber(person),
186 				DL_ID_NUMBER_ELEMENT);
187 		addToElementsList(elementsList, DocumentConverterUtil
188 				.getFormattedCurrentName(person), FULL_NAME_ELEMENT);
189 		addToElementsList(elementsList, DocumentConverterUtil
190 				.getFormattedAddress(address), MAIL_ADDRESS_ELEMENT);
191 		addToElementsList(elementsList, getApplicantMailingCity(address),
192 				MAIL_CITY_ELEMENT);
193 		addToElementsList(elementsList, getApplicantMailingState(address),
194 				MAIL_STATE_ELEMENT);
195 		addToElementsList(elementsList, getApplicantMailingZip(address),
196 				MAIL_ZIP_ELEMENT);
197 		DecimalFormat formatter = new DecimalFormat("####0.00");
198 		InvoiceConverterUtility.populateInvoice(elementsList,
199 				ftpFtaCashElements, ftpFtaDebitElements, ftpFtaCreditElements,
200 				invoice, formatter, userContext.getWorkDate(), false);
201 		addCitations(elementsList, citationDetails, 0, END_PAGE1_CITATIONS);
202 		addToElementsList(elementsList, convertDateToString(userContext
203 				.getWorkDate(), FORMATTER_MM_DD_YY), DATE_ELEMENT);//TODO:  Correct date???
204 		addToElementsList(elementsList, DocumentConverterUtil
205 				.getOfficeNameFromId(userContext.getOfficeId()),
206 				OFFICE_NAME_ELEMENT);
207 		addToElementsList(elementsList, convertValueOrBlankForNull(userContext
208 				.getOfficeId()), OFFICE_ID_ELEMENT);
209 		addToElementsList(elementsList, convertDateToString(userContext
210 				.getWorkDate(), FORMATTER_MM_DD_YY), WORK_DATE_ELEMENT);
211 		addToElementsList(elementsList, convertValueOrBlankForNull(userContext
212 				.getTechId()), KEYERS_ID_ELEMENT);
213 		addToElementsList(elementsList, userContext
214 				.getControlCashierSeqNumber().toString(),
215 				CASHIER_SEQUENCE_ELEMENT);
216 		String xmlData = null;
217 		if (citationDetails.size() > END_PAGE1_CITATIONS) {
218 			addToElementsList(elementsList, PAGE_TEXT, PAGE_ELEMENT);
219 			List <String> pageTwoElementsList = new ArrayList <String>();
220 			List <List <String>> documentsList = new ArrayList <List <String>>();
221 			List <String> templateNames = new ArrayList <String>();
222 			documentsList.add(elementsList);
223 			templateNames.add(DOCUMENT_TEMPLATE_NAME);
224 			documentsList.add(pageTwoElementsList);
225 			templateNames.add(PAGE_TWO_TEMPLATE_NAME);
226 			addToElementsList(pageTwoElementsList, FILE_COPY_TEXT,
227 					FILE_COPY_ELEMENT);//TODO:  Always File Copy?
228 			addToElementsList(pageTwoElementsList, "", TRAINING_ELEMENT);
229 			addToElementsList(pageTwoElementsList, getDlorIdNumber(person),
230 					DL_ID_NUMBER_ELEMENT);
231 			addToElementsList(pageTwoElementsList, DocumentConverterUtil
232 					.getFormattedCurrentName(person), FULL_NAME_ELEMENT);
233 			addCitations(pageTwoElementsList, citationDetails,
234 					START_PAGE2_CITATIONS, END_PAGE2_CITATIONS);
235 			addToElementsList(pageTwoElementsList, convertDateToString(
236 					userContext.getWorkDate(), FORMATTER_MM_DD_YY),
237 					DATE_ELEMENT);
238 			addToElementsList(pageTwoElementsList, DocumentConverterUtil
239 					.getOfficeNameFromId(userContext.getOfficeId()),
240 					OFFICE_NAME_ELEMENT);
241 			addToElementsList(pageTwoElementsList,
242 					convertValueOrBlankForNull(userContext.getOfficeId()),
243 					OFFICE_ID_ELEMENT);
244 			addToElementsList(pageTwoElementsList, convertDateToString(
245 					userContext.getWorkDate(), FORMATTER_MM_DD_YY),
246 					WORK_DATE_ELEMENT);//TODO:  Correct date???
247 			addToElementsList(pageTwoElementsList,
248 					convertValueOrBlankForNull(userContext.getTechId()),
249 					KEYERS_ID_ELEMENT);
250 			addToElementsList(pageTwoElementsList, userContext
251 					.getControlCashierSeqNumber().toString(),
252 					CASHIER_SEQUENCE_ELEMENT);
253 			xmlData = DocumentConverterUtil.convertToXml(documentsList,
254 					templateNames);
255 		}
256 		else {
257 			addToElementsList(elementsList, "", PAGE_ELEMENT);
258 			xmlData = DocumentConverterUtil.createXmlForSingleDocument(
259 					elementsList, DOCUMENT_TEMPLATE_NAME);
260 		}
261 		return xmlData;
262 	}

264 	/***
265 	 * Adds the citations.
266 	 *
267 	 * @param elementsList the elements list
268 	 * @param citationDetails the citation details
269 	 */
270 	private void addCitations(List <String> elementsList,
271 			List <CitationDetail> citationDetails, int startIndex, int endIndex) {
272 		for (int index = startIndex; index < citationDetails.size()
273 				&& index < endIndex; ++index) {
274 			CitationDetail citationDetail = citationDetails.get(index);
275 			addToElementsList(elementsList, citationDetail.getCourtCode()
276 					.getCode(), COURT_ELEMENTS[index]);
277 			addToElementsList(elementsList, citationDetail.getDocketNumber(),
278 					DOCKET_NUMBER_ELEMENTS[index]);
279 			addToElementsList(elementsList, convertDateToString(citationDetail
280 					.getViolationDate(), FORMATTER_MM_DD_YY),
281 					VIOATION_DATE_ELEMENTS[index]);
282 			DecimalFormat format = new DecimalFormat("0000");
283 			addToElementsList(elementsList, format.format(citationDetail
284 					.getFineAmount()), FINE_AMOUNT_ELEMENTS[index]);
285 		}
286 	}

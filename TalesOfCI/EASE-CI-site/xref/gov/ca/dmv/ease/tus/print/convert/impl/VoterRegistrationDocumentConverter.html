46  	 * Populate outgoing XML message for Voter Registration document.
47  	 * 
48  	 * @param requestFromClient the request from client
49  	 * 
50  	 * @return the xml string
51  	 */
52  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
53  		if (requestFromClient == null) {
54  			throw new MessageConversionException(
55  					IPrintServiceErrorMessageConstants.NULL_VOTER_REG_REQUEST);
56  		}
57  		VoterRegistrationReceiptRequestTus request = (VoterRegistrationReceiptRequestTus) requestFromClient;
58  		IUserContext userContext = request.getUserContext();
59  		//DlApplication dlApplication = request.getDlApplication();
60  		Person person = request.getPerson();
61  		Address currentResidenceAddress = person
62  				.getAddress(AddressType.RESIDENTIAL);
63  		Address currentMailingAddress = person.getAddress(AddressType.MAILING);
64  		Address existingResidenceAddress = person
65  				.getExistingAddress(AddressType.RESIDENTIAL);
66  		Address existingMailingAddress = person
67  				.getExistingAddress(AddressType.MAILING);
68  		String voterRegInd = request.getVoterRegInd();
69  		List <String> elementsList = new ArrayList <String>();
70  		//add header elements
71  		elementsList.addAll(addHeaderElements(userContext, person));
72  		//add name elements
73  		elementsList.addAll(addNameElements(userContext, person));
74  		// add current address
75  		elementsList.addAll(addCurrentAddressElements(currentResidenceAddress,
76  				currentMailingAddress));
77  		// add prior address
78  		elementsList.addAll(addPriorAddressElements(currentResidenceAddress,
79  				currentMailingAddress, existingResidenceAddress,
80  				existingMailingAddress));
81  		//add birth date element
82  		elementsList.addAll(addBirthDateElement(person));
83  		//add voter registration elements
84  		elementsList.addAll(addVoterRegIndicatorElements(voterRegInd));
85  		//add footer elements
86  		elementsList.addAll(addFooterElements(userContext));
87  		//add document name element
88  		String xmlData = createXmlForSingleDocument(elementsList,
89  				IPrintConverterConstants.VOTER_REG_DOCUMENT_NAME);
90  		return xmlData;
91  	}

117 		List <String> elementsList = new ArrayList <String>();
118 		if (person.getBirthDate() != null) {
119 			addToElementsList(elementsList, convertDateToString(person

135 		addToElementsList(elementsList, userContext.getOfficeId(),
136 				IPrintConverterConstants.VOTER_REG_ZBPOFFID);

139 				IPrintConverterConstants.VOTER_REG_ZBPCSXDT);

145 	/***
146 	 * Adds the voter reg indicator elements.
147 	 * 
148 	 * @param voterRegInd the voter reg ind
149 	 * 
150 	 * @return the list< string>

152 	private List <String> addVoterRegIndicatorElements(String voterRegInd) {
153 		List <String> elementsList = new ArrayList <String>();
154 		addToElementsList(elementsList, voterRegInd,
155 				IPrintConverterConstants.VOTER_REG_ZBPVOTER);
156 		if (IPrintConverterConstants.VOTER_REG_IND_Y
157 				.equalsIgnoreCase(voterRegInd)) {
158 			addToElementsList(
159 					elementsList,
160 					IPrintConverterConstants.VOTER_REG_ZBPVCTXT_MSG_IF_VOTER_REG_IND_Y,
161 					IPrintConverterConstants.VOTER_REG_ZBPVCTXT);
162 		}
163 		else if (IPrintConverterConstants.VOTER_REG_IND_C
164 				.equalsIgnoreCase(voterRegInd)) {
165 			addToElementsList(
166 					elementsList,
167 					IPrintConverterConstants.VOTER_REG_ZBPVCTXT_MSG_IF_VOTER_REG_IND_C,
168 					IPrintConverterConstants.VOTER_REG_ZBPVCTXT);
169 		}
170 		else if (IPrintConverterConstants.VOTER_REG_IND_S
171 				.equalsIgnoreCase(voterRegInd)) {
172 			addToElementsList(
173 					elementsList,
174 					IPrintConverterConstants.VOTER_REG_ZBPVCTXT_MSG_IF_VOTER_REG_IND_S,
175 					IPrintConverterConstants.VOTER_REG_ZBPVCTXT);
176 		}
177 		return elementsList;
178 	}

195 						getFormattedCurrentName(person))) {

197 					IPrintConverterConstants.VOTER_REG_ZBPRNAMR);

204 	 * 
205 	 * @param currentResidenceAddress the current residence address
206 	 * @param currentMailingAddress the current mailing address
207 	 * 
208 	 * @return the list< string>

210 	private List <String> addCurrentAddressElements(
211 			Address currentResidenceAddress, Address currentMailingAddress) {
212 		List <String> elementsList = new ArrayList <String>();
213 		if (!EaseUtil
214 				.isNullOrBlank(getFormattedAddress(currentResidenceAddress))) {
215 			addToElementsList(elementsList,
216 					getFormattedAddress(currentResidenceAddress),
217 					IPrintConverterConstants.VOTER_REG_ZBPPMADR);
218 			addToElementsList(elementsList, currentResidenceAddress.getCity(),
219 					IPrintConverterConstants.VOTER_REG_ZBPCMCTY);
220 			addToElementsList(elementsList, currentResidenceAddress.getState(),
221 					IPrintConverterConstants.VOTER_REG_ZBPCMSTA);
222 			addToElementsList(elementsList, currentResidenceAddress.getZip(),
223 					IPrintConverterConstants.VOTER_REG_ZBPCMZIP);
224 		}
225 		else {
226 			addToElementsList(elementsList,
227 					getFormattedAddress(currentMailingAddress),
228 					IPrintConverterConstants.VOTER_REG_ZBPPMADR);
229 			addToElementsList(elementsList, currentMailingAddress.getCity(),
230 					IPrintConverterConstants.VOTER_REG_ZBPCMCTY);
231 			addToElementsList(elementsList, currentMailingAddress.getState(),
232 					IPrintConverterConstants.VOTER_REG_ZBPCMSTA);
233 			if (!IPrintConverterConstants.VOTER_REG_ZIP_99999
234 					.equals(currentResidenceAddress.getZip())) {
235 				addToElementsList(elementsList, currentResidenceAddress
236 						.getZip(), IPrintConverterConstants.VOTER_REG_ZBPCMZIP);
237 			}
238 		}
239 		return elementsList;
240 	}

242 	/***
243 	 * Adds the prior address elements.
244 	 * 
245 	 * @param currentResidenceAddress the current residence address
246 	 * @param currentMailingAddress the current mailing address
247 	 * @param existingResidenceAddress the existing residence address
248 	 * @param existingMailingAddress the existing mailing address
249 	 * 
250 	 * @return the list< string>
251 	 */
252 	private List <String> addPriorAddressElements(
253 			Address currentResidenceAddress, Address currentMailingAddress,
254 			Address existingResidenceAddress, Address existingMailingAddress) {
255 		List <String> elementsList = new ArrayList <String>();
256 		if (!EaseUtil
257 				.isNullOrBlank(getFormattedAddress(existingResidenceAddress))) {
258 			if (compareAddresses(existingResidenceAddress,
259 					currentResidenceAddress)) {
260 				addToElementsList(elementsList,
261 						getFormattedAddress(existingResidenceAddress),
262 						IPrintConverterConstants.VOTER_REG_ZBPRMADR);
263 				addToElementsList(elementsList, existingResidenceAddress
264 						.getCity(), IPrintConverterConstants.VOTER_REG_ZBPRMCTY);
265 				addToElementsList(elementsList, existingResidenceAddress
266 						.getState(),
267 						IPrintConverterConstants.VOTER_REG_ZBPRMSTA);
268 				addToElementsList(elementsList, existingResidenceAddress
269 						.getZip(), IPrintConverterConstants.VOTER_REG_ZBPRMZIP);
270 			}
271 		}
272 		else {
273 			if (!EaseUtil
274 					.isNullOrBlank(getFormattedAddress(existingMailingAddress))) {
275 				if (compareAddresses(existingMailingAddress,
276 						currentMailingAddress)) {
277 					addToElementsList(elementsList,
278 							getFormattedAddress(existingMailingAddress),
279 							IPrintConverterConstants.VOTER_REG_ZBPRMADR);
280 					addToElementsList(elementsList, existingMailingAddress
281 							.getCity(),
282 							IPrintConverterConstants.VOTER_REG_ZBPRMCTY);
283 					addToElementsList(elementsList, existingMailingAddress
284 							.getState(),
285 							IPrintConverterConstants.VOTER_REG_ZBPRMSTA);
286 					if (!IPrintConverterConstants.VOTER_REG_ZIP_99999
287 							.equalsIgnoreCase(existingMailingAddress.getZip())) {
288 						addToElementsList(elementsList, existingMailingAddress
289 								.getZip(),
290 								IPrintConverterConstants.VOTER_REG_ZBPRMZIP);
291 					}
292 				}
293 			}
294 		}
295 		return elementsList;
296 	}

257 			sr22Not1 = true;

264 				.getDiscretionarySuspensionOrRevocation();

267 				.getUnableToCalculateEligibilityCode();

270 			if (mandatoryNonProofActionCode != null) {
271 				if (discretionarySuspensionOrRevocation != null) {
272 					customerMessages.append("01,");
273 				}
274 				if (frPenaltyFeeDueUntilDate != null
275 						&& (unableToCalculateEligibilityCode == null || unableToCalculateEligibilityCode
276 								.getCode().length() == 0)) {
277 					customerMessages.append("02,");
278 					if (dlmRecord.getProofEligibilityDate() != null) {
279 						addToElementsList(doc1ElementsList,
280 								convertDateToString(dlmRecord

285 				}

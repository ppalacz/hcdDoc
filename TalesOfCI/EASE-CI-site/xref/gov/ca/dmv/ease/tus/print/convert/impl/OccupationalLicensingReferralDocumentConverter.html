81  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
82  		validateClientRequest(requestFromClient);
83  		// XML Element Names
84  		// Setting xml elements
85  		OccupationalLicensingReferralRequestTus request = (OccupationalLicensingReferralRequestTus) requestFromClient;
86  		IUserContext userContext = request.getUserContext();
87  		SalesPersonApplication salesPersonApplication = request
88  				.getSalesPersonApplication();
89  		VehicleSalesPerson vehicleSalesPerson = salesPersonApplication
90  				.getApplicant();
91  		SalesPersonLicense salesPersonLicense = vehicleSalesPerson
92  				.getSalesPersonLicense();
93  		Address vehicleSalesPersonMailingAddress = vehicleSalesPerson
94  				.getAddress(AddressType.MAILING);
95  		CustomerOrganization vehicleSalesOrganization = salesPersonLicense
96  				.getVehicleSalesOrganization();
97  		List <String> elementsList = new ArrayList <String>();
98  		addToElementsList(elementsList, salesPersonApplication
99  				.getApplicationTypeText(), getTypeApplicationTextElement());
100 		addToElementsList(elementsList, getDlorIdNumber(vehicleSalesPerson),
101 				getDlIdNumberElement());
102 		addToElementsList(elementsList, getLicenseNumber(salesPersonLicense),
103 				getApplicantLicenseNumberElement()); // Assumption: Applicant's license number should be formatted based on some condition given in requirement document. 
104 		//Assuming that this condition will be handled by some other layer.
105 		addToElementsList(elementsList, DocumentConverterUtil
106 				.getFormattedCurrentName(vehicleSalesPerson),
107 				getPhotoNameElement());
108 		addToElementsList(elementsList,
109 				getApplicantMailingAddress(vehicleSalesPersonMailingAddress),
110 				getPhotoAddressElement());
111 		addToElementsList(elementsList,
112 				getApplicantMailingCity(vehicleSalesPersonMailingAddress),
113 				getApplicantMailingCityElement());
114 		addToElementsList(elementsList,
115 				getApplicantMailingState(vehicleSalesPersonMailingAddress),
116 				getApplicantMailingStateElement());
117 		addToElementsList(elementsList,
118 				getApplicantMailingZip(vehicleSalesPersonMailingAddress),
119 				getApplicantMailingZipElement());
120 		addToElementsList(elementsList, getBirthDate(vehicleSalesPerson,
121 				FORMATTER_MM_DD_CCYY), getBirthDateElement());
122 		addToElementsList(elementsList, getPhoneAreaCode(vehicleSalesPerson),
123 				getPhoneAreaCodeElement());
124 		addToElementsList(elementsList,
125 				getPhoneNumberPrefix(vehicleSalesPerson),
126 				getPhonePrefixElement());
127 		addToElementsList(elementsList,
128 				getPhoneNumberSuffix(vehicleSalesPerson),
129 				getPhoneSuffixElement());
130 		addToElementsList(elementsList,
131 				getEmployerLicenseNumber(vehicleSalesOrganization),
132 				getEmployerLicNumberElement());
133 		addToElementsList(elementsList,
134 				getOriginalAppTtc(salesPersonApplication),
135 				getOriginalAppTtcElement());
136 		addToElementsList(elementsList,
137 				getOriginalAppOfficeId(salesPersonApplication),
138 				getOriginalAppOfficeIdElement());
139 		addToElementsList(elementsList, getOriginalApplicationDate(
140 				salesPersonApplication, FORMATTER_MM_dd_yy),
141 				getOriginalAppDateElement());
142 		addToElementsList(elementsList,

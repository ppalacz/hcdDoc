80  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
81  		VerbalNoticeOfActionDocumentRequestTus request = (VerbalNoticeOfActionDocumentRequestTus) requestFromClient;
82  		IUserContext userContext = request.getUserContext();
83  		ServiceOfOrderApplication serviceOfOrderApplication = request
84  				.getServiceOfOrderApplication();
85  		Person person = serviceOfOrderApplication.getApplicant();
86  		List <ServiceOrder> seviceOrders = serviceOfOrderApplication
87  				.getServiceOrders();
88  		List <String> elementsList = new ArrayList <String>();
89  		addToElementsList(elementsList, DocumentConverterUtil
90  				.getFormattedCurrentName(person), FULL_NAME_ELEMENT);
91  		addToElementsList(elementsList, convertDateToString(person
92  				.getBirthDate(), FORMATTER_MM_dd_yy), DATE_OF_BIRTH_ELEMENT);
93  		addToElementsList(elementsList, DocumentConverterUtil
94  				.getDlorIdNumber(person), DL_ID_NUMBER_ELEMENT);
95  		addLicenseLocation(elementsList, userContext,
96  				serviceOfOrderApplication, person);
97  		addServiceOrders(elementsList, seviceOrders);
98  		Date currentDate = new Date();
99  		addToElementsList(elementsList, convertDateToString(currentDate,
100 				FORMATTER_MM_dd_yy), SYSTEM_DATE_ELEMENT);
101 		addToElementsList(elementsList, convertDateToString(currentDate,
102 				FORMATTER_HH_mm), SYSTEM_TIME_ELEMENT);
103 		addToElementsList(elementsList, userContext.getTechId(),
104 				KEYERS_ID_ELEMENT);
105 		addToElementsList(elementsList, DocumentConverterUtil
106 				.getOfficeNameFromId(userContext.getOfficeId()),
107 				OFFICE_NAME_ELEMENT);
108 		String xmlData = DocumentConverterUtil.createXmlForSingleDocument(
109 				elementsList, DOCUMENT_TEMPLATE_NAME);
110 		return xmlData;
111 	}

140 	/***
141 	 * Adds the service orders.
142 	 *
143 	 * @param elementsList the elements list
144 	 * @param seviceOrders the sevice orders
145 	 */
146 	private void addServiceOrders(List <String> elementsList,
147 			List <ServiceOrder> seviceOrders) {
148 		for (int index = 0; index < seviceOrders.size(); ++index) {
149 			ServiceOrder serviceOrder = seviceOrders.get(index);
150 			String action = serviceOrder.getAction();
151 			if (action != null && action.length() > 0) {
152 				addToElementsList(elementsList, action, ACTION_ELEMENT
153 						+ (index + 1));
154 				addSections(elementsList, serviceOrder, (index + 1));
155 				addToElementsList(elementsList, convertDateToString(
156 						(serviceOrder.getEffectiveDate()), FORMATTER_MM_dd_yy),
157 						EFFECTIVE_DATE_ELEMENT + (index + 1));
158 				addToElementsList(elementsList, convertDateToString(
159 						(serviceOrder.getServiceDate()), FORMATTER_MM_dd_yy),
160 						SERVICE_DATE_ELEMENT + (index + 1));
161 			}
162 		}
163 	}

166 	 * Adds the sections.
167 	 *
168 	 * @param elementsList the elements list
169 	 * @param serviceOrder the service order
170 	 * @param countOrders the count orders
171 	 */
172 	private void addSections(List <String> elementsList,
173 			ServiceOrder serviceOrder, int countOrders) {
174 		for (int index = 0; index < serviceOrder.getVehicleCodeSectionCodes()
175 				.size(); ++index) {
176 			String code = serviceOrder.getVehicleCodeSectionCodes().get(index)
177 					.getCode();
178 			if (code != null && code.length() > 0) {
179 				addToElementsList(elementsList, code, CODE_SECTION_ELEMENT
180 						+ countOrders + (index + 1));
181 			}
182 		}
183 	}

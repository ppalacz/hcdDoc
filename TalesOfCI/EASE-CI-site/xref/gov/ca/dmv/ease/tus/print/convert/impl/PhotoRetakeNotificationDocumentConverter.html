71  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
72  		PhotoRetakeNotificationDocumentRequestTus request = (PhotoRetakeNotificationDocumentRequestTus) requestFromClient;
73  		IUserContext userContext = request.getUserContext();
74  		DriverRecordUpdateApplication driverRecordApplication = request
75  				.getDriverRecordApplication();
76  		Person person = driverRecordApplication.getApplicant();
77  		Address currentMailingAddress = person.getAddress(AddressType.MAILING);
78  		List <String> elementsList = new ArrayList <String>();
79  		List <CodeSetElement> licenseClasses = person.getDriverLicense()
80  				.getExistingLicenseClasses();
81  		String previousProcessHistoryTtc = "";
82  		IProcessHistory previousProcessHistory = driverRecordApplication
83  				.getPreviousProcessHistory();
84  		if (previousProcessHistory != null) {
85  			previousProcessHistoryTtc = previousProcessHistory.getUserContext()
86  					.getTtc();
87  		}
88  		boolean isDriverLicense = isIdentificationCardApplicationTtc(previousProcessHistoryTtc);
89  		String processedText = IDENTIFICATION_CARD_TEXT;
90  		String classText = "";
91  		String class1 = "";
92  		String class2 = "";
93  		if (isDriverLicense) {
94  			processedText = DRIVER_LICENSE_TEXT;
95  			classText = IPrintConverterConstants.DL_REC_ZBPCLASS_MSG;
96  			class1 = convertValueOrBlankForNull(licenseClasses, 0);
97  			class2 = convertValueOrBlankForNull(licenseClasses, 1);
98  		}
99  		addToElementsList(elementsList, processedText, CAN_BE_PROCESSED_ELEMENT);
100 		addToElementsList(elementsList, getDlorIdNumber(person),
101 				DL_ID_NUMBER_ELEMENT);
102 		String aClass = convertValueOrBlankForNull(licenseClasses, 0);
103 		if (aClass != null && aClass.length() > 0) {
104 			addToElementsList(elementsList, aClass, DL_REC_ZBPCLASS_ELEMENT);
105 		}
106 		if (class1 != null && class1.length() > 0) {
107 			addToElementsList(elementsList, class1, DL_REC_ZBPCLASS1_ELEMENT);
108 		}
109 		if (class2 != null && class2.length() > 0) {
110 			addToElementsList(elementsList, class2, DL_REC_ZBPCLASS2_ELEMENT);
111 		}
112 		addToElementsList(elementsList, DocumentConverterUtil
113 				.getReverseFormattedName(person.getCurrentName()),
114 				NAME_REVERSED_ELEMENT);
115 		addToElementsList(elementsList,
116 				getFormattedAddress(currentMailingAddress),
117 				PHOTO_ADDRESS_ELEMENT);
118 		addToElementsList(elementsList, currentMailingAddress.getCity(),
119 				MAILING_CITY_ELEMENT);
120 		DocumentConverterUtil.addToElementsList(elementsList,
121 				convertValueOrBlankForNull(userContext.getOfficeId()),
122 				OFFICE_ID_ELEMENT);
123 		DocumentConverterUtil.addToElementsList(elementsList, userContext
124 				.getTechId(), KEYERS_ID_ELEMENT);
125 		DocumentConverterUtil.addToElementsList(elementsList,
126 				convertDateToString(driverRecordApplication
127 						.getDateFeePaidOverride(), //TODO:  Is this the correct date
128 						FORMATTER_MM_SLASH_DD_SLASH_YY), FEE_PAID_DATE_ELEMENT);
129 		String xmlData = DocumentConverterUtil.createXmlForSingleDocument(
130 				elementsList, DOCUMENT_TEMPLATE_NAME);
131 		return xmlData;
132 	}

134 	/***
135 	 * Convert value or blank for null.
136 	 *
137 	 * @param value the value
138 	 * @return the string
139 	 */
140 	private String convertValueOrBlankForNull(String value) {
141 		if (value == null) {
142 			return "";
143 		}
144 		return value;
145 	}
146 
147 	private String convertValueOrBlankForNull(List <CodeSetElement> elements,
148 			int index) {
149 		if (elements == null) {
150 			return "";
151 		}
152 		if (index > elements.size()) {
153 			return "";
154 		}
155 		CodeSetElement element = elements.get(index);
156 		if (element == null) {
157 			return "";
158 		}
159 		String value = element.getCode();
160 		if (value == null) {
161 			return "";
162 		}
163 		return value;
164 	}

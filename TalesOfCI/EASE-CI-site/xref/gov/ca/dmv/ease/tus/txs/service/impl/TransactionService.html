74  	 * @see gov.ca.dmv.ease.tus.txs.service.impl.ITransactionService#execute(gov.ca.dmv.ease.tus.txs.request.TransactionRequest)

76  	public TransactionResponse execute(AbstractTransactionRequest request) {
77  		try {
78  			TransactionResponse response = new TransactionResponse();
79  			ITransactionForTrxService aTransaction = getTransactionObject(request);
80  			IPersistenceServiceRequest saveBusinessObjectRequest = persistenceServiceRequestFactory
81  					.createSaveOrUpdateBusinessObjectRequest(request
82  							.getUserContext(), aTransaction);
83  			PersistenceServiceResponse aPersistenceResponse = saveBusinessObjectRequest
84  					.execute();
85  			if (aPersistenceResponse.getErrorCollector().hasErrors()) {
86  				response.setErrorCollector(aPersistenceResponse
87  						.getErrorCollector());
88  			}
89  			else {
90  				response.setTransaction(getTransaction(request));
91  			}
92  			return response;
93  		}
94  		catch (Exception ex) {
95  			return new TransactionResponse(ex);
96  		}
97  	}

99  	/***
100 	 * This method for reading/getting the persisted transaction for downstream
101 	 * functions like sending to DMVA using the bridge code.
102 	 * 
103 	 * @param transactionRequest the transaction request
104 	 * 
105 	 * @return the transaction
106 	 */
107 	private Transaction getTransaction(
108 			AbstractTransactionRequest transactionRequest) {
109 		DlProcessContext dlContext = (DlProcessContext) transactionRequest
110 				.getContext();
111 		Transaction aTransaction = new Transaction();
112 		//For DLA we have the Transaction Identifier in the context
113 		if (dlContext.getDafRecordNumber() > 0) {
114 			aTransaction.setTransactionIdentifier(dlContext
115 					.getDafRecordNumber());
116 		}
117 		//For CDA and DAF we get it from the existing Transaction object
118 		else {
119 			aTransaction
120 					.setTransactionIdentifier(((DlProcessTransactionRequest) transactionRequest)
121 							.getTransaction().getTransactionIdentifier());
122 		}
123 		IUserContext userContext = dlContext.getUserContext();
124 		aTransaction.setOfficeId(userContext.getOfficeId());
125 		aTransaction.setEmployeeId(userContext.getTechId());
126 		String licenseName = null;
127 		String licenseNumber = null;
128 		if (dlContext instanceof SpProcessContext) {
129 			SpProcessContext spProcessContext = (SpProcessContext) dlContext;
130 			VehicleSalesPerson applicant = spProcessContext
131 					.getSalesPersonApplication().getApplicant();
132 			SalesPersonLicense salesPersonLicense = applicant
133 					.getSalesPersonLicense();
134 			licenseName = applicant.getCurrentName().getFullName();
135 			licenseNumber = salesPersonLicense.getLicenseNumber();
136 		}
137 		else {
138 			licenseName = dlContext.getDlApplicant().getCurrentName()
139 					.getFullName();
140 			licenseNumber = dlContext.getDriverLicense().getLicenseNumber();
141 		}
142 		aTransaction.setLicenseName(licenseName);
143 		aTransaction.setLicenseNumber(licenseNumber);
144 		RetrieveBusinessObjectRequest retrieveBusinessObjectRequest = (RetrieveBusinessObjectRequest) persistenceServiceRequestFactory
145 				.createRetrieveBusinessObjectRequest(
146 						dlContext.getUserContext(), aTransaction);
147 		RetrieveBusinessObjectResponse retrieveBusinessObjectResponse = retrieveBusinessObjectRequest
148 				.execute();
149 		if (retrieveBusinessObjectResponse.getErrorCollector().hasErrors()) {
150 			return aTransaction;
151 		}
152 		return (Transaction) retrieveBusinessObjectResponse.getResults().get(0);
153 	}

192 	private Transaction getTransactionObject(
193 			AbstractTransactionRequest transactionRequest) {
194 		Transaction transaction = ((DlProcessTransactionRequest) transactionRequest)
195 				.getTransaction();
196 		ProcessContext processContext = transactionRequest.getContext();
197 		IUserContext userContext = processContext.getUserContext();
198 		transaction.setOfficeId(userContext.getOfficeId());
199 		transaction.setEmployeeId(userContext.getTechId());
200 		DlProcessContext dlContext = (DlProcessContext) transactionRequest
201 				.getContext();
202 		transaction.setTypeTransactionCode(dlContext.getUserContext().getTtc());
203 		transaction.setTransactionIdentifier(dlContext.getDafRecordNumber());
204 		if (dlContext.isCNACondition()) {
205 			transaction.setCnaStatus("Y");
206 		}
207 		else {
208 			transaction.setCnaStatus("N");
209 		}
210 		if (dlContext instanceof SpProcessContext) {
211 			SpProcessContext spProcessContext = (SpProcessContext) dlContext;
212 			VehicleSalesPerson applicant = spProcessContext
213 					.getSalesPersonApplication().getApplicant();
214 			SalesPersonLicense salesPersonLicense = applicant
215 					.getSalesPersonLicense();
216 			transaction
217 					.setLicenseName(applicant.getCurrentName().getFullName());
218 			transaction.setLicenseNumber(salesPersonLicense.getLicenseNumber());
219 			transaction.setLicenseType(LicenseType.SALES_PERSON_LICENSE);
220 			transaction.setBirthDate(applicant.getBirthDate());
221 			assignBlob(spProcessContext.getSalesPersonApplication(),
222 					transaction);
223 		}
224 		else if (dlContext instanceof DlaProcessContext) {
225 			transaction.setLicenseName(dlContext.getDlApplicant()
226 					.getCurrentName().getFullName());
227 			transaction.setLicenseNumber(dlContext.getDriverLicense()
228 					.getLicenseNumber());
229 			transaction.setLicenseType(LicenseType.DRIVER_LICENSE);
230 			transaction.setBirthDate(dlContext.getDlApplicant().getBirthDate());
231 			assignBlob(dlContext.getDlApplication(), transaction);
232 		}
233 		else if (dlContext instanceof IdProcessContext) {
234 			transaction.setLicenseName(dlContext.getDlApplicant()
235 					.getCurrentName().getFullName());
236 			transaction.setLicenseNumber(dlContext.getDlApplicant().getIdCard()
237 					.getIdNumber());
238 			transaction.setLicenseType(LicenseType.ID_CARD);
239 			transaction.setBirthDate(dlContext.getDlApplicant().getBirthDate());
240 			assignBlob(dlContext.getDlApplication(), transaction);
241 		}
242 		validateDlTransaction(transaction);
243 		return (transaction);
244 	}

247 	 * This method is for getting max DAF number and increment that by one
248 	 * for any given office.
249 	 * 
250 	 * @param processContext the process context
251 	 * @param userContext the user context
252 	 * 
253 	 * @return the daf number
254 	 */
255 	private Integer getDafNumber(ProcessContext processContext,
256 			IUserContext userContext) {
257 		Integer transactionIdentifier = new Integer("001");
258 		//Getting the Max DAF number in the Database
259 		IPersistenceServiceRequest request = persistenceServiceRequestFactory
260 				.createRetrieveCriteriaObjectRequest(processContext
261 						.getUserContext(), Transaction.class);
262 		RetrieveCriteriaObjectResponse response = (RetrieveCriteriaObjectResponse) request
263 				.execute();
264 		if (!response.hasErrors()
265 				&& response.getCriteriaInstance().list().size() > 0) {
266 			Criteria criteria = response.getCriteriaInstance();
267 			criteria.add(Restrictions.like("officeId", userContext
268 					.getOfficeId()));
269 			ProjectionList projList = Projections.projectionList();
270 			projList.add(Projections.max("transactionIdentifier"));
271 			criteria.setProjection(projList);
272 			List listOfBusinessObjects = criteria.list();
273 			for (Object obj : listOfBusinessObjects) {
274 				transactionIdentifier = (Integer) obj;
275 			}
276 		}
277 		if (transactionIdentifier != null)
278 			return transactionIdentifier + 001;
279 		else
280 			return transactionIdentifier;
281 	}

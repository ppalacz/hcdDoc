40  	public String cancelPrintJob(String jobClientId) {
41  		if (LOGGER.isDebugEnabled()) {
42  			LOGGER.debug("cancelPrintJob(String jobClientId=" + jobClientId
43  					+ ") - start");
44  		}
45  		//Build the HP Output Server Query XML
46  		String cancelPrintJobContract = HpOsQueryXMLBuilder
47  				.createCancelPrintJobContract(jobClientId);
48  		//Invoke HPOS Web Service Proxy and get Response
49  		String cancelPrintJobResponse = printHpOsWebService
50  				.cancelPrintJob(cancelPrintJobContract);
51  		//Return the Job Status
52  		String returnString = HpOsPrintUtility.queryPrintResponseByKey(
53  				HpOsPrintUtility.STATUS_CODE, cancelPrintJobResponse);
54  		if (LOGGER.isDebugEnabled()) {
55  			LOGGER.debug("cancelPrintJob(String) - end");
56  		}
57  		return returnString;
58  	}

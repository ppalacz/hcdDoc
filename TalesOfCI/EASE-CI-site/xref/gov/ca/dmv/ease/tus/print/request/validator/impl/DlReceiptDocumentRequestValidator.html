45  	/***
46  	 * This method validates {@link DlReceiptDocumentRequestTus}.
47  	 * 
48  	 * @param request The DlReceiptDocumentRequestTus to validate
49  	 * @param aCollector the a collector
50  	 * 
51  	 * @return <code>IErrorCollector</code> containing Validation Messages.
52  	 * 
53  	 * @see IErrorCollector
54  	 * @see DlReceiptDocumentRequestTus
55  	 */
56  	@Override
57  	public void validateRequest(IPrintServiceRequest request,
58  			IErrorCollector aCollector) {
59  		if (!(request instanceof DlReceiptDocumentRequestTus)) {
60  			addValidationExceptionInErrorCollector("IMPROPER_REQUEST_TYPE",
61  					new String[] { "DlReceiptDocumentRequestTus" }, aCollector);
62  		}
63  		else {
64  			DlReceiptDocumentRequestTus dlReceiptDocumentRequest = (DlReceiptDocumentRequestTus) request;
65  			Application application = dlReceiptDocumentRequest.getApplication();
66  			if (application == null) {
67  				addValidationExceptionInErrorCollector("NULL_APPLICATION",
68  						null, aCollector);
69  			}
70  			else {
71  				Person person = null;
72  				if (application instanceof DlApplication) {
73  					DlApplication dlApplication = (DlApplication) application;
74  					person = dlApplication.getApplicant();
75  				}
76  				else if (application instanceof SalesPersonApplication) {
77  					SalesPersonApplication salesPersonApplication = (SalesPersonApplication) application;
78  					person = salesPersonApplication.getApplicant();
79  				}
80  				validatePerson(person, aCollector);
81  			}
82  		}
83  	}

88  	@Override
89  	protected void validatePerson(Person person, IErrorCollector aCollector) {
90  		if (person == null) {
91  			addValidationExceptionInErrorCollector("NULL_PERSON", null,
92  					aCollector);
93  		}
94  		else {
95  			validatePersonBirthDate(person, aCollector);
96  			validateLicense(person, aCollector);
97  			validateMailingAddress(person, aCollector);
98  			validatePersonName(person.getCurrentName(), aCollector);
99  		}
100 	}

102 	/***
103 	 * Validate mailing address.
104 	 * 
105 	 * @param person the person
106 	 * @param aCollector the a collector
107 	 */
108 	private void validateMailingAddress(Person person,
109 			IErrorCollector aCollector) {
110 		Address mailingAddress = person.getAddress(AddressType.MAILING);
111 		if (isNullOrBlank(mailingAddress)) {
112 			addValidationExceptionInErrorCollector("NULL_MAILING_ADDRESS",
113 					null, aCollector);
114 		}
115 		else {
116 			if (isNullOrBlank(mailingAddress.getLine1())
117 					&& isNullOrBlank(mailingAddress.getLine2())) {
118 				addValidationExceptionInErrorCollector("EMPTY_MAILING_ADDRESS",
119 						null, aCollector);
120 			}
121 			if (isNullOrBlank(mailingAddress.getCity())) {
122 				addValidationExceptionInErrorCollector(
123 						"NULL_OR_EMPTY_MAILING_CITY", null, aCollector);
124 			}
125 			if (isNullOrBlank(mailingAddress.getState())) {
126 				addValidationExceptionInErrorCollector(
127 						"NULL_OR_EMPTY_MAILING_STATE", null, aCollector);
128 			}
129 			if (isNullOrBlank(mailingAddress.getZip())) {
130 				addValidationExceptionInErrorCollector(
131 						"NULL_OR_EMPTY_MAILING_ZIP", null, aCollector);
132 			}
133 		}
134 	}

136 	/***
137 	 * Validate Person Current Name.
138 	 * 
139 	 * @param currentName the current name
140 	 * @param aCollector the Error Collector
141 	 */
142 	private void validatePersonName(PersonName currentName,
143 			IErrorCollector aCollector) {
144 		if (isNullorEmpty(currentName)) {
145 			addValidationExceptionInErrorCollector("NULL_CURRENT_NAME", null,
146 					aCollector);
147 		}
148 		else if (isNullorEmpty(currentName.getFullName())) {
149 			addValidationExceptionInErrorCollector("NULL_CURRENT_NAME", null,
150 					aCollector);
151 		}
152 	}

154 	/***
155 	 * Validate license.
156 	 * 
157 	 * @param aCollector the a collector
158 	 * @param person the person
159 	 */
160 	private void validateLicense(Person person, IErrorCollector aCollector) {
161 		DriverLicense driverLicense = person.getDriverLicense();
162 		IdCard idCard = person.getIdCard();
163 		if (isNullorEmpty(driverLicense) && isNullorEmpty(idCard)) {
164 			addValidationExceptionInErrorCollector(
165 					"NULL_DRIVERLICENSE_AND_IDCARD", null, aCollector);
166 		}
167 		else if (!isNullorEmpty(driverLicense)) {
168 			validateLicenseNumber(driverLicense.getLicenseNumber(), aCollector);
169 		}
170 		else if (!isNullorEmpty(idCard)) {
171 			validateLicenseNumber(idCard.getLicenseNumber(), aCollector);
172 		}
173 	}

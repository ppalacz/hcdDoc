47  	/*** The Constant WORK_DATE_ELEMENT. */

51  	/*** The Constant DOCUMENT_TEMPLATE_NAME. */

54  	/* (non-Javadoc)
55  	 * @see gov.ca.dmv.ease.tus.print.converter.IDocumentConverter#convertToDocumentBinding(gov.ca.dmv.ease.tus.print.request.impl.PrintServiceRequest)
56  	 */
57  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
58  		PhotoRetakeCNADocumentRequestTus request = (PhotoRetakeCNADocumentRequestTus) requestFromClient;
59  		IUserContext userContext = request.getUserContext();
60  		DriverRecordUpdateApplication driverRecordUpdateApplication = request
61  				.getDriverRecordUpdateApplication();
62  		Person person = driverRecordUpdateApplication.getApplicant();
63  		List <String> elementsList = new ArrayList <String>();
64  		addToElementsList(elementsList, getDlorIdNumber(person),
65  				DL_ID_NUMBER_ELEMENT);
66  		addToElementsList(elementsList, convertDateToString(person
67  				.getBirthDate(), FORMATTER_MM_dd_yy), DATE_OF_BIRTH_ELEMENT);
68  		addToElementsList(elementsList, DocumentConverterUtil
69  				.getFormattedCurrentName(person.getCurrentName()),
70  				PHOTO_NAME_ELEMENT);
71  		String dafRecordNumber = getDafRecordNumber(driverRecordUpdateApplication);
72  		if (dafRecordNumber != null && dafRecordNumber.length() > 0) {
73  			addToElementsList(elementsList, dafRecordNumber, RECORD_ELEMENT);
74  		}
75  		addToElementsList(elementsList, userContext.getOfficeId(),
76  				OFFICE_ID_ELEMENT);
77  		addToElementsList(elementsList, convertDateToString(userContext
78  				.getWorkDate(), FORMATTER_MM_DD_YY), WORK_DATE_ELEMENT);
79  		addToElementsList(elementsList, userContext.getTechId(),
80  				KEYERS_ID_ELEMENT);
81  		String xmlData = DocumentConverterUtil.createXmlForSingleDocument(
82  				elementsList, DOCUMENT_TEMPLATE_NAME);
83  		return xmlData;
84  	}
85  }

45  	/***
46  	 * This method validates {@link InstructionPermitDocumentRequestTus}
47  	 * 
48  	 * @param request the InstructionPermitDocumentRequestTus to validate
49  	 * @return <code>IErrorCollector</code> containing Validation Messages.
50  	 * @see IErrorCollector
51  	 * @see InstructionPermitDocumentRequestTus
52  	 */
53  	@Override
54  	public void validateRequest(IPrintServiceRequest request, IErrorCollector collector) {
55  		if (request == null) {
56  			addValidationExceptionInErrorCollector("NULL_REQUEST",
57  					new String[] { "InstructionPermitDocumentRequestTus" },
58  					collector);
59  		}
60  		if (!(request instanceof InstructionPermitDocumentRequestTus)) {
61  			addValidationExceptionInErrorCollector("IMPROPER_REQUEST_TYPE",
62  					new String[] { "InstructionPermitDocumentRequestTus" },
63  					collector);
64  		}
65  		InstructionPermitDocumentRequestTus instructionPermitDocumentRequest = (InstructionPermitDocumentRequestTus) request;
66  		validateUserContext(instructionPermitDocumentRequest.getUserContext(),
67  				collector);
68  		DlApplication dlApplication = instructionPermitDocumentRequest
69  				.getDlApplication();
70  		Office office = instructionPermitDocumentRequest.getOffice();
71  		if (isNullorEmpty(dlApplication)) {
72  			addValidationExceptionInErrorCollector("NULL_DLAPPLICATION",
73  					new String[] { "InstructionPermitDocumentRequestTus" },
74  					collector);
75  		}
76  		else if (isNullorEmpty(office)) {
77  			addValidationExceptionInErrorCollector("NULL_OFFICE",
78  					new String[] { "InstructionPermitDocumentRequestTus" },
79  					collector);
80  		}
81  		else {
82  			Person person = dlApplication.getApplicant();
83  			if (isNullorEmpty(person)) {
84  				addValidationExceptionInErrorCollector(
85  						"PERSON_IS_NULL_ERROR_MESSAGE",
86  						new String[] { "InstructionPermitDocumentRequestTus" },
87  						collector);
88  			}
89  			else {
90  				DriverLicense driverLicense = person.getDriverLicense();
91  				if (isNullorEmpty(driverLicense)) {
92  					addValidationExceptionInErrorCollector(
93  							"NULL_DRIVERLICENSE",
94  							new String[] { "InstructionPermitDocumentRequestTus" },
95  							collector);
96  				}
97  				else {
98  					PhotoInformation photoInformation = person
99  							.getDriverLicense().getPhotoInformation();
100 					if (isNullorEmpty(photoInformation)) {
101 						addValidationExceptionInErrorCollector(
102 								"NULL_PHOTO_INFORMATION",
103 								new String[] { "InstructionPermitDocumentRequestTus" },
104 								collector);
105 					}
106 					validatePhotoInformation(photoInformation, collector);
107 				}
108 				validateDriverLicenseForInstructionPermit(driverLicense,
109 						collector);
110 			}
111 			validatePerson(person, collector);
112 			validatePersonalInformation(person, collector);
113 			validatePersonBirthDate(person, collector);
114 			
115 			if (isNullorEmpty(dlApplication.getOriginalProcessHistory())) {
116 				addValidationExceptionInErrorCollector(
117 						"NULL_ORIGINAL_PROCESS_HISTORY",
118 						new String[] { "InstructionPermitDocumentRequestTus" },
119 						collector);
120 			}
121 		}
122 	}

124 	/***
125 	 * Validates driver license object as required for instruction permit converter.
126 	 * @param driverLicense
127 	 * @param collector
128 	 */
129 	private void validateDriverLicenseForInstructionPermit(
130 			DriverLicense driverLicense, IErrorCollector collector) {
131 		
132 		if (isNullorEmpty(driverLicense.getLicenseNumber())) {
133 			addValidationExceptionInErrorCollector(
134 					"DL_NUMBER_IS_EMPTY_ERROR_MESSAGE",
135 					new String[] { "InstructionPermitDocumentRequestTus" },
136 					collector);			
137 		}
138 		if (isNullorEmpty(driverLicense.getEffectiveDate())) {
139 				addValidationExceptionInErrorCollector(
140 						"NULL_OR_EMPTY_LICENSE_ISSUE_DATE",
141 						new String[] { "InstructionPermitDocumentRequestTus" },
142 						collector);	
143 		}
144 		if (isNullorEmpty(driverLicense.getExpirationDate())) {
145 					addValidationExceptionInErrorCollector(
146 							"NULL_OR_EMPTY_LICENSE_EXPIRATION_DATE",
147 							new String[] { "InstructionPermitDocumentRequestTus" },
148 							collector);	
149 		}
150 		List <InterimPermit> interimPermits = driverLicense
151 							.getInterimPermits(InterimPermitType.INSTRUCTION_PERMIT);
152 		if (isNullorEmpty(interimPermits)) {
153 						addValidationExceptionInErrorCollector(
154 								"NULL_OR_EMPTY_INTERIM_PERMITS",
155 								new String[] { "InstructionPermitDocumentRequestTus" },
156 								collector);
157 		}
158 		InterimPermit interimPermit = interimPermits.get(0);
159 		if (isNullorEmpty(interimPermit)) {
160 							addValidationExceptionInErrorCollector(
161 									"ATLEAST_ONE_INTERIM_PERMIT_MUST_BE_PRESENT",
162 									new String[] { "InstructionPermitDocumentRequestTus" },
163 									collector);
164 		}
165 		validateInterimPermitForInstructionPermit(
166 								interimPermit, collector);
167 	}

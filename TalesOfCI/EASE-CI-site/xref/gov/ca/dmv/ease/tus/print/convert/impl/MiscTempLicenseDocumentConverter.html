62  	/***
63  	 * This method formats the Endorsements.
64  	 * 
65  	 * @param endorsements the Existing Endorsements
66  	 * @param driverLicense the driver license
67  	 * 
68  	 * @return formatted endorsements
69  	 */
70  	public static String convertEndorsements(
71  			List <CodeSetElement> endorsements, DriverLicense driverLicense) {
72  		String formattedEndorsements = AbstractDocumentConverter.BLANK;
73  		if (!EaseUtil.isNullOrBlank(endorsements)) {
74  			for (CodeSetElement endorsement : endorsements) {
75  				if (EaseUtil.isNullOrBlank(endorsement)) {
76  					formattedEndorsements += EcsConverterHelper.HYPEN;
77  				}
78  				else {
79  					formattedEndorsements += replaceEndorsementCode(endorsement
80  							.getCode(), driverLicense);
81  					formattedEndorsements += EcsConverterHelper.HYPEN;
82  				}
83  			}
84  		}
85  		return formattedEndorsements;
86  	}

112 	/***
113 	 * This method converts 2 Byte Commercial Endorsement Code into single Byte.
114 	 * 
115 	 * @param endorsementCode the Endorsement Code
116 	 * @param driverLicense the Driver License
117 	 * 
118 	 * @return the Printable Endorsement Value
119 	 */
120 	private static String replaceEndorsementCode(String endorsementCode,
121 			DriverLicense driverLicense) {
122 		String replaceEndorsementCode = EcsConverterHelper.HYPEN;
123 		CodeSetElement backGroundCheckIndicator = driverLicense
124 				.getBackgroundCheckIndicator();
125 		Date backGroundCheckExpDate = driverLicense
126 				.getBackgroundCheckExpirationDate();
127 		if (getCommercialEndorsementCodes().containsKey(endorsementCode)) {
128 			replaceEndorsementCode = getCommercialEndorsementCodes().get(
129 					endorsementCode);
130 			if (!EaseUtil.isNullOrBlank(backGroundCheckIndicator)) {
131 				if (backGroundCheckIndicator.getCode().equalsIgnoreCase(
132 						BACK_GRND_CHECK_N)) {
133 					if (endorsementCode
134 							.equalsIgnoreCase(COM_ENDORSEMENT_TRAILER)) {
135 						replaceEndorsementCode = CONST_X;
136 					}
137 					else {
138 						replaceEndorsementCode = CONST_H;
139 					}
140 				}
141 				if (backGroundCheckIndicator.getCode().equalsIgnoreCase(
142 						BACK_GRND_CHECK_S)) {
143 					if (endorsementCode
144 							.equalsIgnoreCase(COM_ENDORSEMENT_TRAILER)) {
145 						replaceEndorsementCode = CONST_N;
146 					}
147 					else {
148 						replaceEndorsementCode = AbstractDocumentConverter.BLANK;
149 					}
150 				}
151 				else {
152 					if ((EaseUtil.isNotNull(backGroundCheckExpDate) && backGroundCheckExpDate
153 							.before(new Date()))
154 							|| backGroundCheckIndicator.getCode()
155 									.equalsIgnoreCase(BACK_GRND_CHECK_D)) {
156 						if (endorsementCode
157 								.equalsIgnoreCase(COM_ENDORSEMENT_TRAILER)) {
158 							replaceEndorsementCode = CONST_N;
159 						}
160 						else {
161 							replaceEndorsementCode = AbstractDocumentConverter.BLANK;
162 						}
163 					}
164 				}
165 			}
166 		}
167 		return replaceEndorsementCode;
168 	}

177 	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
178 		validateClientRequest(requestFromClient);
179 		MiscTempLicenseDocumentRequestTus request = (MiscTempLicenseDocumentRequestTus) requestFromClient;
180 		IUserContext userContext = request.getUserContext();
181 		Person person = request.getDlApplicant();
182 		Address mailingAddress = person.getAddress(AddressType.MAILING);
183 		DriverLicense driverLicense = person.getDriverLicense();
184 		String dlNumber = driverLicense.getLicenseNumber();
185 		List <CodeSetElement> licenseClasses = driverLicense
186 				.getExistingLicenseClasses();
187 		List <CodeSetElement> endorsements = driverLicense
188 				.getExistingEndorsements();
189 		List <DrivingRestriction> restrictions = driverLicense
190 				.getExistingRestrictions();
191 		// Setting xml elements
192 		List <String> elementsList = new ArrayList <String>();
193 		if (EcsConverterHelper.isAnyCommercialLicenseClassExist(licenseClasses)) {
194 			addToElementsList(elementsList,
195 					MSC_TEMP_LICENSE_COMMERCIAL_TRAINING_HEADER_MSG,
196 					MSC_TEMP_LICENSE_TEXT);
197 		}
198 		else {
199 			addToElementsList(elementsList,
200 					MSC_TEMP_LICENSE_TRAINING_HEADER_MSG, MSC_TEMP_LICENSE_TEXT);
201 		}
202 		addToElementsList(elementsList, dlNumber, MSC_TEMP_LICENSE_DL_NUMBER);
203 		if (licenseClasses.size() > 0) {
204 			//Add Class 1
205 			addToElementsList(elementsList, licenseClasses.get(0).getCode(),
206 					MSC_TEMP_LICENSE_CLASS_1);
207 			//Ampersand
208 			addToElementsList(elementsList, MSC_TEMP_LICENSE_AMP_TEXT,
209 					MSC_TEMP_LICENSE_AMP);
210 		}
211 		//Add Class 2
212 		if (licenseClasses.size() == 2) {
213 			addToElementsList(elementsList, licenseClasses.get(1).getCode(),
214 					MSC_TEMP_LICENSE_CLASS_2);
215 		}
216 		//Endorsements
217 		String formattedEndorsements = convertEndorsements(endorsements,
218 				driverLicense);
219 		if (!EaseUtil.isNullOrBlank(formattedEndorsements)) {
220 			addToElementsList(elementsList, formattedEndorsements,
221 					MSC_TEMP_LICENSE_ENDORSEMENTS);
222 		}
223 		//TODO ZBPRGBON
224 		addToElementsList(elementsList, IPrintConverterConstants.BLANK,
225 				ZBPRGBON);
226 		//TODO ZBPHZMAT
227 		addToElementsList(elementsList, MSC_TEMP_LICENSE_HAZMAT_TEXT_MSG,
228 				MSC_TEMP_LICENSE_HAZMAT_TEXT);
229 		//TODO ZBPRGBOF
230 		addToElementsList(elementsList, IPrintConverterConstants.BLANK,
231 				ZBPRGBOF);
232 		//Provisional License Check
233 		if (EcsConverterHelper.isApplicantBelow18(person.getBirthDate())) {
234 			addToElementsList(elementsList,
235 					MSC_TEMP_LICENSE_PROVISIONAL_TEXT_MSG,
236 					MSC_TEMP_LICENSE_PROVISIONAL_TEXT);
237 		}
238 		//Issue Date - TODO Need to use Logic
239 		addToElementsList(elementsList, convertDateToString(driverLicense
240 				.getIssuedDate(), FORMATTER_MM_dd_yy),
241 				MSC_TEMP_LICENSE_ISSUE_DATE);
242 		//Office Id
243 		addToElementsList(elementsList, userContext.getOfficeId(),
244 				MSC_TEMP_LICENSE_OFFICE_ID);
245 		//Employee Id
246 		addToElementsList(elementsList, userContext.getTechId(),
247 				MSC_TEMP_LICENSE_EMPLOYEE_ID);
248 		//License Expiry Date
249 		addToElementsList(elementsList, convertDateToString(driverLicense
250 				.getExpirationDate(), FORMATTER_MM_dd_yy),
251 				MSC_TEMP_LICENSE_EXPIRY_DATE);
252 		//Person Name
253 		addToElementsList(elementsList, person.getCurrentName().getFullName(),
254 				MSC_TEMP_LICENSE_NAME);
255 		//Gender
256 		addToElementsList(elementsList, person.getGenderCode().getCode(),
257 				MSC_TEMP_LICENSE_GENDER);
258 		//Hair Color
259 		addToElementsList(elementsList, person.getHairColorCode().getCode(),
260 				MSC_TEMP_LICENSE_HAIR_COLOR);
261 		//Eye Color
262 		addToElementsList(elementsList, person.getEyeColorCode().getCode(),
263 				MSC_TEMP_LICENSE_EYE_COLOR);
264 		//Address
265 		addToElementsList(elementsList, mailingAddress.getLine1()
266 				+ mailingAddress.getLine2(), MSC_TEMP_LICENSE_MAILING_ADDRESS);
267 		//Height in Feet
268 		addToElementsList(elementsList, person.getHeightInFeet().toString(),
269 				MSC_TEMP_LICENSE_HEIGHT_IN_FEET);
270 		//Height in Inches
271 		addToElementsList(elementsList, person.getHeightInInch().toString(),
272 				MSC_TEMP_LICENSE_HEIGHT_IN_INCHES);
273 		//Weight in Pounds
274 		addToElementsList(elementsList, person.getWeight().toString(),
275 				MSC_TEMP_LICENSE_WEIGHT_IN_POUNDS);
276 		//Birth Date
277 		addToElementsList(elementsList, convertDateToString(person
278 				.getBirthDate(), FORMATTER_MM_dd_yy),
279 				MSC_TEMP_LICENSE_BIRTH_DATE);
280 		//Mailing City
281 		addToElementsList(elementsList, mailingAddress.getCity(),
282 				MSC_TEMP_LICENSE_MAILING_CITY);
283 		//Mailing State
284 		addToElementsList(elementsList,
285 				mailingAddress.getStateCode().getCode(),
286 				MSC_TEMP_LICENSE_MAILING_STATE);
287 		//Mailing Zip
288 		addToElementsList(elementsList, mailingAddress.getZip(),
289 				MSC_TEMP_LICENSE_MAILING_ZIP);
290 		//Under 21 Message
291 		if (EaseUtil.isNotNull(DateUtil.isMinorForSpecifiedAgeLimit(person
292 				.getBirthDate(), 18, 0))) {
293 			//Add Logic to Check the Year of 21
294 			addToElementsList(elementsList, MSC_TEMP_LICENSE_AGE_21_TEXT_MSG,
295 					MSC_TEMP_LICENSE_AGE_21_TEXT);
296 		}
297 		//Check Restrictions
298 		if (null != restrictions && !restrictions.isEmpty()) {
299 			//Restriction Check
300 			addToElementsList(elementsList,
301 					MSC_TEMP_LICENSE_RESTICTION_TEXT_MSG,
302 					MSC_TEMP_LICENSE_RESTICTION_TEXT);
303 			//Restriction 1
304 			if (restrictions.size() > 0) {
305 				addToElementsList(elementsList, restrictions.get(0)
306 						.getRestrictionTypeCode().getDescription(),
307 						MSC_TEMP_LICENSE_RESTICTION1_TEXT);
308 			}
309 			//Restriction 2
310 			if (restrictions.size() > 1) {
311 				addToElementsList(elementsList, restrictions.get(1)
312 						.getRestrictionTypeCode().getDescription(),
313 						MSC_TEMP_LICENSE_RESTICTION2_TEXT);
314 			}
315 		}
316 		//Office Id
317 		addToElementsList(elementsList, userContext.getOfficeId(),
318 				MSC_TEMP_LICENSE_OFFICE_ID);
319 		//Issue Date
320 		addToElementsList(elementsList, convertDateToString(driverLicense
321 				.getIssuedDate(), FORMATTER_MM_dd_yy),
322 				MSC_TEMP_LICENSE_ISSUE_DATE);
323 		//Employee Id
324 		addToElementsList(elementsList, userContext.getTechId(),
325 				MSC_TEMP_LICENSE_EMPLOYEE_ID);
326 		//Sequence Number
327 		addToElementsList(elementsList, driverLicense.getSeqNumber(),
328 				MSC_TEMP_LICENSE_SEQ);
329 		String xmlData = createXmlForSingleDocument(elementsList,
330 				MSC_TEMP_LICENSE_DOC_NAME);
331 		return xmlData;
332 	}

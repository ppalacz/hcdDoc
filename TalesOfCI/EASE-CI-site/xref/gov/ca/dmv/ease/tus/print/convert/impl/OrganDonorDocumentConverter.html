59  public class OrganDonorDocumentConverter extends AbstractDocumentConverter {
60  	/***
61  	 * Populates outgoing XML message for Organ Donor Receipt.
62  	 * 
63  	 * @param requestFromClient the request from client
64  	 * 
65  	 * @return the string
66  	 */
67  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
68  		validateClientRequest(requestFromClient);
69  		// XML Element Constant Values
70  		final String URL_ADDRESS1 = "www.donateLIFEcalifornia.org";
71  		final String URL_ADDRESS2 = "www.doneVIDAcalifornia.org";
72  		OrganDonorDocumentRequestTus request = (OrganDonorDocumentRequestTus) requestFromClient;
73  		IUserContext userContext = request.getUserContext();
74  		boolean isNew = request.isNew();
75  		Person person = request.getPerson();
76  		Address mailingAddress = person.getAddress(AddressType.MAILING);
77  		// Setting xml elements
78  		List <String> elementsList = new ArrayList <String>();
79  		addToElementsList(elementsList, getTrainingHeader(userContext),
80  				getTrainingHeaderElement());
81  		addToElementsList(elementsList, getDlorIdNumber(person),
82  				getDlIdNumberElement());
83  		addToElementsList(elementsList, getDonorName(person.getCurrentName()),
84  				getPhotoNameElement());
85  		addToElementsList(elementsList, getDonorAddress(mailingAddress),
86  				getPhotoAddressElement());
87  		addToElementsList(elementsList,
88  				getApplicantMailingCity(mailingAddress),
89  				getApplicantMailingCityElement());
90  		addToElementsList(elementsList, getApplicantMailingState(
91  				mailingAddress, false), getApplicantMailingStateElement());
92  		addToElementsList(elementsList, getApplicantMailingZip(mailingAddress),
93  				getApplicantMailingZipElement());
94  		addToElementsList(elementsList, getBirthDate(person,
95  				FORMATTER_MM_HYPHEN_DD_HYPHEN_CCYY), getBirthDateElement());
96  		addToElementsList(elementsList, URL_ADDRESS1, getUrlAddress1Element());
97  		addToElementsList(elementsList, URL_ADDRESS2, getUrlAddress2Element());
98  		addToElementsList(elementsList, getOfficeId(userContext),
99  				getOfficeIdElement());
100 		addToElementsList(elementsList, getWorkDate(userContext,
101 				FORMATTER_MM_SLASH_DD_SLASH_YY), getWorkDateElement());
102 		addToElementsList(elementsList, getTechId(userContext),
103 				getEmployeeIdElement());
104 		String documentTemplateName = getDocumentName(isNew);
105 		String xmlData = createXmlForSingleDocument(elementsList,
106 				documentTemplateName);
107 		return xmlData;
108 	}
109 
110 	/***
111 	 * Gets the document name.
112 	 * 
113 	 * @param isNew the is new
114 	 * 
115 	 * @return the document name
116 	 */
117 	private String getDocumentName(boolean isNew) {
118 		if (isNew) {
119 			return getDocumentTemplateForNewApplication();
120 		}
121 		else {
122 			return getDocumentTemplateForRenewalApplication();
123 		}
124 	}
125 
126 	/***
127 	 * Gets the zip code.
128 	 * 
129 	 * @param mailingAddress the mailing address
130 	 * 
131 	 * @return the zip code
132 	 */
133 	private String getApplicantMailingZip(Address mailingAddress) {
134 		if ("99999".equalsIgnoreCase(mailingAddress.getZip())) {
135 			return "00000";
136 		}
137 		else {
138 			return mailingAddress.getZip();
139 		}
140 	}

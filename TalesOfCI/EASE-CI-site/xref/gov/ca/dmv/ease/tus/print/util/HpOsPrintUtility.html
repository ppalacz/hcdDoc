45  	/*** The Success Status Code. */
46  	public static final String SUCCESS_STATUS_CODE = "0";

48  	/***
49  	 * This method fetches the Print Status from the Response XML 
50  	 * String returned from Print Web Service call.
51  	 * 
52  	 * @param queryKey the Query Key
53  	 * @param responseXmlString the XML Response String
54  	 * 
55  	 * @return the Print Response for Query Key
56  	 */
57  	public static String queryPrintResponseByKey(String queryKey,
58  			String responseXmlString) {
59  		String printStatusCode = null;
60  		try {
61  			DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory
62  					.newInstance();
63  			DocumentBuilder docBuilder = docBuilderFactory.newDocumentBuilder();
64  			InputSource inStream = new InputSource();
65  			inStream.setCharacterStream(new StringReader(responseXmlString));
66  			Document document = docBuilder.parse(inStream);
67  			printStatusCode = document.getElementsByTagName(queryKey).item(0)
68  					.getTextContent();
69  		}
70  		catch (Exception e) {
71  			throw new EcsPrintServiceResponseException(
72  					"Exception occured while parsing Print Service Response "
73  							+ responseXmlString + e.getMessage());
74  		}
75  		return printStatusCode;
76  	}
77  
78  	/***
79  	 * This method checks if Print Job was successfully completed.
80  	 * 
81  	 * @param responseXmlString the XML Response String
82  	 * 
83  	 * @return true if Successful
84  	 */
85  	public static boolean isPrintJobCompleted(
86  			String responseXmlString) {
87  		String printStatusCode = queryPrintResponseByKey(HpOsPrintUtility.STATUS_CODE,
88  				responseXmlString);
89  		return (!EaseUtil.isNullOrBlank(printStatusCode) && printStatusCode
90  				.equals(SUCCESS_STATUS_CODE)) ? true : false;
91  	}

93  	/***
94  	 * This method will fetch the File and return in byte[] format.
95  	 * 
96  	 * @param printDocument the input file
97  	 * @return the file as byte[]
98  	 */
99  	public static byte[] createByteArray(File printDocument) {
100 		byte[] b = new byte[(int) printDocument.length()];
101 		try {
102 			FileInputStream fileInputStream = new FileInputStream(printDocument);
103 			fileInputStream.read(b);
104 			for (int i = 0; i < b.length; i++) {
105 				System.out.print((char) b[i]);
106 			}
107 		}
108 		catch (FileNotFoundException e) {
109 			System.out.println("File Not Found.");
110 			e.printStackTrace();
111 		}
112 		catch (IOException e1) {
113 			System.out.println("Error Reading The File.");
114 			e1.printStackTrace();
115 		}
116 		return b;
117 	}

27  	public static final String DESTINATION_BEGIN = "<destination>";

31  	public static final String REQUESTED_ATTRBS_BEGIN = "<requested-attributes>";
32  	public static final String REQUESTED_ATTRBS_END = "</requested-attributes>";
33  	public static final String DOC_FILENAME_BEGIN = "<document-filename>";
34  	public static final String DOC_FILENAME_END = "</document-filename>";
35  	public static final String COPY_COUNT_BEGIN = "<copy-count>";
36  	public static final String COPY_COUNT_END = "</copy-count>";
37  	public static final String BATCH_SIZE_BEGIN = "<batch-size>";

42  	public static final String JOB_CUSTOM_2_END = "</job-custom-2>";

51  	/***
52  	 * This method creates the XML Response for Deliver Print Job.
53  	 * 
54  	 * @param printerInfo the Printer Information
55  	 * 
56  	 * @return the string
57  	 */
58  	public static String createDeliverPrintJobContract(Location printerInfo) {
59  		StringBuilder deliverPrintJobContract = new StringBuilder();
60  		//Send Primary & Secondary Printer Information
61  		String primaryPrinter = printerInfo.getStation().getPrimaryPrinterId();
62  		//		String alternatePrinter = printerInfo.getStation().getAlternatePrinterId();
63  		deliverPrintJobContract.append(REQUEST_BEGIN).append(
64  				OPERATION_ATTRBS_BEGIN).append(DESTINATION_BEGIN).append(
65  				primaryPrinter.trim()).append(DESTINATION_END).append(
66  				OPERATION_ATTRBS_END).append(REQUEST_ATTRBS_BEGIN).append(
67  				DOC_FILENAME_BEGIN).append(DOC_FILENAME).append(
68  				DOC_FILENAME_END).append(COPY_COUNT_BEGIN).append(COPY_COUNT)
69  				.append(COPY_COUNT_END)
70  				//Un-comment the code to use Fail over printer strategy
71  				//							   .append(JOB_CUSTOM_2_BEGIN)
72  				//							   .append(alternatePrinter.trim())
73  				//							   .append(JOB_CUSTOM_2_END)
74  				.append(REQUEST_ATTRBS_END).append(REQUEST_END);
75  		return deliverPrintJobContract.toString();
76  	}

94  	/***
95  	 * This method creates the XML Response for Query Print Job.
96  	 * 
97  	 * @param jobClientId the Client Job Id
98  	 * 
99  	 * @return the XML Response for Query Print Job
100 	 */
101 	public static String createQueryPrintJobContract(String jobClientId) {
102 		StringBuilder queryPrintJobContract = new StringBuilder();
103 		queryPrintJobContract.append(REQUEST_BEGIN).append(
104 				OPERATION_ATTRBS_BEGIN).append(BATCH_SIZE_BEGIN).append(
105 				BATCH_SIZE).append(BATCH_SIZE_END).append(JOB_CLIENT_ID_BEGIN)
106 				.append(jobClientId).append(JOB_CLIENT_ID_END).append(
107 						OPERATION_ATTRBS_END).append(REQUEST_ATTRBS_BEGIN)
108 				.append(REQUESTED_ATTRBS_BEGIN).append(CURRENT_JOB_STATE)
109 				.append(REQUESTED_ATTRBS_END).append(REQUESTED_ATTRBS_BEGIN)
110 				.append(JOB_COMPLETION_STATUS).append(REQUESTED_ATTRBS_END)
111 				.append(REQUESTED_ATTRBS_BEGIN).append(QUEUE_ASSIGNED).append(
112 						REQUESTED_ATTRBS_END).append(REQUEST_ATTRBS_END)
113 				.append(REQUEST_END);
114 		return queryPrintJobContract.toString();
115 	}

117 	/***

119 	 *

122 	 * 
123 	 * @return the XML Response for Resubmit Print Job

125 	public static String createResubmitPrintJobContract(String jobClientId,

127 		StringBuilder resubmitPrintJobContract = new StringBuilder();

130 		//String alternatePrinter = printerInfo.getStation().getAlternatePrinterId();
131 		resubmitPrintJobContract.append(REQUEST_BEGIN).append(
132 				OPERATION_ATTRBS_BEGIN).append(JOB_CLIENT_ID_BEGIN).append(
133 				jobClientId).append(JOB_CLIENT_ID_END)
134 				.append(DESTINATION_BEGIN).append(primaryPrinter.trim())
135 				.append(DESTINATION_END).append(OPERATION_ATTRBS_END).append(
136 						REQUEST_ATTRBS_BEGIN).append(COPY_COUNT_BEGIN).append(
137 						COPY_COUNT).append(COPY_COUNT_END).append(
138 						REQUEST_ATTRBS_END).append(REQUEST_END);
139 		return resubmitPrintJobContract.toString();
140 	}

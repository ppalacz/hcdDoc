214 		processSectionsViolated(citationDetail.getSectionsViolated(),
215 				elementsList);

218 	/***
219 	 * Sets the address name.
220 	 *
221 	 * @param elementsList the elements list
222 	 * @param person the person
223 	 * @param application
224 	 */
225 	protected void setAddressName(List <String> elementsList, Person person,
226 			FtaFtpFinePaymentClearanceApplication application) {
227 		Address mailingAddress = person.getExistingAddress(AddressType.MAILING);
228 		Address otherAddress = application.getOtherAddress();
229 		if (otherAddress != null) {
230 			addToElementsList(elementsList, getFormattedAddress(otherAddress),
231 					OTHER_ADDRESS_ELEMENT);
232 			addToElementsList(elementsList, convertDateToString(otherAddress
233 					.getCreatedDate(), FORMATTER_MM_dd_yy),
234 					SAVED_AS_OF_DATE_ELEMENT);
235 		}
236 		addToElementsList(elementsList, getFormattedAddress(mailingAddress),
237 				MAILING_ADDRESS_ELEMENT);
238 		if (otherAddress != null) {
239 			addToElementsList(elementsList, convertDateToString(otherAddress
240 					.getEffectiveDate(), FORMATTER_MM_dd_yy),
241 					ADDRESS_AS_OF_DATE_ELEMENT);
242 			addToElementsList(elementsList, DocumentConverterUtil
243 					.getApplicantMailingCity(otherAddress), OTHER_CITY_ELEMENT);
244 			addToElementsList(elementsList, DocumentConverterUtil
245 					.getApplicantMailingState(otherAddress),//TODO: Requirements
246 					OTHER_STATE_ELEMENT);
247 		}
248 		addToElementsList(elementsList, DocumentConverterUtil
249 				.getApplicantMailingCity(mailingAddress), MAIL_CITY_ELEMENT);
250 		addToElementsList(elementsList, DocumentConverterUtil
251 				.getApplicantMailingState(mailingAddress), MAIL_STATE_ELEMENT);
252 		addToElementsList(elementsList, DocumentConverterUtil
253 				.getFormattedAkaName(person), ALSO_KNOWN_AS_ELEMENT);
254 	}

279 	/***
280 	 * Process sections violated.
281 	 *
282 	 * @param sectionsViolated the sections violated
283 	 * @param elementsList the elements list
284 	 */
285 	private void processSectionsViolated(
286 			List <SectionViolated> sectionsViolated, List <String> elementsList) {
287 		for (int indexSections = 0; indexSections < sectionsViolated.size(); ++indexSections) {
288 			addToElementsList(elementsList,
289 					convertValueOrBlankForNull(sectionsViolated.get(
290 							indexSections).getStatuteCode().getCode()),
291 					STATUTE_CODE_ELEMENTS[indexSections]);
292 			addToElementsList(elementsList,
293 					convertValueOrBlankForNull(sectionsViolated.get(
294 							indexSections).getSectionViolatedNumber()),
295 					SECTION_VIOLATED_CODE_ELEMENTS[indexSections]);
296 		}

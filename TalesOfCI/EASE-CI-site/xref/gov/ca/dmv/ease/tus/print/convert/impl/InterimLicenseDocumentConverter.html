67  		if (requestFromClient == null) {
68  			throw new MessageConversionException(
69  					"Invalid Interim License Receipt Request: Unable to process on null request");
70  		}
71  		boolean isDla = false;
72  		boolean isFeeReasonCodeK = false;
73  		InterimLicenseDocumentRequestTus request = (InterimLicenseDocumentRequestTus) requestFromClient;
74  		IUserContext userContext = request.getUserContext();
75  		DlApplication dlApplication = request.getDlApplication();
76  		Person person = dlApplication.getApplicant();
77  		Address currentMailingAddress = person.getAddress(AddressType.MAILING);
78  		Address currentResidenceAddress = person
79  				.getAddress(AddressType.RESIDENTIAL);
80  		CommercialTestResultGroup dlctr = person.getDriverLicense()
81  				.getCommercialTestResultGroup();
82  		StringBuilder hazmatFlag = new StringBuilder(
83  				IPrintConverterConstants.SPACE);
84  		if (userContext.getTtc().equals(IPrintConverterConstants.TTC_DLA)) {
85  			isDla = true;
86  		}
87  		if (!isNullOrBlank(dlApplication.getFeeRequiredReasonCode())
88  				&& IPrintConverterConstants.FEE_REASON_CODE_K
89  						.equalsIgnoreCase(dlApplication
90  								.getFeeRequiredReasonCode().getCode())) {

92  		}
93  		String endorsements = new EndorsementEvaluatorForInterimLicenseDocument()
94  				.checkCommercialEndorsements(
95  						dlApplication.getApplicant().getDriverLicense(),
96  						hazmatFlag, isDla, isFeeReasonCodeK).trim();

98  				.getApplicant().getDriverLicense().getExistingLicenseClasses();
99  		List <DrivingRestriction> newRestrictions = person.getDriverLicense()
100 				.getNewRestrictions();
101 		List <InterimPermit> interimPermits = person.getDriverLicense()
102 				.getInterimPermits(InterimPermitType.TEMPORARY);
103 		List <String> elementsList = new ArrayList <String>();
104 		elementsList
105 				.add(IPrintConverterConstants.INT_LIC_DOC_VERIFICATION_MESSAGE
106 						+ FIELD_DELIMITER
107 						+ IPrintConverterConstants.INT_LIC_DOC_VERIFICATION_ELEMENT);

112 					+ IPrintConverterConstants.INT_LIC_DOC_HEADER_ELEMENT);

115 				+ IPrintConverterConstants.INT_LIC_DOC_DL_OR_ID_NUMBER_ELEMENT);
116 		if (isNotNull(exisingLicenseClasses)
117 				&& !exisingLicenseClasses.isEmpty()) {
118 			if (!isNullOrBlank(exisingLicenseClasses.get(0))) {
119 				elementsList
120 						.add(DocumentConverterUtil.getClassType(
121 								exisingLicenseClasses.get(0).getCode(),
122 								newRestrictions)
123 								+ FIELD_DELIMITER
124 								+ IPrintConverterConstants.INT_LIC_DOC_CLS_OF_RECORD_1_ELEMENT);
125 			}
126 			if (exisingLicenseClasses.size() > 1
127 					&& !isNullOrBlank(exisingLicenseClasses.get(1))
128 					&& !isNullOrBlank(DocumentConverterUtil.getClassType(
129 							exisingLicenseClasses.get(1).getCode(),
130 							newRestrictions))) {
131 				elementsList.add(IPrintConverterConstants.AMPERSAND
132 						+ FIELD_DELIMITER
133 						+ IPrintConverterConstants.AMPERSAND_ELEMENT);
134 				elementsList
135 						.add(DocumentConverterUtil.getClassType(
136 								exisingLicenseClasses.get(1).getCode(),
137 								newRestrictions)
138 								+ FIELD_DELIMITER
139 								+ IPrintConverterConstants.INT_LIC_DOC_CLS_OF_RECORD_2_ELEMENT);
140 			}
141 		}
142 		if (!isNullOrBlank(endorsements)) {
143 			elementsList.add(endorsements + FIELD_DELIMITER
144 					+ IPrintConverterConstants.INT_LIC_DOC_ENDORSEMENT_ELEMENT);
145 		}
146 		//elementsList.add("ZBPBLDON" + FIELD_DELIMITER + "ZBPBLDON");
147 		if (hazmatFlag.equals(IPrintConverterConstants.VALUE_Y)) {
148 			elementsList.add(IPrintConverterConstants.INT_LIC_DOC_HM_MESSAGE
149 					+ FIELD_DELIMITER
150 					+ IPrintConverterConstants.INT_LIC_DOC_HM_MESSAGE_ELEMENT);
151 		}
152 		//elementsList.add("ZBPBLDOF" + FIELD_DELIMITER + "ZBPBLDOF");
153 		if (IPrintConverterConstants.STATE_CODE_MX
154 				.equalsIgnoreCase(currentResidenceAddress.getState())) {
155 			elementsList
156 					.add(IPrintConverterConstants.INT_LIC_DOC_NON_RESIDENT_MESSAGE
157 							+ FIELD_DELIMITER
158 							+ IPrintConverterConstants.INT_LIC_DOC_PROV_STMT_ELEMENT);
159 		}
160 		else {
161 			if (!isXyearsOld(person, 18)) {
162 				elementsList
163 						.add(IPrintConverterConstants.INT_LIC_DOC_PROV_UNTIL_18_MESSAGE
164 								+ FIELD_DELIMITER
165 								+ IPrintConverterConstants.INT_LIC_DOC_PROV_STMT_ELEMENT);
166 			}
167 		}
168 		elementsList.add(convertDateToString(person.getDriverLicense()
169 				.getEffectiveDate(), FORMATTER_MM_dd_yy)
170 				+ FIELD_DELIMITER
171 				+ IPrintConverterConstants.INT_LIC_DOC_ISSUE_DATE_ELEMENT);
172 		elementsList.add(userContext.getOfficeId() + FIELD_DELIMITER
173 				+ IPrintConverterConstants.INT_LIC_DOC_OFFICE_ID_ELEMENT);
174 		elementsList.add(userContext.getTechId() + FIELD_DELIMITER
175 				+ IPrintConverterConstants.INT_LIC_DOC_KEYERS_ID_ELEMENT);
176 		if (!isNullOrBlank(dlctr)) {
177 			List <CommercialTestResult> performanceTestResults = dlctr
178 					.getCommercialTestResults(CommercialTestResultType.PERFORMANCE_TEST_RESULT);
179 			String examinerId = BLANK;
180 			if (isNotNull(performanceTestResults)
181 					&& !performanceTestResults.isEmpty()) {
182 				examinerId = performanceTestResults.get(0).getExaminerID();
183 			}
184 			if (!isNullOrBlank(examinerId)) {
185 				elementsList.add(examinerId + FIELD_DELIMITER
186 						+ IPrintConverterConstants.INT_LIC_DOC_LRE_ID_ELEMENT);
187 			}
188 		}
189 		elementsList.add(convertDateToString(person.getDriverLicense()
190 				.getExpirationDate(), FORMATTER_MM_dd_yy)
191 				+ FIELD_DELIMITER
192 				+ IPrintConverterConstants.INT_LIC_DOC_EXPIRATION_DATE_ELEMENT);
193 		if (isNotNull(person.getCurrentName())) {
194 			elementsList.add(getFormattedPhotoName(person.getCurrentName()
195 					.getFullName())
196 					+ FIELD_DELIMITER
197 					+ IPrintConverterConstants.INT_LIC_DOC_PHOTO_NAME_ELEMENT);
198 		}
199 		elementsList.add(person.getGenderCode().getCode() + FIELD_DELIMITER
200 				+ IPrintConverterConstants.INT_LIC_DOC_GENDER_ELEMENT);
201 		elementsList.add(person.getHairColorCode().getCode() + FIELD_DELIMITER
202 				+ IPrintConverterConstants.INT_LIC_DOC_HAIR_CODE_ELEMENT);
203 		elementsList.add(person.getEyeColorCode().getCode() + FIELD_DELIMITER
204 				+ IPrintConverterConstants.INT_LIC_DOC_EYE_CODE_ELEMENT);
205 		elementsList.add(getFormattedAddress(person.getDriverLicense()
206 				.getPhotoInformation().getPhotoAddress())
207 				+ FIELD_DELIMITER
208 				+ IPrintConverterConstants.INT_LIC_DOC_PHOTO_ADDRESS_ELEMENT);
209 		elementsList.add(person.getHeightInFeet() + FIELD_DELIMITER
210 				+ IPrintConverterConstants.INT_LIC_DOC_HEIGHT_IN_FEET_ELEMENT);
211 		elementsList.add(person.getHeightInInch() + FIELD_DELIMITER
212 				+ IPrintConverterConstants.INT_LIC_DOC_HEIGHT_IN_INCH_ELEMENT);
213 		elementsList.add(person.getWeight() + FIELD_DELIMITER
214 				+ IPrintConverterConstants.INT_LIC_DOC_WEIGHT_ELEMENT);
215 		elementsList.add(convertDateToString(person.getBirthDate(),
216 				FORMATTER_MM_dd_yy)
217 				+ FIELD_DELIMITER
218 				+ IPrintConverterConstants.INT_LIC_DOC_BIRTHDATE_ELEMENT);
219 		if (!isXyearsOld(person, 21)) {
220 			elementsList.add(IPrintConverterConstants.ageMessage
221 					+ when21yearsOld(person) + FIELD_DELIMITER
222 					+ IPrintConverterConstants.INT_LIC_DOC_AGE_21_ELEMENT);
223 		}
224 		elementsList.add(currentMailingAddress.getCity() + FIELD_DELIMITER
225 				+ IPrintConverterConstants.INT_LIC_DOC_MAIL_CITY_ELEMENT);
226 		if (!"OT".equalsIgnoreCase(currentMailingAddress.getState())) {
227 			elementsList.add(currentMailingAddress.getState() + FIELD_DELIMITER
228 					+ IPrintConverterConstants.INT_LIC_DOC_MAIL_STATE_ELEMENT);
229 		}
230 		if (!IPrintConverterConstants.ZIP_99999.equals(currentMailingAddress
231 				.getZip())) {
232 			elementsList.add(currentMailingAddress.getZip() + FIELD_DELIMITER
233 					+ IPrintConverterConstants.INT_LIC_DOC_MAIL_ZIP_ELEMENT);
234 		}
235 		if (isNotNull(newRestrictions)) {
236 			elementsList
237 					.add(IPrintConverterConstants.RESTRICTION_HEADER
238 							+ FIELD_DELIMITER
239 							+ IPrintConverterConstants.INT_LIC_DOC_RESTRICTION_HEADER_ELEMENT);
240 			List <String> restrictions = getRestrictions(newRestrictions);
241 			if (restrictions
242 					.contains(IPrintConverterConstants.RESTRICTION_CODE1)) {
243 				elementsList
244 						.add(IPrintConverterConstants.INT_LIC_DOC_CORR_LENS_MESSAGE
245 								+ FIELD_DELIMITER
246 								+ IPrintConverterConstants.INT_LIC_DOC_RESTRICTION_1_ELEMENT);
247 				elementsList
248 						.add(IPrintConverterConstants.INT_LIC_DOC_SEE_OVER_1_MESSAGE
249 								+ FIELD_DELIMITER
250 								+ IPrintConverterConstants.INT_LIC_DOC_RESTRICTION_3_ELEMENT);
251 			}
252 			else {
253 				elementsList
254 						.add(IPrintConverterConstants.INT_LIC_DOC_SEE_OVER_2_MESSAGE
255 								+ FIELD_DELIMITER
256 								+ IPrintConverterConstants.INT_LIC_DOC_RESTRICTION_1_ELEMENT);
257 			}
258 		}
259 		LegalPresenceDocument legalPresenceDocument = dlApplication
260 				.getLegalPresenceDocument();
261 		if (isNotNull(legalPresenceDocument)
262 				&& !isNullOrBlank(legalPresenceDocument.getVerificationCode())
263 				&& IPrintConverterConstants.LEGAL_PRESENCE_VERIFICATION_CODE_N
264 						.equalsIgnoreCase(dlApplication
265 								.getLegalPresenceDocument()
266 								.getVerificationCode().getCode())) {
267 			elementsList
268 					.add(IPrintConverterConstants.INT_LIC_DOC_BDLP_VERIFY_MESSAGE
269 							+ FIELD_DELIMITER
270 							+ IPrintConverterConstants.INT_LIC_DOC_BDLP_VERIFY_ELEMENT);
271 		}
272 		if (isNotNull(dlApplication.getOriginalProcessHistory())) {
273 			if (isNotNull(dlApplication.getOriginalProcessHistory()
274 					.getUserContext())) {
275 				elementsList
276 						.add(dlApplication.getOriginalProcessHistory()
277 								.getUserContext().getOfficeId()
278 								+ FIELD_DELIMITER
279 								+ IPrintConverterConstants.INT_LIC_DOC_ORG_OFF_ID_ELEMENT);
280 				elementsList
281 						.add(convertDateToString(dlApplication
282 								.getApplicationDate(), FORMATTER_MM_dd_yy)
283 								+ FIELD_DELIMITER
284 								+ IPrintConverterConstants.INT_LIC_DOC_ORG_APP_DATE_ELEMENT);
285 				elementsList
286 						.add(dlApplication.getOriginalProcessHistory()
287 								.getUserContext().getTechId()
288 								+ FIELD_DELIMITER
289 								+ IPrintConverterConstants.INT_LIC_DOC_ORG_TECH_ID_ELEMENT);
290 			}
291 			elementsList
292 					.add(dlApplication.getOriginalProcessHistory()
293 							.getControlCashierSeqNumber()
294 							+ FIELD_DELIMITER
295 							+ IPrintConverterConstants.INT_LIC_DOC_CONTROL_CASHIER_SEQ_ELEMENT);
296 		}
297 		if (dlApplication.isCNACondition()) {
298 			elementsList.add(IPrintConverterConstants.CNA + FIELD_DELIMITER
299 					+ IPrintConverterConstants.INT_LIC_DOC_CNA_ELEMENT);
300 		}
301 		return createXmlForSingleDocument(elementsList,
302 				IPrintConverterConstants.INT_LIC_DOC_TEMPLATE_NAME);
303 	}

305 	/***
306 	 * Gets the Interim License Receipt header - TODO update logic to derive document header.
307 	 * 
308 	 * Per requirement, If VCSPRTCL_PrintClassCode = 61(INSTRUCTION PERMIT) or
309 	 * 62(TEMPORARY LICENSE) or 74(HQ'S ISSUANCE NOTICE) then execute the below
310 	 * logic for retrieving the header message
311 	 * @param dlApplication 
312 	 * @param driverLicense 
313 	 * 
314 	 * @return documentHeader the documentHeader
315 	 */
316 	private String getDocumentHeader(DlApplication dlApplication,
317 			DriverLicense driverLicense, InterimPermit interimPermit) {
318 		String documentHeader = BLANK;
319 		List <String> newCategories = DocumentConverterUtil
320 				.convertCodeSetToStringList(driverLicense.getNewCategories());
321 		List <String> existingLicenseClasses = DocumentConverterUtil
322 				.convertCodeSetToStringList(driverLicense
323 						.getExistingLicenseClasses());
324 		List <String> existingEndorsements = DocumentConverterUtil
325 				.convertCodeSetToStringList(driverLicense
326 						.getExistingEndorsements());
327 		List <String> newEndorsements = DocumentConverterUtil
328 				.convertCodeSetToStringList(driverLicense.getNewEndorsements());
329 		CommercialTestResultGroup commercialTestResultGroup = driverLicense
330 				.getCommercialTestResultGroup();
331 		Date backgroundCheckExpirationDate = driverLicense
332 				.getBackgroundCheckExpirationDate();
333 		String backgroundCheckIndicator = null;
334 		Calendar tmpCalendar = Calendar.getInstance();
335 		tmpCalendar.add(Calendar.DATE, 89);
336 		if (!EaseUtil
337 				.isNullOrBlank(driverLicense.getBackgroundCheckIndicator())) {
338 			backgroundCheckIndicator = driverLicense
339 					.getBackgroundCheckIndicator().getCode();
340 		}
341 		if (isNotNull(interimPermit)
342 				&& InterimPermitType.TEMPORARY.equals(interimPermit
343 						.getPermitType())) {
344 			if (ConverterHelper.isComercialLicenseClass(interimPermit
345 					.getPermitLicenseClassCode().getCode())) {
346 				documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
347 			}
348 			else {
349 				documentHeader = IPrintConverterConstants.TEMPORARY_DRIVER_LICENSE;
350 			}
351 		}
352 		else {
353 			if (ConverterHelper.isComercialLicenseClass(interimPermit
354 					.getPermitLicenseClassCode().getCode())) {
355 				if (newEndorsements
356 						.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)) {
357 					if (existingEndorsements
358 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)) {
359 						if (DocumentConverterUtil
360 								.isAnyCommercialLicenseClassExist(driverLicense
361 										.getExistingLicenseClasses())) {
362 							if (!isNullOrBlank(backgroundCheckIndicator)
363 									&& backgroundCheckIndicator
364 											.equalsIgnoreCase(IPrintConverterConstants.BACKGROUNDCHECK_0)) {
365 								if (isNotNull(backgroundCheckExpirationDate)
366 										&& backgroundCheckExpirationDate
367 												.before(tmpCalendar.getTime())) {
368 									documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
369 								}//End if - Background Check Expiration date before current date <= 89 days
370 								else {
371 									documentHeader = IPrintConverterConstants.INTERIM_DRIVER_LICENSE;
372 								}// End else - Background Check Expiration date before current date <= 89 days
373 							}//End if - Background check indicator is 0
374 							else {
375 								documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
376 							}//End else - Background check indicator is 0
377 						}//End if is existing classes have commercial class
378 						else {
379 							if (!isNullOrBlank(backgroundCheckIndicator)
380 									&& backgroundCheckIndicator
381 											.equalsIgnoreCase(IPrintConverterConstants.BACKGROUNDCHECK_0)) {
382 								if (isNotNull(backgroundCheckExpirationDate)
383 										&& backgroundCheckExpirationDate
384 												.before(tmpCalendar.getTime())) {
385 									documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
386 								}//End if - Background Check Expiration date before current date <= 89 days - Iteration 1
387 								else {
388 									documentHeader = IPrintConverterConstants.INTERIM_COMMERCIAL_DRIVER_LICENSE;
389 								}//End else - Background Check Expiration date before current date <= 89 days - Iteration 1
390 							} //End if - Background check indicator is 0 - Iteration 1
391 							else {
392 								if (!isNullOrBlank(backgroundCheckIndicator)
393 										&& backgroundCheckIndicator
394 												.equalsIgnoreCase(IPrintConverterConstants.BACKGROUNDCHECK_N)) {
395 									if (isNotNull(dlApplication
396 											.getOriginalProcessHistory())
397 											&& isNotNull(dlApplication
398 													.getOriginalProcessHistory()
399 													.getUserContext())
400 											&& IPrintConverterConstants.TTC_DLA
401 													.equalsIgnoreCase(dlApplication
402 															.getOriginalProcessHistory()
403 															.getUserContext()
404 															.getTtc())) {
405 										documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
406 									} //End if - is Original TTC DLA 
407 									else {
408 										documentHeader = IPrintConverterConstants.INTERIM_COMMERCIAL_DRIVER_LICENSE;
409 									} //End else - is Original TTC DLA
410 								} //End if - Background check indicator is N
411 								else {
412 									documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
413 								}//End else - Background check indicator is N
414 							}//End else - Background check indicator is 0 - Iteration 1
415 						}//End else is existing classes have commercial class
416 					}// End if Existing endorsements have HM
417 					else {
418 						if (!isNullOrBlank(backgroundCheckIndicator)
419 								&& backgroundCheckIndicator
420 										.equalsIgnoreCase(IPrintConverterConstants.BACKGROUNDCHECK_0)) {
421 							if (isNotNull(backgroundCheckExpirationDate)
422 									&& backgroundCheckExpirationDate
423 											.before(tmpCalendar.getTime())) {
424 								documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
425 							}//End if - Background Check Expiration date before current date <= 89 days - Iteration 2
426 							else {
427 								documentHeader = IPrintConverterConstants.INTERIM_COMMERCIAL_DRIVER_LICENSE;
428 							}//End else - Background Check Expiration date before current date <= 89 days - Iteration 2
429 						}//End if - Background check indicator is 0 - Iteration 2 
430 						else {
431 							documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
432 						}//End else - Background check indicator is 0 - Iteration 2
433 					}// End else Existing endorsements have HM
434 				}// End if New endorsements have HM 
435 				else {
436 					if (!isNullOrBlank(dlApplication
437 							.getNewUserVerifiedBdlpCode())
438 							&& dlApplication
439 									.getNewUserVerifiedBdlpCode()
440 									.getCode()
441 									.equalsIgnoreCase(
442 											IPrintConverterConstants.PREV_BD_CORR_VERIF_R)
443 							&& isNotNull(dlApplication.isAbsentParent())
444 							&& dlApplication.isAbsentParent()) {
445 						documentHeader = IPrintConverterConstants.TEMPORARY_COMMERCIAL_DRIVER_LICENSE;
446 					}
447 					else {
448 						documentHeader = IPrintConverterConstants.INTERIM_COMMERCIAL_DRIVER_LICENSE;
449 					}
450 				}// End else New endorsements have HM 
451 			}//End If - is Commercial Interim Permit License class 
452 			else {
453 				if (!isNullOrBlank(dlApplication.getNewUserVerifiedBdlpCode())
454 						&& dlApplication
455 								.getNewUserVerifiedBdlpCode()

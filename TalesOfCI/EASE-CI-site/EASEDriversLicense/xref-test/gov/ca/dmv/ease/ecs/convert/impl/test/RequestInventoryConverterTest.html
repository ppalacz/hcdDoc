241 	/***
242 	 * Test create response.
243 	 */
244 	@Test
245 	public void testCreateResponse() {
246 		// Test case for normal response of DL request
247 		ResponseFixtures responseFixture = ResponseFixtures.NORMAL_RESPONSE_FOR_DL_REQUEST;
248 		RequestInventoryResponseEcs response = (RequestInventoryResponseEcs) converter
249 				.createResponse(responseFixture.anInboundMessagePayload);
250 		License license = (DriverLicense) response.getLicense();
251 		assertEquals(responseFixture.itemCode, license.getItemCode());
252 		assertEquals(responseFixture.licenseOrIdNumber, license
253 				.getLicenseNumber());
254 		// Test case for normal response of ID request		
255 		responseFixture = ResponseFixtures.NORMAL_RESPONSE_FOR_ID_REQUEST;
256 		response = getResponse(responseFixture.anInboundMessagePayload);
257 		IdCard idCard = (IdCard) response.getLicense();
258 		assertEquals(responseFixture.itemCode, idCard.getItemCode());
259 		assertEquals(responseFixture.licenseOrIdNumber, idCard
260 				.getLicenseNumber());
261 		// Test case for normal response of Non-DL/ID request
262 		responseFixture = ResponseFixtures.RESPONSE_FOR_NON_DL_ID_REQUEST;
263 		response = getResponse(responseFixture.anInboundMessagePayload);
264 		assertNull(response.getLicense());
265 		// Test case for error response of DL request with error code 12
266 		responseFixture = ResponseFixtures.ERROR_RESPONSE_FOR_DL_REQUEST_WITH_ERROR_CODE_12;
267 		response = getResponse(responseFixture.anInboundMessagePayload);
268 		assertEquals(responseFixture.errorMessage, response.getErrorMessage());
269 		// Test case for error response of DL request with error code 10		
270 		responseFixture = ResponseFixtures.ERROR_RESPONSE_FOR_DL_REQUEST_WITH_ERROR_CODE_10;
271 		response = getResponse(responseFixture.anInboundMessagePayload);
272 		assertEquals(responseFixture.errorMessage, response.getErrorMessage());
273 	}

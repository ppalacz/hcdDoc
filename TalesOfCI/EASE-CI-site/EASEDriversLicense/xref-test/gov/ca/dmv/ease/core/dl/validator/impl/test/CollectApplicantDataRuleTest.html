185 	public void testValidateExpDate() {
186 		assertFalse(errorCollector.hasErrors());
187 		//Set LegalPresenceDocument
188 		CodeSetElement legalPresenceDocumentType = new CodeSetElement();
189 		List <Document> documents = new ArrayList <Document>();
190 		LegalPresenceDocument lpd = new LegalPresenceDocument();
191 		Calendar calendar = new GregorianCalendar();
192 		legalPresenceDocumentType.setCode("A");
193 		lpd.setLegalPresenceDocumentType(legalPresenceDocumentType);
194 		lpd.setDocumentId("INS");
195 		//D862 - ExpDate already expired
196 		calendar.add(Calendar.DAY_OF_MONTH, -1);
197 		lpd.setExpirationDate(calendar.getTime());
198 		documents.add(lpd);
199 		dlApplication.getApplicant().setDocuments(documents);
200 		collectApplicantDataRule.validateExpDate(dlApplication, errorCollector);
201 		assertEquals(errorCollector.getEntries().size(), 1);
202 		assertEquals(
203 				errorCollector.getEntries().get(0).getExceptionMessage(),
204 				IDataCollectionErrorMessageConstants.INS_DOC_EXP_DATE_VS_CURRENT_DATE);
205 		//D860 - ExpDate less than 30 days
206 		calendar.add(Calendar.DAY_OF_MONTH, 29);
207 		lpd.setExpirationDate(calendar.getTime());
208 		documents.add(lpd);
209 		dlApplication.getApplicant().setDocuments(documents);
210 		collectApplicantDataRule.validateExpDate(dlApplication, errorCollector);
211 		assertEquals(errorCollector.getEntries().size(), 2);
212 		assertEquals(
213 				errorCollector.getEntries().get(1).getExceptionMessage(),
214 				IDataCollectionErrorMessageConstants.INS_DOC_EXP_LESS_THAN_30_DAYS);
215 		//D861 - ExpDate less than 60 days
216 		calendar.add(Calendar.DAY_OF_MONTH, 29);
217 		lpd.setExpirationDate(calendar.getTime());
218 		documents.clear();
219 		documents.add(lpd);
220 		dlApplication.getApplicant().setDocuments(documents);
221 		collectApplicantDataRule.validateExpDate(dlApplication, errorCollector);
222 		assertEquals(errorCollector.getEntries().size(), 3);
223 		assertEquals(
224 				errorCollector.getEntries().get(2).getExceptionMessage(),
225 				IDataCollectionErrorMessageConstants.INS_DOC_EXP_LESS_THAN_60_DAYS);
226 		//D863 - ExpDate more than 60 days but documentId is null
227 		calendar.add(Calendar.DAY_OF_MONTH, 29);
228 		lpd.setExpirationDate(calendar.getTime());
229 		lpd.setDocumentId(null);
230 		documents.clear();
231 		documents.add(lpd);
232 		dlApplication.getApplicant().setDocuments(documents);
233 		collectApplicantDataRule.validateExpDate(dlApplication, errorCollector);
234 		assertEquals(errorCollector.getEntries().size(), 4);
235 		assertEquals(
236 				errorCollector.getEntries().get(3).getExceptionMessage(),
237 				IDataCollectionErrorMessageConstants.INS_DOC_EXP_DATE_VS_INS_DOC_NO);
238 	}

241 		assertFalse(errorCollector.hasErrors());
242 		//Set license class to Non-commercial C (F)
243 		DriverLicense driverLicense = new DriverLicense();
244 		List <CodeSetElement> licenseClasses = new ArrayList <CodeSetElement>();
245 		CodeSetElement licenseClass = new CodeSetElement();
246 		licenseClass.setCode("F");
247 		licenseClasses.add(licenseClass);

249 		dlApplication.setLicense(driverLicense);
250 		//Set correct Data entry receipt (Der) values for COUNTER/MAIL modes
251 		String derNumber = "90000000";
252 		dlApplication.setDataEntryReceiptNumber(derNumber);
253 		CodeSetElement derTypeCode = new CodeSetElement();
254 		dlApplication.setDataEntryReceiptTypeCode(null);
255 		collectApplicantDataRule.validateDerAppNumber(dlApplication,
256 				IEaseModes.COUNTER, errorCollector);
257 		assertFalse(errorCollector.hasErrors());
258 		collectApplicantDataRule.validateDerAppNumber(dlApplication,
259 				IEaseModes.MAIL, errorCollector);
260 		assertFalse(errorCollector.hasErrors());
261 		//Set errors for COUNTER or MAIL modes
262 		//0153
263 		derTypeCode.setCode(IDerAppCodeConstants.A_NON_COMMERCIAL_DL);
264 		dlApplication.setDataEntryReceiptTypeCode(derTypeCode);
265 		collectApplicantDataRule.validateDerAppNumber(dlApplication,
266 				IEaseModes.COUNTER, errorCollector);
267 		assertEquals(errorCollector.getEntries().size(), 1);
268 		assertEquals(errorCollector.getEntries().get(0).getExceptionMessage(),
269 				IGeneralErrorMessageConstants.INVALID_FIELD_PRESENT);
270 		//D672
271 		dlApplication.setDataEntryReceiptTypeCode(null);
272 		derNumber = "80000000";
273 		dlApplication.setDataEntryReceiptNumber(derNumber);
274 		collectApplicantDataRule.validateDerAppNumber(dlApplication,
275 				IEaseModes.MAIL, errorCollector);
276 		assertEquals(errorCollector.getEntries().size(), 2);
277 		assertEquals(errorCollector.getEntries().get(1).getExceptionMessage(),
278 				IDataCollectionErrorMessageConstants.DER_OR_APP_NUMBER_RANGE);
279 		
280 		//Set correct Data entry receipt (Der) values for MANUAL/TRAVEL_CREW modes
281 		derTypeCode.setCode(IDerAppCodeConstants.A_NON_COMMERCIAL_DL);

287 		assertEquals(errorCollector.getEntries().size(), 2);
288 		collectApplicantDataRule.validateDerAppNumber(dlApplication,
289 				IEaseModes.TRAVEL_CREW, errorCollector);
290 		assertEquals(errorCollector.getEntries().size(), 2);
291

293 		//D013

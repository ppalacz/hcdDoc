173 	/***
174 	 * Set the mailing address of a person to an existing list.
175 	 * 
176 	 * @param person the person
177 	 * @param mailingAddress the mailing address
178 	 * 
179 	 * @return the mailing address
180 	 */
181 	public static void setMailingAddress(Person person, Address mailingAddress) {
182 		// If the mailing address is null then do nothing
183 		if (mailingAddress == null) {
184 			return;
185 		}
186 		List <Locator> locators = person.getLocators();
187 		if (locators != null && locators.size() > 0) {
188 			boolean exists = false;
189 			for (int index = 0; index < locators.size(); index++) {
190 				Address address = (Address) locators.get(index);
191 				if (address != null) {
192 					AddressType addressType = address.getAddressType();
193 					if (addressType != null
194 							&& ADDRESS_TYPE_MAIL.equals(addressType)) {
195 						locators.remove(index);
196 						locators.add(index, mailingAddress);
197 						exists = true;
198 						break;
199 					}
200 				} // if loop
201 			} // for loop
202 			if (!exists) {
203 				locators.add(mailingAddress);
204 			}
205 		}
206 		else {
207 			locators.add(mailingAddress);
208 		}
209 	}

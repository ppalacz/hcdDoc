32  public class DlInventoryAccessorFromLocalStoreTest extends TestCase {
33  	private DlInventoryAccessorFromLocalStore fixture;
34  	private InventoryItem inventoryItem;
35  	private IUserContext userContext;
36  	private CodeSetElement inventoryCodeSetElement;
37  
38  	@Override
39  	protected void setUp() throws Exception {
40  		userContext = UserContextFixtureFactory.createUserContext();
41  		//Persist and Retrieve Code Set Element
42  		PersistenceServiceRequestFactory factory = (PersistenceServiceRequestFactory) EaseApplicationContext
43  				.getApplicationContext().getBean(
44  						"persistenceServiceRequestFactory");
45  		inventoryCodeSetElement = new CodeSetElement("DriverLicenceNumber",
46  				"A233444", "DriverLicenceNumber");
47  		inventoryCodeSetElement.setCodeCategory("Test_CodeSet");
48  		IPersistenceServiceRequest request = factory
49  				.createSaveOrUpdateBusinessObjectRequest(userContext,
50  						inventoryCodeSetElement);
51  		SaveOrUpdateBusinessObjectResponse codeSetResponse = (SaveOrUpdateBusinessObjectResponse) request
52  				.execute();
53  		//Retrieve the Persisted Code Set Element
54  		request = factory.createRetrieveBusinessObjectRequest(userContext,
55  				inventoryCodeSetElement);
56  		RetrieveBusinessObjectResponse retrieveResponse = (RetrieveBusinessObjectResponse) request
57  				.execute();
58  		inventoryCodeSetElement = (CodeSetElement) retrieveResponse
59  				.getResults().get(0);
60  		//Persist Inventory Item
61  		inventoryItem = new InventoryItem();
62  		inventoryItem.setIssued(false);
63  		inventoryItem.setCreatedBy("CREATED BY");
64  		inventoryItem.setInventoryTypeCode(inventoryCodeSetElement);
65  		request = factory.createSaveOrUpdateBusinessObjectRequest(userContext,
66  				inventoryItem);
67  		SaveOrUpdateBusinessObjectResponse inventoryResponse = (SaveOrUpdateBusinessObjectResponse) request
68  				.execute();
69  	}
70  
71  	@Override
72  	protected void tearDown() throws Exception {
73  		fixture = null;
74  		userContext = null;
75  		//Remove the Inserted Inventory Record
76  		PersistenceServiceRequestFactory factory = (PersistenceServiceRequestFactory) EaseApplicationContext
77  				.getApplicationContext().getBean(
78  						"persistenceServiceRequestFactory");
79  		IPersistenceServiceRequest request = factory
80  				.createDeleteBusinessObjectRequest(userContext, inventoryItem);
81  		DeleteBusinessObjectResponse aResponse = (DeleteBusinessObjectResponse) request
82  				.execute();
83  		request = factory.createDeleteBusinessObjectRequest(userContext,
84  				inventoryCodeSetElement);
85  		aResponse = (DeleteBusinessObjectResponse) request.execute();
86  		super.tearDown();
87  	}
88  
89  	public void testGetDlInventory() {
90  		fixture = new DlInventoryAccessorFromLocalStore();
91  		InventoryItem inventoryItem = fixture.getDlInventory(userContext);
92  		assertNotNull(inventoryItem);
93  		assertNotNull(inventoryItem.getInventoryTypeCode());
94  		assertNotSame("", inventoryItem.getInventoryTypeCode().getCode());
95  	}

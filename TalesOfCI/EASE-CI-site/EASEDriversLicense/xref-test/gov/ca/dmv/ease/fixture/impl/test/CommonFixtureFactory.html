218 	/***
219 	 * Gets the calender.
220 	 * 
221 	 * @return the calender
222 	 */
223 	public static Calendar getCalender() {
224 		// get the supported ids for GMT-08:00 (Pacific Standard Time)
225 		String[] ids = TimeZone.getAvailableIDs(-8 * 60 * 60 * 1000);
226 		// if no ids were returned, something is wrong. get out.
227 		if (ids.length == 0) {
228 			System.exit(0);
229 		}
230 		// begin output
231 		// create a Pacific Standard Time time zone
232 		SimpleTimeZone pdt = new SimpleTimeZone(-8 * 60 * 60 * 1000, ids[0]);
233 		// set up rules for daylight savings time
234 		pdt
235 				.setStartRule(Calendar.APRIL, 1, Calendar.SUNDAY,
236 						2 * 60 * 60 * 1000);
237 		pdt.setEndRule(Calendar.OCTOBER, -1, Calendar.SUNDAY,
238 				2 * 60 * 60 * 1000);
239 		// create a GregorianCalendar with the Pacific Daylight time zone
240 		// and the current date and time
241 		Calendar calendar = new GregorianCalendar(pdt);
242 		Date trialTime = new Date();
243 		calendar.setTime(trialTime);
244 		return calendar;
245 	}

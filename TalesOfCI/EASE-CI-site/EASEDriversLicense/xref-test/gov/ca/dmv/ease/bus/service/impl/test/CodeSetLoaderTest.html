158 	public void testLoadAllCodeSets() {
159 		IUserContext aContext = new UserContext();
160 		ICodeSetRegistrySingleton aRegistry = (CodeSetRegistrySingleton) EaseApplicationContext
161 				.getApplicationContext().getBean("codeSetRegistrySingleton");
162 		loaderFixture.loadAllCodeSets(aContext, aRegistry);
163 		ICodeSet codeSet = aRegistry.getCodeSetNamed("CITY CODES");
164 		ICodeSet aCodeSet = aRegistry.getCodeSetNamed("HAIR CODES");
165 		Assert.assertEquals(2, aRegistry.getAllCodeSets().size());
166 		Assert.assertFalse(aRegistry.isEmpty());
167 		Assert.assertNotNull(codeSet);
168 		Assert.assertNotNull(aCodeSet);
169 		Assert.assertEquals("CITY CODES", codeSet.getCodeSetName());
170 		Assert.assertEquals("HAIR CODES", aCodeSet.getCodeSetName());
171 		ICodeSetElement codeSetElment = codeSet.getElementNamed("Sacramento");
172 		CodeSetElement cityCode = (CodeSetElement) codeSetElment;
173 		Assert.assertEquals("Sacramento", cityCode.getName());
174 		Assert.assertEquals("City of Sacramento", cityCode.getDescription());
175 		Assert.assertEquals("SA", cityCode.getCode());
176 		ICodeSetElement hairCode = aCodeSet.getElementNamed("Black");
177 		Assert.assertEquals("Black", hairCode.getName());
178 		Assert.assertEquals("Black Hair", hairCode.getDescription());
179 		Assert.assertEquals("BLK", hairCode.getCode());

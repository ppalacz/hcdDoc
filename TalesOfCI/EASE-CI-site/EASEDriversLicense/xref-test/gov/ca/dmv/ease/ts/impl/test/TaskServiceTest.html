321 	public void testExecuteValidateAutomatedReceipt() {
322 		ValidateAutomatedReceiptRequestTs tsRequest = new ValidateAutomatedReceiptRequestTs(
323 				VoucherFixtureFactory
324 						.createVouchersForValidateAutomatedReciept(),
325 				CommonFixtureFactory
326 						.createSequenceNumberForValidateAutomatedReciept(),
327 				UserContextFixtureFactory
328 						.createUserContextForValidateAutomatedReciept());
329 		ValidateAutomatedReceiptResponseTs tsResponse = fixture
330 				.execute(tsRequest);
331 		assertNotNull(tsResponse);
332 		if (!tsResponse.hasErrors()) {
333 			assertNotNull(tsResponse.getVoucherReturnCodes());
334 			assertTrue(tsResponse.getVoucherReturnCodes().size() > 0);
335 			VoucherReturnCode voucherReturnCode = tsResponse
336 					.getVoucherReturnCodes().get(0);
337 			assertNotNull(voucherReturnCode);
338 			Voucher voucher = voucherReturnCode.getValidatedVoucher();
339 			assertNotNull(voucherReturnCode.getValidatedVoucher());
340 			assertNotNull(voucher.getEmployee());
341 			assertNotNull(voucher.getEmployee().getEmployeeTechId());
342 			assertEquals("12", voucher.getEmployee().getEmployeeTechId()
343 					.toString());
344 			assertNotNull(voucher.getOffice());
345 			assertNotNull(voucher.getOffice().getOfficeId());
346 			assertEquals("130", voucher.getOffice().getOfficeId().toString());
347 			assertNotNull(voucher.getWorkDate());
348 			assertNotSame(voucher.getWorkDate(), new Date());
349 			assertEquals("9", voucher.getSequenceNumber().toString());
350 			VoucherReturnCode voucherReturnCode1 = tsResponse
351 					.getVoucherReturnCodes().get(1);
352 			Voucher voucher1 = voucherReturnCode1.getValidatedVoucher();
353 			assertNotNull(voucher1);
354 			assertNotNull(voucher1.getEmployee());
355 			assertNotNull(voucher1.getEmployee().getEmployeeTechId().toString());
356 			assertEquals("12", voucher1.getEmployee().getEmployeeTechId());
357 			assertNotNull(voucher1.getOffice());
358 			assertNotNull(voucher1.getOffice().getOfficeId());
359 			assertEquals("130", voucher1.getOffice().getOfficeId().toString());
360 			assertNotNull(voucher1.getWorkDate());
361 			assertNotSame(voucher1.getWorkDate(), new Date());
362 			assertEquals("10", voucher1.getSequenceNumber().toString());
363 			assertNotNull(voucherReturnCode.getReturnCode());
364 			assertNotNull(voucherReturnCode.getReturnCode().getCode());
365 			assertTrue(voucherReturnCode.getReturnCode().getCode().length() > 0);
366 			assertNotNull(voucherReturnCode.getErrorCode());
367 			assertNotNull(voucherReturnCode.getErrorCode().getCode());
368 			assertTrue(voucherReturnCode.getErrorCode().getCode().length() > 0);
369 			assertNotNull(voucherReturnCode1.getReturnCode());
370 			assertNotNull(voucherReturnCode1.getReturnCode().getCode());
371 			assertTrue(voucherReturnCode1.getReturnCode().getCode().length() > 0);
372 			assertNotNull(voucherReturnCode1.getErrorCode());
373 			assertNotNull(voucherReturnCode1.getErrorCode().getCode());
374 			assertTrue(voucherReturnCode1.getErrorCode().getCode().length() > 0);
375 		}

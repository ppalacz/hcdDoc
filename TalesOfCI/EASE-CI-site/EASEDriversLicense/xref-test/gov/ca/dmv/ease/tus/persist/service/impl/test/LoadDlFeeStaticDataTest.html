92  	/***
93  	 * Test retrieve all business objects of DL Fee Static Data type - No filter is set.
94  	 */
95  	@Test
96  	public void testRetrieveAllObjects() {
97  		PersistenceServiceRequestFactory factory = (PersistenceServiceRequestFactory) EaseApplicationContext
98  				.getApplicationContext().getBean(
99  						"persistenceServiceRequestFactory");
100 		IPersistenceServiceRequest request = factory
101 				.createRetrieveAllBusinessObjectsRequest(userContext,
102 						DlFeeStaticData.class);
103 		RetrieveAllBusinessObjectsResponse aResponse = (RetrieveAllBusinessObjectsResponse) request
104 				.execute();
105 		assertNotNull(aResponse);
106 		assertFalse(aResponse.hasErrors());
107 		assertEquals(dlFeeStaticDataList.size(), aResponse.getResults().size());
108 		DlFeeStaticData compareFeeObj = (DlFeeStaticData) dlFeeStaticDataList
109 				.get(0);
110 		DlFeeStaticData compareToFeeObj = (DlFeeStaticData) aResponse
111 				.getResults().get(0);
112 		assertEquals(compareFeeObj.getFee().doubleValue(), compareToFeeObj
113 				.getFee().doubleValue());
114 		assertEquals(compareFeeObj.getTypeOfApplication(), compareToFeeObj
115 				.getTypeOfApplication());
116 		//Test for Failure Scenario
117 		request = factory.createRetrieveAllBusinessObjectsRequest(userContext,
118 				null);
119 		assertTrue(request.execute().hasErrors());
120 	}

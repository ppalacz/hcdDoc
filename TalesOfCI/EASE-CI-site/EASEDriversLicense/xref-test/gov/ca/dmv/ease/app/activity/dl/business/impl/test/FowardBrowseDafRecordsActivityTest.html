79  	public void testInvokeBrowse() {

81  		fixture.setTransitionKey(key);
82  		TransitionsMap transitions = new TransitionsMap();
83  		transitions.addTransitions(key, new EndActivity());
84  		fixture.setTransitions(transitions);
85  		Action action = new Action();
86  		dafProcessContext.setSelectedAction(action);
87  		action.setKey(IDafAppConstants.TRANSITION_KEY_BROWSE);
88  		action.setName("Browse");
89  		dafProcessContext.setPageNumber(2);
90  		List <Transaction> transactions = new ArrayList <Transaction>();
91  		Transaction tx = new Transaction();
92  		tx.setTransactionIdentifier(Integer.parseInt(SELECTED_DAF_NUMBER));
93  		transactions.add(tx);

95  				transactions);
96  		expect(mockDafTaskService.execute(isA(GetDafReocrdsRequestTs.class)))

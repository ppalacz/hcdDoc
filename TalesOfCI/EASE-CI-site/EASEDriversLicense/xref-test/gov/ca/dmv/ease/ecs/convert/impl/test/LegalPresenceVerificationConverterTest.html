84  	 * @param documentId a Document ID
85  	 * @param documentType a Document Type
86  	 * @param aUserVerifiedBdlpVerificationCode a VErification Code
87  	 * @param passportNumber a Passport Number
88  	 * @param aDriverLicenceNumber a DriverLicence Number
89  	 * @param lastName a Last Name.
90  	 * @param firstName a First Name
91  	 * @param middleName a Middle Name
92  	 * @param birthDate a Birth Date
93  	 * @param expirationDate An Expiration Date
94  	 */
95  	private void createFixture(String documentId, String documentType,
96  			String aUserVerifiedBdlpVerificationCode, String passportNumber,
97  			String aDriverLicenceNumber, String lastName, String firstName,
98  			String middleName, String birthDate, String expirationDate) {
99  		CodeSetElement userVerifiedBdlpVerificationCode = new CodeSetElement();
100 		userVerifiedBdlpVerificationCode
101 				.setCode(aUserVerifiedBdlpVerificationCode);
102 		dlApplication.setNewUserVerifiedBdlpCode(userVerifiedBdlpVerificationCode);
103 		Person applicant = dlApplication.getApplicant();
104 		PersonName personName = applicant.getCurrentName();
105 		LegalPresenceDocument legalPresenceDocument;
106 		List <Document> documents = new ArrayList <Document>();
107 		if (passportNumber != null) {
108 			legalPresenceDocument = new Passport();
109 			((Passport) legalPresenceDocument)
110 					.setPassportNumber(passportNumber);
111 		}
112 		else {
113 			legalPresenceDocument = new LegalPresenceDocument();
114 		}
115 		documents.add(legalPresenceDocument);
116 		applicant.setDocuments(documents);
117 		CodeSetElement documentTypeCodeSet = new CodeSetElement();
118 		documentTypeCodeSet.setCode(documentType);
119 		legalPresenceDocument.setLegalPresenceDocumentType(documentTypeCodeSet);
120 		legalPresenceDocument.setDocumentId(documentId);
121 		DriverLicense driverLicense = applicant.getDriverLicense();
122 		driverLicense.setLicenseNumber(aDriverLicenceNumber);
123 		personName.setLastName(lastName);
124 		personName.setFirstName(firstName);
125 		personName.setMiddleName(middleName);
126 		applicant.setBirthDate(EcsConverterHelper.convertStringToDate(
127 				birthDate,
128 				EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY));
129 		legalPresenceDocument
130 				.setExpirationDate(EcsConverterHelper
131 						.convertStringToDate(
132 								expirationDate,
133 								EcsConverterHelper.FORMATTER_MM_SLASH_DD_SLASH_CCYY));
134 	}

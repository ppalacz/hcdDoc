85  	@SuppressWarnings("unchecked")
86  	public void testExecuteGetDafRecords() {
87  		List <Transaction> transactions = new ArrayList <Transaction>();
88  		Transaction transaction = new Transaction();
89  		transaction.setTransactionIdentifier(DAF_NUMBER);
90  		transactions.add(transaction);
91  		List aList = transactions;
92  		expect(
93  				mockPersistenceServiceRequestFactory
94  						.createRetrieveCriteriaObjectRequest(
95  								isA(IUserContext.class), isA(Class.class)))
96  				.andReturn(mockRetrieveCriteriaObjectRequest);
97  		RetrieveCriteriaObjectResponse perResponse = new RetrieveCriteriaObjectResponse(
98  				mockCriteria);
99  		expect(mockRetrieveCriteriaObjectRequest.execute()).andReturn(
100 				perResponse);
101 		expect(mockCriteria.addOrder(isA(Order.class))).andReturn(mockCriteria);
102 		expect(
103 				mockCriteria.setFirstResult(PAGE_NUMBER
104 						* RetrieveDafRecordsService.NUMBER_OF_RECORD_PER_PAGE))
105 				.andReturn(mockCriteria);
106 		expect(
107 				mockCriteria
108 						.setMaxResults(RetrieveDafRecordsService.NUMBER_OF_RECORD_PER_PAGE))
109 				.andReturn(mockCriteria);
110 		expect(mockCriteria.list()).andReturn(aList);
111 		replay(mockPersistenceServiceRequestFactory);
112 		replay(mockRetrieveCriteriaObjectRequest);
113 		replay(mockCriteria);
114 		GetDafRecordsRequestBus busRequest = new GetDafRecordsRequestBus(
115 				userContext, new Transaction(), PAGE_NUMBER);
116 		GetDafRecordsResponseBus busResponse = fixture.execute(busRequest);
117 		assertFalse(busResponse.hasErrors());
118 		verify(mockPersistenceServiceRequestFactory);
119 		verify(mockRetrieveCriteriaObjectRequest);
120 		verify(mockCriteria);
121 		assertEquals(busResponse.getTransactions(), transactions);
122 	}

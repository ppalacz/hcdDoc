91  		Person person = new Person();
92  		PersonName personName = new PersonName();
93  		personName.setFirstName(firstName);
94  		personName.setLastName(lastName);
95  		DriverLicense license = new DriverLicense();
96  		license.setLicenseNumber(licenseNumber);
97  		license.setLicenseState(licenseState);
98  		OutOfStateLicense outOfStateLicense = new OutOfStateLicense();
99  		outOfStateLicense.setLicenseNumber(outStateLicense);
100 		String DATE_FORMAT = "MM/dd/yyyy";
101 		SimpleDateFormat simpleDateFormat = new SimpleDateFormat(DATE_FORMAT);
102 		try {
103 			person.setBirthDate(simpleDateFormat.parse(birthDate));
104 		}
105 		catch (ParseException e) {
106 			e.printStackTrace();
107 		}
108 		person.setCurrentName(personName);
109 		person.setDriverLicense(license);
110 		person.getPreviousLicenses().add(outOfStateLicense);
111 		DlApplication dlApplication = new DlApplication();
112 		dlApplication.setApplicant(person);
113 		dlApplication.setLicense(license);
114 		ecsRequestFactory = new EcsRequestFactory();
115 		sendAndPdpsRequestEcs = ecsRequestFactory
116 				.createSendCdlisAndPdpsRequestForEcs(license, person,
117 						socialSecurityDocument, userContext);
118 		return sendAndPdpsRequestEcs;
119 	}

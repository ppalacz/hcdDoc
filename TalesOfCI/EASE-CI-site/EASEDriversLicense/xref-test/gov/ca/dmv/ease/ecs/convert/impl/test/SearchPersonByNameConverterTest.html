372 	/***
373 	 * Set the mailing address of a person to an existing list.
374 	 * 
375 	 * @return the mailing address
376 	 */
377 	public static void setMailingAddress(Person person, Address mailingAddress) {
378 		// If the mailing address is null then do nothing
379 		if (mailingAddress == null) {
380 			return;
381 		}
382 		List <Locator> locators = person.getLocators();
383 		if (locators != null && locators.size() > 0) {
384 			boolean exists = false;
385 			for (int index = 0; index < locators.size(); index++) {
386 				Address address = (Address) locators.get(index);
387 				if (address != null) {
388 					AddressType addressType = address.getAddressType();
389 					if (addressType != null
390 							&& ADDRESS_TYPE_MAIL.equals(addressType)) {
391 						locators.remove(index);
392 						locators.add(index, mailingAddress);
393 						exists = true;
394 						break;
395 					}
396 				} // if loop
397 			} // for loop
398 			if (!exists) {
399 				locators.add(mailingAddress);
400 			}
401 		}
402 		else {
403 			locators.add(mailingAddress);
404 		}
405 	}

233 	/***
234 	 * Formats the expected result.
235 	 * @param expectedResult
236 	 * @return
237 	 */
238 	private String formatExpectedResult(
239 			List <String> expectedInboundPayloadResponses) {
240 		StringBuffer strBuff = new StringBuffer();
241 		String ebcidic = null;
242 		String screen = null;
243 		String formattedScreen = null;
244 		try {
245 			for (String expectedResult : expectedInboundPayloadResponses) {
246 				ebcidic = new String(expectedResult.getBytes("Cp1047"));
247 				screen = Decoder3270.decode(ebcidic);
248 				formattedScreen = Decoder3270.formatScreen(screen);
249 				strBuff.append(formattedScreen).append(LINE_SEPARATOR);
250 			}
251 		}
252 		catch (UnsupportedEncodingException e) {
253 			throw new MessageConversionException(e);
254 		}
255 		String formatResponse = StringUtils
256 				.trimTrailingWhitespace(removeDuplicateHeader(strBuff));
257 		return formatResponse;
258 	}

260 	/***
261 	 * This method removes duplicate header in given input string. first line of input string treated as a header.
262 	 * 
263 	 * @param buffer the buffer
264 	 * 
265 	 * @return returnString the string
266 	 */
267 	private String removeDuplicateHeader(StringBuffer buffer) {
268 		String string = buffer.toString();
269 		if (isNullOrBlank(string)) {
270 			return string;
271 		}
272 		StringTokenizer tokenizer = new StringTokenizer(string, LINE_SEPARATOR);
273 		int headerLengh = tokenizer.nextToken().length();
274 		String header = string.substring(0, headerLengh);
275 		String body = string.replace(header, "");
276 		StringBuffer stringBuffer = new StringBuffer();
277 		if (!body.contains(header)) {
278 			stringBuffer.append(header).append(LINE_SEPARATOR);
279 		}
280 		stringBuffer.append(body);
281 		String returnString = stringBuffer.toString();
282 		return returnString;
283 	}

168 	/***
169 	 * This method removes duplicate header in given input string. first line of input string treated as a header.
170 	 * 
171 	 * @param buffer the buffer
172 	 * 
173 	 * @return returnString the string
174 	 */
175 	public String removeDuplicateHeader(StringBuffer buffer) {
176 		String string = buffer.toString();
177 		if (isNullOrBlank(string)) {
178 			return string;
179 		}
180 		StringTokenizer tokenizer = new StringTokenizer(string, LINE_SEPARATOR);
181 		int headerLengh = tokenizer.nextToken().length();
182 		String header = string.substring(0, headerLengh);
183 		String body = string.replace(header, "");
184 		StringBuffer stringBuffer = new StringBuffer();
185 		if (!body.contains(header)) {
186 			stringBuffer.append(header).append(LINE_SEPARATOR);
187 		}
188 		stringBuffer.append(body);
189 		String returnString = stringBuffer.toString();
190 		return returnString;
191 	}

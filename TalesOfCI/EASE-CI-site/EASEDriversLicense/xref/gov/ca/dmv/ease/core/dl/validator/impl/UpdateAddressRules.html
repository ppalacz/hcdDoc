28  public class UpdateAddressRules extends AbstractValidateRule {
29  	/* (non-Javadoc)
30  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
31  	 */
32  	public void validate(ProcessContext processContext) {
33  		CdlProcessContext cdlProcessContext = (CdlProcessContext) processContext;
34  		IErrorCollector errorCollector = new ErrorCollector();
35  		Address mailingAddress = null;
36  		Address residentialAddress = null;
37  		if (!ValidationRulesUtil.isNullorEmpty(cdlProcessContext
38  				.getDlApplicant())) {
39  			mailingAddress = cdlProcessContext.getDlApplicant()
40  					.getExistingAddress(AddressType.MAILING);
41  			residentialAddress = cdlProcessContext.getDlApplicant()
42  					.getExistingAddress(AddressType.RESIDENTIAL);
43  		}
44  		//Validations
45  		if ((!ValidationRulesUtil.isNullorEmpty(mailingAddress)
46  				&& !ValidationRulesUtil.isNullorEmpty(mailingAddress
47  						.getServiceVerificationCode()) && ValidationRulesUtil
48  				.isNullorEmpty(mailingAddress.getServiceVerificationCode()
49  						.getCode()))) {
50  			/*ERROR CODE - D696 - it receives a response from the back end 
51  			 * system for a ZIP/County Code Inquiry request sent () */
52  			errorCollector.register(new ApplicationException("D696"));
53  		}
54  		if ((!ValidationRulesUtil.isNullorEmpty(residentialAddress)
55  				&& !ValidationRulesUtil.isNullorEmpty(residentialAddress
56  						.getServiceVerificationCode()) && ValidationRulesUtil
57  				.isNullorEmpty(residentialAddress.getServiceVerificationCode()
58  						.getCode()))) {
59  			/*ERROR CODE - D696 - it receives a response from the back end 
60  			 * system for a ZIP/County Code Inquiry request sent () */
61  			errorCollector.register(new ApplicationException("D696"));
62  		}
63  		if (!ValidationRulesUtil.isNullorEmpty(mailingAddress)
64  				&& !ValidationRulesUtil.isNullorEmpty(mailingAddress
65  						.getServiceVerificationCode())
66  				&& mailingAddress.getServiceVerificationCode().getCode()
67  						.equalsIgnoreCase("1")) {
68  			/*ERROR CODE - D720 - the response received from the ZIP/COUNTY CODE 
69  			 * Inquiry, that the request was a ZIP Code request and not a County 
70  			 * Code request. The System shall also interpret the inquiry response 
71  			 * return code and display an error message if the Residence City Name 
72  			 * entered for the inquiry was invalid according to the City and 
73  			 * County Code Book ()*/
74  			errorCollector.register(new ApplicationException("D720"));
75  		}
76  		if (!ValidationRulesUtil.isNullorEmpty(residentialAddress)
77  				&& !ValidationRulesUtil.isNullorEmpty(residentialAddress
78  						.getServiceVerificationCode())
79  				&& residentialAddress.getServiceVerificationCode().getCode()
80  						.equalsIgnoreCase("1")) {
81  			/*ERROR CODE - D721 for Residential address */
82  			errorCollector.register(new ApplicationException("D721"));
83  		}
84  		//Setting the error collector
85  		setErrorCollector(errorCollector);
86  	}
87  }

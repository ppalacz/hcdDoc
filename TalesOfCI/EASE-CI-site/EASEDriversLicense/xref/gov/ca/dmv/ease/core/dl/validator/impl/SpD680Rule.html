14  /***
15   * Description: The name must be shortened to two 27-position
16   * lines to be printed on the photo document. This
17   * may be done by changing the middle name to an
18   * initial, dropping the suffix, and/or changing the
19   * first name to an initial.
20   * If one word of the name is longer than 27
21   * positions or if the middle name is to be divided
22   * to print on both lines rather than be shortened to
23   * an initial, key a slash (/) where that word should
24   * be divided for printing.
25   * File: SpD680Rule.java
26   * Module:  gov.ca.dmv.ease.core.dl.validator.impl
27   * Created: Aug 5, 2010 
28   * @author MWCYL  
29   * @version $Revision: 1.2 $
30   * Last Changed: $Date: 2010/08/14 18:16:07 $
31   * Last Changed By: $Author: mwyxg1 $
32   */
33  public class SpD680Rule extends AbstractSpRule {
34  	/* (non-Javadoc)
35  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
36  	 */
37  	public void validate(ProcessContext processContext) {
38  		int fNameLen = 0;
39  		int lNameLen = 0;
40  		int mNameLen = 0;
41  		int suffixLen = 0;
42  		Person applicant = getApplicant(processContext);
43  		if (applicant == null)
44  			return;
45  		PersonName name = applicant.getCurrentName();
46  		fNameLen = name.getFirstName() != null ? name.getFirstName().trim()
47  				.length() : 0;
48  		lNameLen = name.getLastName() != null ? name.getLastName().trim()
49  				.length() : 0;
50  		mNameLen = name.getMiddleName() != null ? name.getMiddleName().trim()
51  				.length() : 0;
52  		suffixLen = name.getSuffix() != null ? name.getSuffix().trim().length()
53  				: 0;
54  		// TODO: This error does not really make sense in the long run, 
55  		// just coding for the maximum length for now. Print services will 
56  		// be able to handle printing longer names appropriately. 
57  		// if the name is > 27 * 2 then throw the error
58  		if (fNameLen + lNameLen + mNameLen + suffixLen > 54) {
59  			IErrorCollector errorCollector = new ErrorCollector();
60  			errorCollector
61  					.register(new ApplicationException(
62  							IDataCollectionErrorMessageConstants.NAME_MUST_BE_SHORTENED_OR_DIVIDED_FOR_PRINTING));
63  			setErrorCollector(errorCollector);
64  		}
65  	}
66  }

38  	private static final int CASH_REFUND_ALERT_AMOUNT = 20;
39  	private static final int HEAD_QUARTER_ALERT_AMOUNT = 100;

61  	/* (non-Javadoc)
62  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
63  	 */
64  	public void validate(ProcessContext processContext) {
65  		IErrorCollector errorCollector = new ErrorCollector();
66  		DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) processContext;
67  		Invoice invoice = dlFeeProcessContext.getDlApplication().getInvoice();
68  		UserContext userContext = (UserContext) dlFeeProcessContext
69  				.getUserContext();
70  		Action selectedAction = dlFeeProcessContext.getSelectedAction();
71  		if (selectedAction != null
72  				&& selectedAction.equals(getActionsRegistry().getAction(
73  						IActionsRegistry.ENTER_ACTION))) {
74  			String mode = userContext.getOperationalMode().getCode();
75  			double invoiceTotal = invoice.getInvoiceTotal().doubleValue();
76  			double netPayment = invoice.getNetAmount().doubleValue();
77  			double totalPayment = invoice.getPaymentTotal().doubleValue();
78  			double manualPayment = invoice.getTotaDebitPaymentsAmount()
79  					.doubleValue();
80  			double cashPayment = 0.0d;
81  			if (invoice.getCashPayment() != null
82  					&& invoice.getCashPayment().getAmount() != null) {
83  				cashPayment = invoice.getCashPayment().getAmount()
84  						.doubleValue();
85  			}
86  			double checkPayment = invoice.getTotalChecksAmount().doubleValue();
87  			double creditPayment = 0.0d;
88  			if (invoice.getCreditPayment() != null
89  					&& invoice.getCreditPayment().getAmount() != null) {
90  				creditPayment = invoice.getCreditPayment().getAmount()
91  						.doubleValue();
92  			}
93  			double vouchersPayment = invoice.getTotalVouchersAmount()
94  					.doubleValue();
95  			double nonCashPayment = manualPayment + checkPayment
96  					+ creditPayment;
97  			String feeActionCode = null;
98  			if (invoice.getFeeRefundActionCode() != null) {
99  				feeActionCode = invoice.getFeeRefundActionCode().getCode();
100 			}
101 			//F025
102 			if (IEaseModes.MAIL.equals(mode)) {
103 				errorCollector
104 						.register(new ApplicationException(
105 								IFeeAndPaymentErrorMessageConstants.CAN_ONLY_ISSUE_CHANGE_IN_COUNTER_MANUAL_OR_TRAVEL_CREW_MODE));
106 			}
107 			else {
108 				//F021
109 				if (EaseUtil.isNullOrBlank(feeActionCode)) {
110 					errorCollector
111 							.register(new ApplicationException(
112 									IFeeAndPaymentErrorMessageConstants.MUST_SPECIFY_HOW_TO_HANDLE_NET_OVER_OR_UNDER));
113 				}
114 				else if (IDlMessageConstants.HEAD_QUARTERS_REFUND
115 						.equals(feeActionCode)) {
116 					//F027
117 					if (!hasPassword(dlFeeProcessContext)) {
118 						if (vouchersPayment > HEAD_QUARTER_ALERT_AMOUNT
119 								|| cashPayment > HEAD_QUARTER_ALERT_AMOUNT
120 								|| nonCashPayment > HEAD_QUARTER_ALERT_AMOUNT
121 								|| netPayment > HEAD_QUARTER_ALERT_AMOUNT) {
122 							errorCollector
123 									.register(new ApplicationException(
124 											IFeeAndPaymentErrorMessageConstants.REQUIRES_APPROVAL_AUTHORITY_PASSWORD_ON_HQS_REFUND_OVER_$100_00));
125 						}
126 					}
127 					//F026
128 					if (netPayment < 0) {
129 						errorCollector
130 								.register(new ApplicationException(
131 										IFeeAndPaymentErrorMessageConstants.CAN_ONLY_ISSUE_CHANGE_ON_OVERPAYMENT));
132 					}
133 				}
134 				else if (IDlMessageConstants.CASH.equals(feeActionCode)) {
135 					//F027
136 					if (!hasPassword(dlFeeProcessContext)) {
137 						if (vouchersPayment > CASH_REFUND_ALERT_AMOUNT
138 								&& vouchersPayment <= HEAD_QUARTER_ALERT_AMOUNT) {
139 							errorCollector
140 									.register(new ApplicationException(
141 											IFeeAndPaymentErrorMessageConstants.REQUIRES_APPROVAL_AUTHORITY_PASSWORD_ON_HQS_REFUND_OVER_$100_00));
142 						}
143 						if (nonCashPayment > CASH_REFUND_ALERT_AMOUNT
144 								&& nonCashPayment <= HEAD_QUARTER_ALERT_AMOUNT) {
145 							errorCollector
146 									.register(new ApplicationException(
147 											IFeeAndPaymentErrorMessageConstants.REQUIRES_APPROVAL_AUTHORITY_PASSWORD_ON_HQS_REFUND_OVER_$100_00));
148 						}
149 					}
150 					if (manualPayment >= invoiceTotal) {
151 						//F041
152 						if (cashPayment > 0) {
153 							errorCollector
154 									.register(new ApplicationException(
155 											IFeeAndPaymentErrorMessageConstants.CASH_AMT_INVALID_WHEN_RECEIPT_TOTAL_GT_ADJ_FEE_DUE));
156 						}
157 						//F039
158 						else if (checkPayment > 0) {
159 							errorCollector
160 									.register(new ApplicationException(
161 											IFeeAndPaymentErrorMessageConstants.CHECK_AMT_INVALID_WHEN_RECEIPT_TOTAL_GT_ADJ_FEE_DUE));
162 						}
163 					}
164 					else {
165 						//F029
166 						if (totalPayment > HEAD_QUARTER_ALERT_AMOUNT) {
167 							errorCollector
168 									.register(new ApplicationException(
169 											IFeeAndPaymentErrorMessageConstants.CAN_ONLY_ISSUE_HQS_REFUND_ON_OVER_PAYMENT));
170 						}
171 						//F023
172 						/*if (netPayment > HEAD_QUARTER_ALERT_AMOUNT) {
173 							errorCollector.register(new ApplicationException(
174 									IFeeAndPaymentErrorMessageConstants.CANNOT_ISSUE_CHANGE_REQUEST_HEADQUARTERS_REFUND));
175 						}
176 						//F022
177 						else if (netPayment > CASH_REFUND_ALERT_AMOUNT) {
178 							errorCollector.register(new ApplicationException(
179 									IFeeAndPaymentErrorMessageConstants.REQUIRES_APPROVAL_AUTHORITY_PASSWORD_FOR_CHANGE_OVER_$20_00));
180 						}*/
181 					}
182 				}
183 			}
184 		}
185 		setErrorCollector(errorCollector);
186 	}

188 	/***
189 	 * @param processContext
190 	 * @return
191 	 */
192 	private boolean hasPassword(DlFeeProcessContext dlFeeProcessContext) {
193 		boolean hasPassword = false;
194 		String userName = null;
195 		if (dlFeeProcessContext.getAuthorizingUserId() != null) {
196 			userName = dlFeeProcessContext.getAuthorizingUserId().toLowerCase();
197 		}
198 		String password = dlFeeProcessContext.getAuthorizingPassword();
199 		if (userName != null || password != null) {
200 			try {
201 				if (getAuthAndAuthService().isAuthorizedApprover(userName,
202 						password)) {
203 					hasPassword = true;
204 				}
205 			}
206 			catch (LoginException e) {
207 			}
208 		}
209 		return hasPassword;
210 	}

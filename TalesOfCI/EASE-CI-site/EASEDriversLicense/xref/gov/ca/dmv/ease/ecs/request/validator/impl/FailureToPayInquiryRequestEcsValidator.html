26  public class FailureToPayInquiryRequestEcsValidator extends EcsRequestValidator {
27  	/*** The Constant serialVersionUID. */
28  	private static final long serialVersionUID = 7109344413801420479L;
29  	/*** The DLL TCODE. */
30  	private static final String DLL_TCODE = "DLL";
31  	/*** The Request. */
32  	private static final String REQUEST = "FailureToPayInquiryRequest";
33  	/*** The Converter. */
34  	private static final String CONVERTER = "FailureToPayInquiryConverter";
35  	/*** The TTC. */
36  	private static final String TTC = "FCP";
37  
38  	/***
39  	 * Constructor.
40  	 */
41  	public FailureToPayInquiryRequestEcsValidator() {
42  		super();
43  	}
44  
45  	/* (non-Javadoc)
46  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
47  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
48  	 */
49  	@Override
50  	protected void validateRequest(IEcsRequest request,
51  			IErrorCollector aCollector) {
52  		if (request == null) {
53  			this.addValidationExceptionInErrorCollector(NULL_REQUEST,
54  					new String[] { DLL_TCODE }, aCollector);
55  		}
56  		else {
57  			if (!(request instanceof FailureToPayInquiryRequestEcs)) {
58  				this.addValidationExceptionInErrorCollector(
59  						IMPROPER_REQUEST_TYPE, new String[] { REQUEST },
60  						aCollector);
61  			}
62  			else {
63  				if (!(request.getMessageConverter() instanceof FailureToPayInquiryConverter)) {
64  					this.addValidationExceptionInErrorCollector(
65  							IMPROPER_MESSAGE_CONVERTER_ERROR_MESSAGE,
66  							new String[] { CONVERTER }, aCollector);
67  				}
68  				if (!(StringUtils
69  						.equals(request.getUserContext().getTtc(), TTC))) {
70  					this.addValidationExceptionInErrorCollector(
71  							INVALID_TTC_ERROR_MESSAGE, new String[] { TTC },
72  							aCollector);
73  				}
74  			}//End of Check for Request
75  		}//End of Check for Request Type
76  	}
77  }

146 	/***
147 	 * 
148 	 * Gets the employer name.
149 	 * 
150 	 * @param employer the employer
151 	 * 
152 	 * @return the employer name
153 	 */
154 	private String getEmployerName(CustomerOrganization employer) {
155 		String employerName = BLANK;
156 		if (!isNull(employer)) {
157 			String employerLicenseNumber = employer.getLicenseNumber();
158 			if (!isNull(employerLicenseNumber)) {
159 				if (employerLicenseNumber
160 						.equals(IPrintConverterConstants.EMPLOYER_LIC_NUMBER_99999)
161 						|| employerLicenseNumber.matches("[0]*")) {
162 					employerName = IPrintConverterConstants.EMPLOYER_NAME_WITH_LIC_NUMBER_99999;
163 				}
164 				else {
165 					OrganizationName organizationName = employer
166 							.getCurrentName();
167 					if (!isNull(organizationName)
168 							&& !isNull(organizationName.getName())) {
169 						employerName = organizationName.getName();
170 					}
171 				}
172 			}
173 		}
174 		return employerName;
175 	}

178 	 * Gets the employer state.
179 	 * 
180 	 * @param employer the employer
181 	 * 
182 	 * @return the employer state
183 	 */
184 	private String getEmployerState(CustomerOrganization employer) {
185 		String employerSate = BLANK;
186 		if (!isNull(employer)) {
187 			Address employerAddress = employer.getAddress(AddressType.MAILING);
188 			if (!isNull(employerAddress)) {
189 				String employerLicNumber = employer.getLicenseNumber();
190 				if (!isNull(employerLicNumber)) {
191 					if (employerLicNumber
192 							.equals(IPrintConverterConstants.STATE_99999)
193 							|| employerLicNumber.matches("[0]*")) {
194 						employerSate = IPrintConverterConstants.STATE_CA;
195 					}
196 					else {
197 						String state = employerAddress.getState();
198 						if (!isNull(state)) {
199 							employerSate = state;
200 						}
201 					}
202 				}
203 			}
204 		}
205 		return employerSate;
206 	}

48  	@Override
49  	protected void invokeService(ProcessContext context) {
50  		OliProcessContext oliProcessContext = (OliProcessContext) context;
51  		DlTaskServiceRequestFactory dlTaskReqFactory = new DlTaskServiceRequestFactory();
52  		GetDatabaseStatusRequestTs request = dlTaskReqFactory
53  				.createDatabaseStatusInquiryRequest(oliProcessContext
54  						.getUserContext());
55  		GetDatabaseStatusResponseTs response = occupationalLicensingInquiryTaskService
56  				.execute(request);
57  		if (response.getErrorCollector() != null
58  				&& response.getErrorCollector().hasErrors()) {
59  			this.populateErrorMessages(oliProcessContext, response
60  					.getErrorCollector());
61  		}
62  		else {
63  			String databaseStatus = response.getDatabaseStatus();
64  			oliProcessContext.setDataStream(databaseStatus);
65  		}
66  	}
67  }

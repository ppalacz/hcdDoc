51  	@Override
52  	public String createMessage(IEcsRequest requestFromClient) {
53  		ReturnDriverInformationAndFeeIndicatorRequestEcs request = (ReturnDriverInformationAndFeeIndicatorRequestEcs) requestFromClient;
54  		UserContext userContext = (UserContext) request.getUserContext();
55  		String ttc = userContext.getTtc();
56  		String officeId = userContext.getOfficeId();
57  		String employeeId = userContext.getTechId();
58  		Person person = request.getPerson();
59  		String dlIdNumber = ConverterHelper.getDlorIdNumber(person);
60  		String lastName = ConverterHelper.getFirstThreeCharsOfLastName(person);
61  		String dateFeesReceived = convertDateToString(request
62  				.getDateFeesReceived());
63  		Object[] message = new Object[15];
64  		message[0] = IEcsConverterConstants.TCODE_DS5;
65  		message[1] = filler(ttc, 3);
66  		message[2] = filler(IEcsConverterConstants.TRANSACTION_SUB_CODE_I, 1);
67  		message[3] = filler(IEcsConverterConstants.REQUESTER_CODE_PREFIX, 2);
68  		message[4] = filler(officeId, 3);
69  		message[5] = filler(employeeId, 2);
70  		message[6] = filler(dlIdNumber, 8);
71  		message[7] = filler(lastName, 3);
72  		message[8] = filler(IEcsConverterConstants.HEX_NULL_N, 1);
73  		message[9] = filler(getDrugCodeIndicator(ttc), 1);
74  		message[10] = filler(getPasCodeIndicator(ttc), 1);
75  		message[11] = filler(IEcsConverterConstants.SPACE, 1);
76  		message[12] = filler(IEcsConverterConstants.HEX_NULL_N, 1);
77  		message[13] = IEcsConverterConstants.HEX_FILL;
78  		message[14] = filler(dateFeesReceived, 6);
79  		return createMessageFormat(message);
80  	}

114 		if (ConverterHelper.isFeeIndicator(pasCodeFeeInd)) {
115 			addRateStatusToDlFee(
116 					pasCodeFeeInd,
117 					IEcsConverterConstants.PRELIMINARY_ALCOHOL_SCREENING_REISSUE_FEE_RATE_CODE,

119 		}
120 		String reissueFeeInd = anInboundMessagePayload.substring(28, 29);

133 			}
134 			else if (feeInd == 4) {
135 				code = IEcsConverterConstants.DRIVER_LICENSE_REISSUE_TYPE_04_FEE_RATE_CODE;
136 			}
137 			else if (feeInd == 5) {
138 				code = IEcsConverterConstants.DRIVER_LICENSE_REISSUE_TYPE_05_FEE_RATE_CODE;
139 			}
140 			FeeRateCode feeRateCode = new FeeRateCode();
141 			feeRateCode.setCode(code);
142 			DriverLicenseRateStatus dlRateStatus = new DriverLicenseRateStatus();
143 			dlRateStatus.setFeeRatecode(feeRateCode);
144 			dlFee.addDriverLicenseRateStatus(dlRateStatus);
145 		}
146 		String addRestFeeInd = anInboundMessagePayload.substring(29, 30);
147 		if (ConverterHelper.isFeeIndicator(addRestFeeInd)) {
148 			addRateStatusToDlFee(
149 					addRestFeeInd,
150 					IEcsConverterConstants.ADD_RESTRICTION_IGNITION_INTERLOCK_DEVICE_FEE_RATE_CODE,
151 					dlFee);

154 		invoiceItems.add(invoiceItem);
155 		invoice.setInvoiceItems(invoiceItems);
156 		Date dateFeesPaid = convertStringToDate(anInboundMessagePayload
157 				.substring(32, 38), EcsConverterHelper.FORMATTER_MM_DD_YY);
158 		response.setDateFeesPaid(dateFeesPaid);
159 		CodeSetElement cse = getCodeSetElement(
160 				ICodeSetNameConstants.TCODE_DS5_CODE, responseErrCode);
161 		response.setErrorCode(cse);
162 		if (IEcsConverterConstants.RES_ERR_CODE_00
163 				.equalsIgnoreCase(responseErrCode)
164 				|| IEcsConverterConstants.RES_ERR_CODE_R0
165 						.equalsIgnoreCase(responseErrCode)

168 			String dlNumber = anInboundMessagePayload.substring(40, 48);
169 			if (!IEcsConverterConstants.SPACE.equalsIgnoreCase(dlNumber
170 					.substring(0, 1))) {
171 				driverLicense.setLicenseNumber(dlNumber);
172 			}

175 			person.setCurrentName(createPersonName(names, personName));
176 			String addressLine = anInboundMessagePayload.substring(83, 118);
177 			String addressCity = anInboundMessagePayload.substring(118, 131);
178 			String addressZip = anInboundMessagePayload.substring(131, 136);
179 			Address mailingAddress = setMailingAddress(addressLine,

182 			String resLine = anInboundMessagePayload.substring(142, 177);

195 	 * Add DriverLicenseRateStatus to DriverLicensefee based on indicator value.
196 	 * 
197 	 * @param indicator
198 	 * @param rateCode
199 	 * @param dlFee
200 	 */
201 	private void addRateStatusToDlFee(String indicator, String rateCode,
202 			DriverLicenseFee dlFee) {
203 		if (!isHexNull(indicator)) {
204 			int feeInd = Integer.parseInt(indicator);
205 			if (feeInd == 0) {
206 				// In back end indicator 0 is equal to 10 (to keep it as a byte).
207 				feeInd = 10;
208 			}
209 			FeeRateCode feeRateCode = new FeeRateCode();
210 			feeRateCode.setCode(rateCode);
211 			for (int i = 0; i < feeInd; i++) {
212 				DriverLicenseRateStatus dlRateStatus = new DriverLicenseRateStatus();
213 				dlRateStatus.setFeeRatecode(feeRateCode);
214 				dlFee.addDriverLicenseRateStatus(dlRateStatus);
215 			}
216 		}
217 	}

252 	/***
253 	 * Parse the result.
254 	 * 
255 	 * @param names the names
256 	 * 
257 	 * @return personName the PersonName
258 	 */
259 	private PersonName createPersonName(String names, PersonName personName) {
260 		StringTokenizer token = new StringTokenizer(names);
261 		int count = 0;
262 		while (token.hasMoreElements()) {
263 			String str = ((String) token.nextElement()).trim();
264 			if (count == 0) {
265 				personName.setLastName(str);
266 			}
267 			else if (count == 1) {
268 				personName.setFirstName(str);
269 			}
270 			else if (count == 2) {
271 				if ((IEcsConverterConstants.SUFFIX_JR.equalsIgnoreCase(str))
272 						|| (IEcsConverterConstants.SUFFIX_SR
273 								.equalsIgnoreCase(str)
274 								|| (IEcsConverterConstants.SUFFIX_II
275 										.equalsIgnoreCase(str))
276 								|| (IEcsConverterConstants.SUFFIX_III
277 										.equalsIgnoreCase(str)) || (IEcsConverterConstants.SUFFIX_IV
278 								.equalsIgnoreCase(str)))) {
279 					personName.setSuffix(str);
280 				}
281 				else {
282 					personName.setMiddleName(str);
283 				}
284 			}
285 			else if (count == 3) {
286 				personName.setSuffix(str);
287 			}
288 			count++;
289 		}
290 		return personName;
291 	}

336 		String stateCode = city.substring(city.length() - 2, city.length());

339 		}

342 			stateCode = IEcsConverterConstants.STATE_CODE_04; // statecode = 04 for CA

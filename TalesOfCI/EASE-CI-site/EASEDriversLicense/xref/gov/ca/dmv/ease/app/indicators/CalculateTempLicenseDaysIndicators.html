28  public class CalculateTempLicenseDaysIndicators implements IndicatorsConstants{
29  	/***
30  	 * This method Calculates Temporary License Days.
31  	 * 
32  	 * @param originalTtc the Original TTC
33  	 * @param originalApplicationDate the Original Application Date
34  	 * @param suspenseDate the Suspense Date
35  	 * @param backgroundExpDate the Back Ground Check Expiration Date
36  	 * @param isPreviousPendingClassExist true if Previous Pending Class Exist
37  	 * @param backgroundCheckCode the Back Ground Check Code  //O,P,T
38  	 * @param dlExpirationDate the DL Expiration Date
39  	 * @param applicationClasses the List of Application Classes
40  	 * @param Hazmat Endorsement the Endorsement 
41  	 * @param Background Record Check Expiration 
42  	 * @param Background Record Check Indicator
43  	 * @param AbsentParentCondition the Absent Parent Condition
44  	 * @param Suspense Due Date
45  	 * @return the Temporary License Days
46  	 */
47  	public int calculateTempLicenseDays(String originalTtc,
48  			Date originalApplicationDate, boolean isPreviousPendingClassExist,
49  			CodeSetElement backgroundCheckCode, Date dlExpirationDate,
50  			List <CodeSetElement> applicationClasses, String endorsement,
51  			Date backgroundCheckExpirationDate,
52  			String backgroundCheckIndicator, boolean absentParentCondition,
53  			Date suspenseDueDate) {
54  		int tempLicenseDays = 0;
55  		String backgroundCheckIndicator_P = "P";
56  		String backgroundCheckIndicator_T = "T";
57  		String backgroundCheckIndicator_O = "O";
58  
59  		Date currentDate = new Date();
60  		//Scenario 1 - when Hazmat Endorsement is being applied for and there is an Absent Parent condition with a Suspense Date
61  		if (HAZARDOUS_MATERIAL.equalsIgnoreCase(endorsement)) {
62  			if ((absentParentCondition)
63  					&& (!EaseUtil.isNullOrBlank(suspenseDueDate))) {
64  				//For TTC - DLA/DLE
65  				if ((TTC_DLA.equalsIgnoreCase(originalTtc))
66  						|| (TTC_DLE.equalsIgnoreCase(originalTtc)))
67  				//Step 1 - If Suspense Date > Original Application Date + 1 Year then the Number of Temporary Days is set to 31.
68  				{
69  					if (suspenseDueDate.after(DateUtil.addToSpecifiedDate(
70  							originalApplicationDate, 1, 0, 0))) {
71  						tempLicenseDays = 31;
72  					}
73  					//Step 2- If Suspense Date less than or equal to 1 year past Original Application Date then the Number of Temporary Days is set to 61.
74  					else if (suspenseDueDate.before(DateUtil
75  							.addToSpecifiedDate(originalApplicationDate, 1, 0,
76  									0))) {
77  						tempLicenseDays = 61;
78  					}
79  					return tempLicenseDays;
80  				}
81  			}
82  			//	return tempLicenseDays;
83  		}
84  		//Scenario 2 - when Hazmat Endorsement  is being applied for in cases not handled above
85  		//If the Previous Pending Application Class is present
86  		if (HAZARDOUS_MATERIAL.equalsIgnoreCase(endorsement) && isPreviousPendingClassExist) {
87  			//And Background Record Check Expiration Date is present 
88  			if (!EaseUtil.isNullOrBlank(backgroundCheckExpirationDate)) {
89  				// and Expired
90  				if (backgroundCheckExpirationDate.before(DateUtil
91  						.addToSpecifiedDate(currentDate, 0, 0, 1)))
92  				//And the Background Record Check Indicator is P,T
93  				{
94  					if (backgroundCheckIndicator_P
95  							.equalsIgnoreCase(backgroundCheckIndicator)
96  							|| backgroundCheckIndicator_T
97  									.equalsIgnoreCase(backgroundCheckIndicator)) {
98  						//Then Set the Number of Temporary Days to 60
99  						tempLicenseDays = 60; //3
100 					}
101 					else {
102 						//else Set the Number of Temporary Days to 90   
103 						tempLicenseDays = 90; //4
104 					}
105 					return tempLicenseDays;
106 				}
107 				//Background Record Check Expiration Date is  Not Expired
108 				else {
109 					//And the Background Record Check Indicator is O
110 					if ((backgroundCheckIndicator_O
111 							.equalsIgnoreCase(backgroundCheckIndicator))) {
112 						//And the Days until Background Record check expires are Less than or Equal to 90
113 						if (backgroundCheckExpirationDate.before(DateUtil
114 								.addToSpecifiedDate(currentDate, 0, 0, 89))) {
115 							tempLicenseDays = 90; //5
116 						}
117 						// the Days until Background Record check expires are greater than 90
118 						else {
119 							tempLicenseDays = 60; //6
120 						}
121 					}
122 					//And the Background Record Check Indicator is Not O   	
123 					else {
124 						//And the Days until Background Record check expires are Less than 90
125 						if (backgroundCheckExpirationDate.before(DateUtil
126 								.addToSpecifiedDate(currentDate, 0, 0, 89))) {
127 							//Days until Background Record Check Expires ?????
128 							long milis1 = backgroundCheckExpirationDate
129 									.getTime();
130 							long milis2 = currentDate.getTime();
131 							tempLicenseDays = (int) ((milis1 - milis2) / (24 * 60 * 60 * 1000)); //7
132 						}
133 						else {
134 							// else Set the Number of Temporary Days to 90   //8
135 							tempLicenseDays = 90;
136 						}
137 					} //else 
138 				}
139 			}
140 			//If the Previous Pending Application Class is present and Background Record Check Expiration Date is not present          
141 			else {
142 				if (EaseUtil.isNullOrBlank(backgroundCheckExpirationDate)) {
143 					tempLicenseDays = 90; //9
144 				}
145 			}
146 		}
147 		//If the Previous Pending Application Class is Not present
148 		else {
149 			//And the DL Expiration Date is Not present
150 			if (EaseUtil.isNullOrBlank(dlExpirationDate)) {
151 				//And the Application Class is D,E,F
152 				for (CodeSetElement applicationClass : applicationClasses) {
153 					if ((CLASS_D.equalsIgnoreCase(applicationClass
154 							.getCode()))
155 							|| (CLASS_E
156 									.equalsIgnoreCase(applicationClass
157 											.getCode()))
158 							|| (CLASS_F
159 									.equalsIgnoreCase(applicationClass
160 											.getCode()))) {
161 						tempLicenseDays = 60; //10
162 						break;
163 					}
164 					else
165 					// Application Class is  Not D,E,F And the Background Record Check Indicator is Not O
166 					{
167 						if ((!backgroundCheckIndicator_O
168 								.equalsIgnoreCase(backgroundCheckIndicator))) {
169 							tempLicenseDays = 90; //11
170 							break;
171 						}
172 						// Application Class is  Not D,E,F And the Background Record Check Indicator is O        			
173 						else //And the Days until Background Record check expires are is less than 90
174 						{
175 							if (backgroundCheckExpirationDate.before(DateUtil
176 									.addToSpecifiedDate(currentDate, 0, 0, 89))) {
177 								tempLicenseDays = 90; //12				 
178 							}
179 							else //And the Days until Background Record check expires are is Greater than or Equal to 90
180 							{
181 								tempLicenseDays = 60; //13   					 
182 							}
183 						}
184 					}
185 				}
186 			}
187 		}
188 		return tempLicenseDays;
189 	} //calculateTempLicenseDays
190 } //CalculateTempLicenseDaysIndicators

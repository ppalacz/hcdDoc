43  public class MedicalReportUpdateDocumentConverter extends
44  		AbstractDocumentConverter {
45  	/***
46  	 * Populates outgoing XML message for Medical Report Update.
47  	 * 
48  	 * @param requestFromClient the request from client
49  	 * 
50  	 * @return the string
51  	 */
52  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
53  		validateClientRequest(requestFromClient);
54  		MedicalReportUpdateDocumentRequestTus request = (MedicalReportUpdateDocumentRequestTus) requestFromClient;
55  		IUserContext userContext = request.getUserContext();
56  		PersonName personName = request.getPersonName();
57  		String dlNumber = request.getDlNumber();
58  		Date medicalExpirationDate = request.getMedicalExpirationDate();
59  		CodeSetElement attachmentCode = request.getAttachmentCode();
60  		// Setting xml elements
61  		List <String> elementsList = new ArrayList <String>();
62  		addToElementsList(elementsList, dlNumber, MED_REPORT_DL_NUMBER);
63  		addToElementsList(elementsList, personName.getFullName(),
64  				MED_REPORT_PERSON_NAME);
65  		addToElementsList(elementsList, convertDateToString(
66  				medicalExpirationDate, FORMATTER_MM_SLASH_DD_SLASH_YY),
67  				MED_REPORT_MED_EXP_DATE);
68  		addToElementsList(elementsList, attachmentCode.getCode(),
69  				MED_REPORT_ATTACH_CODE);
70  		addToElementsList(elementsList, MED_REPORT_MSG_MEDICAL_DATE_UPDATED,
71  				MED_REPORT_MSG_MEDICAL_DATE);
72  		addToElementsList(elementsList, userContext.getOfficeId(),
73  				MED_REPORT_OFFICE_ID);
74  		addToElementsList(elementsList, userContext.getTechId(),
75  				MED_REPORT_EMPLOYEE_ID);
76  		addToElementsList(elementsList, convertDateToString(new Date(),
77  				FORMATTER_MM_SLASH_DD_SLASH_YY), MED_REPORT_CURRENT_DATE);
78  		addToElementsList(elementsList, convertDateToString(new Date(),
79  				FORMATTER_HH_mm), MED_REPORT_CURRENT_TIME);
80  		String xmlData = createXmlForSingleDocument(elementsList,
81  				MED_REPORT_DOC_NAME);
82  		return xmlData;
83  	}
84  }

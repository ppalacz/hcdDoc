84  	public SsnInquiryProcessContext() {
85  		this(new DlApplication());
86  		getDlApplication().setNewLicenseNumberRequired(false);
87  		aniResults = new ArrayList <AniRecord>();
88  		Person person = new Person();
89  		person.setCurrentName(new PersonName());
90  		List <PersonName> personNames = new ArrayList <PersonName>();
91  		person.setPersonNames(personNames);
92  		person.setDriverLicense(new DriverLicense());
93  		person.getDriverLicense().setAppliedLicenseClasses(
94  				new ArrayList <CodeSetElement>());
95  		OutOfStateLicense outOfStateLicense = new OutOfStateLicense();
96  		outOfStateLicense.setName(new PersonName());
97  		person.setPreviousLicenses(new ArrayList <OutOfStateLicense>());
98  		person.addPreviousLicense(outOfStateLicense);
99  		getDlApplication().setApplicant(person);
100 		AniRecord aniRecord = new AniRecord();
101 		aniResults.add(aniRecord);
102 		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
103 		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
104 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
105 		legalPresenceDocument.setVerificationCode(new CodeSetElement());
106 		List <Document> list = new ArrayList <Document>();
107 		Passport passport = new Passport();
108 		list.add(ssnDocument);
109 		list.add(legalPresenceDocument);
110 		list.add(passport);
111 		person.setDocuments(list);
112 		getDlApplication().setApplicationTypeCode(
113 				new CodeSetElement("", "", ""));
114 		Address mailingAddress = new Address(AddressType.MAILING);
115 		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
116 		person.addAddress(mailingAddress);
117 		person.addAddress(residenceAddress);
118 		Invoice invoice = new Invoice();
119 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
120 		invoice.setInvoiceItems(items);
121 		List <Payment> payments = new ArrayList <Payment>();
122 		invoice.setPayments(payments);
123 		CashPayment cashPayment = new CashPayment();
124 		payments.add(cashPayment);
125 		CheckPayment checkPayment = new CheckPayment();
126 		payments.add(checkPayment);
127 		DebitPayment debitPayment = new DebitPayment();
128 		payments.add(debitPayment);
129 		getDlApplication().setInvoice(invoice);
130 		changedName = new PersonName();
131 	}

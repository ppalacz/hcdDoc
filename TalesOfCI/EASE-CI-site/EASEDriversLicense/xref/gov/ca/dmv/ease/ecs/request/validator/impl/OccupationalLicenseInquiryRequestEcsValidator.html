28  public class OccupationalLicenseInquiryRequestEcsValidator extends
29  		EcsRequestValidator {
30  	/*** The Converter. */
31  	private static final String CONVERTER = "OccupationalLicenseInquiryConverter";
32  	/*** The Request. */
33  	private static final String REQUEST = "OccupationalLicenseInquiryRequestEcs";
34  	/*** The Constant serialVersionUID. */
35  	private static final long serialVersionUID = 7336290598585640020L;
36  	/*** The Constant TCODE LI1. */
37  	private static final String TCODE_LI1 = "LI1";
38  	/*** The Constant TCODE LP5. */
39  	private static final String TCODE_LP5 = "LP5";
40  
41  	/* (non-Javadoc)
42  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
43  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
44  	 */
45  	@Override
46  	protected void validateRequest(IEcsRequest request,
47  			IErrorCollector aCollector) {
48  		if (!(request instanceof OccupationalLicenseInquiryRequestEcs)) {
49  			addValidationExceptionInErrorCollector(
50  					REQUEST_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
51  					new String[] { REQUEST }, aCollector);
52  			return;
53  		}
54  		if (!(request.getMessageConverter() instanceof OccupationalLicenseInquiryConverter)) {
55  			addValidationExceptionInErrorCollector(
56  					CONVERTER_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
57  					new String[] { CONVERTER }, aCollector);
58  			return;
59  		}
60  		OccupationalLicenseInquiryRequestEcs occupationalLicenseInquiryRequestEcs = (OccupationalLicenseInquiryRequestEcs) request;
61  		if (!EaseUtil.isNotNull(occupationalLicenseInquiryRequestEcs
62  				.getIsCustomerCopy())) {
63  			addValidationExceptionInErrorCollector(
64  					IS_CUSTOMER_COPY_IS_NULL_ERROR_MESSAGE, null,
65  					aCollector);
66  			return;
67  		}
68  /*		if (!(StringUtils.equals(occupationalLicenseInquiryRequestEcs
69  				.getTransactionCode(), TCODE_LI1) || StringUtils.equals(
70  				occupationalLicenseInquiryRequestEcs.getTransactionCode(),
71  				TCODE_LP5))) {
72  			addValidationExceptionInErrorCollector(
73  					INVALID_TRANSACTION_CODE_ERROR_MESSAGE, null, aCollector);
74  			return;
75  		}*/
76  		if (!EaseUtil.isNotNull(occupationalLicenseInquiryRequestEcs
77  				.getOccupationalLicenseInquiry())) {
78  			addValidationExceptionInErrorCollector(
79  					OCCUPATIONAL_LICENSE_INQUIRT_IS_EMPTY_ERROR_MESSAGE, null,
80  					aCollector);
81  			return;
82  		}
83  		OccupationalLicenseInquiryRequestType requestType = occupationalLicenseInquiryRequestEcs
84  				.getRequestTypeForOccupationalLicenseInquiry();
85  		if (!EaseUtil.isNotNull(requestType)) {
86  			addValidationExceptionInErrorCollector(
87  					OCCUPATIONAL_LICENSE_INQUIRT_REQUEST_TYPE_IS_EMPTY_ERROR_MESSAGE,
88  					null, aCollector);
89  			return;
90  		}
91  		if (requestType == OccupationalLicenseInquiryRequestType.REQUEST_BY_ALPHA_XREF) {
92  			if (EaseUtil.isNullOrBlank(occupationalLicenseInquiryRequestEcs
93  					.getOccupationalLicenseInquiry().getAlphaXrefFileCode())) {
94  				addValidationExceptionInErrorCollector(
95  						ALPHA_XREF_FILE_CODE_IS_EMPTY_ERROR_MESSAGE, null,
96  						aCollector);
97  				return;
98  			}
99  		}
100 		if (requestType == OccupationalLicenseInquiryRequestType.REQUEST_BY_FIRM) {
101 			if (EaseUtil.isNullOrBlank(occupationalLicenseInquiryRequestEcs
102 					.getOccupationalLicenseInquiry()
103 					.getSpecialFirmCategoryCode())
104 					&& !EaseUtil
105 							.isNullOrBlank(occupationalLicenseInquiryRequestEcs
106 									.getOccupationalLicenseInquiry()
107 									.getSpecialFirmBasicRecordCode())) {
108 				addValidationExceptionInErrorCollector(
109 						SPECIAL_FIRM_CATEGORY_CODE_IS_EMPTY_ERROR_MESSAGE,
110 						null, aCollector);
111 				return;
112 			}
113 		}
114 	}
115 }

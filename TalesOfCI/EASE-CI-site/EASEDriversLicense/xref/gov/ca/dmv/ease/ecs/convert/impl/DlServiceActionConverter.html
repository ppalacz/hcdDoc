29  public class DlServiceActionConverter extends AbstractEcsMessageConverter {
30  	/*** The TCode. */
31  	private static final String TCODE = "DLS";
32  	/*** The Transaction Sub Type. */
33  	private static final String TRANSACTION_SUBTYPE = "U80";
34  	/*** The Type Code. */
35  	private static final String TYPE_CODE = "S";
36  	
37  	/* (non-Javadoc)
38  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractUpdateHeadquartersSystemConverter
39  	 * #createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
40  	 */
41  	@Override
42  	public String createMessage(IEcsRequest request) {
43  		DlServiceActionRequestEcs dlServiceActionRequestEcs = (DlServiceActionRequestEcs) request;
44  		IUserContext userContext = dlServiceActionRequestEcs.getUserContext();
45  		List <ServiceOrder> serviceOrderList = dlServiceActionRequestEcs
46  				.getServiceOrderList();
47  		String licenseNumber = dlServiceActionRequestEcs.getLicenseNumber();
48  		String lastName = dlServiceActionRequestEcs.getLastName();
49  		CodeSetElement locationCode = dlServiceActionRequestEcs.getLocationCode();
50  		Object[] values = new Object[34];
51  		values[0] = TCODE; 															//MLSTCODE	3	HQ TRANSACTION CODE
52  		values[1] = getTtc(userContext); 											//MLSTTC	3	TYPE TRANSACTION CODE
53  		values[2] = TRANSACTION_SUBTYPE; 											//MLSTRNSC	1	TRANSACTION SUB-CODE
54  		values[3] = getOfficeId(userContext); 										//MLSOFFID	3	OFFICE ID (REQUESTER CODE SUFFIX)
55  		values[4] = getEmployeeID(userContext); 									//MLSEMPID	2	EMPLOYEE ID
56  		values[5] = filler(licenseNumber, 8); 										//MLSDLID#	8	DL/ID NUMBER
57  		values[6] = lastName.substring(0, 3);										//MLS3LNME	3	FIRST 3 POSITIONS LAST NAME;
58  		values[7] = TYPE_CODE; 														//MLSTPCDE	1	TYPE CODE
59  		//Process Service Orders
60  		int index = 8;
61  		for (ServiceOrder serviceOrder : serviceOrderList) {
62  			this.processServiceOrder(serviceOrder, values, index);
63  			index = index + 5;
64  		}
65  		values[33] = getCodeSetElementValue(locationCode, 2);						//MLSLILOC	2	LICENSE LOCATION CODE
66  		return this.createMessageFormat(values);
67  	}
68  
69  	/***
70  	 * This method processes Service Order.
71  	 * 
72  	 * @param serviceOrder the Service Order
73  	 * @param values the values
74  	 * @param index the index
75  	 */
76  	private void processServiceOrder(ServiceOrder serviceOrder,
77  			Object[] values, int index) {
78  		values[index++] = getCodeSetElementValue(serviceOrder.getActionCode(),
79  				2); 																//MLSACTCD	2	TYPE ACTION CODE
80  		values[index++] = getCodeSetElementValue(serviceOrder
81  				.getActionReasonCode(), 3); 										//MLSREACD	3	ACTION REASON CODE
82  		values[index++] = filler(convertDateToString(serviceOrder
83  				.getEffectiveDate()), 6); 											//MLSEFFDT	6	EFFECTIVE DATE 
84  		values[index++] = getCodeSetElementValue(serviceOrder.getServiceCode(),
85  				1); 																//MLSSERCD	1	SERVICE CODE
86  		values[index++] = filler(convertDateToString(serviceOrder
87  				.getServiceDate()), 6); 											//MLSSERDT	6	SERVICE DATE 
88  	}

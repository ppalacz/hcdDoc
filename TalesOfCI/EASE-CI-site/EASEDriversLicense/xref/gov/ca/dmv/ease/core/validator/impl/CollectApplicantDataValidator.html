34  public class CollectApplicantDataValidator extends AbstractBusValidator {
35  	/***The constant BD/LP verification code*/
36  	private static final String BD_LP_VERIF_CODE_I = "I";
37  	/***The constant BD/LP verification code*/
38  	private static final String BD_LP_VERIF_J = "J";
39  	/***The constant BD/LP verification code*/
40  	private static final String BD_LP_VERIF_CODE_K = "K";
41  	/***The constant BD/LP verification code*/
42  	private static final String BD_LP_VERIF_CODE_L = "L";
43  	/***The constant BD/LP verification code*/
44  	private static final String BD_LP_VERIF_CODE_M = "M";
45  	/***The constant BD/LP verification code*/
46  	private static final String BD_LP_VERIF_CODE_Q = "Q";
47  	/***The constant BD/LP verification code*/
48  	private static final String BD_LP_VERIF_CODE_S = "S";
49  	/***The constant BD/LP verification code*/
50  	private static final String BD_LP_VERIF_CODE_V = "V";
51  	/***The constant BD/LP verification code*/
52  	private static final String BD_LP_VERIF_CODE_W = "W";
53  	/***The constant BD/LP verification code*/
54  	private static final String BD_LP_VERIF_CODE_X = "X";
55  	/*** The Constant serialVersionUID. */
56  	private static final long serialVersionUID = -2863057429226371165L;
57  
58  	/***
59  	 * This method validates DlApplication <code>DlApplication</code>.
60  	 * 
61  	 * @param validatables The DlApplication to validate
62  	 * @return <code>IErrorCollector</code> containing Validation Messages.
63  	 * @see IErrorCollector
64  	 * @see DlApplication
65  	 */
66  	@Override
67  	public void validateValidatable(IValidatable validatable,
68  			IErrorCollector collector) {
69  		try {
70  			if (validatable instanceof DlApplication) {
71  				DlApplication dlApplication = (DlApplication) validatable;
72  				//Validation to check Driver Education and Training is complete in case of a minor - Start
73  				String strDeNumber = dlApplication
74  						.getDriverEducationRecordNumber();
75  				String strDtNumber = dlApplication
76  						.getDriverTrainingRecordNumber();
77  				String strBdLpReq = dlApplication.getLegalPresenceDocument()
78  						.getVerificationCode().getCode();
79  				String strDocumentNumber = dlApplication
80  						.getLegalPresenceDocument().getDocumentId();
81  				Date strDocumentExpiration = dlApplication
82  						.getLegalPresenceDocument().getExpirationDate();
83  				Address mailingAddress = DummySubjectHelper
84  						.getMailingAddress(null);
85  				Address residenceAddress = DummySubjectHelper.getResidenceAddress(dlApplication.getApplicant());
86  				if (dlApplication.getApplicant().getMinor()
87  						&& (isNullOrBlank(strDeNumber) || isNullOrBlank(strDtNumber))) {
88  					collector
89  							.register(new EaseValidationException(
90  									SpringUtils
91  											.getMessage(DRIVER_EDUCATION_TRAINING_NUMBER_IS_INVALID_ERROR_MESSAGE)));
92  				}//Validation to check Driver Education and Training is complete in case of a minor - End
93  				//Validation to check whether INS# is provided when BD/LP code selected is  (I,J,K,L,M,Q,S,V,W,C) - Start
94  				if ((!isNullOrBlank(strBdLpReq) && (strBdLpReq
95  						.equalsIgnoreCase(BD_LP_VERIF_CODE_I)
96  						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_J)
97  						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_K)
98  						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_L)
99  						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_M)
100 						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_Q)
101 						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_S)
102 						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_V)
103 						|| strBdLpReq.equalsIgnoreCase(BD_LP_VERIF_CODE_W) || strBdLpReq
104 						.equalsIgnoreCase(BD_LP_VERIF_CODE_X)))
105 						&& (isNullOrBlank(strDocumentNumber) || isNullOrBlank(strDocumentExpiration))) {
106 					collector
107 							.register(new EaseValidationException(
108 									SpringUtils
109 											.getMessage(LEGAL_PRESENCE_DOCUMENT_IS_INVALID_ERROR_MESSAGE)));
110 				}//Validation to check whether INS# is provided when BD/LP code selected is  (I,J,K,L,M,Q,S,V,W,C) - End
111 				//Validation to check Mailing address and Residence address are not same - Start
112 				if ((!isNullOrBlank(mailingAddress) && DummyLocatorHelper.isSameAs(mailingAddress, residenceAddress))) {
113 					collector.register(new EaseValidationException(SpringUtils
114 							.getMessage(ADDRESS_IS_INVALID_ERROR_MESSAGE)));
115 				}
116 				//Validation to check Mailing address and Residence address are not same - End
117 			}
118 		}
119 		catch (Exception e) {
120 			collector.register(e);
121 		}
122 	}
123 }

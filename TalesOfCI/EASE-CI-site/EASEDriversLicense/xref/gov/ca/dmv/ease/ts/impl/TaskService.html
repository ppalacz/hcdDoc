118 	/***
119 	 * This method invokes Approve Authorization Service.
120 	 * 
121 	 * @param tsRequest
122 	 *            the Approve Authorization request
123 	 * @return the Approve Authorization response
124 	 */
125 	public ApproveAuthorizationResponseTs execute(
126 			ApproveAuthorizationRequestTs tsRequest) {
127 		ApproveAuthorizationResponseTs tsResponse = null;
128 		ITaskServiceResponseFactory responseFactory = new TaskServiceResponseFactoryImpl();
129 		List <IAuthorizable> managers = new ArrayList <IAuthorizable>();
130 		IAuthAndAuthService authService = new AuthAndAuthService();
131 		boolean isAuthorized = false;
132 		IErrorCollector collector = new ErrorCollector();
133 		try {
134 			collector = tsRequest.validate();
135 			if (collector.hasErrors()) {
136 				tsResponse = responseFactory
137 						.createApproveAuthorizationResponseForTs(collector);
138 				return tsResponse;
139 			}
140 			if ((tsRequest.getRole() == null)
141 					|| (tsRequest.getRole().trim().length() == 0)) {
142 				isAuthorized = authService
143 						.isUserAuthorized(tsRequest.getRole());
144 				tsResponse = responseFactory
145 						.createApproveAuthorizationResponseForTs(isAuthorized);
146 			}
147 			else {
148 				// Load all the managers for the user and verify them using
149 				// Authorization Service:
150 				RetrieveBusinessObjectResponse perResponse = null;
151 				Office office = new Office();
152 				office.setOfficeId(tsRequest.getUserContext().getOfficeId());
153 				IPersistenceServiceRequest request = this.persistenceServiceRequestFactory
154 						.createRetrieveBusinessObjectRequest(tsRequest
155 								.getUserContext(), office);
156 				perResponse = (RetrieveBusinessObjectResponse) request
157 						.execute();
158 				if ((perResponse.getResults() != null)
159 						&& (perResponse.getResults().size() > 0)) {
160 					office = (Office) perResponse.getResults().get(0);
161 					for (IAuthorizable authorizable : office.getManagers()) {
162 						managers.add(authorizable);
163 					}
164 					isAuthorized = authService.authorizeManager(managers,
165 							tsRequest.getPassword());
166 				}
167 				tsResponse = responseFactory
168 						.createApproveAuthorizationResponseForTs(isAuthorized);
169 			}
170 		}
171 		catch (Exception e) {
172 			// Register exception in error collector and assign it to response
173 			// object.
174 			collector.register(e);
175 			tsResponse = responseFactory
176 					.createApproveAuthorizationResponseForTs(collector);
177 			return tsResponse;
178 		}
179 		return tsResponse;
180 	}

234 	 * Calls PrintService to print the requested documents. Returns the
235 	 * PrintDocumentsResponse type object, which gives the details of the print
236 	 * job.
237 	 * 
238 	 * @param request
239 	 *            may contain Person, Application, Receipt, Record and
240 	 *            Transaction as input parameter
241 	 * 
242 	 * @return the print documents response
243 	 */
244 	public GenerateDocumentResponseTs execute(GenerateDocumentRequestTs request) {
245 		IssueDlTaskService dlTaskService = new IssueDlTaskService();
246 		GenerateDocumentResponseTs docResponse = new GenerateDocumentResponseTs();
247 		if ((request.getCdlisRecord() != null)
248 				|| (request.getPdpsRecord() != null)) {
249 			DlTaskServiceRequestFactory factory = new DlTaskServiceRequestFactory();
250 			DlApplication dlApplication = request.getDlApplication();
251 			CdlisRecord cdlisRecord = request.getCdlisRecord();
252 			PdpsRecord pdpsRecord = request.getPdpsRecord();
253 			IUserContext userContext = request.getUserContext();
254 			DlCdlisReceiptRequestTs cdlisReceiptRequest = factory
255 					.createDlCdlisReceiptRequest(dlApplication, cdlisRecord,
256 							pdpsRecord, userContext);
257 			DlCdlisReceiptResponseTs response = dlTaskService
258 					.execute(cdlisReceiptRequest);
259 			if (response.hasErrors()) {
260 				docResponse.register(new EaseException(CDLIS_RECEIPT_PRINTED));
261 			}
262 		}
263 		if (request.getDlApplication().isOrganDonor()) {
264 			DlTaskServiceRequestFactory factory = new DlTaskServiceRequestFactory();
265 			DlApplication dlApplication = request.getDlApplication();
266 			IUserContext userContext = request.getUserContext();
267 			DlOrganDonorReceiptRequestTs organDonorReceiptRequest = factory
268 					.createDlOrganDonorReceiptRequest(dlApplication,
269 							userContext);
270 			DlOrganDonorReceiptResponseTs response = dlTaskService
271 					.execute(organDonorReceiptRequest);
272 			if (response.hasErrors()) {
273 				docResponse.register(new EaseException(
274 						ORGAN_DONOR_RECEIPT_PRINTED));
275 			}
276 		}
277 		if (EaseUtil.isNotNull(request.getDlApplication()
278 				.getVoterRegistationRequested().getCode())
279 				&& request.getDlApplication().getVoterRegistationRequested()
280 						.getCode().equalsIgnoreCase("Y")) {
281 			DlTaskServiceRequestFactory factory = new DlTaskServiceRequestFactory();
282 			DlApplication dlApplication = request.getDlApplication();
283 			IUserContext userContext = request.getUserContext();
284 			DlVoterRegistrationReceiptRequestTs voterRegistrationReceiptRequest = factory
285 					.createDlVoterRegistrationReceiptRequest(dlApplication,
286 							userContext);
287 			DlVoterRegistrationReceiptResponseTs response = dlTaskService
288 					.execute(voterRegistrationReceiptRequest);
289 			if (response.hasErrors()) {
290 				docResponse.register(new EaseException(
291 						VOTER_REGISTRATION_RECEIPT_PRINTED));
292 			}
293 		}
294 		if (request.getDlApplication().getInvoice().getPaymentTotal() != null) {
295 			DlTaskServiceRequestFactory factory = new DlTaskServiceRequestFactory();
296 			DlApplication dlApplication = request.getDlApplication();
297 			IUserContext userContext = request.getUserContext();
298 			Invoice invoice = request.getInvoice();
299 			DlPaymentReceiptRequestTs paymentReceiptRequest = factory
300 					.createDlPaymentReceiptRequest(dlApplication.getLicense(),
301 							invoice, userContext);
302 			DlPaymentReceiptResponseTs response = dlTaskService
303 					.execute(paymentReceiptRequest);
304 			if (response.hasErrors()) {
305 				docResponse.register(new EaseException(
306 						PAYMENT_RECEIPT_GENERATION_FAILED));
307 			}
308 		}
309 		return docResponse;
310 	}

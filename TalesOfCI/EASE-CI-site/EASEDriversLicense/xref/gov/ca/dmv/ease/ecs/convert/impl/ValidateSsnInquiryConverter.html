43  	 */
44  	@Override
45  	public String createMessage(IEcsRequest requestFromClient) {
46  		ValidateSsnRequestEcs request = (ValidateSsnRequestEcs) requestFromClient;
47  		Person person = request.getPerson();
48  		SocialSecurityDocument document = person.getSocialSecurityDocument();
49  		DriverLicense driverLicense = person.getDriverLicense();
50  		String dlId = ""; // dlId is optional, so use dlId only if it is available		
51  		this.licenseClassCode = request.getAppliedLicenseClass();
52  		//If Any Non Commercial License Class Exists
53  		if (!EaseUtil.isNullOrBlank(licenseClassCode)
54  				&& !this.isCommercialLicenseClassCode(this.licenseClassCode
55  						.getCode())) {
56  			//Set the First Non Commercial Category
57  			this.nonCommercialCategoryCode = request.getLicenseCategory();
58  		}
59  		if (isNotNull(driverLicense)
60  				&& isNotNull(driverLicense.getLicenseNumber())) {
61  			dlId = driverLicense.getLicenseNumber();
62  		}
63  		Object[] message = new Object[8];
64  		message[0] = IEcsConverterConstants.TCODE_SS5;
65  		message[1] = filler(document.getSsn(), 9);
66  		message[2] = getPersonName(person);
67  		message[3] = convertDateToString(person.getBirthDate(),
68  				EcsConverterHelper.FORMATTER_CCYY_MM_DD);
69  		message[4] = filler(dlId, 8);
70  		message[5] = getOfficeId(request.getUserContext());
71  		message[6] = getEmployeeID(request.getUserContext());
72  		message[7] = EcsConverterHelper.EMPTY_FILLER;
73  		return createMessageFormat(message);
74  	}

79  	@Override
80  	public IEcsResponse createResponse(String ssnMessage) {
81  		ValidateSsnResponseEcs validateSsnResponseEcs = new ValidateSsnResponseEcs();
82  		if (isNullOrBlank(ssnMessage)) {
83  			throw new MessageConversionException(
84  					"Invalid Response: Unable to process on null message");
85  		}
86  		// TODO converter spec has SSN_MESSAGE_LENGTH = 268.
87  		/*if (ssnMessage.length() != SSN_MESSAGE_LENGTH) {
88  			throw new MessageConversionException(
89  					"Invalid Response: Message size is: " + ssnMessage.length()
90  							+ " expected: " + SSN_MESSAGE_LENGTH);
91  		}*/
92  		// request part of response
93  		SocialSecurityDocument socialSecurityDocument = new SocialSecurityDocument();
94  		socialSecurityDocument.setSsn(ssnMessage.substring(3, 12).trim());
95  		// Set person's birth date
96  		validateSsnResponseEcs.setBirthDate(getBirthDate(ssnMessage.substring(
97  				47, 55).trim()));
98  		// actual response		
99  		String returnCodeFromSSA = ssnMessage.substring(68, 69).trim();
100 		CodeSetElement ssnVerificationCode = this
101 				.getSsnVerificationCode(returnCodeFromSSA);
102 		//ssnVerificationCode.setCode(ssnMessage.substring(68, 69).trim());
103 		ssnVerificationCode.setDescription(getDescription(returnCodeFromSSA));
104 		socialSecurityDocument
105 				.setSsnServiceVerificationCode(ssnVerificationCode);
106 		validateSsnResponseEcs
107 				.setSocialSecurityDocument(socialSecurityDocument);
108 		return validateSsnResponseEcs;
109 	}

125 	 * This method creates SSN Verification Code from the SSA Return Code.
126 	 * 
127 	 * @param returnCodeFromSSA the SSA Return Code
128 	 * 
129 	 * @return the SSN Verification Code
130 	 */
131 	private CodeSetElement getSsnVerificationCode(String returnCodeFromSSA) {
132 		CodeSetElement ssnVerificationCode = new CodeSetElement();
133 		if (IEcsConverterConstants.SSN_VERIF_CODE_1.equals(returnCodeFromSSA)) {
134 			ssnVerificationCode
135 					.setCode(IEcsConverterConstants.SSN_VERIF_CODE_A);
136 		}
137 		else if (IEcsConverterConstants.SSN_VERIF_CODE_9
138 				.equals(returnCodeFromSSA)) {
139 			//If Applied Class is A/B || Non Commercial Category1 = 1 or 2
140 			if ((!EaseUtil.isNullOrBlank(this.licenseClassCode) && (this.licenseClassCode
141 					.getCode().equals("A") || this.licenseClassCode.getCode()
142 					.equals("B")))
143 					|| (!EaseUtil.isNullOrBlank(this.nonCommercialCategoryCode) && (this.nonCommercialCategoryCode
144 							.getCode().equals("1") || this.nonCommercialCategoryCode
145 							.getCode().equals("2")))) {
146 				ssnVerificationCode
147 						.setCode(IEcsConverterConstants.SSN_VERIF_CODE_D);
148 			}
149 			else {
150 				ssnVerificationCode
151 						.setCode(IEcsConverterConstants.SSN_VERIF_CODE_X);
152 			}
153 		}
154 		else {
155 			ssnVerificationCode
156 					.setCode(IEcsConverterConstants.SSN_VERIF_CODE_R);
157 		}
158 		//FIXME - Remove this when the codes are added to DB
159 		//ssnVerificationCode = this.getCodeSetElement(ICodeSetNameConstants.SSN_VERIFICATION_CODE, ssnVerificationCode.getCode());
160 		return ssnVerificationCode;
161 	}

195 	/***
196 	 * Get the person name.
197 	 * 
198 	 * @param person the person
199 	 * 
200 	 * @return the person name
201 	 */
202 	private String getPersonName(Person person) {
203 		String reFormattedName = null;
204 		final int ALLOWED_NAME_SIZE = 35;
205 		PersonName personName = person.getCurrentName();
206 		String lastName = getNonNullString(personName.getLastName()).trim();
207 		String firstName = getNonNullString(personName.getFirstName()).trim();
208 		// only last name(single name) is available
209 		if (!lastName.equals(BLANK) && firstName.equals(BLANK)) {
210 			if (lastName.length() < 32) {
211 				reFormattedName = filler(lastName + "@UNK@@", ALLOWED_NAME_SIZE);
212 			}
213 			else if (lastName.length() == 32) {
214 				reFormattedName = filler(lastName.substring(0, 29) + "@"
215 						+ lastName.charAt(0) + "@@", ALLOWED_NAME_SIZE);
216 			}
217 			else if (lastName.length() == 33) {
218 				reFormattedName = lastName + "@U";
219 			}
220 			else if ((lastName.length() == 34) || (lastName.length() == 35)) {
221 				reFormattedName = lastName.substring(0, 33) + "@"
222 						+ lastName.charAt(0);
223 			}
224 		}
225 		// only last name and first name are available
226 		if (!lastName.equals(BLANK) && !firstName.equals(BLANK)) {
227 			reFormattedName = lastName + "@" + firstName + "@@";
228 			if (reFormattedName.length() > 35) {
229 				reFormattedName = reFormattedName.substring(0, 35);
230 			}
231 			else {
232 				reFormattedName = filler(reFormattedName, ALLOWED_NAME_SIZE);
233 			}
234 		}
235 		return reFormattedName;
236 	}

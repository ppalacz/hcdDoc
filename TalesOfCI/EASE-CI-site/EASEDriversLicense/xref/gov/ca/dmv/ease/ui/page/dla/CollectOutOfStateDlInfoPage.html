168 	/***
169 	 * Checks if is empty out of state license.
170 	 * 
171 	 * @param outOfStateLicenseBean the out of state license bean
172 	 * 
173 	 * @return true, if checks if is empty out of state license
174 	 */
175 	public boolean isEmptyOutOfStateLicense(
176 			OutOfStateLicenseBean outOfStateLicenseBean) {
177 		boolean hasState = (outOfStateLicenseBean.getOutOfStateLicense()
178 				.getIssuedByState() != null && outOfStateLicenseBean
179 				.getOutOfStateLicense().getIssuedByState().getCode() != null) ? true
180 				: false;
181 		boolean hasDlNumber = (outOfStateLicenseBean.getOutOfStateLicense()
182 				.getLicenseNumber() != null) ? true : false;
183 		boolean hasName = (outOfStateLicenseBean.getOutOfStateLicense()
184 				.getName().getLastName() != null) ? true : false;
185 		boolean hasBirthdate = (outOfStateLicenseBean.getOutOfStateLicense()
186 				.getBirthDate() != null) ? true : false;
187 		boolean hasGender = (outOfStateLicenseBean.getOutOfStateLicense()
188 				.getGenderCode() != null && outOfStateLicenseBean
189 				.getOutOfStateLicense().getGenderCode().getCode() != null) ? true
190 				: false;
191 		boolean hasSsn = (outOfStateLicenseBean.getOutOfStateLicense().getSsn() != null) ? true
192 				: false;
193 		if (hasState && hasDlNumber && hasName && hasBirthdate && hasGender
194 				&& hasSsn) {
195 			return false;
196 		}
197 		else {
198 			return true;
199 		}
200 	}

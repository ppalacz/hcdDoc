35  	public void validate(ProcessContext processContext) {
36  		SpProcessContext spProcessContext = (SpProcessContext) processContext;		
37  		Person applicant = getApplicant(processContext); 
38  		if (isEnterAction(processContext) && applicant != null) {
39  			String mailingCityCode = spProcessContext.getMailingCityCode();
40  			String mailingCityName = applicant.getAddress(
41  					AddressType.MAILING).getCity();
42  			String mailingStateName = applicant.getAddress(
43  					AddressType.MAILING).getState();
44  			String mailingStateZip = applicant.getAddress(
45  					AddressType.MAILING).getZip();
46  			validateUnableToGenerateZipOrCountyCode(getErrorCollector(),
47  					mailingCityCode, mailingCityName, mailingStateName,
48  					mailingStateZip);
49  		}			
50  		else if(applicant == null){
51  			reportMissingApplicant();
52  		}
53  	}

62  	public void validateUnableToGenerateZipOrCountyCode(
63  			IErrorCollector errorCollector, String mailCityCode,
64  			String mailCityName, String mailState, String mailZipCode) {
65  		if (!isNullOrBlank(mailCityName) && !isNullOrBlank(mailState)
66  				&& isNullOrBlank(mailCityCode)) {
67  			if (isNullOrBlank(mailZipCode) || !isZipValid(mailZipCode)) {
68  				getErrorCollector()

70  								IDataCollectionErrorMessageConstants.UNABLE_TO_GENERATE_ZIP_OR_COUNTY_CODE_PLEASE_VERIFY));
71  			}
72  		}

38  	 * Populates outgoing XML message for Dl Receipt Document
39  	 * 
40  	 * @param requestFromClient the request from client
41  	 * @return the string
42  	 */
43  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
44  		IUserContext currentUserContext = requestFromClient.getUserContext();
45  		Person applicant = null;
46  		Application application = null;
47  		DlApplication dlApplication = null;
48  		SalesPersonApplication spApplication = null;
49  		IssuanceReferralDoublesDocumentRequestTus request = (IssuanceReferralDoublesDocumentRequestTus) requestFromClient;
50  		application = request.getApplication();
51  		if (application instanceof SalesPersonApplication) {
52  			spApplication = (SalesPersonApplication) application;
53  			applicant = spApplication.getApplicant();
54  		}
55  		else if (application instanceof DlApplication) {
56  			dlApplication = (DlApplication) application;
57  			applicant = dlApplication.getApplicant();
58  		}
59  		List <String> elementsList = super.getCommonElementsList(
60  				requestFromClient, application, applicant);
61  		if (!EaseUtil.isNullOrBlank(application.getIssuanceReferralMessages())
62  				&& application.getIssuanceReferralMessages().contains(
63  						IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED)) {
64  			addLineHeaderToElementList(elementsList, dlApplication, applicant); //Add Header.
65  			addOldDlNumberToElementList(elementsList, dlApplication, applicant); //DL # Message and Doubled –Old Driver License #
66  			addInitialDlNumber(elementsList, applicant); //Doubled –Initial Driver License# - For 'X' type license number.
67  			addChangedDlNumber(elementsList, applicant); //Doubled –Changed Driver License#
68  			addPossibleDoubleDlNumbers(elementsList, applicant); //DL# marked as possible double.
69  		}
70  		addOfficeDetailsToElementList(elementsList, currentUserContext); //Office ID, current date & time.
71  		return createXmlForSingleDocument(elementsList,
72  				IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_TEMPLATE);
73  	}

76  	 * Adds the line header to element list.
77  	 * 
78  	 * @param elementsList the elements list
79  	 * @param dlApplication the dl application
80  	 * @param applicant the applicant
81  	 */
82  	private void addLineHeaderToElementList(List <String> elementsList,
83  			DlApplication dlApplication, Person applicant) {
84  		//If Records to be combined Issuance Referral message is present. 
85  		if (!EaseUtil.isNullOrBlank(applicant
86  				.getLicenseNumbersOfDuplicateAniRecords())
87  				|| (EaseUtil.isNotNull(dlApplication)
88  						&& EaseUtil.isNotNull(dlApplication
89  								.isNewLicenseNumberRequired())
90  						&& dlApplication.isNewLicenseNumberRequired() && !EaseUtil
91  						.isNullOrBlank(applicant.getPreviousLicenseNumber()))
92  				|| !EaseUtil.isNullOrBlank(applicant
93  						.getDuplicateLicenseNumberOfAniChangeInquiry())
94  				|| (!EaseUtil.isNullOrBlank(applicant
95  						.getPreviousLicenseNumber()) && DocumentConverterUtil
96  						.isXTypeLicense(applicant.getPreviousLicenseNumber()))) {
97  			//If there are Possible Doubles or if new # is requested 
98  			//or if Ani change inquiry with single hit marked as possible double 
99  			// or if HQ inquiry was made with X type of Driver License.  
100 			addToElementsList(
101 					elementsList,
102 					IPrintConverterConstants.HEADER_LINE_1,
103 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_HEADER_LINE_1); //Header Line 1
104 			addToElementsList(
105 					elementsList,
106 					IPrintConverterConstants.HEADER_LINE_2,
107 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_HEADER_LINE_2); //Header Line 2
108 			addToElementsList(
109 					elementsList,
110 					IPrintConverterConstants.HEADER_LINE_3,
111 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_HEADER_LINE_3); //Header Line 3
112 		}
113 		else {
114 			addToElementsList(
115 					elementsList,
116 					IPrintConverterConstants.HEADER_LINE_1,
117 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_HEADER_LINE_1);//Header Line 1
118 			addToElementsList(
119 					elementsList,
120 					IPrintConverterConstants.HEADER_LINE_2,
121 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_HEADER_LINE_2); //Header Line 2
122 			addToElementsList(
123 					elementsList,
124 					IPrintConverterConstants.HEADER_LINE_3,
125 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_HEADER_LINE_3); //Header Line 3
126 		}
127 	}

139 				&& EaseUtil.isNotNull(dlApplication

142 				&& !(EaseUtil.isNullOrBlank(applicant

144 			//If New DL# is requested display message

147 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_DL_MSG); //DL# Message
148 			addToElementsList(elementsList, applicant
149 					.getPreviousLicenseNumber(),
150 					IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_OLD_DL); //Doubled –Old Driver License #

189 	/***
190 	 * Adds the possible double dl numbers.
191 	 * 
192 	 * @param elementsList the elements list
193 	 * @param applicant the applicant
194 	 */
195 	private void addPossibleDoubleDlNumbers(List <String> elementsList,
196 			Person applicant) {
197 		int count = 1;
198 		if (!EaseUtil.isNullOrBlank(applicant
199 				.getLicenseNumbersOfDuplicateAniRecords())) {
200 			for (count = 1; count < applicant
201 					.getLicenseNumbersOfDuplicateAniRecords().size() + 1; count++) {
202 				String Zero = "";
203 				if (count <= 9) {
204 					Zero = "0";
205 				}
206 				addToElementsList(
207 						elementsList,
208 						applicant.getLicenseNumbersOfDuplicateAniRecords().get(
209 								count - 1),
210 						IPrintConverterConstants.ISSUANCE_REFERRAL_DOUBLES_LICENSE_NUMBER
211 								.concat(Zero).concat(Integer.toString(count))); //Doubled –Changed Driver License#
212 				if (count == 30) {
213 					//Print only up to 30 Possible Double DL#
214 					break;
215 				}
216 			}
217 		}
218 	}

67  	@Override
68  	public String createMessage(IEcsRequest request) {
69  		DlInquiryHomeAddressRequestEcs dlInquiryHomeAddressRequestEcs = (DlInquiryHomeAddressRequestEcs) request;
70  		IUserContext userContext = dlInquiryHomeAddressRequestEcs
71  				.getUserContext();
72  		String fileCode = dlInquiryHomeAddressRequestEcs.getFileCode();
73  		String vehicleLicenseNumber = dlInquiryHomeAddressRequestEcs
74  				.getVehicleLicenseNumber();
75  		Object[] values = new Object[8];
76  		values[0] = TCODE; //M52TC	3	TRANSACTION CODE           
77  		values[1] = getOfficeId(userContext); //M52OFFKY	3	OFFICE ID
78  		values[2] = getWorkDate(userContext); //M52WRKDT	6	WORK DATE (YYMMDD)
79  		values[3] = getEmployeeID(userContext); //M52EMPID	2	EMPLOYEE ID
80  		values[4] = filler(fileCode, 1); //M52FILCD	1	FILE CODE
81  		values[5] = filler(vehicleLicenseNumber, 7); //M52VLIC#	7	VEHICLE LICENSE NUMBER
82  		values[6] = filler(BLANK, 3); //M52L3VIN	3	LAST 3 POS. VIN/HIN OR SPACES
83  		values[7] = getTtc(userContext); //M52TTC	3	TYPE TRANSACTION CODE
84  		return createMessageFormat(values);
85  	}

90  	@Override
91  	public IEcsResponse createResponse(String responseMessage) {
92  		if (isNullOrBlank(responseMessage)) {
93  			throw new MessageConversionException(
94  					"Invalid Response: Unable to process on null message");
95  		}
96  		//Check for Response Size
97  		this.checkResponseSize(responseMessage);
98  		//Process Response
99  		DlInquiryHomeAddressResponseEcs response = new DlInquiryHomeAddressResponseEcs();
100 		//Trim the Request Echo
101 		responseMessage = responseMessage.substring(29);
102 		//If Size is 2 then it contains Error Code
103 		if (responseMessage.length() == ERROR_RESPONSE_SIZE) {
104 			//Process Error Code
105 			this.processErrorCode(response, responseMessage);
106 		}
107 		else if (responseMessage.length() == SECTION_L_RESPONSE_SIZE) {
108 			//Process Section L
109 			this.processSectionL(response, responseMessage);
110 		}
111 		else if (responseMessage.length() == SECTION_C_RESPONSE_SIZE) {
112 			//Process Section C
113 			this.processSectionC(response, responseMessage);
114 		}
115 		else if (responseMessage.length() == SECTION_S_RESPONSE_SIZE) {
116 			//Process Section S
117 			this.processSectionS(response, responseMessage);
118 		}
119 		else if (responseMessage.length() == SECTION_L_S_RESPONSE_SIZE) {
120 			//Process Section L & S
121 			//Process Section L
122 			this.processSectionL(response, responseMessage);
123 			//Process Section S
124 			this.processSectionS(response, responseMessage
125 					.substring(SECTION_L_RESPONSE_SIZE - 1));
126 		}
127 		else if (responseMessage.length() == SECTION_L_C_RESPONSE_SIZE) {
128 			//Process Section L & C
129 			//Process Section L
130 			this.processSectionL(response, responseMessage);
131 			//Process Section C
132 			this.processSectionC(response, responseMessage
133 					.substring(SECTION_L_RESPONSE_SIZE - 1));
134 		}
135 		else if (responseMessage.length() == SECTION_L_C_S_RESPONSE_SIZE) {
136 			//Process Section L C & S
137 			//Process Section L
138 			this.processSectionL(response, responseMessage);
139 			//Process Section C
140 			this.processSectionC(response, responseMessage
141 					.substring(SECTION_L_RESPONSE_SIZE - 1));
142 			//Process Section S
143 			this.processSectionS(response, responseMessage
144 					.substring(SECTION_C_RESPONSE_SIZE - 1));
145 		}
146 		else {
147 			//Invalid Response Length
148 			response
149 					.register(new MessageConversionException(
150 							"Invalid Response: Unable to process on Invalid Response Message"));
151 		}
152 		return response;
153 	}

242 	private void checkResponseSize(String responseMessage) {
243 		int actualMessageSize = responseMessage.length();
244 		if (!(EXPECTED_SECTION_L_RESPONSE_SIZE == actualMessageSize
245 				|| EXPECTED_SECTION_C_RESPONSE_SIZE == actualMessageSize
246 				|| EXPECTED_SECTION_S_RESPONSE_SIZE == actualMessageSize
247 				|| EXPECTED_SECTION_L_S_RESPONSE_SIZE == actualMessageSize
248 				|| EXPECTED_SECTION_L_C_RESPONSE_SIZE == actualMessageSize
249 				|| EXPECTED_SECTION_L_C_S_RESPONSE_SIZE == actualMessageSize || EXPECTED_ERROR_RESPONSE_SIZE == actualMessageSize)) {
250 			throw new MessageConversionException(
251 					"Invalid Response: Unable to process on hit count in message");
252 		}
253 	}

255 	/***
256 	 * This method checks if the Start Byte of a Section Contains Error.
257 	 * If Error then set it to E.
258 	 * 
259 	 * @param startByte the Start Byte of a Section
260 	 * 
261 	 * @return processed start byte
262 	 */
263 	private String checkForError(String startByte) {
264 		String section = BLANK;
265 		if (startByte.equalsIgnoreCase(SECTION_ERROR)) {
266 			section = SECTION_ERROR;
267 		}
268 		else {
269 			section = startByte;
270 		}
271 		return section;
272 	}

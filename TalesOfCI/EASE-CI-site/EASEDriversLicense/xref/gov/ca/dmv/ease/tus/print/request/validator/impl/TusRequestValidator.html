80  	/* (non-Javadoc)
81  	 * @see gov.ca.dmv.ease.fw.validate.impl.AbstractBaseValidator#validateValidatable(gov.ca.dmv.ease.fw.validate.IValidatable, gov.ca.dmv.ease.fw.error.IErrorCollector)
82  	 */
83  	@Override
84  	public void validateValidatable(IValidatable validatable,
85  			IErrorCollector collector) {
86  		if (validatable == null) {
87  			collector.register(new EcsValidationException(
88  					NULL_REQUEST_ERROR_MESSAGE));
89  			return;
90  		}
91  		else {
92  			if (validatable instanceof IPrintServiceRequest) {
93  				//Validate User Context
94  				validateUserContext(((IPrintServiceRequest) validatable)
95  						.getUserContext(), collector);
96  				//Validate Request
97  				validateRequest((IPrintServiceRequest) validatable, collector);
98  			}
99  			else {
100 				collector.register(new EcsValidationException(
101 						WRONG_REQUEST_TYPE_ERROR_MESSAGE
102 								+ validatable.getClass().getName()));
103 			}
104 		}
105 	}

107 	/***
108 	 * Validates user context.
109 	 * 
110 	 * @param userContext the user context
111 	 * @param aCollector the a collector
112 	 * 
113 	 * @throws EcsValidationException the validation exception
114 	 */
115 	protected static void validateUserContext(IUserContext userContext,
116 			IErrorCollector aCollector) throws EcsValidationException {
117 		if (null != userContext) {
118 			String strEmpId = userContext.getTechId();
119 			String strOfficeId = userContext.getOfficeId();
120 			String strTtc = userContext.getTtc();
121 			Date workDate = userContext.getWorkDate();
122 			String strRacfId = userContext.getRacfId();
123 			String ipAddress = userContext.getIpAddress();
124 			String userName = userContext.getUserName();
125 			//Validating Employee Id
126 			if (isNullOrBlank(strEmpId)) {
127 				aCollector.register(new EcsValidationException(
128 						"Invalid Employee Id"));
129 			}
130 			//Validating Office Id
131 			if (isNullOrBlank(strOfficeId)) {
132 				aCollector.register(new EcsValidationException(
133 						"Invalid Office Id"));
134 			}
135 			//Validating TTC
136 			if (isNullOrBlank(strTtc)) {
137 				aCollector.register(new EcsValidationException("Invalid TTC"));
138 			}
139 			//Validating Work date
140 			if (workDate == null) {
141 				aCollector.register(new EcsValidationException(
142 						"Invalid Work date"));
143 			}
144 			//Validating RACF Id
145 			if (isNullOrBlank(strRacfId)) {
146 				aCollector.register(new EcsValidationException(
147 						"Invalid RACF Id"));
148 			}
149 			//Validating IP Address
150 			if (isNullOrBlank(ipAddress)) {
151 				aCollector.register(new EcsValidationException(
152 						"Invalid IP address"));
153 			}
154 			//Validating User Name
155 			if (isNullOrBlank(userName)) {
156 				aCollector.register(new EcsValidationException(
157 						"Invalid User Name"));
158 			}
159 		}
160 		else {
161 			aCollector.register(new EcsValidationException(
162 					"Invalid UserContext : NULL"));
163 		}
164 	}

181 	 * Validates person.
182 	 * 
183 	 * @param aPerson the a person
184 	 * @param aCollector the a collector
185 	 */
186 	protected void validatePerson(Person aPerson, IErrorCollector aCollector) {
187 		if (aPerson == null) {
188 			aCollector.register(new EcsValidationException("Person is null"));
189 			return;
190 		}
191 		PersonName personName = aPerson.getCurrentName();
192 		if (personName == null) {
193 			aCollector.register(new EcsValidationException(
194 					"Person name is null"));
195 		}
196 		if (aPerson.getBirthDate() == null) {
197 			aCollector
198 					.register(new EcsValidationException("Person DOB is null"));
199 		}
200 		String lastName = personName.getLastName();
201 		if (isNullOrBlank(lastName)) {
202 			aCollector.register(new EcsValidationException(
203 					"Person's last name is not set"));
204 		}
205 		String firstName = personName.getFirstName();
206 		if (isNullOrBlank(firstName)) {
207 			aCollector.register(new EcsValidationException(
208 					"Person's first name is not set"));
209 		}
210 		Address address = aPerson.getAddress(AddressType.MAILING);
211 		if (address == null) {
212 			aCollector.register(new EcsValidationException("Address is null"));
213 		}
214 		String city = address.getCity();
215 		if (isNullOrBlank(city)) {
216 			aCollector.register(new EcsValidationException(
217 					"City name is not set"));
218 		}
219 	}

284 	 * Validate license number.
285 	 * 
286 	 * @param licenseNumber the license number
287 	 * @param aCollector the a collector
288 	 */
289 	protected void validateLicenseNumber(String licenseNumber,
290 			IErrorCollector aCollector) {
291 		if (EaseUtil.isNullOrBlank(licenseNumber)) {
292 			addValidationExceptionInErrorCollector(
293 					"NULL_OR_EMPTY_LICENSE_NUMBER", null, aCollector);
294 		}
295 		//Validating License Number		
296 		else if (!validateLength(licenseNumber, DL_MIN_LENGTH, DL_MAX_LENGTH)
297 				|| !licenseNumber.matches("[a-zA-Z]//d{7}")) {
298 			addValidationExceptionInErrorCollector(
299 					"INVALID_LICENSE_NUMBER_FORMAT", null, aCollector);
300 		}
301 	}

367 	 * Validates vehicule Sales person.
368 	 * 
369 	 * @param aPerson 
370 	 * @param aCollector 
371 	 */
372 	protected void validateVehicleSalesPerson(VehicleSalesPerson aPerson,
373 			IErrorCollector aCollector) {
374 		if (aPerson == null) {
375 			addValidationExceptionInErrorCollector("SALES_PERSON_IS_NULL",
376 					new String[] { "validateVehicleSalesPerson" }, aCollector);
377 			return;
378 		}
379 		PersonName personName = aPerson.getCurrentName();
380 		if (personName == null) {
381 			addValidationExceptionInErrorCollector("SALES_PERSON_NAME_IS_NULL",
382 					new String[] { "validateVehicleSalesPerson" }, aCollector);
383 		}
384 		if (aPerson.getBirthDate() == null) {
385 			addValidationExceptionInErrorCollector("SALES_PERSON_DOB_IS_NULL",
386 					new String[] { "validateVehicleSalesPerson" }, aCollector);
387 		}
388 		String lastName = personName.getLastName();
389 		if (isNullOrBlank(lastName)) {
390 			addValidationExceptionInErrorCollector(
391 					"SALES_PERSON_LAST_NAME_IS_NOT_SET",
392 					new String[] { "validateVehicleSalesPerson" }, aCollector);
393 		}
394 		String firstName = personName.getFirstName();
395 		if (isNullOrBlank(firstName)) {
396 			addValidationExceptionInErrorCollector(
397 					"SALES_PERSON_FIRST_NAME_IS_NOT_SET",
398 					new String[] { "validateVehicleSalesPerson" }, aCollector);
399 		}
400 		Address address = aPerson.getAddress(AddressType.MAILING);
401 		if (address == null) {
402 			addValidationExceptionInErrorCollector(
403 					"SALES_PERSON_ADDRESS_IS_NULL",
404 					new String[] { "validateVehicleSalesPerson" }, aCollector);
405 		}
406 		String city = address.getCity();
407 		if (isNullOrBlank(city)) {
408 			addValidationExceptionInErrorCollector(
409 					"SALES_PERSON_CITY_NAME_IS_NULL",
410 					new String[] { "validateVehicleSalesPerson" }, aCollector);
411 		}
412 		if (aPerson.getSalesPersonLicense().getPhotoInformation() == null) {
413 			addValidationExceptionInErrorCollector(
414 					"SALES_PERSON_PHOTO_INFORMATION_IS_NULL",
415 					new String[] { "validateVehicleSalesPerson" }, aCollector);
416 		}
417 	}

486 	/***
487 	 * Validate address.
488 	 * 
489 	 * @param collector the collector
490 	 * @param aAddress the address to be validated
491 	 * 
492 	 */
493 	protected void validateAddress(IErrorCollector collector, Address aAddress,
494 			String errorString) {
495 		if (aAddress == null) {
496 			collector.register(new EcsValidationException(errorString
497 					+ " is Null"));
498 			return;
499 		}
500 		if (isNullorEmpty(aAddress.getLine1())
501 				&& isNullorEmpty(aAddress.getLine2())) {
502 			collector.register(new EcsValidationException(errorString
503 					+ " Address is NULL or empty"));
504 		}
505 		if (isNullorEmpty(aAddress.getCity())) {
506 			collector.register(new EcsValidationException(errorString
507 					+ " City is NULL or empty"));
508 		}
509 		if (isNullorEmpty(aAddress.getState())) {
510 			collector.register(new EcsValidationException(errorString
511 					+ " State is NULL or empty"));
512 		}
513 		if (isNullorEmpty(aAddress.getZip())) {
514 			collector.register(new EcsValidationException(errorString
515 					+ " Zip is NULL or empty"));
516 		}
517 	}

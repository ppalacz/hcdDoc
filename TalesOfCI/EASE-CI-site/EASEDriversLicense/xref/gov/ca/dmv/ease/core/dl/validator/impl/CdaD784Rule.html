62  	public void validate(ProcessContext processContext) {
63  		DlProcessContext dlProcessContext = (DlProcessContext) processContext;
64  		CommercialTestResultGroup commercialTestResultGroup = dlProcessContext
65  				.getCommercialTestResultGroup();
66  		IErrorCollector errorCollector = new ErrorCollector();
67  		if (!EaseUtil.isNullOrBlank(commercialTestResultGroup)) {
68  			List<CommercialTestResult> performanceTestResults = commercialTestResultGroup
69  					.getCommercialTestResults(CommercialTestResultType.PERFORMANCE_TEST_RESULT);
70  			List<CommercialTestResult> preTripTestResults = commercialTestResultGroup
71  					.getCommercialTestResults(CommercialTestResultType.PRETRIP_TEST_RESULT);
72  			if (!EaseUtil.isNullOrBlank(performanceTestResults)
73  					&& performanceTestResults.size() > 1) {
74  				int lastIndex = performanceTestResults.size() - 1;
75  
76  				Action selectedAction = dlProcessContext.getSelectedAction();
77  				DriverLicense driverLicense = dlProcessContext
78  						.getDriverLicense();
79  				CommercialTestResult oldPerformanceTestResult = performanceTestResults
80  						.get(lastIndex - 1);
81  				CommercialTestResult performanceTestResult = performanceTestResults
82  						.get(lastIndex);
83  
84  				D784(performanceTestResult, oldPerformanceTestResult,
85  						preTripTestResults, selectedAction, driverLicense,
86  						errorCollector);
87  			}
88  		}
89  		setErrorCollector(errorCollector);
90  	}

93  	 * D784 - if replacing Performance Class C
94  	 * 
95  	 * if replacing Result W X or C
96  	 * 
97  	 * if replacement class present and not *
98  	 * 
99  	 * if Save Performance Result = C W X or space
100 	 * 
101 	 * or ------------------
102 	 * 
103 	 * if replacing Performance Class not C
104 	 * 
105 	 * if replacing result W X or C if
106 	 * 
107 	 * performance Result NE Pretrip Result
108 	 * 
109 	 * display D784

111 	 * @param driverLicense
112 	 * @param commercialTestResultGroup

115 	public void D784(CommercialTestResult performanceTestResult,
116 			CommercialTestResult oldPerformanceTestResult,
117 			List preTripTestResults, Action selectedAction,
118 			DriverLicense driverLicense, IErrorCollector errorCollector) {
119 		if (selectedAction != null
120 				&& selectedAction.equals(getActionsRegistry().getAction(
121 						IActionsRegistry.ENTER_ACTION))) {
122 			if (!EaseUtil.isNullOrBlank(performanceTestResult)
123 					&& !EaseUtil.isNullOrBlank(oldPerformanceTestResult)) {
124 				String licenseClass = CodeUtil
125 						.getCodeFromCodeSetElement(performanceTestResult
126 								.getLicenseClassCode());
127 				String oldLicenseClass = CodeUtil
128 						.getCodeFromCodeSetElement(oldPerformanceTestResult
129 								.getLicenseClassCode());
130 				String result = CodeUtil
131 						.getCodeFromCodeSetElement(performanceTestResult
132 								.getTestResultCode());
133 				String oldResult = CodeUtil
134 						.getCodeFromCodeSetElement(oldPerformanceTestResult
135 								.getTestResultCode());
136 				// if changing lic class
137 				// and its not changed to * (this indicates delete)
138 				if (!EaseUtil.isNullOrBlank(licenseClass)
139 						&& !IGeneralTestConstants.CORRECTION
140 								.equalsIgnoreCase(licenseClass)
141 						&& !EaseUtil.isNullOrBlank(oldLicenseClass)
142 						&& licenseClass.equalsIgnoreCase(oldLicenseClass)) {
143 					// if existing result is C,W,X
144 					if (!EaseUtil.isNullOrBlank(result)
145 							&& !EaseUtil.isNullOrBlank(oldResult)
146 							&& !result.equalsIgnoreCase(oldResult)
147 							&& (ArrayUtils.contains(oldResult, IDrivingTestResultsConstants.CERTS_OR_WAIVE))) {
148 						// if existing lic class is C
149 						// and result is replaced with C,W,X
150 						if ("C".equalsIgnoreCase(oldLicenseClass)) {
151 							if (result.equalsIgnoreCase(IDrivingTestResultsConstants.CERTIFICATE)
152 									|| result
153 											.equalsIgnoreCase(IDrivingTestResultsConstants.WAIVED)
154 									|| result
155 											.equalsIgnoreCase(IDrivingTestResultsConstants.EXPERIENCE_OR_TRAINING_CERT)) {
156 								errorCollector
157 										.register(new EaseValidationException(
158 												IDataCollectionErrorMessageConstants.WAIVE_OR_EXPERIENCE_MUST_BE_ERASED_FROM_PRE_TRIP_FIELDS));
159 
160 							}
161 						} else { // if existing lic class other than C
162 							if (!EaseUtil.isNullOrBlank(preTripTestResults)
163 									&& preTripTestResults.size() > 0) {
164 								CommercialTestResult preTripTestResult = (CommercialTestResult) preTripTestResults
165 										.get(preTripTestResults.size() - 1);
166 								String preTripResult = CodeUtil
167 										.getCodeFromCodeSetElement(preTripTestResult
168 												.getTestResultCode());
169 								// if new result NE Pretrip Result
170 								if (!EaseUtil.isNullOrBlank(preTripResult)
171 										&& !result
172 												.equalsIgnoreCase(preTripResult)) {
173 									errorCollector
174 											.register(new EaseValidationException(
175 													IDataCollectionErrorMessageConstants.WAIVE_OR_EXPERIENCE_MUST_BE_ERASED_FROM_PRE_TRIP_FIELDS));
176 
177 								}
178 							}
179 						}
180 
181 					}
182 				}
183 			}
184 		}

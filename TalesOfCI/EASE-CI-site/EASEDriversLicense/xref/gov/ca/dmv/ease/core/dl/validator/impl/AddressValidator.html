94  		String line2 = address.getLine2();
95  		//String line3 = address.getLine3();

99  				|| (EaseUtil.isNullOrBlank(line3) && !Character.isDigit(line1.charAt(0)))) {

131 		if (!EaseUtil.isNullOrBlank(cityCode)) {

133 				return Boolean.FALSE;
134 			}
135 		}
136 		if (!EaseUtil.isNullOrBlank(cityName)) {
137 			if (!CALIFORNIA.equals(state) || !EaseUtil.isNullOrBlank(cityCode)) {

284 		if (ArrayUtils.contains(aMailCity, FLEET_POST_OFFICE, ARMY_POST_OFFICE)) {

286 					ARMED_FORCES_EUROPE, ARMED_FORCES_PACIFIC))) {

289 		}
290 		if (ArrayUtils.contains(mailState, ARMED_FORCES_AMERICAS,
291 				ARMED_FORCES_EUROPE, ARMED_FORCES_PACIFIC)) {
292 			if (!ArrayUtils.contains(aMailCity, FLEET_POST_OFFICE,
293 					ARMY_POST_OFFICE)) {

17  /***
18   * Description: This activity class collects the required Ssn data.
19   * File: CollectSsnDataActivity.java
20   * Module:  gov.ca.dmv.ease.app.activity.dl.interaction.impl

25   * Last Changed By: $Author: mwyxg1 $
26   */
27  public class CollectSsnDataActivity extends InteractionActivity implements
28  		ICnaCapable {
29  	/*** The Constant serialVersionUID. */
30  	private static final long serialVersionUID = 2080866235849675160L;
31  	/*** The CdlisAndPdpsInquiryRules validator */
32  	private CdlisAndPdpsInquiryRules cdlisAndPdpsInquiryRules;
33  
34  	/* (non-Javadoc)
35  	 * @see gov.ca.dmv.ease.app.activity.impl.Activity#postCondition(gov.ca.dmv.ease.app.context.impl.ProcessContext)
36  	 */
37  	@Override
38  	protected void postCondition(ProcessContext processContext) {
39  		String ttc = processContext.getUserContext().getTtc();
40  		if (ArrayUtils.contains(ttc, ITtcConstants.DLA, ITtcConstants.DLC,
41  				ITtcConstants.DLD, ITtcConstants.DLE, ITtcConstants.DLP,
42  				ITtcConstants.DRT)) {
43  			if (isNotByPassValidations(processContext)) {
44  				cdlisAndPdpsInquiryRules.validate(processContext);
45  				populateErrorMessages(processContext, cdlisAndPdpsInquiryRules
46  						.getErrorCollector());
47  			}
48  		}
49  	}
50  
51  	/***
52  	 * Set cdlisAndPdpsInquiryRules.
53  	 * 
54  	 * @param cdlisAndPdpsInquiryRules the cdlisAndPdpsInquiryRules to set
55  	 */
56  	public void setCdlisAndPdpsInquiryRules(
57  			CdlisAndPdpsInquiryRules aCdlisAndPdpsInquiryRules) {
58  		cdlisAndPdpsInquiryRules = aCdlisAndPdpsInquiryRules;
59  	}
60  
61  	/* (non-Javadoc)
62  	 * @see gov.ca.dmv.ease.app.activity.ICnaCapable#isCnaActionAllowed(gov.ca.dmv.ease.app.context.impl.ProcessContext)
63  	 */
64  	public boolean isCnaActionAllowed(ProcessContext processContext) {
65  		CustomerInfoProcessContext pc = (CustomerInfoProcessContext) processContext;
66  		if (!pc.isAniDatabaseAvailable() && isNameOrBDChanged(pc)) {
67  			return true;
68  		}
69  		return false;
70  	}
71  
72  	private boolean isNameOrBDChanged(CustomerInfoProcessContext pc) {
73  		boolean isNameOrBirthDateChanged = false;
74  		if (!pc.getDlApplicant().getCurrentName().equals(
75  				pc.getDlApplicant().getPreviousName())) {
76  			isNameOrBirthDateChanged = true;
77  		}
78  		else if (!(pc.getDlApplicant().getBirthDate().equals(pc
79  				.getChangedBirthDate()))) {
80  			isNameOrBirthDateChanged = true;
81  		}
82  		return isNameOrBirthDateChanged;
83  	}
84  }

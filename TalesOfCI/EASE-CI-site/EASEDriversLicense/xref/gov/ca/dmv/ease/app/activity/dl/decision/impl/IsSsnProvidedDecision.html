24  public class IsSsnProvidedDecision extends DecisionActivity implements
25  		IDlaDecisionActivityConstantsForBpDefinition {
26  	/*** The Constant serialVersionUID. */
27  	private static final long serialVersionUID = -6030488606112569318L;
28  
29  	/***
30  	 * This method evaluates if Ssn is provided or not.
31  	 * If provided it returns true, if not returns false.
32  	 */
33  	@Override
34  	protected String evaluate(ProcessContext processContext) {
35  		SsnInquiryProcessContext ssnInquiryProcessContext = (SsnInquiryProcessContext) processContext;
36  		Boolean isSsnEmpty = EaseUtil.isNullOrBlank(ssnInquiryProcessContext
37  				.getSocialSecurityDocument().getSsn());
38  		return isSsnEmpty ? SSN_NOT_PROVIDED : SSN_PROVIDED;
39  	}
40  }

38  	public void preValidate(ProcessContext processContext) {
39  		CdlProcessContext cdlProcessContext = (CdlProcessContext) processContext;
40  		IErrorCollector errorCollector = new ErrorCollector();
41  		String lastName = null;
42  		String echoedLastName = null;
43  		if (!isNullorEmpty(cdlProcessContext.getDlApplicant())) {
44  			//Getting Last name
45  			if (!isNullorEmpty(cdlProcessContext.getDlApplicant()
46  					.getCurrentName())) {
47  				lastName = cdlProcessContext.getDlApplicant().getCurrentName()
48  						.getLastName();
49  			}
50  		}
51  		//Getting echoed Last name
52  		if (!isNullorEmpty(cdlProcessContext.getEchoedPerson())
53  				&& !isNullorEmpty(cdlProcessContext.getEchoedPerson()
54  						.getCurrentName())) {
55  			echoedLastName = cdlProcessContext.getEchoedPerson()
56  					.getCurrentName().getLastName();
57  		}
58  		//Validations
59  		if (!isNullorEmpty(lastName)
60  				&& !(echoedLastName.equalsIgnoreCase(lastName.substring(0,
61  						echoedLastName.length())))) {
62  			/*ERROR CODE - QO76 - the first three characters for the applicant’s 
63  			 last name in the applicant’s DL Master File record match with the 
64  			 3 POS Name field entered*/
65  			errorCollector.register(new ApplicationException("QO76"));
66  		}
67  		//Setting the error collector
68  		setErrorCollector(errorCollector);
69  	}

96  	public void validateMailingAddress(CdlProcessContext cdlProcessContext,
97  			IErrorCollector errorCollector) {
98  		Address mailingAddress = cdlProcessContext.getDlApplicant().getAddress(
99  				AddressType.MAILING);
100 		String mailStreetName = mailingAddress.getLine1();
101 		String mailCityCode = cdlProcessContext.getMailingCityCode();
102 		String mailCityName = mailingAddress.getCity();
103 		String mailState = mailingAddress.getState();
104 		String mailZipCode = mailingAddress.getZip();
105 		String mailCountyCode = mailingAddress.getCounty();
106 		if (!isNullorEmpty(mailingAddress)) {
107 			// D506
108 			validateMailingStreetNameForSpaceOrSlash(errorCollector,
109 					mailStreetName);
110 			// D053, D071, D560, D720, D085 validation rules - for DLA
111 			if (ITtcConstants.DLA.equals(cdlProcessContext.getUserContext()
112 					.getTtc())) {
113 				validateMailingForDla(errorCollector, mailStreetName,
114 						mailCityCode, mailCityName, mailState, mailZipCode,
115 						mailCountyCode);
116 			}
117 			// D008
118 			validateCityLengthVsState(errorCollector, mailCityName, mailState);
119 			// D519
120 			validateMailingCityNameStartsWithSpaceOrSlash(errorCollector,
121 					mailCityName);
122 			// D526 - TODO
123 			if (false) {
124 				errorCollector
125 						.register(new EaseValidationException(
126 								IDataCollectionErrorMessageConstants.UNABLE_TO_ACCESS_CITY_NAME_FILE_KEY_CITY));
127 			}
128 			// D720, D696
129 			if (!isNullorEmpty(mailingAddress.getServiceVerificationCode())) {
130 				// D696
131 				if (isNullorEmpty(mailingAddress.getServiceVerificationCode()
132 						.getCode())) {
133 					errorCollector
134 							.register(new EaseValidationException(
135 									IDataCollectionErrorMessageConstants.UNABLE_TO_VERIFY_MAIL_AND_OR_RESIDENCE_CITY));
136 				}
137 				// D720
138 				if ("1".equals(mailingAddress.getServiceVerificationCode()
139 						.getCode())) {
140 					errorCollector
141 							.register(new EaseValidationException(
142 									IDataCollectionErrorMessageConstants.MAIL_CITY_INVALID_VERIFY_REKEY_OR_IF_CORRECT_PRESS_ENTER));
143 				}
144 			}
145 			// D698 - TODO
146 			if (false) {
147 				errorCollector
148 						.register(new EaseValidationException(
149 								IDataCollectionErrorMessageConstants.MAIL_AND_RES_CITIES_INVALID_VERIFY_REKEY_OR_IF_CORRECT_PRESS_ENTER));
150 			}
151 			// D095
152 			validateOutOfState(errorCollector, mailState, mailZipCode);
153 			// D853
154 			if (!isNullorEmpty(mailCityName)
155 					&& (mailCityName.startsWith("APO") || mailCityName
156 							.startsWith("FPO"))) {
157 				if (!ArrayUtils.contains(mailState, "AA", "AE", "AP")) {
158 					errorCollector
159 							.register(new EaseValidationException(
160 									IDataCollectionErrorMessageConstants.MAIL_CITY_VS_MAIL_STATE));
161 				}
162 			}
163 			// D898
164 			if (!isNullorEmpty(cdlProcessContext.getDlApplication()
165 					.getLicense())
166 					&& !isNullorEmpty(cdlProcessContext.getDlApplication()
167 					.getLicense().getAppliedLicenseClasses())) {
168 				if (ArrayUtils.contains(DlApplicationUtil
169 						.getAppliedLicenseClassCodes(cdlProcessContext
170 								.getDlApplication().getLicense()), "A", "B",
171 						"C")
172 						&& !IStateCodeConstants.CALIFORNIA.equals(mailState)) {
173 					errorCollector
174 							.register(new EaseValidationException(
175 									IDataCollectionErrorMessageConstants.CANNOT_CONTINUE_MUST_BE_A_CA_RESIDENT));
176 				}
177 			}
178 			// D630 - TODO
179 			if (false) {
180 				errorCollector
181 						.register(new EaseValidationException(
182 								IDataCollectionErrorMessageConstants.UNABLE_TO_GENERATE_ZIP_OR_COUNTY_CODE_PLEASE_VERIFY));
183 			}
184 		}
185 	}

188 	 * D531, D550 validation rules
189 	 * 
190 	 * @param cdlProcessContext
191 	 * @param errorCollector
192 	 */
193 	public void validateResidenceAddress(CdlProcessContext cdlProcessContext,
194 			IErrorCollector errorCollector) {
195 		Address residenceAddress = cdlProcessContext.getDlApplicant()
196 				.getAddress(AddressType.RESIDENTIAL);
197 		String residenceStreet = residenceAddress.getLine1();
198 		String residenceCityCode = cdlProcessContext.getResidenceCityCode();
199 		String residenceCity = residenceAddress.getCity();
200 		String residenceState = residenceAddress.getState();
201 		String residenceZipCode = residenceAddress.getZip();
202 		String residenceCountyCode = residenceAddress.getCounty();
203 		if (!isNullorEmpty(residenceAddress)) {
204 			// D529
205 			validateResidenceStreetNameForSpaceOrSlash(errorCollector,
206 					residenceStreet);
207 			// D530
208 			validateResidenceCityNameStartsWithSpaceOrSlash(errorCollector,
209 					residenceCity, residenceState);
210 			// D528, D560, D721, D550, D085 validation rules - for DLA
211 			if (ITtcConstants.DLA.equals(cdlProcessContext.getUserContext()
212 					.getTtc())) {
213 				validateResidenceForDla(errorCollector, residenceStreet,
214 						residenceCityCode, residenceCity, residenceState,
215 						residenceZipCode, residenceCountyCode);
216 				// D721
217 				if (!isNullorEmpty(residenceAddress
218 						.getServiceVerificationCode())
219 						&& "1".equals(residenceAddress
220 								.getServiceVerificationCode().getCode())) {
221 					errorCollector
222 							.register(new EaseValidationException(
223 									IDataCollectionErrorMessageConstants.RES_CITY_INVALID_VERIFY_REKEY_OR_IF_CORRECT_PRESS_ENTER));
224 				}
225 			}
226 
227 			// D008
228 			validateCityLengthVsState(errorCollector, residenceCity,
229 					residenceState);
230 			// D531
231 			Address mailingAddress = cdlProcessContext.getDlApplicant()
232 					.getAddress(AddressType.MAILING);
233 			if (!isNullorEmpty(mailingAddress)) {
234 				String mailingStreet = mailingAddress.getLine1();
235 				if ((isNullorEmpty(residenceStreet) && isNullorEmpty(residenceState))
236 						&& (mailingStreet.indexOf("PO BX") >= 0
237 								|| mailingStreet.indexOf("PO DRW") >= 0 || mailingStreet
238 								.indexOf("GEN DEL") >= 0)) {
239 					errorCollector
240 							.register(new EaseValidationException(
241 									IDataCollectionErrorMessageConstants.RESIDENCE_ADDRESS_REQUIRED));
242 				}
243 			}
244 			// D550
245 			if (!isNullorEmpty(residenceCity)
246 					&& (residenceCity.startsWith("APO") || residenceCity
247 							.startsWith("FPO"))) {
248 				if (!ArrayUtils.contains(residenceState, "AA", "AE", "AP")) {
249 					errorCollector
250 							.register(new EaseValidationException(
251 									IDataCollectionErrorMessageConstants.RESIDENCE_CITY_VS_RESIDENCE_STATED));
252 				}
253 			}
254 			// D009
255 			validateCountyCodeVsOutOfState(errorCollector, residenceCountyCode,
256 					residenceState);
257 			// D507
258 			if (!isNullorEmpty(residenceCountyCode)
259 					&& Integer.parseInt(residenceCountyCode) == 59) {
260 				errorCollector
261 						.register(new EaseValidationException(
262 								IDataCollectionErrorMessageConstants.INVALID_COUNTY_CODE));
263 			}
264 		}
265 	}

270 	 * @param cdlProcessContext
271 	 * @param errorCollector
272 	 */
273 	public void validateMailingAndResidenceAddress(
274 			CdlProcessContext cdlProcessContext, IErrorCollector errorCollector) {
275 		Address mailingAddress = cdlProcessContext.getDlApplicant().getAddress(
276 				AddressType.MAILING);
277 		String mailingStreet = mailingAddress.getLine1();
278 		String mailingCity = mailingAddress.getCity();
279 		String mailingState = mailingAddress.getState();
280 		Address residenceAddress = cdlProcessContext.getDlApplicant()
281 				.getAddress(AddressType.RESIDENTIAL);
282 		String residenceStreet = residenceAddress.getLine1();
283 		String residenceCity = residenceAddress.getCity();
284 		String residenceState = residenceAddress.getState();
285 		if (!isNullorEmpty(residenceStreet) && !isNullorEmpty(mailingStreet)
286 				&& !isNullorEmpty(residenceCity) && !isNullorEmpty(mailingCity)
287 				&& !isNullorEmpty(residenceState)
288 				&& !isNullorEmpty(mailingState)) {
289 			if (residenceStreet.equals(mailingStreet)
290 					&& residenceCity.equals(mailingCity)
291 					&& residenceState.equals(mailingState)) {
292 				errorCollector
293 						.register(new EaseValidationException(
294 								IDataCollectionErrorMessageConstants.MAILING_AND_RESIDENCE_ADDRESSES_CANNOT_BE_THE_SAME));
295 			}
296 		}
297 	}

300 	 * D528, D560, D665, D529, D550 validation rules
301 	 * 
302 	 * @param errorCollector
303 	 * @param street
304 	 * @param cityCode
305 	 * @param city
306 	 * @param state
307 	 * @param zipCode
308 	 * @param county
309 	 */
310 	public void validateResidenceForDla(IErrorCollector errorCollector,
311 			String street, String cityCode, String city, String state,
312 			String zipCode, String county) {
313 		if (!isNullorEmpty(street)) {
314 			// D528
315 			if (isNullorEmpty(cityCode) && isNullorEmpty(city)) {
316 				errorCollector
317 						.register(new EaseValidationException(
318 								IDataCollectionErrorMessageConstants.RESIDENCE_ADDRESS_VS_CITY));
319 			}
320 			if (!isNullorEmpty(cityCode) && !isNullorEmpty(city)) {
321 				errorCollector
322 						.register(new EaseValidationException(
323 								IDataCollectionErrorMessageConstants.RESIDENCE_ADDRESS_VS_CITY));
324 			}
325 			// D560
326 			validateCityCodeVsState(errorCollector, cityCode, city, state);
327 			// D665
328 			if (street.indexOf("PO BX") >= 0 || street.indexOf("PO DRW") >= 0
329 					|| street.indexOf("GEN DEL") >= 0) {
330 				errorCollector
331 						.register(new EaseValidationException(
332 								IDataCollectionErrorMessageConstants.POST_OFFICE_ADDRESSES_NOT_ALLOWED_AS_RESIDENCE_ADDRESS));
333 			}
334 			// D529
335 			if (street.startsWith(" ") || street.indexOf("  ") >= 0) {
336 				errorCollector
337 						.register(new EaseValidationException(
338 								IDataCollectionErrorMessageConstants.RESIDENCE_ADDRESS_INVALID));
339 			}
340 		}
341 		else {
342 			// D528
343 			if (isNullorEmpty(cityCode) && !isNullorEmpty(city)) {
344 				errorCollector
345 						.register(new EaseValidationException(
346 								IDataCollectionErrorMessageConstants.RESIDENCE_ADDRESS_VS_CITY));
347 			}
348 		}
349 		// D550
350 		if (!isNullorEmpty(state) && isNullorEmpty(cityCode)
351 				&& isNullorEmpty(city)) {
352 			errorCollector
353 					.register(new EaseValidationException(
354 							IDataCollectionErrorMessageConstants.RESIDENCE_CITY_VS_RESIDENCE_STATED));
355 		}
356 		// D085
357 		validateUnableToGenerateZipOrCountyCode(errorCollector, cityCode, city,
358 				state, zipCode, county);
359 	}

414 						.register(new EaseValidationException(
415 								IDataCollectionErrorMessageConstants.RESIDENCE_CITY_INVALID));
416 			}
417 			if ((ArrayUtils.contains(state, "AA", "AE", "AP"))
418 					&& (!city.startsWith("APO") && !city.startsWith("FPO"))) {
419 				errorCollector
420 						.register(new EaseValidationException(

456 	public void validateOutOfState(IErrorCollector errorCollector,

462 					errorCollector

464 									IDataCollectionErrorMessageConstants.ZIP_CODE_VS_OUT_OF_STATE));
465 				}
466 			}
467 			else {
468 				if (IStateCodeConstants.CALIFORNIA.equals(state)) {
469 					errorCollector
470 							.register(new EaseValidationException(
471 									IDataCollectionErrorMessageConstants.ZIP_CODE_VS_OUT_OF_STATE));
472 				}
473 			}
474 		}

32  public class PrintDlRequestDocumentsActivity extends BusinessActivity {
33  	private static final long serialVersionUID = -4329589333738924781L;
34  	private CdaTaskService cdaTaskService;
35  
36  	@Override
37  	protected void invokeService(ProcessContext context) {
38  		DlProcessContext dlProcessContext = (DlProcessContext) context;
39  		PrintRequestTs tsRequest = new PrintRequestTs(dlProcessContext
40  				.getUserContext(), dlProcessContext.getDlApplication());
41  		Set <PrintRequestTypes> printRequestTypes = new HashSet <PrintRequestTypes>();
42  		if (dlProcessContext.getDlApplication().getDlPrintRequestDocumentCode() != null) {
43  			String selectedCode = dlProcessContext.getDlApplication()
44  					.getDlPrintRequestDocumentCode().getCode();
45  			if (selectedCode != null) {
46  				if ("1".equals(selectedCode)) {
47  					printRequestTypes
48  							.add(PrintRequestTypes.TEMP_OR_INTERIM_LICENSE_LDD0602);
49  				}
50  				if ("2".equals(selectedCode)) {
51  					printRequestTypes
52  							.add(PrintRequestTypes.INSTRUCTION_PERMIT_LDD0611);
53  				}
54  				if ("3".equals(selectedCode)) {
55  					printRequestTypes
56  							.add(PrintRequestTypes.TEMP_OR_INTERIM_LICENSE_LDD0602);
57  				}
58  				if ("4".equals(selectedCode)) {
59  					printRequestTypes
60  							.add(PrintRequestTypes.INSTRUCTION_PERMIT_LDD0611);
61  					printRequestTypes
62  							.add(PrintRequestTypes.TEMP_OR_INTERIM_LICENSE_LDD0602);
63  				}
64  				//5 Receipt only
65  			}
66  			tsRequest.setPrintRequestTypes(printRequestTypes);
67  			PrintResponseTs tsResponse = cdaTaskService
68  					.processTestResultsAndIssuanceReferralDocuments(tsRequest);
69  			if (tsResponse.hasErrors()) {
70  				populateErrorMessages(dlProcessContext, tsResponse
71  						.getErrorCollector());
72  			}
73  		}
74  	}
75  
76  	/***
77  	 * @param cdaTaskService the cdaTaskService to set
78  	 */
79  	public void setCdaTaskService(CdaTaskService cdaTaskService) {
80  		this.cdaTaskService = cdaTaskService;
81  	}
82  }

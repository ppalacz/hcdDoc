84  	public CdlisAndPdpsHistoryResponseBus createCdlisAndPdpsHistoryResponseForBus(
85  			CdlisAndPdpsResponseEcs ecsResponse) {
86  		CdlisAndPdpsHistoryResponseBus cdlisAndPdpsHistoryResponseBus = null;
87  		// Added NULL check condition because the back-end system is not ready
88  		// and NULL is returned as response
89  		// When the back-end system is ready exception is thrown and we can
90  		// remove the check and make modifications accordingly
91  		if (ecsResponse != null) {
92  			cdlisAndPdpsHistoryResponseBus = new CdlisAndPdpsHistoryResponseBus(
93  					ecsResponse.getCdlisHits(), ecsResponse.getPdpsHits(),
94  					ecsResponse.getCdlisRecordHits(), ecsResponse
95  							.getPdpsRecordHits());
96  			cdlisAndPdpsHistoryResponseBus.setErrorCollector(ecsResponse
97  					.getErrorCollector());
98  		}
99  		else {
100 			//(ecsResponse == null)
101 			//TODO masking null response as an empty response is most likely wrong - should use exception instead
102 			List <CdlisRecord> cdlisRecords = new ArrayList <CdlisRecord>();
103 			List <PdpsRecord> pdpsRecords = new ArrayList <PdpsRecord>();
104 			List <DrivingRecord> cdlisResultHits = new ArrayList <DrivingRecord>();
105 			List <DrivingRecord> pdpsResultHits = new ArrayList <DrivingRecord>();
106 			cdlisAndPdpsHistoryResponseBus = new CdlisAndPdpsHistoryResponseBus(
107 					cdlisRecords, pdpsRecords, cdlisResultHits, pdpsResultHits);
108 		}
109 		return cdlisAndPdpsHistoryResponseBus;
110 	}

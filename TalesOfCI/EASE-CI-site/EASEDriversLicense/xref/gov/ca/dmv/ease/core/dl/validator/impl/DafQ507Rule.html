29  public class DafQ507Rule extends AbstractDafRule {
30  	private CdaTaskService cdaTaskService;// Injected by spring
31  
32  	/***
33  	 * Sets the cda task service.
34  	 * 
35  	 * @param cdaTaskService the new cda task service
36  	 */
37  	public void setCdaTaskService(CdaTaskService cdaTaskService) {
38  		this.cdaTaskService = cdaTaskService;
39  	}

41  	/* (non-Javadoc)
42  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
43  	 * The System shall validate if an inactive DAF record is entered and a Browse, Display or Purge is invoked 
44  	 */
45  	public void validate(ProcessContext processContext) {
46  		DafProcessContext dafProcessContext = (DafProcessContext) processContext;		
47  		IErrorCollector errorCollector = new ErrorCollector();					
48  		Integer selectedDafRecordNumber = getDafNumber(dafProcessContext);		
49  		String activityName = getCurrentActivityName(dafProcessContext);
50  		
51  		if (selectedDafRecordNumber != null) {// If it is null then daf record is not entered on screen
52  			if (activityName != null && activityName.equalsIgnoreCase("DafMaintenanceActivity")) {			
53  				if (compareSelectedAction(new Action[] {
54  						getAction(IActionsRegistry.BROWSE_ACTION),
55  						getAction(IActionsRegistry.DISPLAY_ACTION),
56  						getAction(IActionsRegistry.PURGE_ACTION)},
57  						dafProcessContext)) {						
58  					if(!isDafActive(dafProcessContext, selectedDafRecordNumber, errorCollector, cdaTaskService)) {
59  						errorCollector.register(new ApplicationException("Q507"));
60  					}						
61  				} 
62  			} 
63  			else if  (activityName != null && activityName.equalsIgnoreCase("BrowseDailyAppFileActivity")) {
64  				Transaction dafRecord = null;				
65  				if (compareSelectedAction(new Action[] {
66  						getAction(IActionsRegistry.BROWSE_FWD_ACTION),
67  						getAction(IActionsRegistry.BROWSE_BWD_ACTION),
68  						getAction(IActionsRegistry.DISPLAY_ACTION),
69  						getAction(IActionsRegistry.PURGE_ACTION)},					
70  						dafProcessContext)) {
71  					dafRecord = getDafRecord(dafProcessContext, selectedDafRecordNumber, errorCollector, cdaTaskService);
72  					if(!isDafActive(dafRecord, selectedDafRecordNumber)) {
73  						errorCollector.register(new ApplicationException("Q507"));
74  					}					
75  					if(compareSelectedAction(new Action[] {						
76  							getAction(IActionsRegistry.DISPLAY_ACTION),
77  							getAction(IActionsRegistry.PURGE_ACTION)},					
78  							dafProcessContext)) {
79  						if(dafRecord.getCnaStatus().equalsIgnoreCase("Y")) {
80  							errorCollector.register(new ApplicationException("Q507"));
81  						}
82  					}
83  				} 
84  			}
85  		}		
86  		setErrorCollector(errorCollector);
87  	}
88

90  	/***
91  	 * Checks if is daf active.
92  	 * 
93  	 * @param dafProcessContext the daf process context
94  	 * @param selectedDafRecordNumber the selected daf record number
95  	 * @param errorCollector the error collector
96  	 * @param cdaTaskService the cda task service
97  	 * 
98  	 * @return true, if is daf active
99  	 */
100 	private boolean isDafActive(DafProcessContext dafProcessContext, Integer selectedDafRecordNumber, IErrorCollector errorCollector, CdaTaskService cdaTaskService) {
101 		Transaction tempDafRecord = retrieveDafFromDatabase(dafProcessContext, selectedDafRecordNumber, errorCollector, cdaTaskService);
102 		if(tempDafRecord != null && tempDafRecord.getTransactionIdentifier()== selectedDafRecordNumber.intValue()) {
103 			return true;
104 		}				
105 		return false;
106 	}
107 	
108 	/***
109 	 * Checks if is daf active.
110 	 * 
111 	 * @param tempDafRecord the temp daf record
112 	 * @param selectedDafRecordNumber the selected daf record number
113 	 * 
114 	 * @return true, if is daf active
115 	 */
116 	private boolean isDafActive(Transaction tempDafRecord, Integer selectedDafRecordNumber) {		
117 		if(tempDafRecord != null && tempDafRecord.getTransactionIdentifier()== selectedDafRecordNumber.intValue()) {
118 			return true;
119 		}				
120 		return false;
121 	}

124 	/***
125 	 * Gets the daf record.
126 	 * 
127 	 * @param dafProcessContext the daf process context
128 	 * @param selectedDafRecordNumber the selected daf record number
129 	 * @param errorCollector the error collector
130 	 * @param cdaTaskService the cda task service
131 	 * 
132 	 * @return the daf record
133 	 */
134 	private Transaction getDafRecord(DafProcessContext dafProcessContext, Integer selectedDafRecordNumber, IErrorCollector errorCollector, CdaTaskService cdaTaskService) {
135 		return retrieveDafFromDatabase(dafProcessContext, selectedDafRecordNumber, errorCollector, cdaTaskService);
136 	}

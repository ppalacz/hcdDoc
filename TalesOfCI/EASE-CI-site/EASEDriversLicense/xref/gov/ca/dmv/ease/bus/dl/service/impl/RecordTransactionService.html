30  public class RecordTransactionService {
31  	private IEcsRequestFactory ecsRequestFactory;
32  
33  	public void setEcsRequestFactory(IEcsRequestFactory ecsRequestFactory) {
34  		this.ecsRequestFactory = ecsRequestFactory;
35  	}
36  
37  	public RecordDlReviewResponseBus execute(
38  			RecordDlReviewRequestBus recordDlReviewRequest) {
39  		RecordDlReviewResponseBus busResponse = new RecordDlReviewResponseBus();
40  		RecordDlReviewRequestEcs ecsRequest = null;
41  		IErrorCollector errorCollector = null;
42  		try {
43  			errorCollector = recordDlReviewRequest.validate();
44  			if (errorCollector.hasErrors()) {
45  				// Set the Validation Messages to BusinessServiceResponse
46  				busResponse.setErrorCollector(errorCollector);
47  			}
48  			else {
49  				ecsRequest = ecsRequestFactory.createRecordDlReviewRequestEcs(
50  						recordDlReviewRequest.getUserContext(),
51  						recordDlReviewRequest.getTransaction()
52  								.getDlApplication());
53  				ecsRequest.execute();
54  			}
55  		}
56  		catch (Exception e) {
57  			busResponse.register(e);
58  		}
59  		return busResponse;
60  	}
61  
62  	public RecordCashierTransactionResponseBus execute(
63  			RecordCashierTransactionRequestBus recordCashierTransactionRequest) {
64  		RecordCashierTransactionResponseBus busResponse = new RecordCashierTransactionResponseBus();
65  		RecordCashierTransactionRequestEcs ecsRequest = null;
66  		IErrorCollector errorCollector = null;
67  		try {
68  			errorCollector = recordCashierTransactionRequest.validate();
69  			if (errorCollector.hasErrors()) {
70  				// Set the Validation Messages to BusinessServiceResponse
71  				busResponse.setErrorCollector(errorCollector);
72  			}
73  			else {
74  				ecsRequest = ecsRequestFactory
75  						.createRecordCashierTransactionRequestEcs(
76  								recordCashierTransactionRequest
77  										.getUserContext(),
78  								recordCashierTransactionRequest
79  										.getTransaction().getDlApplication());
80  				ecsRequest.execute();
81  			}
82  		}
83  		catch (Exception e) {
84  			busResponse.register(e);
85  		}
86  		return busResponse;
87  	}
88  }

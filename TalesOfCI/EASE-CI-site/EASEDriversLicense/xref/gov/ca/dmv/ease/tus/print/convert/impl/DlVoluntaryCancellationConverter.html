152 		}

185 						LIC_LOCATION_ELEMENT);

193 			}
194 			else if (LIC_LOCATION_CODE_S.equalsIgnoreCase(person

197 						LIC_LOCATION_ELEMENT);
198 			}
199 			else if (LIC_LOCATION_CODE_N.equalsIgnoreCase(person
200 					.getDriverLicense().getLicenseLocation().getCode())) {
201 				addToElementsList(elementsList, LIC_LOCATION_CODE_N_TEXT,
202 						LIC_LOCATION_ELEMENT);
203 			}
204 			else if (LIC_LOCATION_CODE_W.equalsIgnoreCase(person
205 					.getDriverLicense().getLicenseLocation().getCode())) {
206 				addToElementsList(elementsList, LIC_LOCATION_CODE_W_TEXT,
207 						LIC_LOCATION_ELEMENT);
208 			}
209 		}
210 		Date currentDate = new Date();
211     	addToElementsList(elementsList, convertDateToString(currentDate,

214 		addToElementsList(elementsList, convertDateToString(currentDate,

216 				CURRENT_SYSTEM_TIME);
217 		addToElementsList(elementsList, userContext.getTechId(),
218 				KEYED_BY_ELEMENT);
219 		addToElementsList(elementsList, userContext.getOfficeId(),

221         addToElementsList(elementsList,getOfficeNameFromId(userContext.getOfficeId()), OFFICE_NAME_ELEMENT);
222 		String xmlData = createXmlForSingleDocument(elementsList,

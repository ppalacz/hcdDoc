56  		TransactionalPayload(String initPayload, int seqNum) {
57  			payloads = new TreeMap <Integer, String>();
58  			payloads.put(new Integer(seqNum), initPayload);

61  		/***
62  		 * Gets the complete payload.
63  		 * 
64  		 * @return the complete payload
65  		 */
66  		String getCompletePayload() {
67  			StringBuilder strBuilder = new StringBuilder();
68  			for (Integer in : payloads.keySet()) {
69  				strBuilder.append(payloads.get(in));
70  			}
71  			return strBuilder.toString();
72  		}
73  
74  		/***
75  		 * Updates the payload.
76  		 * 
77  		 * @param payload the payload
78  		 * @param seqNum the seq num
79  		 */
80  		void update(String payload, int seqNum) {
81  			payloads.put(new Integer(seqNum), payload);
82  		}
83  	}

40  	/***
41  	 * Populate outgoing XML message for PDPS NDR Matches Report Document.
42  	 * 
43  	 * @param requestFromClient the request from client
44  	 * @return the outgoing XML message
45  	 */
46  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
47  		if (requestFromClient == null) {
48  			throw new MessageConversionException(
49  					IPrintServiceErrorMessageConstants.NULL_PDPS_NDR_MATCHES_REQUEST);
50  		}
51  		PdpsNdrReportDocumentRequestTus request = (PdpsNdrReportDocumentRequestTus) requestFromClient;
52  		IUserContext userContext = request.getUserContext();
53  		DlApplication dlApplication = request.getDlApplication();
54  		Person person = dlApplication.getApplicant();
55  		List <String> elementsList = new ArrayList <String>();
56  		addToElementsList(elementsList, dlApplication.getApplicationType()
57  				.getCode(), IPrintConverterConstants.APPLICATION_TYPE);
58  		addToElementsList(elementsList, getDlorIdNumber(person),
59  				IPrintConverterConstants.DRIVER_LICENSE_ID_NUMBER);
60  		addToElementsList(elementsList, getFormattedCurrentName(person),
61  				IPrintConverterConstants.CURRENT_NAME);
62  		addToElementsList(elementsList, convertDateToString(person
63  				.getBirthDate(), FORMATTER_MM_dd_yy),
64  				IPrintConverterConstants.BIRTH_DATE);
65  		addToElementsList(elementsList, person.getSocialSecurityDocument()
66  				.getSsn(), IPrintConverterConstants.SSN);
67  		List <PdpsRecord> pdpsRecords = dlApplication.getPdpsRecords();
68  		if (pdpsRecords != null && pdpsRecords.size() > 0) {
69  			for (int count = 0; count < 15; count++) {
70  				PdpsRecord pdpsRecord = pdpsRecords.get(count);
71  				if (count < 9) {
72  					addToElementsList(elementsList, pdpsRecord
73  							.getDesignateAction().getCode(),
74  							IPrintConverterConstants.PDPS_DESIGNATE
75  									+ IPrintConverterConstants.ZERO + ++count);
76  					if (!IPrintConverterConstants.STATE_CODE_XX
77  							.equalsIgnoreCase(pdpsRecord
78  									.getCurrentPrimaryDlNumberState().getCode())
79  							&& IPrintConverterConstants.STATE_CODE_ZZ
80  									.equalsIgnoreCase(pdpsRecord
81  											.getCurrentPrimaryDlNumberState()
82  											.getCode())) {
83  						addToElementsList(elementsList,
84  								IPrintConverterConstants.PDPS_TYPE_N,
85  								IPrintConverterConstants.PDPS_TYPE
86  										+ IPrintConverterConstants.ZERO + count);
87  					}
88  					else if (!IPrintConverterConstants.STATE_CODE_XX
89  							.equalsIgnoreCase(pdpsRecord
90  									.getCurrentPrimaryDlNumberState().getCode())
91  							&& !IPrintConverterConstants.STATE_CODE_ZZ
92  									.equalsIgnoreCase(pdpsRecord
93  											.getCurrentPrimaryDlNumberState()
94  											.getCode())) {
95  						addToElementsList(elementsList,
96  								IPrintConverterConstants.PDPS_TYPE_P,
97  								IPrintConverterConstants.PDPS_TYPE
98  										+ IPrintConverterConstants.ZERO + count);
99  					}
100 					addToElementsList(
101 							elementsList,
102 							getFormattedCurrentName(pdpsRecord.getCurrentName()),
103 							IPrintConverterConstants.PDPS_NAME
104 									+ IPrintConverterConstants.ZERO + count);
105 					addToElementsList(elementsList, pdpsRecord
106 							.getCurrentPrimaryDlNumberState().getCode(),
107 							IPrintConverterConstants.PDPS_CODE
108 									+ IPrintConverterConstants.ZERO + count);
109 					addToElementsList(elementsList, pdpsRecord
110 							.getLicenseNumber(),
111 							IPrintConverterConstants.PDPS_DL
112 									+ IPrintConverterConstants.ZERO + count);
113 					if (!EaseUtil.isNullOrBlank(pdpsRecord
114 							.getCurrentPrimarySsn())
115 							&& !IPrintConverterConstants.ZERO.equals(pdpsRecord
116 									.getCurrentPrimarySsn().trim())) {
117 						addToElementsList(elementsList, pdpsRecord
118 								.getCurrentPrimarySsn(),
119 								IPrintConverterConstants.PDPS_SSN
120 										+ IPrintConverterConstants.ZERO + count);
121 					}
122 					count--;
123 				}
124 				else {
125 					addToElementsList(elementsList, pdpsRecord
126 							.getDesignateAction().getCode(),
127 							IPrintConverterConstants.PDPS_DESIGNATE + ++count);
128 					if (!IPrintConverterConstants.STATE_CODE_XX
129 							.equalsIgnoreCase(pdpsRecord
130 									.getCurrentPrimaryDlNumberState().getCode())
131 							&& IPrintConverterConstants.STATE_CODE_ZZ
132 									.equalsIgnoreCase(pdpsRecord
133 											.getCurrentPrimaryDlNumberState()
134 											.getCode())) {
135 						addToElementsList(elementsList,
136 								IPrintConverterConstants.PDPS_TYPE_N,
137 								IPrintConverterConstants.PDPS_TYPE + count);
138 					}
139 					else if (!IPrintConverterConstants.STATE_CODE_XX
140 							.equalsIgnoreCase(pdpsRecord
141 									.getCurrentPrimaryDlNumberState().getCode())
142 							&& !IPrintConverterConstants.STATE_CODE_ZZ
143 									.equalsIgnoreCase(pdpsRecord
144 											.getCurrentPrimaryDlNumberState()
145 											.getCode())) {
146 						addToElementsList(elementsList,
147 								IPrintConverterConstants.PDPS_TYPE_P,
148 								IPrintConverterConstants.PDPS_TYPE + count);
149 					}
150 					addToElementsList(
151 							elementsList,
152 							getFormattedCurrentName(pdpsRecord.getCurrentName()),
153 							IPrintConverterConstants.PDPS_NAME + count);
154 					addToElementsList(elementsList, pdpsRecord
155 							.getCurrentPrimaryDlNumberState().getCode(),
156 							IPrintConverterConstants.PDPS_CODE + count);
157 					addToElementsList(elementsList, pdpsRecord
158 							.getLicenseNumber(),
159 							IPrintConverterConstants.PDPS_DL + count);
160 					if (!EaseUtil.isNullOrBlank(pdpsRecord
161 							.getCurrentPrimarySsn())
162 							&& !IPrintConverterConstants.ZERO.equals(pdpsRecord
163 									.getCurrentPrimarySsn().trim())) {
164 						addToElementsList(elementsList, pdpsRecord
165 								.getCurrentPrimarySsn(),
166 								IPrintConverterConstants.PDPS_SSN + count);
167 					}
168 					count--;
169 				}
170 			}
171 			if (pdpsRecords.size() > 15) {
172 				addToElementsList(elementsList, (pdpsRecords.size() - 15)
173 						+ IPrintConverterConstants.RECORD_MSG,
174 						IPrintConverterConstants.OVERFLOW_MESSAGE);
175 			}
176 		}
177 		addToElementsList(elementsList, userContext.getOfficeId(),
178 				IPrintConverterConstants.OFFICE_ID);
179 		addToElementsList(elementsList, convertDateToString(dlApplication
180 				.getApplicationDate(), FORMATTER_MM_dd_yy),
181 				IPrintConverterConstants.APPLICATION_DATE);
182 		addToElementsList(elementsList, convertDateToString(Calendar
183 				.getInstance().getTime(), FORMATTER_MM_dd_yy),
184 				IPrintConverterConstants.CURRENT_SYSTEM_DATE);
185 		addToElementsList(elementsList, convertDateToString(Calendar
186 				.getInstance().getTime(), FORMATTER_HH_mm),
187 				IPrintConverterConstants.CURRENT_TIME);
188 		String xmlData = createXmlForSingleDocument(elementsList,
189 				IPrintConverterConstants.PDPS_DOCUMENT_NAME);
190 		return xmlData;
191 	}

41  @Aspect
42  public class DlReviewAspect extends AbstractAspect {
43  	/*** The logger. */
44  	private static final Log LOGGER = LogFactory.getLog(DlReviewAspect.class);
45  
46  	/***
47  	 * The purpose of this point-cut is to mark the end of drivers license business process.
48  	 */
49  	@Pointcut("!inEaseLogAppenderExecution() && execution(* gov.ca.dmv.ease.app.context.dl.impl.DlProcessContext.end())")
50  	public void endOfBusinessProcess() {
51  	}
52  
53  	/***
54  	 * The purpose of this advise is to define the behavior at the end of business process for the fraud protection.
55  	 * This method constructs a message for fraud protection and then passes it across the bridge code.
56  	 * This method also constructs a message for system management log and passes it across the ecs service.
57  	 * 
58  	 * @param jp the joint point at the end of business process
59  	 */
60  	@Before("endOfBusinessProcess()")
61  	public void endOfBusinessProcess(JoinPoint jp) {
62  		Object obj = (Object) EaseProxyUtils.getProxyObject(jp.getTarget());
63  		if (isMajorBusinessProcess(obj)) {
64  			ProcessContext pc = (ProcessContext) obj;
65  			if (!pc.getSelectedAction().getKey().equals(
66  					ActionsRegistry.CANCEL_ACTION)) {
67  				String bridgeMessage = processDlReview(pc);
68  				IUserContext userContext = pc.getUserContext();
69  				StringBuilder logString = this.generateHeaderSection(jp,
70  						userContext);
71  				logString.append(LOG_ENTRY + NAME_VALUE_SEPERATOR);
72  				logString.append(StringUtils.rightPad(userContext
73  						.getIpAddress(), 15)
74  						+ " ; ");
75  				logString.append("FRAUD_STAT_MSG" + NAME_VALUE_SEPERATOR);
76  				logString.append(bridgeMessage + " ; ");
77  				String message = logString.toString();
78  				this.processAuditLogRequestMessage(message, userContext);
79  			}
80  		}
81  	}
82  
83  	/***
84  	 * The purpose of this method is to determine if Context is a major context.
85  	 * 
86  	 * @param Object ChildContext
87  	 */
88  	private boolean isMajorBusinessProcess(Object obj) {
89  		if ((obj != null) && (obj instanceof ChildContext)) {
90  			if ((((ChildContext) obj).getParentProcessContext() != null)
91  					&& ((Object) ((ChildContext) obj).getParentProcessContext() instanceof DlMenuProcessContext)) {
92  				return true;
93  			}
94  		}
95  		return false;
96  	}
97  
98  	/***
99  	 * The purpose of this method is to process the request for dl review.
100 	 * 
101 	 * @param pc the instance of process context
102 	 * 
103 	 * @return the fraud protection message for bridge service
104 	 */
105 	private String processDlReview(ProcessContext pc) {
106 		String messageText = "";
107 		DlProcessContext dlProcessContext = (DlProcessContext) pc;
108 		String ttcCode = pc.getUserContext().getTtc();
109 		if (CollectionUtils.arrayToList(ITtcConstants.TTC_DLREVIEW).contains(
110 				ttcCode)) {
111 			LOGGER.info("TTC: " + ttcCode);
112 			LOGGER.info("Process Context: " + dlProcessContext.toString());
113 			RecordDlReviewRequestEcs recordDlReviewRequestEcs = new RecordDlReviewRequestEcs(
114 					"", dlProcessContext.getUserContext(), dlProcessContext
115 							.getDlApplication());
116 			recordDlReviewRequestEcs.setDlProcessContext(dlProcessContext);
117 			DlReviewConverter dlReviewConverter = new DlReviewConverter();
118 			messageText = dlReviewConverter
119 					.createMessage(recordDlReviewRequestEcs);
120 			LOGGER.info("Message Text: [" + messageText + "]");
121 			//if ((messageText != null) && !messageText.equals("")) {
122 			BridgeRequest bridgeRequest = AbstractAspect.bridgeRequestFactory
123 					.createBridgeRequest(messageText, dlProcessContext
124 							.getUserContext());
125 			bridgeRequest.execute();
126 			//}
127 		}
128 		return messageText;
129 	}

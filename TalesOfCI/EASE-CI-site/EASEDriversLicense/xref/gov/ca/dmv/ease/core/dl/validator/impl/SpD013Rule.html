46  						.isNullorEmpty(getSalesPersonApplication(processContext).getDataEntryReceiptNumber())) {
47  			createError = true; 
48  		}
49  		
50  		// if we have an APP# (dataEntryRecieptNumber) then the mode must be M or T. 
51  		else if ((!ValidationRulesUtil
52  				.isNullorEmpty(getSalesPersonApplication(processContext).getDataEntryReceiptNumber()))
53  				&& !(getMode(processContext).equalsIgnoreCase(IEaseModes.MANUAL) || getMode(
54  						processContext)
55  						.equalsIgnoreCase(IEaseModes.TRAVEL_CREW))) {
56  			createError = true;
57  		}
58  		
59  		if (createError) {
60  			getErrorCollector()
61  					.register(new EaseValidationException(
62  							IDataCollectionErrorMessageConstants.DER_VS_MODE));

64  	}

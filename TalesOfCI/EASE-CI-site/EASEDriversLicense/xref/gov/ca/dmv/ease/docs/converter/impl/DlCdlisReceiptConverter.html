66  	/***
67  	 * Instantiates a new dl cdlis receipt converter.
68  	 * 
69  	 * @param userContext the user context
70  	 * @param appln the appln
71  	 * @param cdlisRecords the cdlis records
72  	 * @param pdpsRecords the pdps records
73  	 */
74  	public DlCdlisReceiptConverter(IUserContext userContext,
75  			DlApplication appln, CdlisRecord cdlisRecord, PdpsRecord pdpsRecord) {
76  		// set the default document template
77  		setTemplate(new DlCdlisReceiptTemplate());
78  		dlNumber = appln.getLicense().getLicenseNumber();
79  		Person person = appln.getApplicant();
80  		ssn = appln.getSocialSecurityDocument().getSsn();
81  		//FIXME Person names have changed....
82  		firstName = person.getCurrentName().getFirstName();
83  		middleName = person.getCurrentName().getMiddleName();
84  		lastName = person.getCurrentName().getLastName();
85  		Address mailingAddress = DummySubjectHelper.getMailingAddress(null);
86  		addressLine1 = mailingAddress.getLine1();
87  		addressLine2 = mailingAddress.getLine2();
88  		Date date = person.getBirthDate();
89  		DateFormat dateFormat1 = new SimpleDateFormat("ddMMyy");
90  		DateFormat dateFormat2 = new SimpleDateFormat("dd-MM-yy");
91  		birthDate = dateFormat2.format(date);
92  		workDate = dateFormat2.format(userContext.getWorkDate());
93  		userId = userContext.getTechId();
94  		officeId = userContext.getOfficeId();
95  		Calendar cal = Calendar.getInstance();
96  		int hour = cal.get(Calendar.HOUR);
97  		int minutes = cal.get(Calendar.MINUTE);
98  		time = hour + ":" + minutes;
99  		StringBuilder cdlisRecordsCodes = new StringBuilder("");
100 		if (cdlisRecord != null) {
101 			cdlisRecordsCodes.append(/***cdlisRecord.getStatusCode() + */
102 			" " + "NH" + " " + cdlisRecord.getLicenseNumber());
103 			cdlisRecordsCodes.append("\t\t");
104 		}
105 		StringBuilder pdpsRecordsCodes = new StringBuilder("");
106 		if (pdpsRecord != null) {
107 			pdpsRecordsCodes.append(/***pdpsRecord.getStatusCode() + */
108 			" " + "ZY" + " " + pdpsRecord.getLicenseNumber());
109 			pdpsRecordsCodes.append("\t\t");
110 		}
111 		cashierSeqNumber = "33";
112 		receiptVarValueMap.put("#DL_NUMBER", dlNumber);
113 		receiptVarValueMap.put("#FNAME", firstName);
114 		receiptVarValueMap.put("#MNAME", middleName);
115 		receiptVarValueMap.put("#LNAME", lastName);
116 		receiptVarValueMap.put("#DOB", birthDate);
117 		receiptVarValueMap.put("#SSN", ssn);
118 		receiptVarValueMap.put("#WORK_DATE", workDate);
119 		receiptVarValueMap.put("#OFFICE_ID", officeId);
120 		receiptVarValueMap.put("#USER_ID", userId);
121 		receiptVarValueMap.put("#TIME", time);
122 		receiptVarValueMap.put("#PDPS_HITS", pdpsRecordsCodes.toString());
123 		receiptVarValueMap.put("#CDLIS_HITS", cdlisRecordsCodes.toString());
124 		receiptVarValueMap.put("#CASHIER_SEQ", cashierSeqNumber);
125 	}
126 }

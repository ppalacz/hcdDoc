64  	@Override
65  	protected void validatePerson(Person person, IErrorCollector aCollector) {
66  		if (person == null) {
67  			addValidationExceptionInErrorCollector("NULL_PERSON", null,
68  					aCollector);
69  		}
70  		else {
71  			validatePersonBirthDate(person, aCollector);
72  			validateLicense(person, aCollector);
73  			validatePhotoInformation(person, aCollector);
74  		}
75  	}

77  	/***
78  	 * Validate license.
79  	 * 
80  	 * @param aCollector the a collector
81  	 * @param person the person
82  	 */
83  	private void validateLicense(Person person, IErrorCollector aCollector) {
84  		DriverLicense driverLicense = person.getDriverLicense();
85  		IdCard idCard = person.getIdCard();
86  		if (isNullorEmpty(driverLicense) && isNullorEmpty(idCard)) {
87  			addValidationExceptionInErrorCollector(
88  					"NULL_DRIVERLICENSE_AND_IDCARD", null, aCollector);
89  		}
90  		else if (!isNullorEmpty(driverLicense)) {
91  			validateLicenseNumber(driverLicense.getLicenseNumber(), aCollector);
92  		}
93  		else if (!isNullorEmpty(idCard)) {
94  			validateLicenseNumber(idCard.getLicenseNumber(), aCollector);
95  		}
96  	}

98  	/***
99  	 * Validate photo information.
100 	 * 
101 	 * @param person the person
102 	 * @param aCollector the a collector
103 	 */
104 	private void validatePhotoInformation(Person person,
105 			IErrorCollector aCollector) {
106 		PhotoInformation photoInformation = person.getDriverLicense()
107 				.getPhotoInformation();
108 		if (isNullorEmpty(photoInformation)) {
109 			addValidationExceptionInErrorCollector("NULL_PHOTO_INFORMATION",
110 					null, aCollector);
111 		}
112 		else {
113 			if (isNullorEmpty(photoInformation.getPhotoName())) {
114 				addValidationExceptionInErrorCollector(
115 						"NULL_OR_EMPTY_PHOTO_NAME", null, aCollector);
116 			}
117 		}
118 	}

31  public class DlDocumentResponse extends SynchronousResponse {
32  	/*** Logger for this class. */
33  	private static final Log LOGGER = LogFactory
34  			.getLog(DlDocumentResponse.class);
35  	/*** The Constant serialVersionUID. */
36  	private static final long serialVersionUID = 6322140539051974005L;
37  	/*** The DL Document response */
38  	private String response;
39  	
40  	/***
41  	 * Instantiates a new DL document receipt response
42  	 */
43  	public DlDocumentResponse() {
44  		super();
45  	}
46  	
47  	/***
48  	 * Instantiates a new DL document receipt response
49  	 */
50  	public DlDocumentResponse(String response) {
51  		super();
52  		this.setReponse(response);
53  	}
54  
55  	/***
56  	 * Instantiates a new DL Document Response.
57  	 * 
58  	 * @param errorCollector the error messages
59  	 */
60  	public DlDocumentResponse(IErrorCollector errorCollector) {
61  		super(errorCollector);
62  	}
63  
64  	/***
65  	 * Gets the DL document response
66  	 *
67  	 * @return the DL document response
68  	 */
69  	public String getResponse() {
70  		return response;
71  	}
72  
73  	/***
74  	 * Sets the DL document response
75  	 *
76  	 * @param response the DL document response
77  	 */
78  	public void setReponse(String response) {
79  		this.response = writeToTargetFile(getTargetFile(), response);
80  	}
81  
82  	/***
83  	 * Creates a target file in the temporary directory
84  	 * 
85  	 * @return the target file
86  	 */
87  	public static File getTargetFile() {
88  		long time = (new Date()).getTime();
89  		File aTargetFile = null;
90  		try {
91  			aTargetFile = File.createTempFile(Long.toString(time), ".ps");
92  		}
93  		catch (IOException e) {
94  			LOGGER.error(e);
95  		}
96  		return aTargetFile;
97  	}
98  
99  	/***
100 	 * Write content to target file
101 	 * 
102 	 * @param targetFile the target file
103 	 * @param content the document content
104 	 * 
105 	 * @return the absolute path of the output document
106 	 */
107 	private String writeToTargetFile(File targetFile, String content) {
108 		try {
109 			OutputStreamWriter outputStream;
110 			outputStream = new OutputStreamWriter(new FileOutputStream(
111 					targetFile));
112 			outputStream.write(content);
113 			outputStream.flush();
114 			outputStream.close();
115 		}
116 		catch (IOException e) {
117 			LOGGER.error(e);
118 		}
119 		return targetFile.getAbsolutePath();
120 	}
121 }

32  public class IsPaymentEqualDecision extends DecisionActivity implements
33  		IDlaDecisionActivityConstantsForBpDefinition {
34  	/*** The Constant serialVersionUID. */
35  	private static final long serialVersionUID = 0L;
36  
37  	/***
38  	 * This method checks if the payment amount is equal to the total invoice amount.
39  	 * If equal it returns true, if not returns false.
40  	 * 
41  	 * @param processContext the context
42  	 * @return the string
43  	 */
44  	@Override
45  	protected String evaluate(ProcessContext processContext) {
46  		DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) processContext;
47  		Float invoiceTotal = dlFeeProcessContext.getInvoice().getInvoiceTotal()
48  				.floatValue();
49  		Invoice invoice = dlFeeProcessContext.getInvoice();
50  		Float totalPayments = getCalculateTotalFee(invoice);
51  		if (totalPayments.floatValue() == invoiceTotal.floatValue()) {
52  			return PAYMENT_EQUAL;
53  		}
54  		else {
55  			if (totalPayments > invoiceTotal) {
56  				return PAYMENT_GREATER_THAN;
57  			}
58  			else {
59  				dlFeeProcessContext
60  						.getMessageCollector()
61  						.addValidationMessage(
62  								"F509",
63  								new ErrorMessage(
64  										"gov.ca.dmv.ease.fw.exception.impl.ApplicationException",
65  										"F509"));
66  				return PAYMENT_LESS_THAN;
67  			}
68  		}
69  	}
70  
71  	/***
72  	 * Gets the calculate total fee.
73  	 * 
74  	 * @param invoice the invoice
75  	 * @return the calculate total fee
76  	 */
77  	private Float getCalculateTotalFee(Invoice invoice) {
78  		Float cashAmount = 0.0f;
79  		if (invoice.getCashPayment() != null) {
80  			if (invoice.getCashPayment().getAmount() != null) {
81  				cashAmount = invoice.getCashPayment().getAmount().floatValue();
82  			}
83  		}
84  		List <CheckPayment> checkPayments = invoice.getCheckPayments();
85  		List <DebitPayment> debitPayments = invoice.getDebitPayments();
86  		Float totalCheckPayments = getTotalCheckPayments(checkPayments);
87  		Float totalDebitPayments = getTotalDebitPayments(debitPayments);
88  		Float totalVoucherPayments = getVoucherPayments(invoice
89  				.getVoucherPayments());
90  		return cashAmount + totalCheckPayments + totalDebitPayments
91  				+ totalVoucherPayments;
92  	}
93  
94  	/***
95  	 * Gets the total check payments.
96  	 * 
97  	 * @param checkPayments the check payments
98  	 * @return the total check payments
99  	 */
100 	private Float getTotalCheckPayments(List <CheckPayment> checkPayments) {
101 		Float totalCheckPayments = 0.0f;
102 		for (CheckPayment checkPayment : checkPayments) {
103 			if (checkPayment.getAmount() != null) {
104 				totalCheckPayments += checkPayment.getAmount().floatValue();
105 			}
106 		}
107 		return totalCheckPayments;
108 	}
109 
110 	/***
111 	 * Gets the total debit payments.
112 	 * 
113 	 * @param debitPayments the debit payments
114 	 * @return the total debit payments
115 	 */
116 	private Float getTotalDebitPayments(List <DebitPayment> debitPayments) {
117 		Float totalDebitPayments = 0.0f;
118 		for (DebitPayment debitPayment : debitPayments) {
119 			if (debitPayment.getAmount() != null) {
120 				totalDebitPayments += debitPayment.getAmount().floatValue();
121 			}
122 		}
123 		return totalDebitPayments;
124 	}
125 
126 	private Float getVoucherPayments(List <Voucher> voucherPayments) {
127 		Float totalVoucherPayments = 0.0f;
128 		for (Voucher voucherPayment : voucherPayments) {
129 			if (voucherPayment.getAmount() != null) {
130 				if (voucherPayment.getStatusCode().getCode().equals("C")) {
131 					totalVoucherPayments += voucherPayment.getAmount()
132 							.floatValue();
133 				}
134 			}
135 		}
136 		return totalVoucherPayments;
137 	}

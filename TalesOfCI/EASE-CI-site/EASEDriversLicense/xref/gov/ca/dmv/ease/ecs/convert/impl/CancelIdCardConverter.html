36  public class CancelIdCardConverter extends AbstractEcsMessageConverter {
37  	/*** Logger for this class */
38  	private static final Logger logger = Logger
39  			.getLogger(SendCdlisAndPdpsConverter.class);
40  	/*** Transaction code */
41  	private static final String TCODE = "DUH";
42  	/*** The Constant Inquiry Transaction Sub Type. */
43  	private static final String INQUIRY_TRANSACTION_SUB_TYPE = "I";
44  	/*** The Delimiter. */
45  	private static final String DELIMITER = " ";

47  	/* (non-Javadoc)
48  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
49  	 */
50  	@Override
51  	public String createMessage(IEcsRequest request) {
52  		CancelIdCardRequestEcs cancelIdCardRequestEcs = (CancelIdCardRequestEcs) request;
53  		List <Object> objectList = new ArrayList <Object>(0);
54  		/**** Begin Conversion ***/
55  		/* HQ TRANSACTION CODE */
56  		objectList.add(filler(TCODE, 3));
57  		/* TYPE TRANS CODE */
58  		objectList.add(filler(super.getTtc(request.getUserContext()), 3));
59  		/* TRANSACTION SUB-CODE */
60  		objectList
61  				.add(filler(cancelIdCardRequestEcs.getTransactionSubCode(), 1));
62  		/* REQUESTER CODE PREFIX */
63  		objectList.add(filler("80", 2));
64  		/* OFFICE ID (REQUESTER CODE SUFFIX) */
65  		objectList.add(filler(super.getOfficeId(request.getUserContext()), 3));
66  		/* EMPLOYEE ID */
67  		objectList
68  				.add(filler(super.getEmployeeID(request.getUserContext()), 2));
69  		/* DL/ID NUMBER */
70  		objectList.add(filler(cancelIdCardRequestEcs.getDlOrIdCardNumber(), 8));
71  		/* FIRST 3 POSITIONS LAST NAME */
72  		objectList.add(filler(cancelIdCardRequestEcs.getLastName(), 3));
73  		/* TYPE UPDATE CD (UPDATING SERVICE CD) */
74  		objectList.add(filler("S", 1));
75  		/* ID SERVICE CODE */
76  		String idServiceCode = "W";
77  		if (StringUtils.equals(cancelIdCardRequestEcs.getCancellationCode(),
78  				"L")) {
79  			idServiceCode = "V";
80  		}
81  		objectList.add(filler(idServiceCode, 1));
82  		/* ID SERVICE DATE */
83  		if (EaseUtil.isNullOrBlank(cancelIdCardRequestEcs.getProcessingDate())) {
84  			objectList.add(filler(super.getWorkDate(request.getUserContext(),
85  					EcsConverterHelper.FORMATTER_MM_DD_YY), 6));
86  		}
87  		else {
88  			objectList.add(filler(
89  					convertDateToString(cancelIdCardRequestEcs
90  							.getProcessingDate(),
91  							EcsConverterHelper.FORMATTER_MM_DD_YY), 6));
92  		}
93  		/* RESPONSE/ERROR CODE */
94  		objectList.add(filler(BLANK, 2));
95  		/* NAME OF RECORD */
96  		if (StringUtils.equals(cancelIdCardRequestEcs.getTransactionSubCode(),
97  				"I")) {
98  			objectList.add(filler(BLANK, 35));
99  		}
100 		/**** End Conversion ***/
101 		return super.createMessageFormat(objectList.toArray());
102 	}

107 	@Override
108 	public IEcsResponse createResponse(String anInboundMessagePayload) {
109 		CancelIdCardResponseEcs cancelIdCardResponseEcs = new CancelIdCardResponseEcs();
110 		Map <String, String> tcodeResponseElementsMap = this
111 				.buildTcodeResponseElementsMap(anInboundMessagePayload);
112 		String aStatusInfo = tcodeResponseElementsMap
113 				.get("RESPONSE_ERROR_CODE");
114 		//		CodeSetElement aStatusInfoCode = getCodeSetElement(
115 		//				ICodeSetNameConstants.DL_RETURN_CODE, aStatusInfo);//MUHRESCD	2	UPDATE CODE
116 		CodeSetElement aStatusInfoCode = new CodeSetElement();
117 		aStatusInfoCode.setCode(aStatusInfo);
118 		cancelIdCardResponseEcs.setResponseInfoCode(aStatusInfoCode);
119 		String transactionSubCode = tcodeResponseElementsMap
120 				.get("TRANSACTION_SUB_CODE");
121 		if (StringUtils
122 				.equals(transactionSubCode, INQUIRY_TRANSACTION_SUB_TYPE)) {
123 			cancelIdCardResponseEcs.setNameOfRecord(ConverterResponseHelper
124 					.buildPersonNameFromString(tcodeResponseElementsMap
125 							.get("NAME_OF_RECORD"), DELIMITER));
126 		}
127 		return cancelIdCardResponseEcs;
128 	}

130 	/***
131 	 * The purpose of this method is to build the tcode response elements map from the given flat message.
132 	 * 
133 	 * @param message the response string to the host 
134 	 * 
135 	 * @return the map of response elements 
136 	 */
137 	private Map <String, String> buildTcodeResponseElementsMap(String message) {
138 		List <TcodeElementNode> tcodeElementNodes = new ArrayList <TcodeElementNode>(
139 				0);
140 		/* HQ TRANSACTION CODE */
141 		tcodeElementNodes.add(new TcodeElementNode("HQ_TRANSACTION_CODE", 3));
142 		/* TYPE TRANS CODE */
143 		tcodeElementNodes.add(new TcodeElementNode("TYPE_TRANS_CODE", 3));
144 		/* TRANSACTION SUB-CODE */
145 		tcodeElementNodes.add(new TcodeElementNode("TRANSACTION_SUB_CODE", 1));
146 		/* REQUESTER CODE PREFIX */
147 		tcodeElementNodes.add(new TcodeElementNode("REQUESTER_CODE_PREFIX", 2));
148 		/* OFFICE ID (REQUESTER CODE SUFFIX) */
149 		tcodeElementNodes.add(new TcodeElementNode("OFFICE_ID", 3));
150 		/* EMPLOYEE ID */
151 		tcodeElementNodes.add(new TcodeElementNode("EMPLOYEE_ID", 2));
152 		/* DL/ID NUMBER */
153 		tcodeElementNodes.add(new TcodeElementNode("DL_ID_NUMBER", 8));
154 		/* FIRST 3 POSITIONS LAST NAME */
155 		tcodeElementNodes.add(new TcodeElementNode(
156 				"FIRST_3_POSITIONS_LAST_NAME", 3));
157 		/* TYPE UPDATE CD (UPDATING SERVICE CD) */
158 		tcodeElementNodes.add(new TcodeElementNode("TYPE_UPDATE_CD", 1));
159 		/* ID SERVICE CODE */
160 		tcodeElementNodes.add(new TcodeElementNode("ID_SERVICE_CODE", 1));
161 		/* ID SERVICE DATE */
162 		tcodeElementNodes.add(new TcodeElementNode("ID_SERVICE_DATE", 6));
163 		/* RESPONSE/ERROR CODE */
164 		tcodeElementNodes.add(new TcodeElementNode("RESPONSE_ERROR_CODE", 2));
165 		/* NAME OF RECORD */
166 		if (message.length() > 35) {
167 			tcodeElementNodes.add(new TcodeElementNode("NAME_OF_RECORD", 35));
168 		}
169 		/* Extract the tcode response elements */
170 		return ConverterResponseHelper.buildTcodeResponseElementsMap(message,
171 				tcodeElementNodes);
172 	}

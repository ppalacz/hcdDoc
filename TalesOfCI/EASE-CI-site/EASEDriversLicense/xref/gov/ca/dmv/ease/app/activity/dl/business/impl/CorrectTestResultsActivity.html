57  			DriveAndLawTestResultType testResultType) {
58  		DriveAndLawTestResult driveAndLawTestResult = testResultListFromCdaProcessContext
59  				.get(testResultListFromCdaProcessContext.size() - 1);
60  		//Checking if the code in the driveAndLawTestResult is *. If yes, proceed to remove the last entered result from

62  		if ((driveAndLawTestResult.getTestResultCode() != null)
63  				&& (driveAndLawTestResult.getTestResultCode().getCode()
64  						.equals("*"))) {
65  			DriverLicense driverLicense = cdaProcessContext.getDriverLicense();
66  			DriveAndLawTestResultGroup driveAndLawTestResultGroupfromDriverLicense = driverLicense
67  					.getDriveAndLawTestResultGroup();
68  			List <DriveAndLawTestResult> testResultListFromDL = driveAndLawTestResultGroupfromDriverLicense
69  					.getDriveAndLawTestResults(testResultType);
70  			int index1 = testResultListFromDL.size();
71  			if (testResultListFromDL != null
72  					&& (testResultListFromDL.size() > 0)) {
73  				DriveAndLawTestResult testResult = testResultListFromDL
74  						.get(index1 - 1);
75  				if (testResult != null) {
76  					driveAndLawTestResultGroupfromDriverLicense
77  							.deleteDriveAndLawTestResult(testResult);
78  					driverLicense
79  							.setDriveAndLawTestResultGroup(driveAndLawTestResultGroupfromDriverLicense);
80  					DlApplication dlApplication = cdaProcessContext
81  							.getDlApplication();
82  					dlApplication.setLicense(driverLicense);
83  					cdaProcessContext.setDlApplication(dlApplication);
84  				}
85  				cdaProcessContext.setDriveAndLawTestResultGroup(null);
86  			}

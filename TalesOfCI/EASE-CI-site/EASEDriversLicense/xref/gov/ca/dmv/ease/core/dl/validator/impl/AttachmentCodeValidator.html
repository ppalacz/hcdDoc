45  	public void validate(DlApplication dlApplication,
46  			DriveAndLawTestResultGroup driveAndLawTestResultGroup,
47  			CommercialTestResultGroup commercialTestResultGroup, String ttc,
48  			IErrorCollector errorCollector) {
49  		Person applicant = dlApplication.getApplicant();
50  		if (!EaseUtil.isNullOrBlank(applicant)) {
51  			DriverLicense driverLicense = applicant.getDriverLicense();
52  			if (!EaseUtil.isNullOrBlank(driverLicense)) {
53  				List <String> attachmentCodes = DlApplicationUtil
54  						.getAllAttachmentCodes(driverLicense);
55  				if (attachmentCodes != null) {
56  					validateAttachmentCodes(attachmentCodes, errorCollector);
57  					validate10Or13NotPresent(attachmentCodes,
58  							driveAndLawTestResultGroup, errorCollector);
59  					for (String attachmentCode : attachmentCodes) {
60  						validateDriverLicense(driverLicense, attachmentCode,
61  								errorCollector);
62  						validateTestResults(applicant, attachmentCode,
63  								driveAndLawTestResultGroup,
64  								commercialTestResultGroup, errorCollector);
65  						validateMedical(applicant, attachmentCode,
66  								errorCollector);
67  						if (ITtcConstants.DLE.equals(ttc)) {
68  							int limTermYears = driverLicense
69  									.getLimitedLicenseTermYears();
70  							validateLimitTerm(attachmentCode, limTermYears,
71  									errorCollector);
72  						}
73  					}
74  				}
75  			}
76  		}
77  	}

79  	/***
80  	 * @param attachmentCodes
81  	 * @param errorCollector
82  	 */
83  	public void validateAttachmentCodes(List <String> attachmentCodes,
84  			IErrorCollector errorCollector) {
85  		//D868
86  		if (ArrayUtils.contains(attachmentCodes, IAttachmentCode.A13)) {
87  			errorCollector.register(new EaseValidationException(
88  					IDataCollectionErrorMessageConstants.CANNOT_BE_KEYED));
89  		}
90  		//D660
91  		if (ArrayUtils.contains(attachmentCodes, IAttachmentCode.A23,
92  				IAttachmentCode.A52)) {
93  			errorCollector
94  					.register(new EaseValidationException(
95  							IDataCollectionErrorMessageConstants.ATTACH_CODE_23_52_OR_RESTR_63_CANNOT_BE_KEYED));
96  		}
97  		//D789
98  		if (ArrayUtils.contains(attachmentCodes, IAttachmentCode.A51)
99  				&& ArrayUtils.contains(attachmentCodes, IAttachmentCode.A54)) {
100 			errorCollector
101 					.register(new EaseValidationException(
102 							IDataCollectionErrorMessageConstants.ATTACHMENT_CODES_51_AND_54_CANNOT_BOTH_BE_PRESENT));
103 		}
104 		//Discontinued code
105 		if (ArrayUtils.contains(attachmentCodes, IAttachmentCode.A12,
106 				IAttachmentCode.A15, IAttachmentCode.A16, IAttachmentCode.A23,
107 				IAttachmentCode.A52, IAttachmentCode.A63, IAttachmentCode.A66,
108 				IAttachmentCode.A94, IAttachmentCode.A95)) {
109 			errorCollector.register(new EaseValidationException(
110 					"Discontinued attachment code used"));
111 		}
112 	}

129 	/***
130 	 * @param attachmentCodes
131 	 * @param driveAndLawTestResults
132 	 * @param errorCollector
133 	 */
134 	public void validate10Or13NotPresent(List <String> attachmentCodes,
135 			DriveAndLawTestResultGroup driveAndLawTestResultGroup,
136 			IErrorCollector errorCollector) {
137 		//D581
138 		if (ArrayUtils.contains(attachmentCodes, IAttachmentCode.A10,
139 				IAttachmentCode.A13)) {
140 			List <DriveAndLawTestResult> driveAndLawTestResults = null;
141 			if (driveAndLawTestResultGroup != null) {
142 				driveAndLawTestResults = driveAndLawTestResultGroup
143 						.getDriveAndLawTestResults(DriveAndLawTestResultType.CLASS_F_DRIVE_TEST_RESULT);
144 			}
145 			if (driveAndLawTestResults != null) {
146 				for (DriveAndLawTestResult driveAndLawTestResult : driveAndLawTestResults) {
147 					String testResultCode = driveAndLawTestResult
148 							.getTestResultCode().getCode();
149 					if (ILawTestConstants.PASS.equalsIgnoreCase(testResultCode)
150 							|| ILawTestConstants.SATISFACTORY
151 									.equalsIgnoreCase(testResultCode)
152 							|| ILawTestConstants.WAIVED
153 									.equalsIgnoreCase(testResultCode)) {
154 						errorCollector
155 								.register(new EaseValidationException(
156 										IDataCollectionErrorMessageConstants.ATTACH_CODE_10_VS_LIMITED_TERM));
157 					}
158 				}
159 			}
160 		}
161 	}

164 	 * @param driverLicense
165 	 * @param attachmentCode
166 	 * @param errorCollector
167 	 */
168 	public void validateDriverLicense(DriverLicense driverLicense,
169 			String attachmentCode, IErrorCollector errorCollector) {
170 		//D811
171 		if (IAttachmentCode.A17.equals(attachmentCode)) {
172 			List <String> appliedLicenseClassCodes = DlApplicationUtil
173 					.getAppliedLicenseClassCodes(driverLicense);
174 			for (String appliedLicenseClassCode : appliedLicenseClassCodes) {
175 				if (!IDlClassTypeConstants.COMMERCIAL_APP_CLASS_A
176 						.equals(appliedLicenseClassCode)
177 						&& !IDlClassTypeConstants.COMMERCIAL_APP_CLASS_B
178 								.equalsIgnoreCase(appliedLicenseClassCode)) {
179 					List <String> categoryCodes = DlApplicationUtil
180 							.getAllCategoryCodes(driverLicense);
181 					for (String categoryCode : categoryCodes) {
182 						if (!IDlClassTypeConstants.FIRE_FIGHTER_CLASSS_A
183 								.equalsIgnoreCase(categoryCode)
184 								&& !IDlClassTypeConstants.FIRE_FIGHTER_CLASSS_B
185 										.equalsIgnoreCase(categoryCode)) {
186 							errorCollector
187 									.register(new EaseValidationException(
188 											IDataCollectionErrorMessageConstants.CERT_OF_TRAINING_ONLY_ALLOWED_FOR_COMMERCIAL_OR_FIREFIGHTER));
189 							break;
190 						}
191 					}
192 				}
193 			}
194 		}
195 	}

203 	public void validateTestResults(Person applicant, String attachmentCode,
204 			DriveAndLawTestResultGroup driveAndLawTestResultGroup,
205 			CommercialTestResultGroup commercialTestResultGroup,
206 			IErrorCollector errorCollector) {
207 		Boolean isMinor = applicant.getMinor();
208 		List <CommercialTestResult> performanceTestResults = null;
209 		if (commercialTestResultGroup != null) {
210 			performanceTestResults = commercialTestResultGroup
211 					.getCommercialTestResults(CommercialTestResultType.PERFORMANCE_TEST_RESULT);
212 		}
213 		if (isNotAllowedForPerformanceTest(attachmentCode,
214 				performanceTestResults)) {
215 			if (Boolean.TRUE.equals(isMinor)) {
216 				//D605
217 				errorCollector
218 						.register(new EaseValidationException(
219 								IDataCollectionErrorMessageConstants.CERT_OF_TRNG_EXP_COMP_NOT_ALLOWED_FOR_MINOR));
220 			}
221 			else {
222 				//D709
223 				errorCollector
224 						.register(new EaseValidationException(
225 								IDataCollectionErrorMessageConstants.ATTACH_CODE_VS_DRIVE_MC_SKILL_RESULT));
226 			}
227 		}
228 		List <DriveAndLawTestResult> motorCycleTestResults = null;
229 		if (driveAndLawTestResultGroup != null) {
230 			motorCycleTestResults = driveAndLawTestResultGroup
231 					.getDriveAndLawTestResults(DriveAndLawTestResultType.CLASS_M_DRIVE_TEST_RESULT);
232 		}
233 		if (EaseUtil.isNullOrBlank(motorCycleTestResults)) {
234 			if (IAttachmentCode.A67.equals(attachmentCode)) {
235 				//D710
236 				errorCollector
237 						.register(new EaseValidationException(
238 								IDataCollectionErrorMessageConstants.ALLOWED_FOR_CLASS_M4_ONLY));
239 			}
240 		}
241 	}

243 	/***
244 	 * @param attachmentCode
245 	 * @param performanceTestResults
246 	 * @param errorCollector
247 	 * @return
248 	 */
249 	private boolean isNotAllowedForPerformanceTest(String attachmentCode,
250 			List <CommercialTestResult> performanceTestResults) {
251 		if (performanceTestResults != null) {
252 			for (CommercialTestResult performanceTestResult : performanceTestResults) {
253 				if (performanceTestResult != null) {
254 					String resultCode = CodeUtil
255 							.getCodeFromCodeSetElement(performanceTestResult
256 									.getTestResultCode());
257 					if ("X".equalsIgnoreCase(resultCode)
258 							&& IAttachmentCode.A17.equals(attachmentCode)) {
259 						return true;
260 					}
261 					if ("C".equalsIgnoreCase(resultCode)
262 							&& IAttachmentCode.A67.equals(attachmentCode)) {
263 						return true;
264 					}
265 				}
266 			}
267 		}
268 		return false;
269 	}

272 	 * @param applicant
273 	 * @param attachmentCode
274 	 * @param errorCollector
275 	 */
276 	public void validateMedical(Person applicant, String attachmentCode,
277 			IErrorCollector errorCollector) {
278 		Date medicalExpDate = applicant.getMedicalExpirationDate();
279 		if (IAttachmentCode.A51.equalsIgnoreCase(attachmentCode)) {
280 			if (medicalExpDate == null) {
281 				//D754
282 				errorCollector
283 						.register(new EaseValidationException(
284 								IDataCollectionErrorMessageConstants.MEDICAL_EXP_DATE_REQUIRED));
285 			}
286 			else {
287 				//D762
288 				Date twoYearsFromNow = DateUtil.addToSpecifiedDate(new Date(),
289 						2, 0, 0);
290 				if (medicalExpDate.after(twoYearsFromNow)) {
291 					errorCollector
292 							.register(new EaseValidationException(
293 									IDataCollectionErrorMessageConstants.MEDICAL_EXP_MUST_BE_WITHIN_2_YEARS));
294 				}
295 			}
296 		}
297 		List <String> attachmentCodes = DlApplicationUtil
298 				.getAllAttachmentCodes(applicant.getDriverLicense());
299 		if (ArrayUtils.contains(attachmentCodes, IAttachmentCode.A71,
300 				IAttachmentCode.A80, IAttachmentCode.A81, IAttachmentCode.A85)) {
301 			if (medicalExpDate == null || medicalExpDate.before(new Date())) {
302 				//D754
303 				errorCollector
304 						.register(new EaseValidationException(
305 								IDataCollectionErrorMessageConstants.MEDICAL_EXP_DATE_REQUIRED));
306 			}
307 		}
308 		else {
309 			//D791
310 			errorCollector
311 					.register(new EaseValidationException(
312 							IDataCollectionErrorMessageConstants.ATTACH_CODE_54_NOT_VALID_FOR_COMM_APP));
313 		}

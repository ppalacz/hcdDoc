50  		if (!(request instanceof OrganDonorDocumentRequestTus)) {
51  			addValidationExceptionInErrorCollector("IMPROPER_REQUEST_TYPE",
52  					new String[] { "OrganDonorDocumentRequestTus" }, aCollector);
53  		}
54  		else {
55  			OrganDonorDocumentRequestTus organDonorDocumentRequest = (OrganDonorDocumentRequestTus) request;
56  			Person person = organDonorDocumentRequest.getPerson();
57  			validatePerson(person, aCollector);
58  		}
59  	}

61  	/***
62  	 * Validate mailing address.
63  	 * 
64  	 * @param person the person
65  	 * @param aCollector the a collector
66  	 */
67  	private void validateMailingAddress(Person person,
68  			IErrorCollector aCollector) {
69  		Address mailingAddress = person.getAddress(AddressType.MAILING);
70  		if (isNullOrBlank(mailingAddress)) {
71  			addValidationExceptionInErrorCollector("NULL_MAILING_ADDRESS",
72  					null, aCollector);
73  		}
74  		else {
75  			if (isNullOrBlank(mailingAddress.getLine1())
76  					&& isNullOrBlank(mailingAddress.getLine2())) {
77  				addValidationExceptionInErrorCollector("EMPTY_MAILING_ADDRESS",
78  						null, aCollector);
79  			}
80  			if (isNullOrBlank(mailingAddress.getCity())) {
81  				addValidationExceptionInErrorCollector(
82  						"NULL_OR_EMPTY_MAILING_CITY", null, aCollector);
83  			}
84  			if (isNullOrBlank(mailingAddress.getState())) {
85  				addValidationExceptionInErrorCollector(
86  						"NULL_OR_EMPTY_MAILING_STATE", null, aCollector);
87  			}
88  			if (isNullOrBlank(mailingAddress.getZip())) {
89  				addValidationExceptionInErrorCollector(
90  						"NULL_OR_EMPTY_MAILING_ZIP", null, aCollector);
91  			}
92  		}
93  	}

98  	@Override
99  	protected void validatePerson(Person person, IErrorCollector aCollector) {
100 		if (person == null) {
101 			addValidationExceptionInErrorCollector("NULL_PERSON", null,
102 					aCollector);
103 		}
104 		else {
105 			validatePersonBirthDate(person, aCollector);
106 			validateLicense(person, aCollector);
107 			validateMailingAddress(person, aCollector);
108 			validatePersonName(person.getCurrentName(), aCollector);
109 		}
110 	}
111 
112 	/***
113 	 * Validate Person Current Name.
114 	 * 
115 	 * @param currentName the current name
116 	 * @param aCollector the Error Collector
117 	 */
118 	private void validatePersonName(PersonName currentName, IErrorCollector aCollector) {
119 		if (isNullorEmpty(currentName)) {
120 			addValidationExceptionInErrorCollector(
121 					"NULL_CURRENT_NAME", null, aCollector);
122 		}
123 		else if (isNullorEmpty(currentName.getFullName())) {
124 			addValidationExceptionInErrorCollector(
125 					"NULL_CURRENT_NAME", null, aCollector);
126 		}
127 	}
128 
129 	/***
130 	 * Validate license.
131 	 * 
132 	 * @param aCollector the a collector
133 	 * @param person the person
134 	 */
135 	private void validateLicense(Person person, IErrorCollector aCollector) {
136 		DriverLicense driverLicense = person.getDriverLicense();
137 		IdCard idCard = person.getIdCard();
138 		if (isNullorEmpty(driverLicense) && isNullorEmpty(idCard)) {
139 			addValidationExceptionInErrorCollector(
140 					"NULL_DRIVERLICENSE_AND_IDCARD", null, aCollector);
141 		}
142 		else if (!isNullorEmpty(driverLicense)) {
143 			validateLicenseNumber(driverLicense.getLicenseNumber(), aCollector);
144 		}
145 		else if (!isNullorEmpty(idCard)) {
146 			validateLicenseNumber(idCard.getLicenseNumber(), aCollector);
147 		}
148 	}

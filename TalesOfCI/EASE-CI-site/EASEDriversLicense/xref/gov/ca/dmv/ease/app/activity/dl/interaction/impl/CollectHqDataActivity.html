29  public class CollectHqDataActivity extends InteractionActivity implements
30  		ICnaCapable {
31  	/*** The Constant serialVersionUID. */
32  	private static final long serialVersionUID = -211632185014100494L;
33  	/*** The headQuartersInquiryRules for validation */
34  	private HeadQuartersInquiryRules headQuartersInquiryRules;
35  
36  	/* (non-Javadoc)
37  	 * @see gov.ca.dmv.ease.app.activity.impl.Activity#postCondition(gov.ca.dmv.ease.app.context.impl.ProcessContext)
38  	 */
39  	@Override
40  	protected void postCondition(ProcessContext processContext) {
41  		String ttc = processContext.getUserContext().getTtc();
42  		if(ArrayUtils.contains(ttc, ITtcConstants.DLA, ITtcConstants.DLC,
43  				ITtcConstants.DLD, ITtcConstants.DLE, ITtcConstants.DLP, ITtcConstants.DRT)) {
44  			if (isNotByPassValidations(processContext)) {
45  				headQuartersInquiryRules.validate(processContext);
46  				populateErrorMessages(processContext, headQuartersInquiryRules
47  						.getErrorCollector());
48  			}
49  		}
50  	}
51  
52  	/***
53  	 * Set headQuartersInquiryRules.
54  	 * 
55  	 * @param headQuartersInquiryRules the headQuartersInquiryRules to set
56  	 */
57  	public void setHeadQuartersInquiryRules(
58  			HeadQuartersInquiryRules aHeadQuartersInquiryRules) {
59  		headQuartersInquiryRules = aHeadQuartersInquiryRules;
60  	}
61  
62  	/* (non-Javadoc)
63  	 * @see gov.ca.dmv.ease.app.activity.ICnaCapable#isCnaActionAllowed(gov.ca.dmv.ease.app.context.impl.ProcessContext)
64  	 */
65  	public boolean isCnaActionAllowed(ProcessContext processContext) {
66  		HqInquiryProcessContext hqInquiryProcessContext = (HqInquiryProcessContext) processContext;
67  		boolean licenseNumberProvided = false;
68  		if (!EaseUtil.isNullOrBlank(hqInquiryProcessContext.getDriverLicense())) {
69  			licenseNumberProvided = !EaseUtil
70  					.isNullOrBlank(hqInquiryProcessContext.getDriverLicense()
71  							.getLicenseNumber());
72  		}
73  		if (licenseNumberProvided && !hqInquiryProcessContext.isCamvAvailable()) {
74  			return true;
75  		}
76  		else if (licenseNumberProvided
77  				&& !hqInquiryProcessContext.isDlAvailable()) {
78  			return true;
79  		}
80  		else if (!licenseNumberProvided
81  				&& !hqInquiryProcessContext.isAniDatabaseAvailable()
82  				&& hqInquiryProcessContext.isInventoryDatabaseAvailable()) {
83  			return true;
84  		}
85  		else {
86  			return false;
87  		}
88  	}

35  	@Override
36  	public void validateRequest(IEcsRequest request, IErrorCollector aCollector) {
37  		if (!(request instanceof UpdateExtendedNameIndexRequestEcs)) {
38  			addValidationExceptionInErrorCollector(
39  					EN2_REQUEST_MISMATCH, aCollector);
40  			return;
41  		}
42  		UpdateExtendedNameIndexRequestEcs updateExtendedNameIndexRequest = (UpdateExtendedNameIndexRequestEcs) request;
43  		Person person = updateExtendedNameIndexRequest.getPerson();
44  		if (isNullorEmpty(person)) {
45  			addValidationExceptionInErrorCollector(
46  					EN2_PERSON_UNAVAILABLE, aCollector);
47  			
48  		}
49  		PersonName name = person.getCurrentName();
50  		if (!(name instanceof PersonName)) {
51  			addValidationExceptionInErrorCollector(
52  					EN2_NAME_CAST_ERROR, aCollector);
53  			return;
54  		}
55  		else {
56  			PersonName personName = name;
57  			if (isNullOrBlank(personName)) {
58  				
59  				addValidationExceptionInErrorCollector(
60  						EN2_PERSON_NAME_UNAVAILABLE, aCollector);
61  			
62  				return;
63  			}
64  		}
65  		PersonName personName = person.getCurrentName();
66  		validatePersonLastFirstMiddleSuffxName(personName, aCollector);
67  		//ValidatorUtil.validatePersonBirthDate(person, aCollector);
68  	}
69  }

27  public class IsVoucherCreditedDecision extends DecisionActivity implements
28  		IDlaDecisionActivityConstantsForBpDefinition {
29  	/*** The Constant serialVersionUID. */
30  	private static final long serialVersionUID = 8258773223543690851L;
31  
32  	/***
33  	 * This method evaluates if fee has been waived. 
34  	 * If the fee is waived it returns true, if not returns false.
35  	 */
36  	@Override
37  	protected String evaluate(ProcessContext context) {
38  		DlaProcessContext dlaProcessContext = (DlaProcessContext) context;
39  		List <Voucher> vouchers = dlaProcessContext.getDlApplication()
40  				.getInvoice().getVoucherPayments();
41  		Boolean isVoucherCredited = false;
42  		for (Voucher voucher : vouchers) {
43  			if (voucher.getAmount() != null
44  					&& ((voucher.getStatusCode().getCode().equals("C")))) {
45  				isVoucherCredited = true;
46  				break;
47  			}
48  		}
49  		return isVoucherCredited ? VOUCHER_CREDITED : VOUCHER_NOT_CREDITED;
50  	}
51  }

19  /***
20   * Description: The purpose of this class is to validate the request for TCODE DCF.
21   * File: FailureToPayDCFRequestEcsValidator.java
22   * Module:  gov.ca.dmv.ease.ecs.request.validator.impl

29  public class FailureToPayDCFRequestEcsValidator extends EcsRequestValidator {
30  	/*** The Constant serialVersionUID. */
31  	private static final long serialVersionUID = 1672596280202930261L;
32  	/*** The DCF TCODE. */
33  	private static final String DCF_TCODE = "DCF";
34  	/*** The Request. */
35  	private static final String REQUEST = "FailureToPayDCFRequestEcs";
36  	/*** The Converter. */
37  	private static final String CONVERTER = "FailureToPayDCFConverter";
38  	/*** The TTC. */
39  	private static final String TTC = "FTP";
40  
41  	/***
42  	 * Constructor.
43  	 */
44  	public FailureToPayDCFRequestEcsValidator() {
45  		super();
46  	}
47  
48  	/* (non-Javadoc)
49  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
50  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
51  	 */
52  	@Override
53  	protected void validateRequest(IEcsRequest request,
54  			IErrorCollector aCollector) {
55  		if (request == null) {
56  			this.addValidationExceptionInErrorCollector(NULL_REQUEST,
57  					new String[] { DCF_TCODE }, aCollector);
58  		}
59  		else {
60  			if (!(request instanceof FailureToPayDCFRequestEcs)) {
61  				this.addValidationExceptionInErrorCollector(
62  						IMPROPER_REQUEST_TYPE, new String[] { REQUEST },
63  						aCollector);
64  			}
65  			else {
66  				if (!(request.getMessageConverter() instanceof FailureToPayDCFConverter)) {
67  					this.addValidationExceptionInErrorCollector(
68  							IMPROPER_MESSAGE_CONVERTER_ERROR_MESSAGE,
69  							new String[] { CONVERTER }, aCollector);
70  				}
71  				if (!(StringUtils
72  						.equals(request.getUserContext().getTtc(), TTC))) {
73  					this.addValidationExceptionInErrorCollector(
74  							INVALID_TTC_ERROR_MESSAGE, new String[] { TTC },
75  							aCollector);
76  				}
77  				FailureToPayDCFRequestEcs failureToPayDCFRequestEcs = (FailureToPayDCFRequestEcs) request;
78  				Person dlApplicant = failureToPayDCFRequestEcs.getDlApplicant();
79  				if (EaseUtil.isNullOrBlank(dlApplicant)) {
80  					this.addValidationExceptionInErrorCollector(
81  							PERSON_IS_EMPTY_ERROR_MESSAGE, aCollector);
82  				}
83  				else {
84  					//Driver Name
85  					if (EaseUtil.isNullOrBlank(dlApplicant.getCurrentName()
86  							.getLastName())) {
87  						this.addValidationExceptionInErrorCollector(
88  								PERSON_LAST_NAME_IS_EMPTY_ERROR_MESSAGE,
89  								aCollector);
90  					}
91  					//Birth Date
92  					if (EaseUtil.isNullOrBlank(dlApplicant.getBirthDate())) {
93  						this.addValidationExceptionInErrorCollector(
94  								PERSON_DATE_OF_BIRTH_IS_NULL_ERROR_MESSAGE,
95  								aCollector);
96  					}
97  					DriverLicense driverLicense = dlApplicant
98  							.getDriverLicense();
99  					//Driver License
100 					if (EaseUtil.isNullOrBlank(driverLicense)) {
101 						this.addValidationExceptionInErrorCollector(
102 								DRIVER_LICENSE_IS_EMPTY_ERROR_MESSAGE,
103 								aCollector);
104 					}
105 					//License Number
106 					else if (EaseUtil.isNullOrBlank(driverLicense
107 							.getLicenseNumber())) {
108 						this.addValidationExceptionInErrorCollector(
109 								DRIVER_LICENSE_NUMBER_IS_EMPTY_ERROR_MESSAGE,
110 								aCollector);
111 					}
112 				}//End of Else for Person Check
113 			}//End of Check for Request
114 		}//End of Check for Request Type
115 	}
116 }

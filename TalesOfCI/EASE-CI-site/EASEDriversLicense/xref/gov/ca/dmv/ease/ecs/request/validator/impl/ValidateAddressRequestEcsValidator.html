35  		implements IRequestValidator, IErrorMessageConstants {
36  	/*** The Constant serialVersionUID. */
37  	private static final long serialVersionUID = 3825034943909846679L;

39  	/***
40  	 * Validates address received in inquiry request.
41  	 * 
42  	 * @param address the address.
43  	 * @param aCollector the Error Collector
44  	 */
45  	private void validateAddress(Address address, IErrorCollector aCollector) {
46  		if (!isNotNull(address)) {
47  			aCollector.register(new EcsValidationException(
48  					VAL_MSG_NULL_ADDRESS_ERROR_MESSAGE));
49  		}
50  		else {
51  			if (!isNotNull(address.getCity())) {
52  				aCollector.register(new EcsValidationException(
53  						VAL_MSG_NULL_OR_EMPTY_CITY_NAME_ERROR_MESSAGE));
54  			}
55  			if (!isNotNull(address.getLine1())) {
56  				aCollector.register(new EcsValidationException(
57  						VAL_MSG_NULL_OR_EMPTY_ADDRESS_LINE1_ERROR_MESSAGE));
58  			}
59  		}
60  	}
61  
62  	/***
63  	 * Validates addresses received in inquiry request.
64  	 * 
65  	 * @param addresses the list of addresses.
66  	 * @param aCollector the Error Collector
67  	 */
68  	private void validateAddresses(List <Address> addresses,
69  			IErrorCollector aCollector) {
70  		if (isNullOrBlank(addresses)) {
71  			aCollector.register(new EcsValidationException(
72  					"A request has empty or null list of addresses"));
73  		}
74  		else if (!(addresses.size() == 1 || addresses.size() == 2)) {
75  			aCollector.register(new EcsValidationException(
76  					VAL_MSG_WRONG_NUMBER_OF_ADDRESSES_ERROR_MESSAGE));
77  		}
78  		else {
79  			for (int i = 0; i < addresses.size() - 1; i++) {
80  				Address address = addresses.get(i);
81  				validateAddress(address, aCollector);
82  			}
83  		}
84  	}
85  
86  	/***
87  	 * Validates the address inquiry request.
88  	 * 
89  	 * @param request the request
90  	 * @param aCollector the Error Collector
91  	 */
92  	@Override
93  	protected void validateRequest(IEcsRequest request,
94  			IErrorCollector aCollector) {
95  		ValidateAddressRequestEcs validateAddressRequestEcs = null;
96  		try {
97  			validateAddressRequestEcs = (ValidateAddressRequestEcs) request;
98  			List <Address> addresses = validateAddressRequestEcs.getAddresses();
99  			validateAddresses(addresses, aCollector);
100 		}
101 		catch (ClassCastException e) {
102 			aCollector.register(new EcsValidationException(
103 					WRONG_REQUEST_TYPE_ERROR_MESSAGE
104 							+ request.getClass().getName()));
105 		}
106 	}
107 }

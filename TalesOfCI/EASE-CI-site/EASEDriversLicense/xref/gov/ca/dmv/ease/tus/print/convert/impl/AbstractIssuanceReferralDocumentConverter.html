48  	/***
49  	 * Gets the list of common elements for Issuance Referral Document.
50  	 * 
51  	 * @param requestFromClient the request from client
52  	 * @param application the application
53  	 * @param applicant the applicant
54  	 * 
55  	 * @return the string
56  	 */
57  	public List <String> getCommonElementsList(
58  			PrintServiceRequest requestFromClient, Application application,
59  			Person applicant) {
60  		IUserContext currentUserContext = requestFromClient.getUserContext();
61  		List <String> elementsList = new ArrayList <String>();
62  		if (!EaseUtil.isNotNull(requestFromClient)) {
63  			throw new MessageConversionException(
64  					IPrintServiceErrorMessageConstants.ISSUANCE_REFERRAL_DOCUMENT_REQUEST_IS_NULL_ERROR_MESSAGE);
65  		}
66  		addToElementsList(elementsList, getIssuanceRefHeader(
67  				currentUserContext, application),
68  				IPrintConverterConstants.ISSUANCE_REFERRAL_HEADER); //Issuance Referral Header
69  		addToElementsList(
70  				elementsList,
71  				DocumentConverterUtil.getApplicationTypeCode(application),
72  				IPrintConverterConstants.ISSUANCE_REFERRAL_TYPE_OF_APPLICATION_TEXT); //Type of Application Text
73  		addLicenseNumberAndHeader(elementsList, application, applicant); //DL/SP # Header and License Number
74  		addToElementsList(elementsList, DocumentConverterUtil
75  				.getDlExpDateOrExpText(applicant.getDriverLicense()),
76  				IPrintConverterConstants.ISSUANCE_REFERRAL_DL_EXP_DATE); //Expiration Date or Text for DL. 
77  		addToElementsList(elementsList, DocumentConverterUtil
78  				.getIdExpDateOrExpText(applicant.getIdCard()),
79  				IPrintConverterConstants.ISSUANCE_REFERRAL_ID_EXP_DATE); //Expiration Date or Text for ID. 
80  		addToElementsList(elementsList, DocumentConverterUtil
81  				.getFormattedCurrentName(applicant.getCurrentName()),
82  				IPrintConverterConstants.ISSUANCE_REFERRAL_CURRENT_NAME); //Applican't Full Name. 
83  		//New Restrictions are present only on the Issuance Referral Document template.
84  		if (requestFromClient instanceof IssuanceReferralDocumentRequestTus) {
85  			addRestrictions(elementsList, applicant); //Restrictions
86  		}
87  		addToElementsList(elementsList, getAkaName(applicant),
88  				IPrintConverterConstants.ISSUANCE_REFERRAL_AKA_NAME); //Applican't AKA Name. 
89  		Address currentMailingAddress = applicant
90  				.getAddress(AddressType.MAILING);
91  		addMailingAddressToElementList(elementsList, currentMailingAddress); //Mailing Address.
92  		Address currentResidentialAddress = applicant
93  				.getAddress(AddressType.RESIDENTIAL);
94  		addResidenceAddressToElementList(elementsList,
95  				currentResidentialAddress); //Residence Address.
96  		//Applicant's physical details such as Sex, Hair color, Eye color, Height, Weight. 
97  		addPersonPhysicalDetailsToElementList(elementsList, applicant);
98  		addToElementsList(elementsList, DocumentConverterUtil.getBirthDate(
99  				applicant, FORMATTER_MM_SLASH_DD_SLASH_YY),
100 				IPrintConverterConstants.ISSUANCE_REFERRAL_PERSON_BIRTH_DATE); //Applican't birth date. 
101 		addIncompleteMsgToElementList(elementsList, application); //Application Incomplete Message.
102 		// Add Photo information such as Photo Office ID, Photo taken date, Micro graphics #,  
103 		addPhotoInfoToElementList(elementsList, applicant, currentUserContext);
104 		addEmployeeIdToElementList(elementsList, application,
105 				currentUserContext); //Employee Id.
106 		//IPrintConverterConstants.ISSUANCE_REFERRAL_MANAGER_ID); //Manager ID is never used in existing system so use blank. 
107 		return elementsList;
108 	}

110 	/***
111 	 * Gets the issuance referral header based on -
112 	 * 1.production or training environment
113 	 * 2.Type of original application in process.
114 	 * 
115 	 * @param currentUserContext the current user context
116 	 * @param application the application
117 	 * 
118 	 * @return the issuance ref header
119 	 */
120 	private String getIssuanceRefHeader(IUserContext currentUserContext,
121 			Application application) {
122 		if (DocumentConverterUtil.isTrainingEnvironment(currentUserContext)) {
123 			//Current environment is the Training environment. 
124 			if (application instanceof DlApplication) {
125 				//Original Application in process is not salesperson application.
126 				return IPrintConverterConstants.TRAINING_NON_OCCUPATIONAL_LICENSE_REFERRAL;
127 			}
128 			else if (application instanceof SalesPersonApplication) {
129 				//Original Application in process is salesperson application.
130 				return IPrintConverterConstants.TRAINING_OCCUPATIONAL_LICENSE_REFERRAL;
131 			}
132 		}
133 		else {
134 			//Current environment is the Production environment.
135 			if (application instanceof DlApplication) {
136 				//Original Application in process is not salesperson application.
137 				return IPrintConverterConstants.PRODUCTION_NON_OCCUPATIONAL_LICENSE_REFERRAL;
138 			}
139 			else if (application instanceof SalesPersonApplication) {
140 				//Original Application in process is salesperson application.
141 				return IPrintConverterConstants.PRODUCTION_OCCUPATIONAL_LICENSE_REFERRAL;
142 			}
143 		}
144 		return BLANK;
145 	}

148 	 * Gets the photo date based on -
149 	 * 1.production or training environment
150 	 * 
151 	 * @param currentUserContext the current user context
152 	 * @param applicant the applicant
153 	 * 
154 	 * @return the photo date
155 	 */
156 	private String getPhotoDate(IUserContext currentUserContext,
157 			Person applicant) {
158 		if (DocumentConverterUtil.isTrainingEnvironment(currentUserContext)) {
159 			//If Current environment is the Training environment then use current system date. 
160 			return convertDateToString(new Date(),
161 					FORMATTER_MM_SLASH_DD_SLASH_YY);
162 		}
163 		else {
164 			//If Current environment is not the Training environment then use photo date.
165 			if (EaseUtil.isNotNull(DocumentConverterUtil
166 					.getPhotoInformation(applicant))) {
167 				return convertDateToString(DocumentConverterUtil
168 						.getPhotoInformation(applicant).getPhotoTakenDate(),
169 						FORMATTER_MM_SLASH_DD_SLASH_YY);
170 			}
171 		}
172 		return BLANK;
173 	}

175 	/***
176 	 * Adds the restrictions.
177 	 * 
178 	 * @param elementsList the elements list
179 	 * @param applicant the applicant
180 	 */
181 	private void addRestrictions(List <String> elementsList, Person applicant) {
182 		if (EaseUtil.isNotNull(applicant.getDriverLicense())
183 				&& !EaseUtil.isNullOrBlank(applicant.getDriverLicense()
184 						.getNewRestrictions())) {
185 			//Display Restrictions headers and all driver license restrictions codes only for Issuance Referral Document.
186 			addToElementsList(
187 					elementsList,
188 					IPrintConverterConstants.RESTIRCTIONS_HEADER,
189 					IPrintConverterConstants.ISSUANCE_REFERRAL_RESTRICTIONS_HEADER); //Restriction Header. 
190 		}
191 		List <DrivingRestriction> newRestrictions = null;
192 		if (EaseUtil.isNotNull(applicant.getDriverLicense())) {
193 			newRestrictions = applicant.getDriverLicense().getNewRestrictions();
194 		}
195 		DocumentConverterUtil.addRestrictionsToElementList(elementsList,
196 				newRestrictions,
197 				IPrintConverterConstants.ISSUANCE_REFERRAL_RESTRICTIONS); //New Restrictions
198 	}

201 	 * Adds the license number and header.
202 	 * 
203 	 * @param elementsList the elements list
204 	 * @param application the application
205 	 * @param applicant the applicant
206 	 */
207 	private void addLicenseNumberAndHeader(List <String> elementsList,
208 			Application application, Person applicant) {
209 		if (application instanceof DlApplication
210 				|| application instanceof DriverRecordUpdateApplication) {
211 			//For DL Application or 30U application header is DL# and number is DL or ID number.
212 			addToElementsList(elementsList,
213 					IPrintConverterConstants.DL_ID_NUMBER_HEADER,
214 					IPrintConverterConstants.ISSUANCE_REFERRAL_DL_OR_SP_HEADER); // DL/SP # Header.
215 			addToElementsList(elementsList, DocumentConverterUtil
216 					.getDlorIdNumber(applicant),
217 					IPrintConverterConstants.ISSUANCE_REFERRAL_DL_OR_SP_NUMBER); // DL/ID Number.
218 		}
219 		else if (EaseUtil
220 				.isNotNull(application instanceof SalesPersonApplication)) {
221 			//For SP Application header is SP# and number is SP license number.
222 			addToElementsList(elementsList,
223 					IPrintConverterConstants.SP_NUMBER_HEADER,
224 					IPrintConverterConstants.ISSUANCE_REFERRAL_DL_OR_SP_HEADER); // DL/SP # Header.
225 			VehicleSalesPerson spApplicant = ((SalesPersonApplication) application)
226 					.getApplicant();
227 			if (EaseUtil.isNotNull(spApplicant)
228 					&& EaseUtil.isNotNull(spApplicant.getSalesPersonLicense())) {
229 				addToElementsList(
230 						elementsList,
231 						spApplicant.getSalesPersonLicense().getLicenseNumber(),
232 						IPrintConverterConstants.ISSUANCE_REFERRAL_DL_OR_SP_NUMBER); //SalesePerson Number.
233 			}
234 		}
235 	}

333 	 * 
334 	 * @param elementsList the elements list
335 	 * @param application the application
336 	 */
337 	private void addIncompleteMsgToElementList(List <String> elementsList,
338 			Application application) {
339 		if (!(application instanceof DlApplication)) {
340 			return;
341 		}
342 		DlApplication dlApplication = (DlApplication) application;
343 		//Application Incomplete message is applicable only for dlApplication.
344 		if (EaseUtil.isNotNull(dlApplication)
345 				&& EaseUtil.isNotNull(dlApplication
346 						.getNewIncompleteApplicationReasonCode())) {
347 			if (!(dlApplication
348 					.getNewIncompleteApplicationReasonCode()
349 					.getCode()
350 					.equalsIgnoreCase(
351 							IPrintConverterConstants.PHOTO_INCOMPLETE_APPLICATION) || dlApplication
352 					.getNewIncompleteApplicationReasonCode()
353 					.getCode()
354 					.equalsIgnoreCase(
355 							IPrintConverterConstants.MAGNETIC_STRIPE_INCOMPLETE_APPLICATION))) {
356 				addToElementsList(
357 						elementsList,
358 						IPrintConverterConstants.APPLICATION_INCOMPLETE_MESSAGE_TEXT,
359 						IPrintConverterConstants.ISSUANCE_REFERRAL_APPLICATION_INCOMPLETE_MESSAGE);
360 			}
361 		}

428 			//If there was no transaction prior to the current transaction then use current employee ID.
429 			if (EaseUtil.isNotNull(currentUserContext)) {
430 				addToElementsList(elementsList, currentUserContext.getTechId(),
431 						IPrintConverterConstants.ISSUANCE_REFERRAL_EMPLOYEE_ID);

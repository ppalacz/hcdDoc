44  	private boolean areRequiredDriveTestPassed(DriverLicense driverLicense,
45  			DriveAndLawTestResultGroup existingDriveAndLawTestResultGroup,
46  			CommercialTestResultGroup existingCommercialTestResultGroup) {
47  		// check if DriveTestReqired = DriveTestPassed
48  		TestResult1RequiredExamsValidator testResult1RequiredExamsValidator = new TestResult1RequiredExamsValidator();
49  		IErrorCollector anErrorCollector = new ErrorCollector();
50  		testResult1RequiredExamsValidator.validateDriveTest(driverLicense,
51  				existingDriveAndLawTestResultGroup, anErrorCollector);
52  		if (!anErrorCollector.isEmpty()) {
53  			return false;
54  		}
55  		TestResults2RequiredExamsValidator testResult2RequiredExamsValidator = new TestResults2RequiredExamsValidator();
56  		testResult2RequiredExamsValidator.validateDriveTest(driverLicense,
57  				existingCommercialTestResultGroup, anErrorCollector);
58  		if (!anErrorCollector.isEmpty()) {
59  			return false;
60  		}
61  		return true;
62  	}

64  	private boolean checkLicenseClasses(List <CodeSetElement> licenseClasses,
65  			String[] licClassesToBeSearched) {
66  		List <String> availableLicClasses = new ArrayList <String>();
67  		for (CodeSetElement licenseClass : licenseClasses) {
68  			availableLicClasses.add(licenseClass.getCode());
69  		}
70  		for (String licClass : licClassesToBeSearched) {
71  			if (availableLicClasses.contains(licClass)) {
72  				return true;
73  			}
74  		}
75  		return false;
76  	}

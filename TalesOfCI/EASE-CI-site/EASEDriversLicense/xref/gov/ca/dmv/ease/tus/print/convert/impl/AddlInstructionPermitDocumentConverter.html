38  public class AddlInstructionPermitDocumentConverter extends
39  		AbstractDocumentConverter {
40  	/***
41  	 * Populates outgoing XML message for Additional Instruction Permit Document
42  	 * 
43  	 * @param requestFromClient
44  	 *            the request from client
45  	 * 
46  	 * @return the string
47  	 */
48  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
49  		if (requestFromClient == null) {
50  			throw new MessageConversionException(
51  					"Invalid Additional Instruction Permit Document Request: Unable to process on null request");
52  		}
53  		AddlInstructionPermitDocumentRequestTus request = (AddlInstructionPermitDocumentRequestTus) requestFromClient;
54  		IUserContext userContext = request.getUserContext();
55  		DlApplication dlApplication = request.getDlApplication();
56  		Person person = dlApplication.getApplicant();
57  		List <String> elementsList = new ArrayList <String>();
58  		elementsList.add(getDlorIdNumber(person) + FIELD_DELIMITER
59  				+ "DRIVER_LICENSE_ID_NUMBER");
60  		elementsList.add(getFormattedPhotoName(person.getDriverLicense()
61  				.getPhotoInformation().getPhotoName())
62  				+ FIELD_DELIMITER + "CURRENT_NAME");
63  		if (isTrainingEnvironment(userContext)) {
64  			elementsList.add(TRAINING_HEADER + FIELD_DELIMITER
65  					+ "TRAINING_HEADER");
66  		}
67  		List <DrivingRestriction> newRestrictions = person.getDriverLicense()
68  				.getNewRestrictions();
69  		if (newRestrictions != null && newRestrictions.size() > 0) {
70  			for (int count = 0; count < newRestrictions.size(); count++) {
71  				elementsList.add(newRestrictions.get(count)
72  						.getRestrictionTypeCode().getCode()
73  						+ FIELD_DELIMITER + "RESTRICTION_TEXT_" + ++count);
74  				count--;
75  			}
76  		}
77  		elementsList.add(convertDateToString(
78  				dlApplication.getApplicationDate(), FORMATTER_MM_dd_yy)
79  				+ FIELD_DELIMITER + "TRANS_DATE");
80  		if (EaseUtil.isNotNull(dlApplication.getOriginalProcessHistory())) {
81  			elementsList.add(dlApplication.getOriginalProcessHistory()
82  					.getUserContext().getOfficeId()
83  					+ FIELD_DELIMITER + "OFFICE_ID");
84  			elementsList.add(dlApplication.getOriginalProcessHistory()
85  					.getUserContext().getTechId()
86  					+ FIELD_DELIMITER + "EMPLOYEE_ID");
87  			elementsList.add(dlApplication.getOriginalProcessHistory()
88  					.getUserContext().getControlCashierSeqNumber()
89  					+ FIELD_DELIMITER + "APP_CASHIER_SEQUENCE");
90  		}
91  		return convertToInputXml(elementsList, "LDD0612");
92  	}
93  }

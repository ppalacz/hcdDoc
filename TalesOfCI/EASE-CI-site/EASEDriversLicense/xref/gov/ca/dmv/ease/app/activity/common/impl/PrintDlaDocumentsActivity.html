53  	@Override
54  	protected void invokeService(ProcessContext processContext) {
55  		PrintRequestTs tsRequest = null;
56  		Set <PrintRequestTypes> printRequestTypes = new HashSet <PrintRequestTypes>();
57  		if (processContext instanceof DlaProcessContext) {
58  			DlaProcessContext dlaProcessContext = (DlaProcessContext) processContext;
59  			tsRequest = new PrintRequestTs(dlaProcessContext.getUserContext(),
60  					dlaProcessContext.getDlApplication());
61  		}
62  		else if (processContext instanceof IdProcessContext) {
63  			IdProcessContext idProcessContext = (IdProcessContext) processContext;
64  			tsRequest = new PrintRequestTs(idProcessContext.getUserContext(),
65  					idProcessContext.getDlApplication());
66  		}
67  		//Getting the document/receipt types to be printed
68  		printRequestTypes = determineDocumentsTobePrinted(processContext,
69  				printRequestTypes);
70  		//By default payment receipt to be printed for every transaction
71  		printRequestTypes.add(PrintRequestTypes.DL_RECEIPT);
72  		//Calling the print service if there are any documents to be printed
73  		if (printRequestTypes.size() > 0) {
74  			tsRequest.setPrintRequestTypes(printRequestTypes);
75  			PrintResponseTs responseTs = issueDlTaskService.execute(tsRequest);
76  			if (responseTs.hasErrors()) {
77  				//this.populateErrorMessages(context, responseTs.getErrorCollector());
78  			}
79  		}
80  	}

82  	/***
83  	 * @param dlaProcessContext
84  	 * @param printRequestTypes
85  	 * @return
86  	 */
87  	private Set <PrintRequestTypes> determineDocumentsTobePrinted(
88  			ProcessContext processContext,
89  			Set <PrintRequestTypes> printRequestTypes) {
90  		if (processContext instanceof DlaProcessContext) {
91  			DlaProcessContext dlaProcessContext = (DlaProcessContext) processContext;
92  			if (dlaProcessContext.getCurrentCDLISRecord() != null) {
93  				printRequestTypes
94  						.add(PrintRequestTypes.CDLIS_AND_NDR_MATCHES_REPORT);
95  			}
96  			if (dlaProcessContext.getCurrentPDPSRecord() != null) {
97  				printRequestTypes
98  						.add(PrintRequestTypes.PDPS_AND_NDR_MATCHES_REPORT);
99  			}
100 			if (dlaProcessContext.getDlApplication().isOrganDonor()) {
101 				printRequestTypes.add(PrintRequestTypes.ORGAN_DONOR_1_LDD0851);
102 				// FIXME how to determine whether its a Non-Renewal Organ Donor
103 				// Receipt
104 				printRequestTypes.add(PrintRequestTypes.ORGAN_DONOR_2_LDD0852);
105 			}
106 			// TODO check with the UI/COdesetElement how to check whether Voter
107 			// registration is requested or not
108 			if (dlaProcessContext.getDlApplication()
109 					.getVoterRegistationRequested().getCode().equals("Y")) {
110 				printRequestTypes
111 						.add(PrintRequestTypes.VOTER_REGISTRATION_LDD0861);
112 			}
113 		}
114 		else if (processContext instanceof IdProcessContext) {
115 			IdProcessContext idProcessContext = (IdProcessContext) processContext;
116 			if (idProcessContext.getDlApplication().isOrganDonor()) {
117 				printRequestTypes.add(PrintRequestTypes.ORGAN_DONOR_1_LDD0851);
118 				// FIXME how to determine whether its a Non-Renewal Organ Donor
119 				// Receipt
120 				printRequestTypes.add(PrintRequestTypes.ORGAN_DONOR_2_LDD0852);
121 			}
122 			// TODO check with the UI/COdesetElement how to check whether Voter
123 			// registration is requested or not
124 			if (idProcessContext.getDlApplication()
125 					.getVoterRegistationRequested().getCode().equals("Y")) {
126 				printRequestTypes
127 						.add(PrintRequestTypes.VOTER_REGISTRATION_LDD0861);
128 			}
129 		}
130 		return printRequestTypes;
131 	}

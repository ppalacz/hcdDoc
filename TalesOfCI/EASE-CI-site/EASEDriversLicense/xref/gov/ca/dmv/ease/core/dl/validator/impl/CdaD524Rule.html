232 	 * Return if PreTrip Inspection Fields Not Blank.
233 	 * 
234 	 * @param commercialPreTripTestResults
235 	 * @return boolean 
236 	 */
237 	private boolean isPreTripInspectionFieldsNotBlank(
238 			List <CommercialTestResult> commercialPreTripTestResults) {
239 		boolean isPreTripInspectionFieldsNotBlank = false;
240 		for (CommercialTestResult commercialPreTripTestResult : commercialPreTripTestResults) {
241 			if (EaseUtil.isNotNull(commercialPreTripTestResult
242 					.getTestFailureDate())
243 					|| EaseUtil.isNotNull(commercialPreTripTestResult
244 							.getTestResultCode().getCode())
245 					|| EaseUtil.isNotNull(commercialPreTripTestResult
246 							.getExaminerID())
247 					|| EaseUtil.isNotNull(commercialPreTripTestResult
248 							.getLicenseClassCode().getCode())) {
249 				isPreTripInspectionFieldsNotBlank = true;
250 				return isPreTripInspectionFieldsNotBlank;
251 			}
252 		}
253 		return isPreTripInspectionFieldsNotBlank;
254 	}

271 				&& commercialGeneralTestResults.size() > 1) {
272 			int lastIndex = commercialGeneralTestResults.size() - 1;
273 			String newTestResultCode = commercialGeneralTestResults.get(
274 					lastIndex).getTestResultCode().getCode();
275 			String previousTestResultCode = commercialGeneralTestResults.get(
276 					lastIndex - 1).getTestResultCode().getCode();
277 			if (EaseUtil.isNotNull(newTestResultCode)
278 					&& EaseUtil.isNotNull(previousTestResultCode)) {
279 				if (newTestResultCode.equalsIgnoreCase(previousTestResultCode)) {
280 					errorCollector
281 							.register(new ApplicationException(
282 									IDataCollectionErrorMessageConstants.CORRECTION_INVALID));

39  	 * D855. non original ID transaction, if the record contains a SSN, the SSN
40  	 * cannot be deleted.
41  	 * 
42  	 * @param applicant
43  	 *            applicant data
44  	 * @param ttc
45  	 *            transaction code
46  	 * @param eliRecord
47  	 *            data from storage
48  	 * @param errorCollector
49  	 *            the error collector
50  	 */
51  	public void D855(Person applicant, String ttc, EliRecord eliRecord,
52  			IErrorCollector errorCollector) {
53  		if (applicant != null) {
54  			SocialSecurityDocument socialSecurityDocument = applicant
55  					.getSocialSecurityDocument();
56  
57  			String ssn = null;
58  			if (socialSecurityDocument != null) {
59  				ssn = socialSecurityDocument.getSsn();
60  			}
61  			// D855 - SSN CANNOT BE DELETED
62  			if (("IDP".equals(ttc) || "IDC".equals(ttc) || "IRT".equals(ttc))
63  					&& EaseUtil.isNullOrBlank(ssn)) {
64  				if (eliRecord != null) {
65  					errorCollector
66  							.register(new EaseValidationException(
67  									IDataCollectionErrorMessageConstants.SSN_CANNOT_BE_DELETED));
68  				}
69  			}
70  		}
71  	}

40  public class RestrictionCodeValidator {
41  	public void validate(DlApplication dlApplication,
42  			DriveAndLawTestResultGroup driveAndLawTestResultGroup,
43  			IErrorCollector errorCollector) {
44  		Person person = dlApplication.getApplicant();
45  		List <String> restrictionCodes = DlApplicationUtil
46  				.getAllRestrictionCodes(person.getDriverLicense());
47  		if (!EaseUtil.isNullOrBlank(restrictionCodes)) {
48  			validateRestrictionCodes(restrictionCodes, errorCollector);
49  			for (String restrictionCode : restrictionCodes) {
50  				if (driveAndLawTestResultGroup != null) {
51  					validateVision(restrictionCode, driveAndLawTestResultGroup
52  							.getVisionTestResultCode(), errorCollector);
53  				}
54  				if (person.getBirthDate() != null) {
55  					validateBirthDate(restrictionCode, person.getBirthDate(),
56  							errorCollector);
57  				}
58  			}
59  		}
60  	}

62  	/***
63  	 * @param restrictionCodes
64  	 * @param errorCollector
65  	 */
66  	public void validateRestrictionCodes(List <String> restrictionCodes,
67  			IErrorCollector errorCollector) {
68  		if (!EaseUtil.isNullOrBlank(restrictionCodes)) {
69  			//0126
70  			Set <String> codeSet = new HashSet <String>(restrictionCodes);
71  			if (codeSet.size() != restrictionCodes.size()) {
72  				errorCollector.register(new EaseValidationException(
73  						IGeneralErrorMessageConstants.DUPLICATE_CODE_SELECTED));
74  			}
75  			//0013
76  			if (ArrayUtils.contains(restrictionCodes, IRestrictionCode.R05,
77  					IRestrictionCode.R14, IRestrictionCode.R15,
78  					IRestrictionCode.R25, IRestrictionCode.R95)) {
79  				errorCollector
80  						.register(new EaseValidationException(
81  								IGeneralErrorMessageConstants.DATA_DOES_NOT_MATCH_VALID_CODES));
82  			}
83  			//D888
84  			if (ArrayUtils.contains(restrictionCodes, IRestrictionCode.R64)
85  					&& ArrayUtils.contains(restrictionCodes,
86  							IRestrictionCode.R65)) {
87  				errorCollector
88  						.register(new EaseValidationException(
89  								IDataCollectionErrorMessageConstants.RESTRICTION_64_VS_65));
90  			}
91  			//D769
92  			int counterForRestr = 0;
93  			for (String restrictionCode : restrictionCodes) {
94  				if (IRestrictionCode.R71.equals(restrictionCode)) {
95  					counterForRestr++;
96  				}
97  				else if (IRestrictionCode.R72.equals(restrictionCode)) {
98  					counterForRestr++;
99  				}
100 				else if (IRestrictionCode.R73.equals(restrictionCode)) {
101 					counterForRestr++;
102 				}
103 				else if (IRestrictionCode.R80.equals(restrictionCode)) {
104 					counterForRestr++;
105 				}
106 				else if (IRestrictionCode.R81.equals(restrictionCode)) {
107 					counterForRestr++;
108 				}
109 				else if (IRestrictionCode.R82.equals(restrictionCode)) {
110 					counterForRestr++;
111 				}
112 				else if (IRestrictionCode.R83.equals(restrictionCode)) {
113 					counterForRestr++;
114 				}
115 				else if (IRestrictionCode.R84.equals(restrictionCode)) {
116 					counterForRestr++;
117 				}
118 				else if (IRestrictionCode.R85.equals(restrictionCode)) {
119 					counterForRestr++;
120 				}
121 				else if (IRestrictionCode.R86.equals(restrictionCode)) {
122 					counterForRestr++;
123 				}
124 				else if (IRestrictionCode.R87.equals(restrictionCode)) {
125 					counterForRestr++;
126 				}
127 				if (counterForRestr > 1) {
128 					errorCollector
129 							.register(new EaseValidationException(
130 									IDataCollectionErrorMessageConstants.CANNOT_ADD_MORE_THAN_ONE_RESTRICTION));
131 					break;
132 				}
133 			}
134 		}
135 	}

143 	public void validateVision(String restrictionCode,
144 			CodeSetElement visionTestResult, IErrorCollector errorCollector) {
145 		//D621
146 		String visionTestResultCode = CodeUtil
147 				.getCodeFromCodeSetElement(visionTestResult);
148 		if (IVisionTestConstants.PASSED_WO_LENSES.equals(visionTestResultCode)
149 				|| IVisionTestConstants.SATISFACTORY_WO_LENSES
150 						.equals(visionTestResultCode)
151 				|| EaseUtil.isNullOrBlank(visionTestResultCode)) {
152 			if (IRestrictionCode.R01.equals(restrictionCode)) {
153 				errorCollector
154 						.register(new EaseValidationException(
155 								IDataCollectionErrorMessageConstants.AGE_VS_RESTRICTION));
156 			}
157 		}
158 	}

164 	 */
165 	public void validateBirthDate(String restrictionCode, Date birthDate,
166 			IErrorCollector errorCollector) {
167 		//D570
168 		if (birthDate != null) {
169 			if (!DateUtil.is18YearsOld(birthDate)) {
170 				if (IRestrictionCode.R34.equals(restrictionCode)
171 						|| IRestrictionCode.R04.equals(restrictionCode)) {
172 					errorCollector
173 							.register(new EaseValidationException(
174 									IDataCollectionErrorMessageConstants.AGE_VS_RESTRICTION));
175 				}
176 			}
177 		}

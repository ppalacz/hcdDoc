29  public class CollectApplicationDataActivity extends InteractionActivity {
30  	/*** The Constant serialVersionUID. */
31  	private static final long serialVersionUID = -4804040937434079954L;
32  	/*** The collectApplicantDataRule for validation. */
33  	private CollectApplicantDataRule collectApplicantDataRule;
34  
35  	/* (non-Javadoc)
36  	 * @see gov.ca.dmv.ease.app.activity.impl.Activity#postCondition(gov.ca.dmv.ease.app.context.impl.ProcessContext)
37  	 */
38  	@Override
39  	protected void postCondition(ProcessContext processContext) {
40  		String ttc = processContext.getUserContext().getTtc();
41  		if (ArrayUtils.contains(ttc, ITtcConstants.DLA, ITtcConstants.DLC,
42  				ITtcConstants.DLD, ITtcConstants.DLE, ITtcConstants.DLP,
43  				ITtcConstants.DRT)) {
44  			if (isNotByPassValidations(processContext)) {
45  				collectApplicantDataRule.validate(processContext);
46  				populateErrorMessages(processContext, collectApplicantDataRule
47  						.getErrorCollector());
48  			}
49  		}
50  	}
51  
52  	/* (non-Javadoc)
53  	 * @see gov.ca.dmv.ease.app.activity.impl.Activity#preCondition(gov.ca.dmv.ease.app.context.impl.ProcessContext)
54  	 */
55  	@Override
56  	protected void preCondition(ProcessContext processContext) {
57  		if (processContext.getUserContext().getTtc().equalsIgnoreCase(
58  				ITtcConstants.DLA)) {
59  			Action selectedAction = processContext.getSelectedAction();
60  			if (selectedAction != null) {
61  				if (!selectedAction.equals(getActionsRegistry().getAction(
62  						IActionsRegistry.RESTART_ACTION))
63  						&& !selectedAction.equals(getActionsRegistry()
64  								.getAction(IActionsRegistry.CANCEL_ACTION))) {
65  					super.preCondition(processContext);
66  					String ttc = processContext.getUserContext().getTtc();
67  					if (ITtcConstants.DLD.equals(ttc)
68  							|| ITtcConstants.DRT.equals(ttc)) {
69  						DlProcessContext dlProcessContext = (DlProcessContext) processContext;
70  						CodeSetElement photoRequiredCode = new CodeSetElement(
71  								"Y", "Y", "Y");
72  						dlProcessContext.getDlApplication()
73  								.setPhotoRequiredCode(photoRequiredCode);
74  					}
75  					if (ITtcConstants.DRT.equals(ttc)) {
76  						DlProcessContext dlProcessContext = (DlProcessContext) processContext;
77  						dlProcessContext.getDlApplication().setFeeRequired(
78  								Boolean.FALSE);
79  					}
80  					collectApplicantDataRule.displayMessage(processContext);
81  				}
82  			}
83  		}
84  	}
85

91  	public void setCollectApplicantDataRule(
92  			CollectApplicantDataRule aCollectApplicantDataRule) {
93  		collectApplicantDataRule = aCollectApplicantDataRule;
94  	}

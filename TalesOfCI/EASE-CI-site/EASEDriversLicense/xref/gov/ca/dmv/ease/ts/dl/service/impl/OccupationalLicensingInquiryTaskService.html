57  	/***
58  	 * Execute.
59  	 * 
60  	 * @param tsRequest the ts request
61  	 * 
62  	 * @return the ol inquiry response ts
63  	 */
64  	public OlInquiryResponseTs execute(OlInquiryRequestTs tsRequest) {
65  		OlInquiryRequestBus busRequest = null;
66  		OlInquiryResponseTs tsResponse = null;
67  		OlInquiryResponseBus busResponse = null;
68  		IBusDlRequestFactory requestFactory = new BusDlRequestFactory();
69  		IDlTaskServiceResponseFactory responseFactory = new DlTaskServiceResponseFactory();
70  		IErrorCollector collector = new ErrorCollector();
71  		try {
72  			collector = tsRequest.validate();
73  			if (collector.hasErrors()) {
74  				tsResponse = responseFactory.createOlInquiryResponse(collector);
75  			}
76  			else {
77  				busRequest = requestFactory.createOlInquiryRequestBus(tsRequest
78  						.getUserContext(), tsRequest
79  						.getOccupationalLicenseInquiry(), tsRequest
80  						.getIsCustomerCopy());
81  				busResponse = occupationalLicensingInquiryService
82  						.execute(busRequest);
83  				if (busResponse.hasErrors()) {
84  					tsResponse = responseFactory
85  							.createOlInquiryResponse(busResponse
86  									.getErrorCollector());
87  				}
88  				else {
89  					tsResponse = responseFactory
90  							.createOlInquiryResponse(busResponse
91  									.getDataStream());
92  				}
93  			}
94  		}
95  		catch (Exception exception) {
96  			collector.register(exception);
97  			tsResponse = responseFactory.createOlInquiryResponse(collector);
98  			return tsResponse;
99  		}
100 		return tsResponse;
101 	}

191 			feeStatusTypes.add(FeeStatusType.FEE_DUE);

206 	/***
207 	 * Gets the Out of State Indicator.
208 	 * 
209 	 * @param dlApplication the dl application
210 	 * @param currentUserContext the current user context
211 	 * @param driverLicense the driver license
212 	 * @param outOfStateLicense the out of state license
213 	 * 
214 	 * @return The Out of State Indicator.
215 	 */
216 	public static String getOutOfStateIndicator(DlApplication dlApplication,
217 			IUserContext currentUserContext, DriverLicense driverLicense,
218 			OutOfStateLicense outOfStateLicense) {
219 		String outOfStateIndicator = "N";
220 		Boolean isCNACondition = dlApplication
221 				.isCNACondition(CnaConditionType.CNA_SEND_CDLIS_PDPS_INQUIRY);
222 		CodeSetElement incompleteAppCode = dlApplication
223 				.getNewIncompleteApplicationReasonCode();
224 		CodeSetElement printRequestDocumentCode = dlApplication
225 				.getDlPrintRequestDocumentCode();
226 		String currentOfficeId = currentUserContext.getOfficeId();
227 		Boolean isOriginalApplicationIndicator = dlApplication
228 				.isOriginalApplicationIndicator();
229 		String pdpsEligibility = null;
230 		//Check if there is no CNA due to CDLIS/PDPS inquiry and there is no incomplete app reason code with value CI.
231 		if (EaseUtil.isNullOrBlank(isCNACondition)
232 				&& (EaseUtil.isNullOrBlank(incompleteAppCode) || (!EaseUtil
233 						.isNullOrBlank(incompleteAppCode) && !"CI"
234 						.equalsIgnoreCase(incompleteAppCode.getCode())))) {
235 			//Check if document 1/2/3/4 has been requested.
236 			if (!EaseUtil.isNullOrBlank(printRequestDocumentCode)
237 					&& ("1".equals(printRequestDocumentCode.getCode())
238 							|| "2".equals(printRequestDocumentCode.getCode())
239 							|| "3".equals(printRequestDocumentCode.getCode()) || "4"
240 							.equals(printRequestDocumentCode.getCode()))) {
241 				//Check if there is no cdlis transfer into or out of california.
242 				if (EaseUtil
243 						.isNullOrBlank(dlApplication.getCdlisTransferType())) {
244 					//Check if user didnt surrender an O/S license to CLD1
245 					if (EaseUtil.isNotNull(outOfStateLicense)
246 							&& (EaseUtil.isNullOrBlank(outOfStateLicense
247 									.getLicenseLocation()) || (!EaseUtil
248 									.isNullOrBlank(outOfStateLicense
249 											.getLicenseLocation()) && !"S"
250 									.equalsIgnoreCase(outOfStateLicense
251 											.getLicenseLocation().getCode())))) {
252 						//Check if there is not valid attachment 17 presented
253 						if (EaseUtil.isNotNull(driverLicense)
254 								&& (EaseUtil.isNotNull(driverLicense
255 										.getNewAttachments())
256 										&& !driverLicense.getNewAttachments()
257 												.isEmpty()
258 										&& !EaseUtil
259 												.isNullOrBlank(driverLicense
260 														.getNewAttachments()
261 														.get(0))
262 										&& !"17".equals(driverLicense
263 												.getNewAttachments().get(0)
264 												.getCode()) || ("225"
265 										.equalsIgnoreCase(currentOfficeId) || "230"
266 										.equalsIgnoreCase(currentOfficeId)))) {
267 							//Check if applied class is commercial
268 							if (EaseUtil.isNotNull(driverLicense
269 									.getAppliedLicenseClasses())
270 									&& !driverLicense
271 											.getAppliedLicenseClasses()
272 											.isEmpty()
273 									&& !EaseUtil.isNullOrBlank(driverLicense
274 											.getAppliedLicenseClasses().get(0))
275 									&& isComercialLicenseClass(driverLicense
276 											.getAppliedLicenseClasses().get(0)
277 											.getCode())) {
278 								//Check if first time CA driver with prior O/S commercial DL.
279 								if (!EaseUtil
280 										.isNullOrBlank(isOriginalApplicationIndicator)
281 										&& isOriginalApplicationIndicator
282 												.booleanValue()
283 										&& EaseUtil
284 												.isNotNull(outOfStateLicense)
285 										&& !EaseUtil
286 												.isNullOrBlank(outOfStateLicense
287 														.isCommercial())
288 										&& outOfStateLicense.isCommercial()
289 												.booleanValue()) {
290 									outOfStateIndicator = "I";
291 								}
292 								else {
293 									outOfStateIndicator = "E";
294 								}
295 							}//End - Check if applied class is commercial
296 							//if applied class is non-commercial
297 							else if (EaseUtil.isNotNull(driverLicense
298 									.getAppliedLicenseClasses())
299 									&& !driverLicense
300 											.getAppliedLicenseClasses()
301 											.isEmpty()
302 									&& !EaseUtil.isNullOrBlank(driverLicense
303 											.getAppliedLicenseClasses().get(0))
304 									&& isNonComercialLicenseClass(driverLicense
305 											.getAppliedLicenseClasses().get(0)
306 											.getCode())) {
307 								//Check if first time CA driver with prior O/S commercial DL.
308 								if (!EaseUtil
309 										.isNullOrBlank(isOriginalApplicationIndicator)
310 										&& isOriginalApplicationIndicator
311 												.booleanValue()
312 										&& EaseUtil
313 												.isNotNull(outOfStateLicense)
314 										&& !EaseUtil
315 												.isNullOrBlank(outOfStateLicense
316 														.isCommercial())
317 										&& outOfStateLicense.isCommercial()
318 												.booleanValue()) {
319 									pdpsEligibility = getPdpsEligibilityIndicator(
320 											dlApplication, currentUserContext);
321 									//Check if with PDPS eligibility issues.
322 									if (!EaseUtil
323 											.isNullOrBlank(pdpsEligibility)
324 											&& "E"
325 													.equalsIgnoreCase(pdpsEligibility)) {
326 										outOfStateIndicator = "I";
327 									}
328 								}
329 							}//End - if applied class is non-commercial
330 						}//End - Check if there is not valid attachment 17 presented
331 					}
332 				}//End - //Check if there is no cdlis transfer into or out of california.
333 			}//Check if document 1/2/3/4 has been requested.
334 		}//End - Check if there is no CNA due to CDLIS/PDPS inquiry and there is no incomplete app reason code with value CI.
335 		return outOfStateIndicator;
336 	}

339 	 * Gets the PDPS Eligibility indicator.
340 	 * 
341 	 * @param dlApplication the dl application
342 	 * @param currentUserContext the current user context
343 	 * 
344 	 * @return PDPS Eligibility indicator
345 	 */
346 	public static String getPdpsEligibilityIndicator(
347 			DlApplication dlApplication, IUserContext currentUserContext) {
348 		String pdpsEligibility = "E";
349 		List <PdpsRecord> pdpsRecords = dlApplication.getPdpsRecords();
350 		String designateAction = BLANK;
351 		String officeId = BLANK;
352 		Boolean isClearedByIssuance = null;
353 		Boolean isCNACondition = null;
354 		if (EaseUtil.isNotNull(pdpsRecords) && !pdpsRecords.isEmpty()
355 				&& EaseUtil.isNotNull(pdpsRecords.get(0))) {
356 			if (!EaseUtil
357 					.isNullOrBlank(pdpsRecords.get(0).getDesignateAction())) {
358 				designateAction = pdpsRecords.get(0).getDesignateAction()
359 						.getCode();
360 			}
361 			officeId = currentUserContext.getOfficeId();
362 			isClearedByIssuance = dlApplication.isClearedByIssuance();
363 			isCNACondition = dlApplication.isCNACondition();
364 			if ((!EaseUtil.isNullOrBlank(officeId) && "I"
365 					.equalsIgnoreCase(officeId.substring(0, 1)))
366 					&& !EaseUtil.isNullOrBlank(isClearedByIssuance)
367 					&& isClearedByIssuance.booleanValue()) {
368 				pdpsEligibility = "E";
369 			}
370 			else if (!EaseUtil.isNullOrBlank(designateAction)
371 					&& ("Q".equalsIgnoreCase(designateAction) || "S"
372 							.equalsIgnoreCase(designateAction))
373 					&& EaseUtil.isNullOrBlank(pdpsRecords.get(0).getPassword())) {
374 				pdpsEligibility = "N";
375 			}
376 			else if (!(EaseUtil.isNullOrBlank(isCNACondition) || isCNACondition
377 					.booleanValue())
378 					|| EaseUtil.isNullOrBlank(isCNACondition)) {
379 				pdpsEligibility = "E";
380 			}
381 		}
382 		return pdpsEligibility;
383 	}

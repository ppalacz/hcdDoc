36  	/*** The Constant serialVersionUID. */
37  	private static final long serialVersionUID = -1105216921695287395L;
38  	/*** The vouchers. */
39  	private List <Voucher> vouchers;
40  
41  	/***
42  	 * Instantiates a new validate automated receipt request ECS.
43  	 * 
44  	 * @param vouchers the vouchers
45  	 * @param userContext the user context
46  	 */
47  	public ValidateAutomatedReceiptRequestEcs(List <Voucher> vouchers,
48  			IUserContext userContext,
49  			ValidateAutomatedReceiptConverter validateAutomatedReceiptConverter) {
50  		super(userContext);
51  		setVouchers(vouchers);
52  		setMessageConverter(validateAutomatedReceiptConverter);
53  	}
54  
55  	/* (non-Javadoc)
56  	 * @see gov.ca.dmv.ease.ecs.request.impl.SynchronousEcsRequest#execute()
57  	 */
58  	@Override
59  	public ValidateAutomatedReceiptResponseEcs execute()
60  			throws EcsServiceException {
61  		ValidateAutomatedReceiptResponseEcs validateAutomatedReceiptResponseEcs = null;
62  		IErrorCollector errorCollector = getValidator().validate(this);
63  		if (errorCollector.hasErrors()) {
64  			validateAutomatedReceiptResponseEcs = new ValidateAutomatedReceiptResponseEcs(
65  					errorCollector);
66  		}
67  		else {
68  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
69  			//In the catch block we need to register the error to the error collector.
70  			try {
71  				validateAutomatedReceiptResponseEcs = (ValidateAutomatedReceiptResponseEcs) super
72  						.execute();
73  			}
74  			catch (EcsResponseIsNullException responseIsNullException) {
75  				errorCollector = new ErrorCollector();
76  				errorCollector.register(new EaseValidationException(
77  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
78  				validateAutomatedReceiptResponseEcs = new ValidateAutomatedReceiptResponseEcs(
79  						errorCollector);
80  			}
81  			catch (SystemException sysEx) {
82  				errorCollector = new ErrorCollector();
83  				errorCollector.register(new EaseValidationException(
84  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
85  				validateAutomatedReceiptResponseEcs = new ValidateAutomatedReceiptResponseEcs(
86  						errorCollector);
87  			}
88  		}
89  		return validateAutomatedReceiptResponseEcs;
90  	}
91

93  	 * @see gov.ca.dmv.ease.fw.service.impl.AbstractRequest#getValidator()
94  	 */
95  	@Override
96  	protected IRequestValidator getValidator() {
97  		return new ValidateAutomatedReceiptRequestEcsValidator();
98  	}

41  	public void validate(ProcessContext processContext) {
42  		CdaProcessContext cdaProcessContext = (CdaProcessContext) processContext;
43  		Action selectedAction = cdaProcessContext.getSelectedAction();
44  		if (selectedAction != null
45  				&& selectedAction.equals(getActionsRegistry().getAction(
46  						IActionsRegistry.ENTER_ACTION))) {
47  			IErrorCollector errorCollector = new ErrorCollector();
48  			//Validate the business objects
49  			validate(cdaProcessContext.getDriverLicense(), cdaProcessContext
50  					.getDlApplicant(), cdaProcessContext.getUserContext(),
51  					errorCollector);
52  		}
53  	}

55  	/***
56  	 * 
57  	 * @param driverLicense
58  	 * @param applicant
59  	 * @param userContext
60  	 */
61  	public void validate(DriverLicense driverLicense, Person applicant,
62  			IUserContext userContext, IErrorCollector errorCollector) {
63  		List<DrivingRestriction> existingRestrictions = driverLicense
64  				.getExistingRestrictions();
65  		List<DrivingRestriction> newRestrictions = driverLicense
66  				.getNewRestrictions();
67  		List<DrivingRestriction> restrictions = new ArrayList<DrivingRestriction>();
68  		if (newRestrictions != null)
69  			restrictions.addAll(newRestrictions);
70  		if (existingRestrictions != null)
71  			restrictions.addAll(existingRestrictions);
72  		if (restrictions != null && restrictions.size() > 0) {
73  			for (Iterator<DrivingRestriction> iterator = restrictions
74  					.iterator(); iterator.hasNext();) {
75  				DrivingRestriction restriction = iterator.next();
76  				String restrictionCode = restriction.getRestrictionTypeCode()
77  						.getCode();
78  				Date birthdate = null;
79  				Date workDate = null;
80  				if (applicant != null) {
81  					birthdate = applicant.getBirthDate();
82  					workDate = userContext.getWorkDate();
83  				}
84  				if (restrictionCode.equalsIgnoreCase("04")
85  						&& ValidateHelper.is18Years(birthdate, workDate)) {
86  					errorCollector.register(new ApplicationException("D669"));
87  				}
88  			}
89  		}
90  		setErrorCollector(errorCollector);
91  	}
92  }

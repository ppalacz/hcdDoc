28  public class SequenceNumberService {
29  	/***
30  	 * This method invokes Business Utility Service to generate the Sequence number for the given
31  	 * transaction. 
32  	 * Returns the <code>GenerateSequenceNoResponseBus</code>; which
33  	 * provides the sequence number.
34  	 * 
35  	 * @param request GenerateSequenceNoRequestBus
36  	 * 
37  	 * @return GenerateSequenceNoRequestBus the generate sequence no response
38  	 */
39  	public GenerateSequenceNoResponseBus execute(
40  			GenerateSequenceNoRequestBus busRequest) {
41  		GenerateSequenceNoResponseBus busResponse = null;
42  		UserContextValidator contextValidator = new UserContextValidator();
43  		IBusDlResponseFactory responseFactory = new BusDlResponseFactory();
44  		GenerateSequenceNumberRequestValidator validator = new GenerateSequenceNumberRequestValidator();
45  		IErrorCollector errorCollector = new ErrorCollector();
46  		try {
47  			//errorCollector = validator.validate(busRequest.getProcessId());
48  			// now validate the userContext
49  			contextValidator.validate(busRequest.getUserContext(),
50  					errorCollector);
51  			if (errorCollector.hasErrors()) {
52  				busResponse = responseFactory
53  						.createGenerateSequenceNoResponseBus(errorCollector);
54  			}
55  			else {
56  /*				Integer sequenceNumber = SequenceNumberAssesser
57  						.generateSequenceNumber(busRequest);*/
58  				busResponse = responseFactory
59  						.createGenerateSequenceNoResponseBus(new Integer(5001));
60  			}
61  		}
62  		catch (Exception e) {
63  			//Register exception in error collector and assign it to response object.
64  			errorCollector.register(e);
65  			return busResponse = responseFactory
66  					.createGenerateSequenceNoResponseBus(errorCollector);
67  		}
68  		return busResponse;
69  	}
70  }

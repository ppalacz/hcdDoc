56  				&& selectedAction == getActionsRegistry().getAction(

59  					dlApplication, errorCollector);

66  	/***
67  	 * @param dlProcessContext
68  	 * @param ttc
69  	 * @param dlApplication
70  	 * @param errorCollector
71  	 */
72  	public void validateResidenceAddress(DlProcessContext dlProcessContext,
73  			String ttc, DlApplication dlApplication,
74  			IErrorCollector errorCollector) {
75  		Address residenceAddress = dlApplication.getApplicant().getAddress(
76  				AddressType.RESIDENTIAL);
77  		String residenceStreetName = residenceAddress.getLine1();
78  		String residenceCityCode = dlProcessContext.getResidenceCityCode();
79  		String residenceCityName = residenceAddress.getCity();
80  		String residenceState = residenceAddress.getState();
81  		String residenceZipCode = residenceAddress.getZip();
82  		String residenceCountyCode = residenceAddress.getCounty();
83  		if (!EaseUtil.isNullOrBlank(residenceAddress)) {
84  			if (ITtcConstants.DLA.equals(ttc)) {
85  				validateResidenceForDla(errorCollector, residenceStreetName,
86  						residenceCityCode, residenceCityName, residenceState,
87  						residenceZipCode);
88  			}
89  			// D531
90  			if (residenceStreetName != null && residenceCityName != null
91  					&& residenceState != null) {
92  				Address mailingAddress = dlApplication.getApplicant()
93  						.getAddress(AddressType.MAILING);
94  				String mailStreetName = mailingAddress.getLine1();
95  				if (mailStreetName.indexOf("PO BX") >= 0
96  						|| mailStreetName.indexOf("PO DRW") >= 0
97  						|| mailStreetName.indexOf("GEN DEL") >= 0) {
98  					errorCollector
99  							.register(new EaseValidationException(
100 									IDataCollectionErrorMessageConstants.RESIDENCE_ADDRESS_REQUIRED));
101 				}
102 				// D693
103 				String mailCityName = mailingAddress.getCity();
104 				String mailState = mailingAddress.getState();
105 				if (residenceStreetName.equals(mailStreetName)
106 						&& residenceCityName.equals(mailCityName)
107 						&& residenceState.equals(mailState)) {
108 					errorCollector
109 							.register(new EaseValidationException(
110 									IDataCollectionErrorMessageConstants.MAILING_AND_RESIDENCE_ADDRESSES_CANNOT_BE_THE_SAME));
111 				}
112 			}
113 			// D560
114 			validateCityCodeVsState(errorCollector, residenceCityCode,
115 					residenceCityName, residenceState);
116 			// D008
117 			validateCityLengthVsState(errorCollector, residenceCityName,
118 					residenceState);
119 			// D009
120 			validateCountyCodeVsOutOfState(errorCollector, residenceCountyCode,
121 					residenceState);
122 			// D550
123 			if ("APO".equals(residenceCityCode)
124 					|| "FPO".equals(residenceCityCode)) {
125 				if (!ArrayUtils.contains(residenceState,
126 						IStateCodeConstants.CALIFORNIA,
127 						IStateCodeConstants.OREGON)) {
128 					errorCollector
129 							.register(new EaseValidationException(
130 									IDataCollectionErrorMessageConstants.RESIDENCE_CITY_VS_RESIDENCE_STATED));
131 				}
132 			}
133 			//D095
134 			validateOutOfState(errorCollector, residenceState, residenceZipCode);
135 		}
136 	}

219 	 * @param dlProcessContext
220 	 * @param ttc
221 	 * @param dlApplication
222 	 * @param errorCollector
223 	 */
224 	public void validateMailingAddress(DlProcessContext dlProcessContext,
225 			String ttc, DlApplication dlApplication,
226 			IErrorCollector errorCollector) {
227 		Address mailingAddress = dlApplication.getApplicant().getAddress(
228 				AddressType.MAILING);
229 		String mailStreetName = mailingAddress.getLine1();
230 		String mailCityCode = dlProcessContext.getMailingCityCode();
231 		String mailCityName = mailingAddress.getCity();
232 		String mailState = mailingAddress.getState();
233 		String mailZipCode = mailingAddress.getZip();
234 		String mailCountyCode = mailingAddress.getCounty();
235 		if (!EaseUtil.isNullOrBlank(mailingAddress)) {
236 			if (ITtcConstants.DLA.equals(ttc)) {
237 				validateMailingForDla(errorCollector, mailStreetName,
238 						mailCityCode, mailCityName, mailState, mailZipCode);
239 			}
240 			// D506
241 			if (mailStreetName != null) {
242 				if (mailStreetName.startsWith(" ")
243 						|| mailStreetName.indexOf("  ") >= 0) {
244 					errorCollector
245 							.register(new EaseValidationException(
246 									IDataCollectionErrorMessageConstants.MAILING_ADDRESS_INVALID));
247 				}
248 			}
249 			// D560
250 			validateCityCodeVsState(errorCollector, mailCityCode, mailCityName,
251 					mailState);
252 			// D008
253 			validateCityLengthVsState(errorCollector, mailCityName, mailState);
254 			// D009
255 			validateCountyCodeVsOutOfState(errorCollector, mailCountyCode,
256 					mailState);
257 			// D853
258 			if ("APO".equals(mailCityCode) || "FPO".equals(mailCityCode)) {
259 				if (!ArrayUtils.contains(mailState,
260 						IStateCodeConstants.CALIFORNIA,
261 						IStateCodeConstants.OREGON)) {
262 					errorCollector
263 							.register(new EaseValidationException(
264 									IDataCollectionErrorMessageConstants.MAIL_CITY_VS_MAIL_STATE));
265 				}
266 			}
267 			// D898
268 			if (!EaseUtil.isNullOrBlank(dlApplication.getLicense())) {
269 				if (!EaseUtil.isNullOrBlank(dlApplication.getLicense()
270 						.getAppliedLicenseClasses())) {
271 					if (ArrayUtils.contains(DlApplicationUtil
272 							.getAppliedLicenseClassCodes(dlApplication
273 									.getLicense()), "A", "B", "C")
274 							&& !IStateCodeConstants.CALIFORNIA
275 									.equals(mailState)) {
276 						errorCollector
277 								.register(new EaseValidationException(
278 										IDataCollectionErrorMessageConstants.CANNOT_CONTINUE_MUST_BE_A_CA_RESIDENT));
279 					}
280 				}
281 			}
282 			//D095
283 			validateOutOfState(errorCollector, mailState, mailZipCode);
284 		}
285 	}

292 	public void validateOutOfState(IErrorCollector errorCollector,
293 			String state, String zipCode) {
294 		if (zipCode != null) {
295 			int zipInt = Integer.parseInt(zipCode);
296 			if (zipInt >= 90000 && zipInt <= 96699) {
297 				if (!IStateCodeConstants.CALIFORNIA.equals(state)) {
298 					errorCollector
299 							.register(new EaseValidationException(
300 									IDataCollectionErrorMessageConstants.ZIP_CODE_VS_OUT_OF_STATE));
301 				}
302 			}
303 			else {
304 				if (IStateCodeConstants.CALIFORNIA.equals(state)) {
305 					errorCollector
306 							.register(new EaseValidationException(
307 									IDataCollectionErrorMessageConstants.ZIP_CODE_VS_OUT_OF_STATE));
308 				}
309 			}
310 		}

402 					errorCollector
403 							.register(new EaseValidationException(
404 									IDataCollectionErrorMessageConstants.COUNTY_CODE_VS_OUT_OF_STATE));

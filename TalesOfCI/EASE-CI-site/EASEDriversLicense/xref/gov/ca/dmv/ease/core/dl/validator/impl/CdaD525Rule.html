52  				validate(commercialTestResultGroup,
53  						CommercialTestResultType.TRAILERCOACH_TEST_RESULT,

57  						errorCollector);
58  				validate(

61  						errorCollector);

79  	/***
80  	 * @param commercialTestResultGroup
81  	 * @param commercialTestResultType
82  	 * @param errorCollector
83  	 */
84  	private void validate(CommercialTestResultGroup commercialTestResultGroup,
85  			CommercialTestResultType commercialTestResultType,
86  			IErrorCollector errorCollector) {
87  		List <CommercialTestResult> commercialGeneralTestResults = commercialTestResultGroup
88  				.getCommercialTestResults(commercialTestResultType);
89  		if (EaseUtil.isNotNull(commercialGeneralTestResults)
90  				&& commercialGeneralTestResults.size() > 3) {
91  			errorCollector
92  					.register(new ApplicationException(
93  							IDataCollectionErrorMessageConstants.THIRD_TEST_PREVIOUSLY_POSTED_ERASE_OR_PRESS_CORRECTION));
94  		}
95  	}

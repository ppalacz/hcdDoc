73  		UpdateDriverSsnResponseEcs updateDriverSsnResponseEcs = null;
74  		IErrorCollector errorCollector = getValidator().validate(this);
75  		if (errorCollector.hasErrors()) {
76  			updateDriverSsnResponseEcs = new UpdateDriverSsnResponseEcs(
77  					errorCollector);
78  		}
79  		else {
80  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
81  			//In the catch block we need to register the error to the error collector.
82  			try {
83  				updateDriverSsnResponseEcs = (UpdateDriverSsnResponseEcs) super
84  						.execute();
85  			}
86  			catch (EcsResponseIsNullException responseIsNullException) {
87  				errorCollector = new ErrorCollector();
88  				errorCollector.register(new EaseValidationException(
89  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
90  				updateDriverSsnResponseEcs = new UpdateDriverSsnResponseEcs(
91  						errorCollector);
92  			}
93  			catch (SystemException sysEx) {
94  				errorCollector = new ErrorCollector();
95  				errorCollector.register(new EaseValidationException(
96  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
97  				updateDriverSsnResponseEcs = new UpdateDriverSsnResponseEcs(
98  						errorCollector);
99  			}
100 		}
101 		return updateDriverSsnResponseEcs;
102 	}

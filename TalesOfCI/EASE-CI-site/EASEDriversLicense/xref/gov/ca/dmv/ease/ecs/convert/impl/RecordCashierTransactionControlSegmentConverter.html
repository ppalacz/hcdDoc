321 	//Get operational mode
322 	//TODO: is there any string format based on possible null or empty strings?
323 	private String getOperationalMode(IUserContext userContext) {
324 		CodeSetElement operationalMode = userContext.getOperationalMode();
325 		CodeSetElement functionalMode = userContext.getFunctionalMode();
326 		if (!EaseUtil.isNullOrBlank(operationalMode)
327 				&& !EaseUtil.isNullOrBlank(functionalMode)) {
328 			return operationalMode.getCode() + functionalMode.getCode();
329 		}
330 		else if (EaseUtil.isNullOrBlank(operationalMode)) {
331 			return functionalMode.getCode();
332 		}
333 		else if (EaseUtil.isNullOrBlank(functionalMode)) {
334 			return operationalMode.getCode();
335 		}
336 		else {
337 			return "";
338 		}

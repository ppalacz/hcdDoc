26  public class IsSsnValidatedDecision extends DecisionActivity implements
27  		IDlaDecisionActivityConstantsForBpDefinition {
28  	/*** The Constant serialVersionUID. */
29  	private static final long serialVersionUID = 1L;
30  	/*** The constant SSN Verification code */
31  	private static final String SSN_VERIFICATION_CODE_A = "A";
32  	/*** The constant SSN Verification code */
33  	private static final String SSN_VERIFICATION_CODE_D = "D";
34  	/*** The constant SSN Verification code */
35  	private static final String SSN_VERIFICATION_CODE_I = "I";
36  
37  	/***
38  	 * This method evaluates if Ssn is validated or not. 
39  	 * If validated it returns true, if not returns false.
40  	 */
41  	@Override
42  	protected String evaluate(ProcessContext processContext) {
43  		SsnInquiryProcessContext ssnInquiryProcessContext = (SsnInquiryProcessContext) processContext;
44  		Boolean isSsnValidated = Boolean.FALSE;
45  		EliRecord eliRecord = ssnInquiryProcessContext.getEliRecord();
46  		if (eliRecord != null) {
47  			CodeSetElement userVerifiedSsnCode = ssnInquiryProcessContext
48  					.getDlApplication().getNewUserVerifiedSsnCode();
49  			String ssnVerified = userVerifiedSsnCode == null ? null
50  					: userVerifiedSsnCode.getCode();
51  			isSsnValidated = SSN_VERIFICATION_CODE_A.equals(ssnVerified);
52  		}
53  		return isSsnValidated ? SSN_VALIDATED : SSN_NOT_VALIDATED;
54  	}
55  }

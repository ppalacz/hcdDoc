117 	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
118 		DlInfoRequestReceiptDocumentRequestTus request = (DlInfoRequestReceiptDocumentRequestTus) requestFromClient;
119 		IUserContext userContext = request.getUserContext();
120 		DlInformationRequest application = request.getApplication();
121 		Invoice invoice = application.getInvoice();
122 		List <List <String>> documentsList = new ArrayList <List <String>>();
123 		List <String> doc1ElementsList = new ArrayList <String>();
124 		documentsList.add(doc1ElementsList);
125 		addToElementsList(doc1ElementsList, convertDateToString(application
126 				.getSystemDate(), FORMATTER_MM_SLASH_DD_SLASH_YY),
127 				SYSTEM_DATE_ELEMENT);
128 		addToElementsList(doc1ElementsList, "", AMOUNT_PAID_ELEMENT);//TODO:  Invoice code
129 		addToElementsList(doc1ElementsList, "", COMMENTS_ELEMENT);
130 		String licenseNumber = DocumentConverterUtil
131 				.getLicenseNumber(application.getApplicant().getDriverLicense());
132 		String licenseText = "";
133 		if (licenseNumber.length() > 0) {
134 			licenseText = LICENSCE_TEXT;
135 		}
136 		addToElementsList(doc1ElementsList, licenseText, LICENSCE_TEXT_ELEMENT);
137 		addToElementsList(doc1ElementsList, licenseNumber, DL_ID_NUMBER_ELEMENT);
138 		DecimalFormat formatter = new DecimalFormat("####0.00");
139 		InvoiceConverterUtility.populateInvoice(doc1ElementsList,
140 				dlInfoRequestCashElements, dlInfoRequestDebitElements,
141 				dlInfoRequestCreditElements, invoice, formatter, userContext
142 						.getWorkDate(), false);
143 		addToElementsList(doc1ElementsList, DocumentConverterUtil
144 				.getOfficeNameFromId(userContext.getOfficeId()),
145 				OFFICE_NAME_ELEMENT);
146 		addToElementsList(doc1ElementsList,
147 				convertValueOrBlankForNull(userContext.getOfficeId()),
148 				OFFICE_ID_ELEMENT);
149 		addToElementsList(doc1ElementsList, convertDateToString(userContext
150 				.getWorkDate(), FORMATTER_MM_DD_YY), WORK_DATE_ELEMENT);
151 		addToElementsList(doc1ElementsList,
152 				convertValueOrBlankForNull(userContext.getTechId()),
153 				KEYERS_ID_ELEMENT);
154 		addToElementsList(doc1ElementsList, userContext
155 				.getControlCashierSeqNumber().toString(),
156 				CASHIER_SEQUENCE_ELEMENT);
157 		addToElementsList(doc1ElementsList,
158 				convertValueOrBlankForNull(userContext.getTtc()), TTC_ELEMENT);
159 		List <String> doc2ElementsList = new ArrayList <String>();
160 		addToElementsList(doc2ElementsList, convertDateToString(application
161 				.getSystemDate(), FORMATTER_MM_SLASH_DD_SLASH_YY),
162 				SYSTEM_DATE_ELEMENT);
163 		addToElementsList(doc2ElementsList, "", COMMENTS_ELEMENT);//TODO:  WHERE COMMENTS
164 		addToElementsList(doc2ElementsList, licenseText, LICENSCE_TEXT_ELEMENT);
165 		addToElementsList(doc2ElementsList, licenseNumber, DL_ID_NUMBER_ELEMENT);
166 		InvoiceConverterUtility.populateInvoice(doc2ElementsList,
167 				dlInfoRequestCashElements, dlInfoRequestDebitElements,
168 				dlInfoRequestCreditElements, invoice, formatter, userContext
169 						.getWorkDate(), false);
170 		addToElementsList(doc2ElementsList, DocumentConverterUtil
171 				.getOfficeNameFromId(userContext.getOfficeId()),
172 				OFFICE_NAME_ELEMENT);
173 		addToElementsList(doc2ElementsList,
174 				convertValueOrBlankForNull(userContext.getOfficeId()),
175 				OFFICE_ID_ELEMENT);
176 		addToElementsList(doc2ElementsList, convertDateToString(userContext
177 				.getWorkDate(), FORMATTER_MM_DD_YY), WORK_DATE_ELEMENT);
178 		addToElementsList(doc2ElementsList,
179 				convertValueOrBlankForNull(userContext.getTechId()),
180 				KEYERS_ID_ELEMENT);
181 		addToElementsList(doc2ElementsList, userContext
182 				.getControlCashierSeqNumber().toString(),
183 				CASHIER_SEQUENCE_ELEMENT);
184 		addToElementsList(doc2ElementsList,
185 				convertValueOrBlankForNull(userContext.getTtc()), TTC_ELEMENT);
186 		documentsList.add(doc2ElementsList);
187 		List <String> templateNames = new ArrayList <String>();
188 		templateNames.add(DOCUMENT_1_TEMPLATE_NAME);
189 		templateNames.add(DOCUMENT_2_TEMPLATE_NAME);
190 		String xmlData = DocumentConverterUtil.convertToXml(documentsList,
191 				templateNames);
192 		return xmlData;
193 	}

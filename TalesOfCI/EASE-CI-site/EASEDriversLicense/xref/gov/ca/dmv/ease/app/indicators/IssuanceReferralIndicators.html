32  public class IssuanceReferralIndicators implements IndicatorsConstants {
33  	/***
34  	 * This method process Issuance Referral Conditions for DLA1 Program.
35  	 * Input is <code>HqInquiryProcessContext</code> Head Quarters Process Context.
36  	 * 
37  	 * @param hqInquiryProcessContext the Head Quarters Process Context
38  	 * 
39  	 * @return list of Issuance Referral Messages
40  	 */
41  	public List <IssuanceReferralMessagesType> processForHq(
42  			HqInquiryProcessContext hqInquiryProcessContext,
43  			IUserContext userContext) {
44  		Set <IssuanceReferralMessagesType> issuanceReferralMessages = new HashSet <IssuanceReferralMessagesType>();
45  		Boolean isNewDLRequested = hqInquiryProcessContext.getDlApplication()
46  				.isNewLicenseNumberRequired();
47  		//		IUserContext userContext = hqInquiryProcessContext.getUserContext();
48  		String officeId = userContext.getOfficeId();
49  		String ttc = userContext.getTtc();
50  		boolean isIssuanceOffice = userContext.isIssuanceOffice();
51  		CodeSetElement operationalMode = userContext.getOperationalMode();
52  		boolean isSalesApplication = !EaseUtil
53  				.isNullOrBlank(hqInquiryProcessContext
54  						.getSalesPersonApplication().getApplicationType());
55  		String licenseNumber = hqInquiryProcessContext.getDlApplication()
56  				.getLicense().getLicenseNumber();
57  		boolean isDlNumberPresent = !EaseUtil.isNullOrBlank(licenseNumber);
58  		boolean isDlNumberContainsX = (licenseNumber.charAt(0) == DLNUMBER_X) ? true
59  				: false;
60  		//TODO NEED TO RECHECK THIS FLAG
61  		Boolean isCNACondition = !hqInquiryProcessContext.isCamvAvailable();
62  		CodeSetElement existingGenderCode = hqInquiryProcessContext
63  				.getDlApplicant().getGenderCode();
64  		CodeSetElement eliGenderCode = hqInquiryProcessContext.getEliRecord()
65  				.getPerson().getGenderCode();
66  		Integer controlCashierSeqNumber = userContext
67  				.getControlCashierSeqNumber();
68  		List <String> duplicateAniRecords = hqInquiryProcessContext
69  				.getDlApplicant().getLicenseNumbersOfDuplicateAniRecords();
70  		boolean isPasswordRequired = hqInquiryProcessContext
71  				.getRecordConditionIndicators().isPasswordRequired();
72  		//New Number Requested Indicator
73  		//EDL VCSI45B3 - Bit 0
74  		//Java IssuanceReferralMessagesType.NEW_NUMBER_REQUESTED
75  		//Step 1 - Is new DL Number Requested
76  		if (EaseUtil.isNotNull(isNewDLRequested)
77  				&& isNewDLRequested.equals(Boolean.TRUE)) {
78  			//Step 2 - If Office Id is not Training Office Id "999"
79  			if (!TRAINING_OFFICE_ID.equals(officeId)) {
80  				//Step 3 - If TTC is not DLD or IDP or DLP or DRT or IRT And
81  				//         Not a Sales Application
82  				if (!(ttc.equalsIgnoreCase(TTC_DLD)
83  						|| ttc.equalsIgnoreCase(TTC_IDP)
84  						|| ttc.equalsIgnoreCase(TTC_DLP)
85  						|| ttc.equalsIgnoreCase(TTC_DRT) || ttc
86  						.equalsIgnoreCase(TTC_IRT))
87  						&& !isSalesApplication) {
88  					//Step 4 - If Manual or Counter Mode
89  					if (!EaseUtil.isNullOrBlank(operationalMode)
90  							&& (MANUAL_MODE.equalsIgnoreCase(operationalMode
91  									.getCode()) || COUNTER_MODE
92  									.equalsIgnoreCase(operationalMode.getCode()))) {
93  						//Step 5 - If DL Number is Present And
94  						//	 	   Not an Issuance Office And     
95  						//		   DL Number First Byte is Not "X"
96  						if (isDlNumberPresent && !isIssuanceOffice
97  								&& !isDlNumberContainsX) {
98  							//Add Issuance Message - New Number Requested
99  							issuanceReferralMessages
100 									.add(IssuanceReferralMessagesType.NEW_NUMBER_REQUESTED);
101 						}
102 					}
103 				}
104 			}
105 		}
106 		//Record Processed with CNA Indicator
107 		//EDL VCSI45B2 - Bit 0
108 		//Java IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA
109 		//Step 1 - Is CNA Condition Exist And
110 		//         Is Sales Person Application
111 		if (EaseUtil.isNotNull(isCNACondition)
112 				&& isCNACondition.equals(Boolean.TRUE) && isSalesApplication) {
113 			//Add Issuance Message - Record Processed with CNA
114 			issuanceReferralMessages
115 					.add(IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA);
116 		}
117 		//Record to be combined and Record Processed with CNA Indicator
118 		//EDL VCSI45B2 - Bit 0 And VCSI45B1 - Bit 0
119 		//Java IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED
120 		//Java IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA
121 		//Step 1 - If TTC - DLA, IDA, IDS And
122 		//         Is CNA Condition
123 		if ((TTC_DLA.equalsIgnoreCase(ttc) || TTC_IDA.equalsIgnoreCase(ttc) || TTC_IDS
124 				.equalsIgnoreCase(ttc))
125 				&& isCNACondition.equals(Boolean.TRUE)) {
126 			//Add Issuance Message - Record Processed with CNA
127 			issuanceReferralMessages
128 					.add(IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA);
129 			//Add Issuance Message - Record to be combined
130 			issuanceReferralMessages
131 					.add(IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED);
132 		}
133 		//Record Processed with CNA Indicator
134 		//EDL VCSI45B2 - Bit 0
135 		//Java IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA
136 		//Step 1 - If CNA Condition
137 		if (isCNACondition.equals(Boolean.TRUE)) {
138 			//Add Issuance Message - Record Processed with CNA
139 			issuanceReferralMessages
140 					.add(IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA);
141 		}
142 		//DLA4/DLA7 - Record to be Combined
143 		//EDL VCSI45B1 - Bit 0
144 		//Java IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED
145 		//Step 1 - If isNewDLRequested And
146 		//         Inventory DL Number is Not Null
147 		if (isNewDLRequested && isDlNumberPresent) {
148 			//Add Issuance Message - Record to be Combined
149 			issuanceReferralMessages
150 					.add(IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED);
151 		}
152 		//LA4A - Record New Number Requested
153 		//EDL VCSI45B3 - Bit 0
154 		//Java IssuanceReferralMessagesType.NEW_NUMBER_REQUESTED
155 		//Step 1 - For TTC - DRT/IRT
156 		//Step 2 - If Gender Code between Existing Record and ELI Record does not Match
157 		if (TTC_DRT.equalsIgnoreCase(ttc) || TTC_IRT.equalsIgnoreCase(ttc)) {
158 			if (!EaseUtil.isNullOrBlank(existingGenderCode)
159 					&& !EaseUtil.isNullOrBlank(eliGenderCode)) {
160 				if (existingGenderCode.getCode().equalsIgnoreCase(
161 						eliGenderCode.getCode())) {
162 					//Remove Issuance Message - Record New Number Requested
163 					issuanceReferralMessages
164 							.remove(IssuanceReferralMessagesType.NEW_NUMBER_REQUESTED);
165 				}
166 			}
167 		}
168 		//LA4A - SUBROUT Z#SMOV04 Move Pending Segments to DWA
169 		//EDL VCSI45B1 - Bit 1 Java - IssuanceReferralMessagesType.STOPS_WERE_BYPASSED
170 		//EDL VCSI45B3 - Bit 1 Java - IssuanceReferralMessagesType.NAME_SEX_DLNUMBER_OR_BIRTHDATE_CHANGED
171 		//Step 1 - If Original Application Cashier SEQ# is present And > 7000
172 		if (EaseUtil.isNotNull(controlCashierSeqNumber)
173 				&& controlCashierSeqNumber > CONTROL_CASHIER_SEQ_MAX) {
174 			//Add Issuance Message - Stops were Bypassed
175 			issuanceReferralMessages
176 					.add(IssuanceReferralMessagesType.STOPS_WERE_BYPASSED);
177 			//Add Issuance Message - Name, Sex, DL Number or BirthDate Changed
178 			issuanceReferralMessages
179 					.add(IssuanceReferralMessagesType.NAME_SEX_DLNUMBER_OR_BIRTHDATE_CHANGED);
180 		}
181 		//DLA7 - If Multiple ANI Duplicated Exist
182 		//EDL VCSI45B1 - Bit 0
183 		//Java - IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED
184 		//Step 1 - If Duplicate ANI Records Exists
185 		if (!EaseUtil.isNullOrBlank(duplicateAniRecords)) {
186 			//Add Issuance Message - Record to be combined
187 			issuanceReferralMessages
188 					.add(IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED);
189 		}
190 		//DLA7 - If Password Required
191 		//EDL VCSI45B1 - Bit 1
192 		//Java - IssuanceReferralMessagesType.STOPS_WERE_BYPASSED
193 		//Step 1 - If Password Required
194 		//Step 2 - If Password is not Required
195 		if (isPasswordRequired) {
196 			//Add Issuance Message - Stops were ByPassed
197 			issuanceReferralMessages
198 					.add(IssuanceReferralMessagesType.STOPS_WERE_BYPASSED);
199 		}
200 		else {
201 			//Remove Issuance Message - Stops were ByPassed
202 			issuanceReferralMessages
203 					.remove(IssuanceReferralMessagesType.STOPS_WERE_BYPASSED);
204 		}
205 		//TODO Requirements SUBROUT  CKYSNOIN CHECK YES NO INDICATORS
206 		//TODO Requirements SUBROUT MVPRIMIN   - Move Prim Info
207 		//TODO Requirements SUBROUT PRIMINIT - Primary Initial
208 		//TODO DLA8
209 		//LA4A - Generic if No CNA then remove the RECORD_PROCESSED_WITH_CNA
210 		//EDL VCSI45B2 Bit 0
211 		//Java - IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA
212 		//Step 1 - If no CNA Condition Exists
213 		if(!isCNACondition)
214 		{
215 			//Remove Issuance Message - Record Processed with CNA
216 			issuanceReferralMessages
217 					.remove(IssuanceReferralMessagesType.RECORD_PROCESSED_WITH_CNA);
218 		}
219 		//Return the List of Applicable Issuance Referral Indicators
220 		Iterator <IssuanceReferralMessagesType> iterator = issuanceReferralMessages
221 				.iterator();
222 		List <IssuanceReferralMessagesType> issuanceReferralMessagesList = new ArrayList <IssuanceReferralMessagesType>();
223 		while (iterator.hasNext()) {
224 			issuanceReferralMessagesList.add(iterator.next());
225 		}
226 		return issuanceReferralMessagesList;
227 	}

53  	/***
54  	 * Evaluates the Endorsements for the interim license document. 
55  	 * @param driverLicense
56  	 * @param hazmatFlag
57  	 * @param isDla
58  	 * @param isFeeReasonCodeK
59  	 * @return
60  	 */
61  	public String checkCommercialEndorsements(DriverLicense driverLicense,
62  			StringBuilder hazmatFlag, boolean isDla, boolean isFeeReasonCodeK) {
63  		hazmatFlag.setCharAt(0, IPrintConverterConstants.HAZMAT_CHAR_VALUE_N);
64  		ArrayList <String> endorsementsList = new ArrayList <String>();
65  		List <String> newCategories = DocumentConverterUtil
66  				.convertCodeSetToStringList(driverLicense.getNewCategories());
67  		List <String> existingLicenseClasses = DocumentConverterUtil
68  				.convertCodeSetToStringList(driverLicense
69  						.getExistingLicenseClasses());
70  		List <String> existingEndorsements = DocumentConverterUtil
71  				.convertCodeSetToStringList(driverLicense
72  						.getExistingEndorsements());
73  		List <String> newEndorsements = DocumentConverterUtil
74  				.convertCodeSetToStringList(driverLicense.getNewEndorsements());
75  		CommercialTestResultGroup commercialTestResultGroup = driverLicense
76  				.getCommercialTestResultGroup();
77  		List <InterimPermit> interimPermits = driverLicense
78  				.getInterimPermits(InterimPermitType.TEMPORARY);
79  		// If class of records have A or B and (if TTC is DLA or fee reason code is K) 
80  		if (existingLicenseClasses
81  				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_A)
82  				|| existingLicenseClasses
83  						.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_B)
84  				&& (isDla || isFeeReasonCodeK)) {
85  			if (newEndorsements
86  					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_PV)
87  					&& existingEndorsements
88  							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_PV)) {
89  				DocumentConverterUtil.addToListWithRedundancyCheck(
90  						endorsementsList,
91  						IPrintConverterConstants.DL_REC_ENDORSEMENT_P);
92  			}
93  			if (newEndorsements
94  					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_PV)
95  					&& newEndorsements
96  							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_SB)
97  					&& existingEndorsements
98  							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_SB)) {
99  				DocumentConverterUtil.addToListWithRedundancyCheck(
100 						endorsementsList,
101 						IPrintConverterConstants.DL_REC_ENDORSEMENT_S);
102 			}
103 			if (newEndorsements
104 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_PV)
105 					&& newEndorsements
106 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_SB)
107 					&& DocumentConverterUtil
108 							.isEndorsementLawTestPassedOrNotRequired(
109 									commercialTestResultGroup,
110 									CommercialTestResultType.PASSENGER_TEST_RESULT)) {
111 				DocumentConverterUtil.addToListWithRedundancyCheck(
112 						endorsementsList,
113 						IPrintConverterConstants.DL_REC_ENDORSEMENT_S);
114 			}
115 			if (newEndorsements
116 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_PV)
117 					&& DocumentConverterUtil
118 							.isEndorsementLawTestPassedOrNotRequired(
119 									commercialTestResultGroup,
120 									CommercialTestResultType.PASSENGER_TEST_RESULT)) {
121 				DocumentConverterUtil.addToListWithRedundancyCheck(
122 						endorsementsList,
123 						IPrintConverterConstants.DL_REC_ENDORSEMENT_P);
124 			}
125 		}
126 		// if class of records have B or C or D and (if TTC is DLA or fee reason code is K) or class of records have A and category 1 FF
127 		if ((((existingLicenseClasses
128 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_B)
129 				|| existingLicenseClasses
130 						.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_C) || existingLicenseClasses
131 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_D))
132 				&& EaseUtil.isNotNull(newCategories)
133 				&& !newCategories.isEmpty() && newCategories
134 				.contains(IPrintConverterConstants.FIRE_FIGHTER_CLASS_A_CATEGORY_1)) || (existingLicenseClasses
135 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_A)))
136 				&& newEndorsements
137 						.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_DT)
138 				&& isDla && isFeeReasonCodeK) {
139 			if (existingEndorsements
140 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_DT)
141 					|| DocumentConverterUtil
142 							.isEndorsementLawTestPassedOrNotRequired(
143 									commercialTestResultGroup,
144 									CommercialTestResultType.DOUBLES_OR_TRIPLES_TEST_RESULT)
145 					|| (!existingEndorsements
146 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_DT) && DocumentConverterUtil
147 							.isEndorsementLawTestPassedOrNotRequired(
148 									commercialTestResultGroup,
149 									CommercialTestResultType.DOUBLES_OR_TRIPLES_TEST_RESULT))) {
150 				DocumentConverterUtil.addToListWithRedundancyCheck(
151 						endorsementsList,
152 						IPrintConverterConstants.DL_REC_ENDORSEMENT_T);
153 			}
154 		}
155 		boolean isCommercialPermitClass = false;
156 		List <CodeSetElement> permitClassCodes = new ArrayList <CodeSetElement>();
157 		if (EaseUtil.isNotNull(interimPermits) && !interimPermits.isEmpty()) {
158 			for (InterimPermit interimPermit : interimPermits) {
159 				if (!EaseUtil.isNullOrBlank(interimPermit
160 						.getPermitLicenseClassCode())) {
161 					permitClassCodes.add(interimPermit
162 							.getPermitLicenseClassCode());
163 				}
164 			}
165 		}
166 		if (ConverterHelper.isAnyCommercialLicenseClassExist(permitClassCodes)) {
167 			isCommercialPermitClass = true;
168 		}
169 		List <String> permitLicenseClasses = DocumentConverterUtil
170 				.convertCodeSetToStringList(permitClassCodes);
171 		//COMHMWTV - CHECK COMM'L HM WITH TV
172 		if (!(permitLicenseClasses
173 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_M) || permitLicenseClasses
174 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_F))
175 				&& isCommercialPermitClass
176 				&& newEndorsements
177 						.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)) {
178 			if (DocumentConverterUtil.isEndorsementLawTestPassedOrNotRequired(
179 					commercialTestResultGroup,
180 					CommercialTestResultType.TANK_TEST_RESULT)
181 					|| (existingEndorsements
182 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV))
183 					&& !DocumentConverterUtil
184 							.isEndorsementLawTestPassedOrNotRequired(
185 									commercialTestResultGroup,
186 									CommercialTestResultType.TANK_TEST_RESULT)) {
187 				checkCommercialHMWithTVEndorsement(endorsementsList,
188 						driverLicense, isDla, hazmatFlag);
189 			}
190 		}
191 		//NCOMHWTV - CHECK NON-COMMERCIAL HM WITH TV
192 		if (!(permitLicenseClasses
193 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_M) || permitLicenseClasses
194 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_F))
195 				&& !isCommercialPermitClass
196 				&& EaseUtil.isNotNull(newCategories)
197 				&& !newCategories.isEmpty()
198 				&& (newCategories
199 						.contains(IPrintConverterConstants.FIRE_FIGHTER_CLASS_A_CATEGORY_1) || newCategories
200 						.contains(IPrintConverterConstants.FIRE_FIGHTER_CLASS_A_CATEGORY_2))) {
201 			checkNonCommercialHMWithTVEndorsements(endorsementsList,
202 					driverLicense);
203 		}
204 		// COMWOTV - CHECK COMMERCIAL HM WITHOUT TV
205 		if (!(permitLicenseClasses
206 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_M) || permitLicenseClasses
207 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_F))
208 				&& isCommercialPermitClass
209 				&& newEndorsements
210 						.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM))
211 			checkCommercialHMWithoutTVEndorsements(endorsementsList,
212 					driverLicense, isDla, hazmatFlag);
213 		// COMWOTV - CHECK NON-COMMERCIAL HM WITHOUT TV		
214 		if (!(permitLicenseClasses
215 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_M) || permitLicenseClasses
216 				.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_F))
217 				&& !isCommercialPermitClass
218 				&& newEndorsements
219 						.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)
220 				&& EaseUtil.isNotNull(newCategories)
221 				&& !newCategories.isEmpty()
222 				&& (newCategories
223 						.contains(IPrintConverterConstants.FIRE_FIGHTER_CLASS_A_CATEGORY_1) || newCategories
224 						.contains(IPrintConverterConstants.FIRE_FIGHTER_CLASS_A_CATEGORY_2))) {
225 			checkNonCommercialHMWithoutTVEndorsements(endorsementsList,
226 					driverLicense);
227 		}
228 		String endorsementsString = DocumentConverterUtil
229 				.convertEndorsementsListToString(endorsementsList);
230 		if (!isNullOrBlank(endorsementsString)
231 				&& !endorsementsString.trim().equalsIgnoreCase(
232 						IPrintConverterConstants.BLANK)) {
233 			return IPrintConverterConstants.ENDORSEMENT_PREFIX
234 					+ endorsementsString.trim();
235 		}
236 		else {
237 			return IPrintConverterConstants.BLANK;
238 		}
239 	}

241 	/***
242 	 * Evaluates COMHMWTV - CHECK COMM'L HM WITH TV Section. 
243 	 * @param endorsementsList 
244 	 * @param driverLicense
245 	 * @param isDla
246 	 * @param hazmatFlag
247 	 */
248 	private void checkCommercialHMWithTVEndorsement(
249 			ArrayList <String> endorsementsList, DriverLicense driverLicense,
250 			boolean isDla, StringBuilder hazmatFlag) {
251 		Date backgroundCheckExpirationDate = driverLicense
252 				.getBackgroundCheckExpirationDate();
253 		List <String> existingEndorsements = DocumentConverterUtil
254 				.convertCodeSetToStringList(driverLicense
255 						.getExistingEndorsements());
256 		List <String> newEndorsements = DocumentConverterUtil
257 				.convertCodeSetToStringList(driverLicense.getNewEndorsements());
258 		String backgroundCheckIndicator = null;
259 		if (!EaseUtil
260 				.isNullOrBlank(driverLicense.getBackgroundCheckIndicator())) {
261 			backgroundCheckIndicator = driverLicense
262 					.getBackgroundCheckIndicator().getCode();
263 		}
264 		Date prevAppWorkDate = DocumentConverterUtil.convertStringToDate(
265 				IPrintConverterConstants.PREVIOUS_APPLICATION_WORK_DATE,
266 				IPrintConverterConstants.PREVIOUS_APPLICATION_WORK_DATE_FORMAT);
267 		if (newEndorsements
268 				.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)) {
269 			if (EaseUtil.isNotNull(backgroundCheckIndicator)
270 					&& backgroundCheckIndicator
271 							.equals(IPrintConverterConstants.BACKGROUNDCHECK_0)) {
272 				DocumentConverterUtil.addToListWithRedundancyCheck(
273 						endorsementsList,
274 						IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
275 			}
276 			if (existingEndorsements
277 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)) {
278 				if ((EaseUtil.isNotNull(backgroundCheckIndicator) && (backgroundCheckIndicator
279 						.equals(IPrintConverterConstants.BACKGROUNDCHECK_N) || backgroundCheckIndicator
280 						.equals(IPrintConverterConstants.BACKGROUNDCHECK_S)))
281 						&& isDla) {
282 					if (EaseUtil.isNotNull(backgroundCheckExpirationDate)
283 							&& backgroundCheckExpirationDate
284 									.before(prevAppWorkDate)) {
285 						if (ConverterHelper
286 								.isAnyCommercialLicenseClassExist(driverLicense
287 										.getExistingLicenseClasses())) {
288 							DocumentConverterUtil
289 									.addToListWithRedundancyCheck(
290 											endorsementsList,
291 											IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
292 						}
293 						else {
294 							hazmatFlag
295 									.setCharAt(
296 											0,
297 											IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
298 							DocumentConverterUtil
299 									.addToListWithRedundancyCheck(
300 											endorsementsList,
301 											IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
302 						}
303 					}
304 					else {
305 						hazmatFlag.setCharAt(0,
306 								IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
307 						DocumentConverterUtil.addToListWithRedundancyCheck(
308 								endorsementsList,
309 								IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
310 					}
311 				}
312 				if ((EaseUtil.isNotNull(backgroundCheckIndicator)
313 						&& backgroundCheckIndicator
314 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_N) && !isDla)) {
315 					DocumentConverterUtil.addToListWithRedundancyCheck(
316 							endorsementsList,
317 							IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
318 				}
319 				if ((EaseUtil.isNotNull(backgroundCheckIndicator)
320 						&& backgroundCheckIndicator
321 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_S) && !isDla)) {
322 					hazmatFlag.setCharAt(0,
323 							IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
324 					DocumentConverterUtil.addToListWithRedundancyCheck(
325 							endorsementsList,
326 							IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
327 				}
328 				if (EaseUtil.isNotNull(backgroundCheckIndicator)
329 						&& (backgroundCheckIndicator
330 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_P) || backgroundCheckIndicator
331 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_T))) {
332 					if (EaseUtil.isNotNull(backgroundCheckExpirationDate)
333 							&& backgroundCheckExpirationDate.before(new Date())) {
334 						hazmatFlag.setCharAt(0,
335 								IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
336 						DocumentConverterUtil.addToListWithRedundancyCheck(
337 								endorsementsList,
338 								IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
339 					}
340 					else {
341 						DocumentConverterUtil.addToListWithRedundancyCheck(
342 								endorsementsList,
343 								IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
344 					}
345 				}
346 				if (!(backgroundCheckIndicator
347 						.equals(IPrintConverterConstants.BACKGROUNDCHECK_0)
348 						|| backgroundCheckIndicator
349 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_N)
350 						|| backgroundCheckIndicator
351 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_P)
352 						|| backgroundCheckIndicator
353 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_S) || backgroundCheckIndicator
354 						.equals(IPrintConverterConstants.BACKGROUNDCHECK_T))) {
355 					if (ConverterHelper
356 							.isAnyCommercialLicenseClassExist(driverLicense
357 									.getExistingLicenseClasses())) {
358 						DocumentConverterUtil.addToListWithRedundancyCheck(
359 								endorsementsList,
360 								IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
361 					}
362 					else {
363 						hazmatFlag.setCharAt(0,
364 								IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
365 						DocumentConverterUtil.addToListWithRedundancyCheck(
366 								endorsementsList,
367 								IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
368 					}
369 				}
370 			}
371 			else {
372 				if (EaseUtil.isNotNull(backgroundCheckIndicator)
373 						&& !backgroundCheckIndicator
374 								.equals(IPrintConverterConstants.BACKGROUNDCHECK_0)) {
375 					hazmatFlag.setCharAt(0,
376 							IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
377 					DocumentConverterUtil.addToListWithRedundancyCheck(
378 							endorsementsList,
379 							IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
380 				}
381 			}
382 		}
383 		else {
384 			if (!existingEndorsements
385 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)) {
386 				hazmatFlag.setCharAt(0,
387 						IPrintConverterConstants.HAZMAT_CHAR_VALUE_Y);
388 				DocumentConverterUtil.addToListWithRedundancyCheck(
389 						endorsementsList,
390 						IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
391 			}
392 		}
393 	}

395 	/***
396 	 * Evaluates NCOMHWTV - CHECK NON-COMMERCIAL HM WITH TV Section.
397 	 * @param endorsementsList
398 	 * @param driverLicense
399 	 */
400 	private void checkNonCommercialHMWithTVEndorsements(
401 			ArrayList <String> endorsementsList, DriverLicense driverLicense) {
402 		List <String> existingLicenseClasses = DocumentConverterUtil
403 				.convertCodeSetToStringList(driverLicense
404 						.getExistingLicenseClasses());
405 		List <String> existingEndorsements = DocumentConverterUtil
406 				.convertCodeSetToStringList(driverLicense
407 						.getExistingEndorsements());
408 		List <String> newEndorsements = DocumentConverterUtil
409 				.convertCodeSetToStringList(driverLicense.getNewEndorsements());
410 		if (!newEndorsements
411 				.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)) {
412 			if ((DocumentConverterUtil.isEndorsementLawTestPassedOrNotRequired(
413 					driverLicense.getCommercialTestResultGroup(),
414 					CommercialTestResultType.TANK_TEST_RESULT) && newEndorsements
415 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV))
416 					|| (existingEndorsements
417 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV) && !DocumentConverterUtil
418 							.isEndorsementLawTestPassedOrNotRequired(
419 									driverLicense
420 											.getCommercialTestResultGroup(),
421 									CommercialTestResultType.TANK_TEST_RESULT))) {
422 				DocumentConverterUtil.addToListWithRedundancyCheck(
423 						endorsementsList,
424 						IPrintConverterConstants.DL_REC_ENDORSEMENT_N);
425 			}
426 		}
427 		else {
428 			if ((newEndorsements
429 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)
430 					&& DocumentConverterUtil
431 							.isEndorsementLawTestPassedOrNotRequired(
432 									driverLicense
433 											.getCommercialTestResultGroup(),
434 									CommercialTestResultType.TANK_TEST_RESULT) && DocumentConverterUtil
435 					.isEndorsementLawTestPassedOrNotRequired(driverLicense
436 							.getCommercialTestResultGroup(),
437 							CommercialTestResultType.HAZMAT_TEST_RESULT))
438 					|| (existingEndorsements
439 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)
440 							&& !DocumentConverterUtil
441 									.isEndorsementLawTestPassedOrNotRequired(
442 											driverLicense
443 													.getCommercialTestResultGroup(),
444 											CommercialTestResultType.TANK_TEST_RESULT) && DocumentConverterUtil
445 							.isEndorsementLawTestPassedOrNotRequired(
446 									driverLicense
447 											.getCommercialTestResultGroup(),
448 									CommercialTestResultType.HAZMAT_TEST_RESULT))) {
449 				DocumentConverterUtil.addToListWithRedundancyCheck(
450 						endorsementsList,
451 						IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
452 			}
453 			if (existingEndorsements
454 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)
455 					&& (existingLicenseClasses
456 							.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_D) || existingLicenseClasses
457 							.contains(IPrintConverterConstants.DL_REC_INPUT_CLASS_E))
458 					&& ((newEndorsements
459 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)
460 							&& DocumentConverterUtil
461 									.isEndorsementLawTestPassedOrNotRequired(
462 											driverLicense
463 													.getCommercialTestResultGroup(),
464 											CommercialTestResultType.TANK_TEST_RESULT) && !DocumentConverterUtil
465 							.isEndorsementLawTestPassedOrNotRequired(
466 									driverLicense
467 											.getCommercialTestResultGroup(),
468 									CommercialTestResultType.HAZMAT_TEST_RESULT)) || (existingEndorsements
469 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)
470 							&& !DocumentConverterUtil
471 									.isEndorsementLawTestPassedOrNotRequired(
472 											driverLicense
473 													.getCommercialTestResultGroup(),
474 											CommercialTestResultType.TANK_TEST_RESULT) && !DocumentConverterUtil
475 							.isEndorsementLawTestPassedOrNotRequired(
476 									driverLicense
477 											.getCommercialTestResultGroup(),
478 									CommercialTestResultType.HAZMAT_TEST_RESULT)))) {
479 				DocumentConverterUtil.addToListWithRedundancyCheck(
480 						endorsementsList,
481 						IPrintConverterConstants.DL_REC_ENDORSEMENT_X);
482 			}
483 			if (!existingEndorsements
484 					.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_HM)
485 					&& ((newEndorsements
486 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)
487 							&& !DocumentConverterUtil
488 									.isEndorsementLawTestPassedOrNotRequired(
489 											driverLicense
490 													.getCommercialTestResultGroup(),
491 											CommercialTestResultType.TANK_TEST_RESULT) && DocumentConverterUtil
492 							.isEndorsementLawTestPassedOrNotRequired(
493 									driverLicense
494 											.getCommercialTestResultGroup(),
495 									CommercialTestResultType.HAZMAT_TEST_RESULT)) || (existingEndorsements
496 							.contains(IPrintConverterConstants.DL_REC_ENDORSEMENT_TV)
497 							&& !DocumentConverterUtil
498 									.isEndorsementLawTestPassedOrNotRequired(

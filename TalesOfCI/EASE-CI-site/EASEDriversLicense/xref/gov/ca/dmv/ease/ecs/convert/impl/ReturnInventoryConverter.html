31  public class ReturnInventoryConverter extends AbstractEcsMessageConverter {
32  	/*** The Constant TCODE. */
33  	private static final String TCODE = "C78";
34  	/*** The Constant SEQ_NUMBER. */
35  	private static final String SEQ_NUMBER_FOR_DL = "4784";
36  	/*** The Constant SEQ_NUMBER_FOR_ID. */
37  	private static final String SEQ_NUMBER_FOR_ID = "0000";

39  	/* (non-Javadoc)
40  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
41  	 */
42  	@Override
43  	public String createMessage(IEcsRequest requestFromClient) {
44  		// NOTE : This Tcode is used in both DL and VR. For now, it is implemented only for DL. In future, it has to be modified to accept VR related information also.
45  		ReturnInventoryRequestEcs request = (ReturnInventoryRequestEcs) requestFromClient;
46  		License license = request.getLicense();
47  		Object[] message = new Object[] {
48  				TCODE, // TCODE
49  				getOfficeId(request.getUserContext()), // OFFICE_ID
50  				getWorkDate(request.getUserContext(),
51  						EcsConverterHelper.FORMATTER_MM_DD_YY), // WORKDATE
52  				getEmployeeID(request.getUserContext()),// TECHNICIAN ID		
53  				// NOTE : For it is not implemented for VR.
54  				getTechSequenceNumber(license), // TECH SEQUENCE NUMBER				
55  				filler(license.getItemCode(), 3), // INVENTORY ITEM CODE
56  				filler(getCentralIssuedLicenseOrIdNumber(license), 8), // CENTRAL ISSUED LICENSE/ID #
57  				// NOTE : For DL, TYPE LICENSE is blank. For VR, it is not blank.
58  				filler(BLANK, 2),// TYPE LICENSE
59  				// NOTE : For DL, VIN is blank. For VR, it is not blank.
60  				filler(BLANK, 30), // VIN
61  				// NOTE : For DL, FIRST 3 POS MAKE is blank, For VR, it is not blank.
62  				filler(BLANK, 3),// FIRST 3 POS MAKE
63  				getTypeTransactionCode(request.getUserContext()), // TYPE TRANS CODE (DMVA)				
64  				filler(BLANK, 7) // FILLER
65  		};
66  		return createMessageFormat(message);
67  	}

69  	/***
70  	 * Gets the type transaction code.
71  	 * 
72  	 * @param userContext the user context
73  	 * @return the type transaction code
74  	 */
75  	private String getTypeTransactionCode(IUserContext userContext) {
76  		String c78ttc = BLANK;
77  		if (userContext.getFunctionalMode() != null
78  				&& userContext.getFunctionalMode().equals("S")
79  				&& !userContext.getTtc().equals("006")) {
80  			c78ttc = userContext.getTtc();
81  		}
82  		else {
83  			c78ttc = userContext.getTtc();
84  		}
85  		if (c78ttc.length() >= 3 && c78ttc.charAt(2) == '1') {
86  			c78ttc = c78ttc.substring(0, 2) + "0" + c78ttc.substring(3);
87  		}
88  		return c78ttc;
89  	}

91  	/***
92  	 * This method will fetch either License Number or ID Card Number.
93  	 * 
94  	 * @param license the License
95  	 * @return either License Number or ID Card Number
96  	 */
97  	private String getTechSequenceNumber(License license) {
98  		String seqNum = BLANK;
99  		if (license instanceof DriverLicense) {
100 			seqNum = SEQ_NUMBER_FOR_DL;
101 		}
102 		else if (license instanceof IdCard) {
103 			seqNum = SEQ_NUMBER_FOR_ID;
104 		}
105 		return seqNum;
106 	}
107 
108 	/***
109 	 * This method retrieves either License or ID Card Number.
110 	 * @param license the License
111 	 * @return either License or ID Card Number.
112 	 */
113 	private String getCentralIssuedLicenseOrIdNumber(License license) {
114 		String issuedLicenseOrIdNumber = BLANK;
115 		if (!isNull(license)) {
116 			issuedLicenseOrIdNumber = license.getLicenseNumber();
117 		}
118 		return issuedLicenseOrIdNumber;
119 	}

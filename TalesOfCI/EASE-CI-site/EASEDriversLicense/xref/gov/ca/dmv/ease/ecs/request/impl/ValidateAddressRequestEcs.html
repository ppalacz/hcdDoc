35  public class ValidateAddressRequestEcs extends SynchronousEcsRequest {
36  	/*** Serial version id for this class. */
37  	private static final long serialVersionUID = 3339478437265761604L;
38  	/*** The addresses. */
39  	private List <Address> addresses;
40  
41  	/***
42  	 * Constructs a validate address ecs request.
43  	 * 
44  	 * @param addresses the addresses.
45  	 * @param userContext the user context
46  	 */
47  	public ValidateAddressRequestEcs(List <Address> addresses,
48  			IUserContext userContext,
49  			ValidateAddressConverter validateAddressConverter) {
50  		super(userContext);
51  		setAddresses(addresses);
52  		setMessageConverter(validateAddressConverter);
53  	}
54  
55  	/* (non-Javadoc)
56  	 * @see gov.ca.dmv.ease.ecs.request.impl.SynchronousEcsRequest#execute()
57  	 */
58  	@Override
59  	public ValidateAddressResponseEcs execute() throws EcsServiceException {
60  		ValidateAddressResponseEcs validateAddressResponseEcs = null;
61  		IErrorCollector errorCollector = getValidator().validate(this);
62  		if (errorCollector.hasErrors()) {
63  			validateAddressResponseEcs = new ValidateAddressResponseEcs(
64  					errorCollector);
65  		}
66  		else {
67  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
68  			//In the catch block we need to register the error to the error collector.
69  			try {
70  				validateAddressResponseEcs = (ValidateAddressResponseEcs) super
71  						.execute();
72  			}
73  			catch (EcsResponseIsNullException responseIsNullException) {
74  				errorCollector = new ErrorCollector();
75  				errorCollector.register(new EaseValidationException(
76  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
77  				validateAddressResponseEcs = new ValidateAddressResponseEcs(
78  						errorCollector);
79  			}
80  			catch (SystemException sysEx) {
81  				errorCollector = new ErrorCollector();
82  				errorCollector.register(new EaseValidationException(
83  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
84  				validateAddressResponseEcs = new ValidateAddressResponseEcs(
85  						errorCollector);
86  			}
87  		}
88  		return validateAddressResponseEcs;
89  	}
90

41  	@Override
42  	public String createMessage(IEcsRequest requestFromClient) {
43  		DriverRecordUpdateRequestEcs driverRecordUpdateRequestEcs = (DriverRecordUpdateRequestEcs) requestFromClient;
44  		UserContext userContext = (UserContext) driverRecordUpdateRequestEcs
45  				.getUserContext();
46  		String ttc = userContext.getTtc();
47  		Person person = driverRecordUpdateRequestEcs.getPerson();
48  		Date birthDate = person.getBirthDate();
49  		String typeOfUpdateCode = null;
50  		if (isNotNull(driverRecordUpdateRequestEcs.getTypeOfUpdate())) {
51  			typeOfUpdateCode = driverRecordUpdateRequestEcs.getTypeOfUpdate()
52  					.getCode();
53  		}
54  		Object[] message = new Object[11];
55  		message[0] = IEcsConverterConstants.TCODE_DL6;
56  		message[1] = filler(ttc, 3);
57  		message[2] = filler(IEcsConverterConstants.TRANSACTION_SUB_CODE_U, 1);
58  		message[3] = filler(IEcsConverterConstants.REQUESTER_CODE_PREFIX, 2);
59  		message[4] = filler(userContext.getOfficeId(), 3);
60  		message[5] = filler(userContext.getTechId(), 2);
61  		message[6] = filler(ConverterHelper.getDlorIdNumber(person), 8);
62  		message[7] = filler(ConverterHelper
63  				.getFirstThreeCharsOfLastName(person), 3);
64  		message[8] = filler(getTypeInputCode(ttc), 1);
65  		if ((isNotNull(typeOfUpdateCode))) {
66  			if ((typeOfUpdateCode
67  					.equalsIgnoreCase(IEcsConverterConstants.INPUT_CODE_1))
68  					|| (typeOfUpdateCode
69  							.equalsIgnoreCase(IEcsConverterConstants.INPUT_CODE_3))) {
70  				message[9] = filler(getFieldMnemonicCode(ttc, typeOfUpdateCode,
71  						birthDate), 2);
72  				message[10] = filler(
73  						getUpdateData(driverRecordUpdateRequestEcs), 28);
74  			}
75  			else {
76  				//NOTE: 4 bytes are moved. 2 bytes overflow into the ML6DATA field. i.e 'field mnemonic code' in written in to 'update data' field
77  				message[9] = filler(getFieldMnemonicCode(ttc, typeOfUpdateCode,
78  						birthDate), 4);
79  				message[10] = filler(
80  						getUpdateData(driverRecordUpdateRequestEcs), 26);
81  			}
82  		}
83  		if (!ttc.equalsIgnoreCase(IEcsConverterConstants.TTC_DPR4_30U)) {
84  			message[9] = filler(getFieldMnemonicCode(ttc, typeOfUpdateCode,
85  					birthDate), 2);
86  			message[10] = filler(getUpdateData(driverRecordUpdateRequestEcs),
87  					28);
88  		}
89  		return createMessageFormat(message);
90  	}

200 	 * Get updateData.
201 	 * 
202 	 * @param 
203 	 * 
204 	 * @return typeInputCode
205 	 */
206 	private String getUpdateData(DriverRecordUpdateRequestEcs request) {
207 		StringBuilder updateData = new StringBuilder();
208 		String ttc = request.getUserContext().getTtc();
209 		if (IEcsConverterConstants.TTC_37U4_37U.equalsIgnoreCase(ttc)) {
210 			updateData.append(request.getEmployingAgency());
211 		}
212 		else if (IEcsConverterConstants.TTC_DPR1_07M.equalsIgnoreCase(ttc)
213 				|| IEcsConverterConstants.TTC_DPR1_14M.equalsIgnoreCase(ttc)) {
214 			Invoice invoice = request.getInvoice();
215 			UserContext userContext = (UserContext) request.getUserContext();
216 			String cashierSeqNo = filler(userContext
217 					.getControlCashierSeqNumber().toString(), 4);
218 			String dateFeesReceived = convertDateToString(request
219 					.getDateFeesReceived(),
220 					EcsConverterHelper.FORMATTER_MM_DD_YY);
221 			String workDate = convertDateToString(userContext.getWorkDate(),
222 					EcsConverterHelper.FORMATTER_MM_DD_YY);
223 			updateData.append(dateFeesReceived);
224 			updateData.append(workDate);
225 			updateData.append(cashierSeqNo);
226 			if (isCreditAmountPresent(invoice)) {
227 				updateData.append("Y");
228 			}
229 			else {
230 				updateData.append(IEcsConverterConstants.SPACE);
231 			}
232 			if (IEcsConverterConstants.TTC_DPR1_14M.equalsIgnoreCase(ttc)) {
233 				List <CodeSetElement> frPenaltyFeeTypeCodes = request
234 						.getFrPenaltyFeeTypeCodes();
235 				for (CodeSetElement frPenaltyFeeTypeCode : frPenaltyFeeTypeCodes) {
236 					if (isNotNull(frPenaltyFeeTypeCode)) {
237 						if (isNotNull(frPenaltyFeeTypeCode.getCode())) {
238 							/////Clarify
239 							updateData.append(frPenaltyFeeTypeCode.getCode());
240 						}
241 					}
242 				}
243 			}
244 		}
245 		else if (IEcsConverterConstants.TTC_DPR4_30U.equalsIgnoreCase(ttc)) {
246 			String typeOfUpdateCode = request.getTypeOfUpdate().getCode();
247 			Person person = request.getPerson();
248 			DriverLicense driverLicense = person.getDriverLicense();
249 			PhotoInformation photoInformation = driverLicense
250 					.getPhotoInformation();
251 			String correctedMgNo = photoInformation
252 					.getBatesMicrographicsNumber();
253 			if (isNotNull(typeOfUpdateCode)) {
254 				if (typeOfUpdateCode
255 						.equalsIgnoreCase(IEcsConverterConstants.INPUT_CODE_1)) {
256 					updateData.append(correctedMgNo);
257 				}
258 				else if (typeOfUpdateCode
259 						.equalsIgnoreCase(IEcsConverterConstants.INPUT_CODE_3)) {
260 					if (isValidAge(person.getBirthDate())) {
261 						updateData
262 								.append(IEcsConverterConstants.UPDATE_DATA_30);
263 					}
264 					else {
265 						updateData
266 								.append(IEcsConverterConstants.UPDATE_DATA_00);
267 					}
268 				}
269 			}
270 		}
271 		return updateData.toString();
272 	}

274 	/***
275 	 * Return true if valid age i.e age >= 16 or age < 18 
276 	 * 
277 	 * @param birthDate
278 	 * @return boolean value
279 	 */
280 	private boolean isValidAge(Date birthDate) {
281 		Calendar birthCalender = Calendar.getInstance();
282 		birthCalender.setTime(birthDate);
283 		Calendar currentCalender = Calendar.getInstance();
284 		int birthYear = birthCalender.get(Calendar.YEAR);
285 		int currentYear = currentCalender.get(Calendar.YEAR);
286 		int yearDiff = currentYear - birthYear;
287 		int birthMonth = birthCalender.get(Calendar.MONTH) + 1;
288 		int currentMonth = currentCalender.get(Calendar.MONTH) + 1;
289 		int monthDiff = currentMonth - birthMonth;
290 		int birthDay = birthCalender.get(Calendar.DAY_OF_MONTH);
291 		int currentDay = currentCalender.get(Calendar.DAY_OF_MONTH);
292 		int dayDiff = currentDay - birthDay;
293 		if ((yearDiff) == 17) {
294 			return true;
295 		}
296 		else if (yearDiff == 16 && monthDiff > 0) {
297 			return true;
298 		}
299 		else if (yearDiff == 16 && monthDiff == 0 && dayDiff >= 0) {
300 			return true;
301 		}
302 		else if (yearDiff == 18 && monthDiff < 0) {
303 			return true;
304 		}
305 		else if (yearDiff == 18 && monthDiff == 0 && dayDiff > 0) {
306 			return true;
307 		}
308 		return false;
309 	}

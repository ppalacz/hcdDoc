49  	public static String getCorrectedMgNumber(
50  			ProcessContext processContext) {
51  		String requiredMgNumber = null;
52  		if (processContext instanceof MiscDlProcessContext) {
53  			MiscDlProcessContext miscDlProcessContext = 
54  				(MiscDlProcessContext) processContext;
55  			if (isDriverRecordUpdateApplication(
56  					miscDlProcessContext)) {
57  				Person applicant = miscDlProcessContext
58  				.getDriverRecordUpdateApplication().getApplicant();
59  				if (applicant != null && applicant
60  						.getDriverLicense() != null 
61  						&& applicant.getDriverLicense()
62  						.getPhotoInformation() != null) {
63  					requiredMgNumber = applicant.getDriverLicense()
64  					.getPhotoInformation().getBatesMicrographicsNumber();
65  				}
66  			}
67  		}
68  		return requiredMgNumber;
69  	}

119 	 * Checks if is mg number in range.
120 	 * 
121 	 * @param mgNumber the mg number
122 	 * 
123 	 * @return true, if is mg number in range
124 	 */
125 	public static boolean isMgNumberInRange(
126 			String mgNumber) {
127 		if (mgNumber != null) {
128 			if (StringUtils.isNumeric(mgNumber)) {
129 				int mgNum = Integer.parseInt(mgNumber);
130 				if ((mgNum > 0 && mgNum < 991) 
131 						|| (mgNum == 997 
132 								|| mgNum == 998)) {
133 					return Boolean.TRUE;
134 				}
135 			}
136 		}
137 		return Boolean.FALSE;
138 	}

150 		String requiredMsg = RecordConditionConstants.BLANK;
151 		if (msg.equalsIgnoreCase(RecordConditionConstants
152 				.SUSP_EFFECTIVE_MSG)

154 						.SUSP_EFFECTIVE_MSG)
155 				|| msg.equalsIgnoreCase(RecordConditionConstants
156 						.SUSP_REV_EFFECTIVE_MSG)) {
157 			requiredMsg = getSuspEffMessage(msg, dlmRecord);
158 		}
159 		else if (msg.equalsIgnoreCase(RecordConditionConstants
160 				.DL_PENDING_APP_MSG)) {
161 			requiredMsg = getDlPendingAppMessage(msg, dlmRecord);
162 		}
163 		else if (msg.equalsIgnoreCase(RecordConditionConstants
164 				.ID_PENDING_APP_MSG)) {
165 			requiredMsg = getIdPendingAppMessage(msg, dlmRecord);
166 		}
167 		else if (msg.equalsIgnoreCase(RecordConditionConstants
168 				.DOUBLE_RECORD_MSG)) {
169 			requiredMsg = getDoubleRecordMessage(msg, dlmRecord);
170 		}
171 		else if (msg.equalsIgnoreCase(RecordConditionConstants
172 				.LIC_LOC_DSR_MSG)) {
173 			requiredMsg = getLicLocDsrMessage(msg, dlmRecord);		
174 		}
175 		else if (msg.equalsIgnoreCase(RecordConditionConstants
176 				.LATEST_DL_ISS)) {
177 			requiredMsg = getLatestDlIssuedMessage(msg, dlmRecord);		
178 		} 
179 		else if (msg.equalsIgnoreCase(RecordConditionConstants
180 				.MG_NUMBER_ON_RECORD)
181 				|| msg.equalsIgnoreCase(RecordConditionConstants
182 						.MG_NUMBERS_EQUAL)) {
183 			requiredMsg = getMgRecordMessage(msg, dlmRecord);		
184 		}
185 		return requiredMsg;
186 	}

238 				//If DL incomplete application reason code or application date

240 				requiredMsg = msg + getIncompleteAppReasonCode(

243 				+ RecordConditionRulesUtil.getDlFormattedAppDate(
244 						reasonCode, appDate);
245 			}
246 		}
247 		return requiredMsg.trim();

422 	 * Sets the record condition indicators.
423 	 * 
424 	 * @param miscDlProcessContext the misc dl process context
425 	 * @param recordConditions the record conditions
426 	 */
427 	public static void setRecordConditionIndicators(
428 			MiscDlProcessContext miscDlProcessContext, 
429 			List <RecordConditionRule> recordConditions) {
430 	 	 isPasswordRequired = false;
431 	     isCannotContinue = false;
432 		if (recordConditions != null) {
433 			for (RecordConditionRule recordCondition 
434 					: recordConditions) {
435 				//Set Non Record Match indicators: password, cannot continue.
436 				setNonRecordMatchIndicators(recordCondition);
437 				//Add non duplicate Action Message to the list.
438 				addActionMessageToList(miscDlProcessContext, 
439 						recordCondition);
440 			}
441 			if (isCannotContinue) {
442 				//If CAN NOT CONTINUE condition is present password is not required.
443 				isPasswordRequired = false;				
444 			}
445 		}
446 		//Set all indicators to HqInquiryProcess Context to use later in the validations.
447 		miscDlProcessContext.getRecordConditionIndicators()
448 				.setCannotContinue(isCannotContinue);
449 		miscDlProcessContext.getRecordConditionIndicators()
450 				.setPasswordRequired(isPasswordRequired);
451 	}

477 	 * Adds the action message to list.
478 	 * 
479 	 * @param miscDlProcessContext the misc dl process context
480 	 * @param recordCondition the record condition
481 	 */
482 	public static void addActionMessageToList(
483 			MiscDlProcessContext miscDlProcessContext, 
484 			RecordConditionRule recordCondition) {
485 		//Add non duplicate messages.
486 		boolean duplicateMessage = false;
487 		Iterator <String> iterator = miscDlProcessContext
488 				.getActionMessages().iterator();
489 		while (iterator.hasNext()) {
490 			String message = iterator.next();
491 			if (message.equals(recordCondition.getActionMessage())) {
492 				duplicateMessage = true;
493 				break;
494 			}
495 		}
496 		if (!duplicateMessage) {
497 			miscDlProcessContext.addActionMessage(
498 					recordCondition.getActionMessage());
499 		}
500 	}

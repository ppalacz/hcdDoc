36  		IDlaDecisionActivityConstantsForBpDefinition {
37  	/*** The Constant serialVersionUID. */
38  	private static final long serialVersionUID = 1938863433476680377L;

40  	/***
41  	 * This method evaluates if there is match for pdps record.
42  	 * If match found it returns true, if not returns false.
43  	 * 
44  	 * @param context the context
45  	 * 
46  	 * @return true, if evaluate
47  	 */
48  	@Override
49  	protected String evaluate(ProcessContext context) {
50  		DlaProcessContext dlaProcessContext = ((DlaProcessContext) context);
51  		List <PdpsRecord> pdpsRecords = dlaProcessContext.getPdpsRecords();
52  		for (PdpsRecord pdpsRecord : pdpsRecords) {
53  			if (SAME.equals(pdpsRecord.getDesignateAction().getCode())) {
54  				return PRIMARY_PDPS;
55  			}
56  		}
57  		return NOT_PRIMARY_PDPS;
58  	}
59  }

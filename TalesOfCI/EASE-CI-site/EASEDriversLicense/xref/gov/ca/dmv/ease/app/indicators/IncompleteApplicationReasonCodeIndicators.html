83  		boolean isAreaDrivingTest = isKeyedInAppReason(keyedInAppReasons,
84  				AREA_DRIVER_TEST);
85  		boolean isApplicationVoid = isKeyedInAppReason(keyedInAppReasons,
86  				APPLICATION_VOIDED);
87  		boolean isCashMailIn = isKeyedInAppReason(keyedInAppReasons,
88  				CASH_MAIL_IN);
89  		boolean isDiscretionaryDriveTest = isKeyedInAppReason(
90  				keyedInAppReasons, DRIVER_DESCRITIONARY);
91  		boolean isDriveTestMandatory = isKeyedInAppReason(keyedInAppReasons,
92  				DRIVER_MANDATORY);
93  		boolean isDriverSafetyRefereeReferral = isKeyedInAppReason(
94  				keyedInAppReasons, DSR_REFERRAL);
95  		boolean isFraudulentApplication = isKeyedInAppReason(keyedInAppReasons,

97  		boolean isTempDrivingTestWaiver = isKeyedInAppReason(keyedInAppReasons,
98  				TEMP_DRIVER_TEST_FEE_WAIVER);
99  		boolean isFingerOrPhotoMismatch = isKeyedInAppReason(keyedInAppReasons,
100 				PHOTO_UNMATCHED);
101 		boolean isHQReferral = isKeyedInAppReason(keyedInAppReasons,
102 				HQ_REFERRAL);
103 		boolean isLegalPresenceExpired = isKeyedInAppReason(keyedInAppReasons,
104 				INSURANCE_DOCUMENTS_EXPIRES_IN_LESSTHAN_60_DAYS);
105 		boolean isMinorCode = isKeyedInAppReason(keyedInAppReasons, MINOR_CODE);
106 		boolean isOutOfStateMailIn = isKeyedInAppReason(keyedInAppReasons,
107 				OUT_OF_STATE_MAIN_IN);
108 		boolean isProcessIncomplete = isKeyedInAppReason(keyedInAppReasons,
109 				PROCESSING_INCOMPLETE);
110 		boolean isRefund = isKeyedInAppReason(keyedInAppReasons, REFUND);
111 		boolean isRejected = isKeyedInAppReason(keyedInAppReasons, REJECTED);
112 		boolean isThumbPrintRefused = isKeyedInAppReason(keyedInAppReasons,
113 				THUMB_PRINT_REFUSED);
114 		boolean isVisionTestReferral = isKeyedInAppReason(keyedInAppReasons,
115 				VISION_REFERRAL);
116 		boolean isTestResult = isKeyedInAppReason(keyedInAppReasons, TEST_RESULTS);

118 		boolean isCNACondition = dlApplication.isCNACondition();
119 		boolean isCDLISInquiryEmpty = EaseUtil.isNullOrBlank(dlApplication

123 		boolean isCDLISResponseEmpty = EaseUtil.isNullOrBlank(dlApplication
124 				.getCdlisResponseCode());
125 		CodeSetElement birthDateVerificationCode = dlApplication
126 				.getNewUserVerifiedBdlpCode();
127 		List <CodeSetElement> attachments = driverLicense.getNewAttachments();
128 		CodeSetElement appPendingReasonCode = dlApplication
129 				.getApplicationPendingIndicatorCode();
130 		Date insExpirationDate = dlApplication.getLegalPresenceDocument()
131 				.getExpirationDate();
132 		CodeSetElement feeReasonCode = dlApplication.getFeeRequiredReasonCode();
133 		boolean isDriveTestFailure = isDriveTestFailure(driverLicense
134 				.getDriveAndLawTestResultGroup())
135 				&& isCommercialDriveTestFailure(driverLicense
136 						.getCommercialTestResultGroup());
137 		CodeSetElement operationalMode = userContext.getOperationalMode();
138 		boolean isManualOrTravelMode = (!EaseUtil
139 				.isNullOrBlank(operationalMode) && (MANUAL_MODE
140 				.equalsIgnoreCase(operationalMode.getCode()) || TRAVEL_MODE
141 				.equalsIgnoreCase(operationalMode.getCode())));
142 		boolean isTravelMode = (!EaseUtil.isNullOrBlank(operationalMode) && TRAVEL_MODE
143 				.equalsIgnoreCase(operationalMode.getCode()));
144 		boolean isLicenseClassABOrC = isLicenseClassPresent(driverLicense
145 				.getAppliedLicenseClasses(), LICENSE_CLASS_A)
146 				|| isLicenseClassPresent(driverLicense
147 						.getAppliedLicenseClasses(), LICENSE_CLASS_B)
148 				|| isLicenseClassPresent(driverLicense
149 						.getAppliedLicenseClasses(), LICENSE_CLASS_C);
150 		boolean isFeeReasonCodeT = (!EaseUtil.isNullOrBlank(feeReasonCode) && IFeeReasonCodeConstants.T_TEMPORARY_WAIVER_FEES_DUE
151 				.equalsIgnoreCase(feeReasonCode.getCode()));
152 		boolean isLicenseClassDOrE = isLicenseClassPresent(driverLicense
153 				.getAppliedLicenseClasses(), LICENSE_CLASS_D)
154 				|| isLicenseClassPresent(driverLicense
155 						.getAppliedLicenseClasses(), LICENSE_CLASS_E);
156 		boolean isNonCommercialLicenseClass = isLicenseClassPresent(
157 				driverLicense.getAppliedLicenseClasses(), LICENSE_CLASS_F)
158 				|| isLicenseClassPresent(driverLicense
159 						.getAppliedLicenseClasses(), LICENSE_CLASS_M);
160 		CodeSetElement existingUserVerifiedSsnCode = dlApplication
161 				.getExistingUserVerifiedSsnCode();
162 		CodeSetElement legalPresenceInquiryIndicator = dlApplication
163 				.getLegalPresenceDocument().getLegalPresenceInquiryIndicator();
164 		boolean isLegalPresenceIndicatorE = (!EaseUtil
165 				.isNullOrBlank(legalPresenceInquiryIndicator) && "E"
166 				.equalsIgnoreCase(legalPresenceInquiryIndicator.getCode()));
167 		boolean isPreviousVerifiedSsnE = !EaseUtil.isNullOrBlank(dlApplication
168 				.getExistingUserVerifiedSsnCode())
169 				&& "R".equalsIgnoreCase(dlApplication
170 						.getExistingUserVerifiedSsnCode().getCode());
171 		boolean isPDPSResponseCodeW = !EaseUtil.isNullOrBlank(dlApplication
172 				.getPdpsResponseCode())
173 				&& "W".equalsIgnoreCase(dlApplication.getPdpsResponseCode()
174 						.getCode());
175 		if (isDlApplicationCancelled) {
176 			//If Application Cancel - AC , 4
177 			incompletedAppReasons.add(new IncompleteApplicationReason(
178 					APPLICATION_CANCELLED, 4));
179 		}
180 		//Check for Class F or M
181 		if (isAreaDrivingTest
182 				&& (isLicenseClassPresent(driverLicense
183 						.getAppliedLicenseClasses(), LICENSE_CLASS_F) || isLicenseClassPresent(
184 						driverLicense.getAppliedLicenseClasses(),
185 						LICENSE_CLASS_M))) {
186 			//AD, 13
187 			incompletedAppReasons.add(new IncompleteApplicationReason(
188 					AREA_DRIVER_TEST, 13));
189 		}
190 		//Check for Application void
191 		if (isApplicationVoid) {
192 			//AV, 5
193 			incompletedAppReasons.add(new IncompleteApplicationReason(
194 					APPLICATION_VOIDED, 5));
195 		}
196 		//Check if Birth Date Verified Indicator = 'N'
197 		if (EaseUtil.isNullOrBlank(birthDateVerificationCode)
198 				&& BIRTH_DATE_NOT_VERIFIED
199 						.equalsIgnoreCase(birthDateVerificationCode.getCode())) {
200 			//BD, 18
201 			incompletedAppReasons.add(new IncompleteApplicationReason(
202 					BIRTH_DATE_VERIFICATION_REQUIRED, 18));
203 		}
204 		//Check for Cash Mail In
205 		if (isCashMailIn) {
206 			//CA, 29
207 			incompletedAppReasons.add(new IncompleteApplicationReason(
208 					CASH_MAIL_IN, 29));
209 		}
210 		//Check if CDLIS Out of State is invalid
211 		if (isCdlisOutofStateInvalid) {
212 			//CC, 33
213 			incompletedAppReasons.add(new IncompleteApplicationReason(
214 					COMMERCIAL_COMPARE, 33));
215 		}
216 		//Check If not an issuance office and (class = A/B or category present) and attachment 17 is present
217 		if (!userContext.isIssuanceOffice()
218 				&& (isLicenseClassPresent(driverLicense
219 						.getAppliedLicenseClasses(), LICENSE_CLASS_A) || isLicenseClassPresent(
220 						driverLicense.getAppliedLicenseClasses(),
221 						LICENSE_CLASS_B)
222 						&& isAttachmentPresent(attachments, CERT_OF_DRIVING_EXP_TRAINING))) {

224 			incompletedAppReasons.add(new IncompleteApplicationReason(
225 					DL_170_VERIFICATION, 37));
226 		}
227 		//Check if In Manual or Travel Mode, Fee Reason = T, Class = A or B or C
228 		if (isManualOrTravelMode && isFeeReasonCodeT && isLicenseClassABOrC) {
229 			//CF,10
230 			incompletedAppReasons.add(new IncompleteApplicationReason(
231 					COMMERCIAL_FEE_DUE, 10));
232 		}
233 		//CDLIS RECORDS ARE EMPTY & cdlisResponseCode iS EMPTY
234 		if (isCNACondition && isCDLISInquiryEmpty && isCDLISResponseEmpty) {
235 			isCommercialInquiryIncomplete = true;
236 		}
237 		//Check for PDPS Records, designateAction & cdlisResponseCode is QUESTIONABLE
238 		if (!isPDPSInquiryEmpty && dlApplication.isQuestionablePdpsIndicator()) {
239 			isCommercialInquiryIncomplete = true;
240 		}
241 		if (isCommercialInquiryIncomplete) {
242 			//CI, 35
243 			incompletedAppReasons.add(new IncompleteApplicationReason(
244 					COMMERCIAL_INQUIRY, 35));
245 		}
246 		if (isDiscretionaryDriveTest) {
247 			//DE,14
248 			incompletedAppReasons.add(new IncompleteApplicationReason(
249 					DRIVER_DESCRITIONARY, 14));
250 		}
251 		if (isDriveTestFailure) {
252 			//DF,16
253 			incompletedAppReasons.add(new IncompleteApplicationReason(
254 					DRIVING_TEST_FAILURE, 16));
255 		}
256 		if (isDriveTestMandatory) {
257 			//DM,15
258 			incompletedAppReasons.add(new IncompleteApplicationReason(
259 					DRIVER_MANDATORY, 15));
260 		}
261 		if (isDriverSafetyRefereeReferral) {
262 			//DS,8
263 			incompletedAppReasons.add(new IncompleteApplicationReason(
264 					DSR_REFERRAL, 8));
265 		}
266 		if (isFraudulentApplication) {
267 			//FA, 2
268 			incompletedAppReasons.add(new IncompleteApplicationReason(
269 					FRAUD_APPLICATION, 2));
270 		}
271 		if ((TTC_DLP.equalsIgnoreCase(userContext.getTtc()) && isFeeReasonCodeT && isDriveTestFailure)

273 			//FD, 9
274 			incompletedAppReasons.add(new IncompleteApplicationReason(
275 					TEMP_DRIVER_TEST_FEE_WAIVER, 2));

289 		}

291 			//HQ, 26
292 			incompletedAppReasons.add(new IncompleteApplicationReason(
293 					HQ_REFERRAL, 26));

303 			incompletedAppReasons.add(new IncompleteApplicationReason(
304 					MINOR_CODE, 12));

306 		if (isMedicalExamRequired(driverLicense.getAppliedLicenseClasses(),
307 				driverLicense.getNewCategories(), driverLicense
308 						.getNewAttachments(), driverLicense

311 			//ME, 24
312 			incompletedAppReasons.add(new IncompleteApplicationReason(

321 		}

328 		if (isOutOfStateMailIn) {

330 			incompletedAppReasons.add(new IncompleteApplicationReason(
331 					OUT_OF_STATE_MAIN_IN, 28));
332 		}
333 		if (isNonCommercialLicenseClass
334 				&& (isPDPSInquiryEmpty

336 						.isNullOrBlank(dlApplication.getPdpsResponseCode()) && "B"

338 								.getCode())))) {
339 			//PD, 36
340 			incompletedAppReasons.add(new IncompleteApplicationReason(

342 		}
343 		if (!EaseUtil.isNotNull(driverLicense.getPhotoInformation()
344 				.getPhotoTakenDate())) {
345 			//PH, 30
346 			incompletedAppReasons.add(new IncompleteApplicationReason(
347 					PHOTO_NEEDED, 30));
348 		}
349 		if (isProcessIncomplete) {
350 			//PI, 34

362 		if (isRejected) {

370 				|| (EaseUtil.isNullOrBlank(existingUserVerifiedSsnCode) && isLegalPresenceIndicatorE)) {
371 			//SN, 22

373 					SOCIAL_SECURITY_NUMBER, 22));
374 		}
375 		if (isPreviousVerifiedSsnE) {
376 			//SR, 23
377 			incompletedAppReasons.add(new IncompleteApplicationReason(
378 					SSN_VERIF_REQUIRED, 23));
379 		}
380 		if (isThumbPrintRefused) {
381 			//TP, 27
382 			incompletedAppReasons.add(new IncompleteApplicationReason(
383 					THUMB_PRINT_REFUSED, 27));
384 		}
385 		if(isTestResult)
386 		{
387 			//TR, 31
388 			incompletedAppReasons.add(new IncompleteApplicationReason(
389 					TEST_RESULTS, 31));
390 		}
391 		if (isVisionTestReferral) {
392 			//VR, 17
393 			incompletedAppReasons.add(new IncompleteApplicationReason(
394 					VISION_REFERRAL, 17));
395 		}
396 		if (isWrittenTestFailed(driverLicense.getDriveAndLawTestResultGroup())) {
397 			//WF, 25
398 			incompletedAppReasons.add(new IncompleteApplicationReason(
399 					WRITTEN_FAILED, 25));
400 		}
401 		if (isNonCommercialLicenseClass && isPDPSResponseCodeW) {
402 			//WP, 38
403 			incompletedAppReasons.add(new IncompleteApplicationReason(
404 					WITHHELD_PDPS_CLEARANCE, 38));
405 		}
406 		if (isThreeTestFailure(driverLicense.getDriveAndLawTestResultGroup(),
407 				driverLicense.getCommercialTestResultGroup())) {
408 			//3F, 7
409 			incompletedAppReasons.add(new IncompleteApplicationReason(
410 					THREE_TIMES_FALIURE, 7));
411 		}
412 		return incompletedAppReasons;
413 	}

415 	/***
416 	 * This method checks if Medical Exam is Required.
417 	 * 
418 	 * @param appliedLicenseClasses the Applied License Classes
419 	 * @param newCategories the New Categories
420 	 * @param newAttachments the New Attachments
421 	 * @param newRestrictions the New Restrictions
422 	 * @param medicalExpirationDate the Medical Expiration Date
423 	 * 
424 	 * @return true if Medical Exam is Required
425 	 */
426 	private boolean isMedicalExamRequired(
427 			List <CodeSetElement> appliedLicenseClasses,
428 			List <CodeSetElement> newCategories,
429 			List <CodeSetElement> newAttachments,
430 			List <DrivingRestriction> newRestrictions,
431 			Date medicalExpirationDate) {
432 		boolean isMedicalExamRequired = false;
433 		boolean isCommercialLicense = false;
434 		boolean isNonCommercialCategory = false;
435 		boolean isAttachment51Or54 = false;
436 		boolean isAttachment51 = false;
437 		boolean isClassAOrB = false;
438 		boolean isCategory1Or2 = false;
439 		boolean isRestriction80Or81 = false;
440 		if (!EaseUtil.isNullOrBlank(appliedLicenseClasses)) {
441 			for (CodeSetElement licenseClass : appliedLicenseClasses) {
442 				if (!(LICENSE_CLASS_F.equalsIgnoreCase(licenseClass.getCode()) || LICENSE_CLASS_M
443 						.equalsIgnoreCase(licenseClass.getCode()))) {
444 					isCommercialLicense = true;
445 					break;
446 				}
447 			}
448 		}
449 		if (!EaseUtil.isNullOrBlank(appliedLicenseClasses)) {
450 			for (CodeSetElement licenseClass : appliedLicenseClasses) {
451 				if (LICENSE_CLASS_A.equalsIgnoreCase(licenseClass.getCode())
452 						|| LICENSE_CLASS_B.equalsIgnoreCase(licenseClass
453 								.getCode())) {
454 					isClassAOrB = true;
455 					break;
456 				}
457 			}
458 		}
459 		if (!EaseUtil.isNullOrBlank(newCategories)) {
460 			for (CodeSetElement category : newCategories) {
461 				if (CATEGORY_1.equalsIgnoreCase(category.getCode())
462 						|| CATEGORY_2.equalsIgnoreCase(category.getCode())
463 						|| CATEGORY_3.equalsIgnoreCase(category.getCode())) {
464 					isNonCommercialCategory = true;
465 					break;
466 				}
467 			}
468 		}
469 		if (!EaseUtil.isNullOrBlank(newCategories)) {
470 			for (CodeSetElement category : newCategories) {
471 				if (CATEGORY_1.equalsIgnoreCase(category.getCode())

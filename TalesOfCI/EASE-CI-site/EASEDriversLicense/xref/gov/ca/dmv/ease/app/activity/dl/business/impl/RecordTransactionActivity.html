51  	/* (non-Javadoc)
52  	 * @see gov.ca.dmv.ease.app.activity.BusinessActivity#invokeService(gov.ca.dmv.ease.app.context.ProcessContext)
53  	 */
54  	@Override
55  	protected void invokeService(ProcessContext context) {
56  		DlProcessTransactionRequest dlaProcessTransactionRequest = new DlProcessTransactionRequest(
57  				context, transactionService);
58  		TransactionResponse transactionResponse = dlaProcessTransactionRequest
59  				.execute();
60  		IErrorCollector errorCollector = new ErrorCollector();
61  		if (!transactionResponse.hasErrors()) {
62  			Transaction transaction = transactionResponse.getTransaction();
63  			//Sending Record DL Review Request
64  			RecordDlReviewRequestTs recordDlReviewRequestTs = new RecordDlReviewRequestTs(
65  					transaction, context.getUserContext());
66  			RecordDlReviewResponseTs recordDlReviewResponseTs = issueDlTaskService
67  					.execute(recordDlReviewRequestTs);
68  			if (recordDlReviewResponseTs.hasErrors()) {
69  				//FIXME Populate error messages
70  				//				errorCollector.register(new EaseException(
71  				//						"Error while sending Record DL Review message"));
72  				//				this.populateErrorMessages(context, errorCollector);
73  			}
74  			else {
75  				//Sending Record Dl Review Request
76  				RecordCashierTransactionRequestTs recordCashierTransactionRequestTs = new RecordCashierTransactionRequestTs(
77  						transaction, context.getUserContext());
78  				RecordCashierTransactionResponseTs recordCashierTransactionResponseTs = issueDlTaskService
79  						.execute(recordCashierTransactionRequestTs);
80  				if (recordCashierTransactionResponseTs.hasErrors()) {
81  					//FIXME Populate error messages
82  					//					errorCollector
83  					//							.register(new EaseException(
84  					//									"Error while sending Record Cashier Transaction message"));
85  					//					this.populateErrorMessages(context, errorCollector);
86  				}
87  			}
88  		}
89  		else {
90  			this.populateErrorMessages(context, transactionResponse
91  					.getErrorCollector());
92  		}
93  	}

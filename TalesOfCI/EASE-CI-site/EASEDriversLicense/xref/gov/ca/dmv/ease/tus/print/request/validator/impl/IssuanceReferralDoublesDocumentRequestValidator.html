37  public class IssuanceReferralDoublesDocumentRequestValidator extends
38  		TusRequestValidator implements IRequestValidator {
39  	/*** The Constant serialVersionUID. */
40  	private static final long serialVersionUID = -6129258770750333848L;
41  	/*** The Request. */
42  	private static final String REQUEST = "IssuanceReferralDocumentRequestTus";
43  	
44  	/***
45  	 * Constructor.
46  	 */
47  	public IssuanceReferralDoublesDocumentRequestValidator() {
48  		super();
49  	}
50  	
51  	/* (non-Javadoc)
52  	 * @see gov.ca.dmv.ease.ecs.request.validator.impl.EcsRequestValidator#validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
53  	 */
54  	@Override
55  	protected void validateRequest(IPrintServiceRequest aRequest,
56  			IErrorCollector aCollector) {
57  		if (!EaseUtil.isNotNull(aRequest)) {
58  			addValidationExceptionInErrorCollector(
59  					IPrintServiceErrorMessageConstants
60  						.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_REQUEST_IS_NULL_ERROR_MESSAGE, 
61  						null,aCollector);
62  			return;
63  		} 
64  		else if (!(aRequest instanceof IssuanceReferralDoublesDocumentRequestTus)) {
65  			addValidationExceptionInErrorCollector(
66  					IPrintServiceErrorMessageConstants
67  						.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_REQUEST_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
68  						new String [] { REQUEST }, aCollector);
69  			return;
70  		}
71  		else {
72  			IssuanceReferralDoublesDocumentRequestTus request
73  				= (IssuanceReferralDoublesDocumentRequestTus)aRequest;
74  			IUserContext userContext = request.getUserContext();
75  			Application application = request.getApplication();
76  			Person applicant = null;
77  			if (!EaseUtil.isNotNull(userContext)) {
78  				addValidationExceptionInErrorCollector(
79  						IPrintServiceErrorMessageConstants
80  							.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_USER_CONTEXT_IS_NULL_ERROR_MESSAGE, null,
81  						aCollector);
82  				return;
83  			}
84  			if (!EaseUtil.isNotNull(application)) {
85  				aCollector
86  				.register(new EcsValidationException(
87  						IPrintServiceErrorMessageConstants
88  							.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_APPLICATION_IS_NULL_ERROR_MESSAGE));
89  				return;
90  			}
91  			if(!(application instanceof DlApplication || application instanceof SalesPersonApplication)) {
92  				aCollector
93  				.register(new EcsValidationException(
94  						IPrintServiceErrorMessageConstants
95  							.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_INVALID_APPLICATION_TYPE_ERROR_MESSAGE));
96  				return;
97  			}
98  			if(!EaseUtil.isNotNull(application.getIssuanceReferralMessages())) {
99  				aCollector
100 				.register(new EcsValidationException(
101 						 IPrintServiceErrorMessageConstants
102 						 	.ISSUANCE_REFERRAL_MESSAGES_ARE_NULL_ERROR_MESSAGE));
103 				return;
104 			}
105 			if(!application.getIssuanceReferralMessages().contains(
106 					IssuanceReferralMessagesType.RECORDS_TO_BE_COMBINED)) {
107 				aCollector
108 				.register(new EcsValidationException(
109 							IPrintServiceErrorMessageConstants
110 								.RECORDS_TO_BE_COMBINED_ISSUANCE_MESSAGE_IS_NOT_PRESENT_ERROR_MESSAGE));
111 				return;
112 			}
113 			if (application instanceof DlApplication) {
114 				applicant = ((DlApplication)application).getApplicant();
115 			}
116 			else if (application instanceof SalesPersonApplication) {
117 				applicant = ((SalesPersonApplication)application).getApplicant();
118 			}
119 			if (!EaseUtil.isNotNull(applicant)) {
120 				aCollector
121 				.register(new EcsValidationException(
122 							 IPrintServiceErrorMessageConstants
123 									.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_PERSON_IS_NULL_ERROR_MESSAGE));
124 				return;
125 			}
126 			if(application instanceof DlApplication) {
127 				if(!
128 					(!EaseUtil.isNullOrBlank(applicant.getLicenseNumbersOfDuplicateAniRecords())
129 						|| (EaseUtil.isNotNull(((DlApplication)application))
130 							&& EaseUtil.isNotNull(((DlApplication)application).isNewLicenseNumberRequired())
131 							&& ((DlApplication)application).isNewLicenseNumberRequired()
132 							&& !EaseUtil.isNullOrBlank(applicant.getPreviousLicenseNumber()))
133 						|| !EaseUtil.isNullOrBlank(applicant.getDuplicateLicenseNumberOfAniChangeInquiry())
134 						|| (!EaseUtil.isNullOrBlank(applicant.getPreviousLicenseNumber())
135 								&& DocumentConverterUtil.isXTypeLicense(applicant.getPreviousLicenseNumber()))
136 					)
137 				  ) {
138 					aCollector
139 					.register(new EcsValidationException(
140 							 	IPrintServiceErrorMessageConstants
141 									.ISSUANCE_REFERRAL_DOCUMENT_DOUBLE_LICENSE_NUMBER_IS_NULL_ERROR_MESSAGE));
142 					return;
143 				}
144 			}
145 			else if(application instanceof SalesPersonApplication) {
146 				if(!(!EaseUtil.isNullOrBlank(applicant
147 						.getLicenseNumbersOfDuplicateAniRecords())
148 						|| !EaseUtil.isNullOrBlank(applicant.getDuplicateLicenseNumberOfAniChangeInquiry())
149 							|| (!EaseUtil.isNullOrBlank(applicant.getPreviousLicenseNumber())
150 									&& DocumentConverterUtil.isXTypeLicense(applicant.getPreviousLicenseNumber())))) {
151 						aCollector
152 						.register(new EcsValidationException(
153 									IPrintServiceErrorMessageConstants
154 										.ISSUANCE_REFERRAL_DOCUMENT_DOUBLE_LICENSE_NUMBER_IS_NULL_ERROR_MESSAGE));
155 						return;
156 					}
157 			}
158 			if (!EaseUtil.isNotNull(applicant.getAddress(AddressType.MAILING))) {
159 				aCollector
160 				.register(new EcsValidationException(
161 							IPrintServiceErrorMessageConstants
162 								.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_MAILING_ADDRESS_IS_NULL_ERROR_MESSAGE));
163 				return;
164 			}
165 			if (!EaseUtil.isNotNull(applicant.getCurrentName())) {
166 				aCollector
167 				.register(new EcsValidationException(
168 							IPrintServiceErrorMessageConstants
169 								.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_PERSON_NAME_IS_NULL_ERROR_MESSAGE));
170 				return;
171 			}
172 			if (application instanceof DlApplication && 
173 					(EaseUtil.isNullOrBlank(DocumentConverterUtil.getDlorIdNumber(applicant)))) {
174 				aCollector
175 				.register(new EcsValidationException(
176 							IPrintServiceErrorMessageConstants
177 								.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_DRIVER_LICENSE_OR_ID_CARD_NUMBER_IS_EMPTY_ERROR_MESSAGE));
178 				return;
179 			}
180 			if (application instanceof SalesPersonApplication) {
181 				VehicleSalesPerson vehicleSalesPerson = (VehicleSalesPerson)applicant;
182 				SalesPersonLicense salesPersonLicense = vehicleSalesPerson.getSalesPersonLicense();
183 				if (!EaseUtil.isNotNull(salesPersonLicense)) {
184 					aCollector
185 					.register(new EcsValidationException(
186 							IPrintServiceErrorMessageConstants
187 								.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_SALESPERSON_LICENSE_IS_NULL_ERROR_MESSAGE));
188 					return;
189 				}
190 				if (EaseUtil.isNullOrBlank(salesPersonLicense.getLicenseNumber())) {
191 					aCollector
192 					.register(new EcsValidationException(
193 							IPrintServiceErrorMessageConstants
194 								.ISSUANCE_REFERRAL_DOUBLES_DOCUMENT_SALESPERSON_LICENSE_NUMBER_IS_EMPTY_ERROR_MESSAGE));
195 					return;
196 				}
197 			}
198 		}
199 	}

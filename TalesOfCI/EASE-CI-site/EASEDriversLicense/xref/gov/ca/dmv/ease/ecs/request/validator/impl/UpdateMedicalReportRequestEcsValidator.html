26  public class UpdateMedicalReportRequestEcsValidator extends EcsRequestValidator implements
27  		IRequestValidator {
28  	/*** The Constant serialVersionUID. */
29  	private static final long serialVersionUID = 3247885913552525887L;
30  	/*** The Request. */
31  	private static final String CONVERTER = "UpdateMedicalReportConverter";
32  	/*** The Request. */
33  	private static final String REQUEST = "UpdateMedicalReportRequestEcs";
34  
35  	/* (non-Javadoc)
36  	 * @see gov.ca.dmv.ease.ecs.request.impl.EcsRequestValidator#
37  	 * validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
38  	 */
39  	@Override
40  	protected void validateRequest(IEcsRequest aRequest,
41  			IErrorCollector aCollector) {
42  		if (!(aRequest instanceof UpdateMedicalReportRequestEcs)) {
43  			addValidationExceptionInErrorCollector(
44  					REQUEST_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
45  					new String[] { REQUEST }, aCollector);
46  		}
47  		else {
48  			if (!(aRequest.getMessageConverter() instanceof UpdateMedicalReportConverter)) {
49  				addValidationExceptionInErrorCollector(
50  						CONVERTER_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
51  						new String[] { CONVERTER }, aCollector);
52  				return;
53  			}
54  			UpdateMedicalReportRequestEcs request = (UpdateMedicalReportRequestEcs) aRequest;
55  			if (EaseUtil.isNullOrBlank(request.getLastName())) {
56  				addValidationExceptionInErrorCollector(
57  						PERSON_LAST_NAME_IS_EMPTY_ERROR_MESSAGE, null,
58  						aCollector);
59  			}
60  			if (EaseUtil.isNullOrBlank(request.getDlOrIdNumber())) {
61  				addValidationExceptionInErrorCollector(
62  						DRIVER_LICENSE_NUMBER_IS_EMPTY_ERROR_MESSAGE, null,
63  						aCollector);
64  			}
65  		}
66  	}
67  }

49  	/***
50  	 * Execute.
51  	 * 
52  	 * @param request the request
53  	 * 
54  	 * @return the generate driver license cdlis receipt response bus
55  	 */
56  	public DlCdlisReceiptResponseTus execute(DlCdlisReceiptRequestTus request) {
57  		DlCdlisReceiptResponseTus response = new DlCdlisReceiptResponseTus();
58  		boolean isValidRequest = validateDlCdlisReceiptRequest(request,
59  				response);
60  		if (!isValidRequest) {
61  			return response;
62  		}
63  		DlCdlisReceiptConverter converter = FACTORY
64  				.createDlCdlisReceiptConverter(request.getUserContext(),
65  						request.getDlAppication(), request.getCdlisRecord(), request
66  								.getPdpsRecord());
67  		converter.setTemplate(new DlCdlisReceiptTemplate());
68  		DocumentToken docToken = new DocumentToken();
69  		docToken.setDocType("txt");
70  		response.setToken(docToken);
71  		try {
72  			converter.generateFileFor(docToken);
73  		}
74  		catch (EaseValidationException e) {
75  			LOGGER.error(e);
76  			response.register(e);
77  		}
78  		return response;
79  	}

159 	/***
160 	 * This method validates the DlCdlisReceiptRequestTus request for its UserContext, DlApplication, CdlisHits, and PdpsHits objects
161 	 * 
162 	 * @param request - the DlCdlisReceipt request 
163 	 * @return true if DlCdlisReceipt request is valid, else returns false
164 	 */
165 	private boolean validateDlCdlisReceiptRequest(
166 			DlCdlisReceiptRequestTus request, DlCdlisReceiptResponseTus response) {
167 		if (request.getUserContext() == null) {
168 			response
169 					.register(new EaseValidationException(INVALID_USER_CONTEXT));
170 		}
171 		if (request.getDlAppication() == null) {
172 			response.register(new EaseValidationException(
173 					INVALID_DL_APPLICATION));
174 		}
175 		if (request.getCdlisRecord() == null && request.getPdpsRecord() == null) {
176 			response.register(new EaseValidationException(
177 					INVALID_CDLIS_PDPS_RECORDS));
178 		}
179 		boolean isValidRequest = !response.hasErrors();
180 		return isValidRequest;
181 	}

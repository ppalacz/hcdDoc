63  	/***
64  	 * Populate ProofOfInsuranceFiling From an Inbound Payload.
65  	 * 
66  	 * @param anInboundMessagePayload
67  	 * @return a ProofOfInsuranceFiling
68  	 */
69  	public List <InsuranceCompany> populateInsuranceCompaniesFromInboundPayload(
70  			String anInboundMessagePayload) {
71  		int noOfRecords = 0;
72  		List <InsuranceCompany> insuranceCompanies = new ArrayList <InsuranceCompany>();
73  		String errorCode = anInboundMessagePayload.substring(19, 21);
74  		// if error code = '00', Record found.
75  		if (IEcsConverterConstants.DHA_DHB_RETURN_CODE_00
76  				.equalsIgnoreCase(errorCode)) {
77  			noOfRecords = Integer.parseInt(anInboundMessagePayload.substring(
78  					21, 23));
79  		}
80  		// if there are hits ...
81  		if (noOfRecords > 0) {
82  			String segment3 = anInboundMessagePayload.substring(24);
83  			for (int i = 0; i < noOfRecords; i++) {
84  				InsuranceCompany insuranceCompany = new InsuranceCompany();
85  				String companyRecord = segment3.substring(0, 256);
86  				String insuranceCompanyNumber = companyRecord.substring(2, 6);
87  				insuranceCompany
88  						.setInsuranceCompanyNumber(insuranceCompanyNumber);
89  				String companyName = companyRecord.substring(6, 66);
90  				OrganizationName organizationName = new OrganizationName();
91  				organizationName.setName(companyName.trim());
92  				insuranceCompany.setCurrentName(organizationName);
93  				Address address = new Address(AddressType.MAILING);
94  				String line1 = companyRecord.substring(66, 138).trim();
95  				address.setLine1(line1);
96  				String city = companyRecord.substring(138, 161).trim();
97  				address.setCity(city);
98  				String codeSetElementCode = companyRecord.substring(161, 163)
99  						.trim();
100 				CodeSetElement stateCode = getCodeSetElement("codeSetname",
101 						codeSetElementCode);
102 				address.setStateCode(stateCode);
103 				String zipCode = companyRecord.substring(163, 172);
104 				address.setZip(zipCode);
105 				insuranceCompany.addAddress(address);
106 				String telNo = companyRecord.substring(179, 189).trim();
107 				NonAddressLocator nonAddresslocator = new NonAddressLocator(
108 						NonAddressType.HOME_PHONE);
109 				nonAddresslocator.setContact(telNo);
110 				insuranceCompany.addNonAddressLocator(nonAddresslocator);
111 				String alphaKey = companyRecord.substring(189, 195).trim();
112 				insuranceCompany.setInsuranceCompanyAbbreviation(alphaKey);
113 				String effectiveDate = companyRecord.substring(241, 247);
114 				if (!isNullOrBlank(effectiveDate)) {
115 					Date effDate = convertStringToDate(effectiveDate,
116 							EcsConverterHelper.FORMATTER_MM_DD_YY);
117 					insuranceCompany.setEffectiveStartDate(effDate);
118 				}
119 				if (segment3.length() > 256) {
120 					segment3 = segment3.substring(256);
121 				}
122 				insuranceCompanies.add(insuranceCompany);
123 			}
124 		}
125 		return insuranceCompanies;
126 	}

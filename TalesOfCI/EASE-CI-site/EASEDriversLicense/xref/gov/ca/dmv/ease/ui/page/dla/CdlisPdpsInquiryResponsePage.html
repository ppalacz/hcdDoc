102 	/***
103 	 * Convert get aka name1.
104 	 */
105 	private void convertGetAkaNames() {
106 		try {
107 			akaName1 = null;
108 			akaName2 = null;
109 			akaName3 = null;
110 			if (getCdlisPdpsProcessContext().getCurrentCDLISRecordHit() != null) {
111 				List <AkaInformation> existingAkaNames = getCdlisPdpsProcessContext()
112 						.getCurrentCDLISRecordHit().getAkaInformation();
113 				if (existingAkaNames != null && existingAkaNames.size() > 0) {
114 					akaName1 = existingAkaNames.get(0).getAkaName()
115 							.getFullName();
116 				}
117 				if (existingAkaNames != null && existingAkaNames.size() > 1) {
118 					akaName2 = existingAkaNames.get(1).getAkaName()
119 							.getFullName();
120 				}
121 				if (existingAkaNames != null && existingAkaNames.size() > 2) {
122 					akaName3 = existingAkaNames.get(2).getAkaName()
123 							.getFullName();
124 				}
125 			}
126 		}
127 		catch (Exception e) {
128 			errorMessage(e);
129 		}
130 	}

261 	 * Gets the driving records.
262 	 * 
263 	 * @return the driving records
264 	 */
265 	public DataModel getDrivingRecords() {
266 		cdlisPdpsInquiryResponseDataModel = new ListDataModel();
267 		allDrivingRecords = new ArrayList <DrivingRecord>();
268 		//cdlisResultHits.clear();
269 		//pdpsResultHits.clear();
270 		try {
271 			for (DrivingRecord p : getCdlisPdpsProcessContext()
272 					.getPdpsResultHitsCdu()) {
273 				p.setRecordTypeOrStateOfOrigin("PDPS");
274 				allDrivingRecords.add(p);
275 			}
276 			for (DrivingRecord c : getCdlisPdpsProcessContext()
277 					.getCdlisResultHitsCdu()) {
278 				c.setRecordTypeOrStateOfOrigin("CDLIS");
279 				allDrivingRecords.add(c);
280 			}
281 			cdlisPdpsInquiryResponseDataModel.setWrappedData(allDrivingRecords);
282 		}
283 		catch (Exception e) {
284 			errorMessage(e);
285 		}
286 		return cdlisPdpsInquiryResponseDataModel;
287 	}

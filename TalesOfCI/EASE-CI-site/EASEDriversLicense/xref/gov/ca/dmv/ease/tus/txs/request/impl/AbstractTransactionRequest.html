32  public abstract class AbstractTransactionRequest extends AbstractRequest {
33  	private static final long serialVersionUID = 795354669532366088L;
34  	private ProcessContext context;
35  	private TransactionService transactionService;
36  
37  	/***
38  	 * Instantiates a new transaction request.
39  	 * 
40  	 * @param userContext the user context
41  	 * @param businessProcess the business process
42  	 * @param complete the complete
43  	 */
44  	public AbstractTransactionRequest(ProcessContext context,
45  			TransactionService transactionService) {
46  		super(context.getUserContext());
47  		this.context = context;
48  		this.transactionService = transactionService;
49  	}
50  
51  	/***
52  	 * Returns true if there are any downstream functions available for the 
53  	 * current transaction
54  	 * @return True if any downstream functions exists for this business process
55  	 */
56  	public boolean areDownstreamFunctionsAvailable() {
57  		return getConverters() != null && getConverters().size() > 0;
58  	}
59  
60  	/***
61  	 * executes the transaction service's execute method, 
62  	 * passing itself as the parameter  
63  	 */
64  	public TransactionResponse execute() {
65  		return getService().execute(this);
66  	}
67  
68  	/***
69  	 * This method gets the list of converters for this current request
70  	 * @return List of converters
71  	 */
72  	public abstract List <IConverter> getConverters();
73  
74  	/***
75  	 * @return the transactionService
76  	 */
77  	public TransactionService getTransactionService() {
78  		return transactionService;
79  	}
80  
81  	/***
82  	 * @param transactionService the transactionService to set
83  	 */
84  	public void setTransactionService(TransactionService transactionService) {
85  		this.transactionService = transactionService;
86  	}
87  
88  	/***
89  	 * Gets the transaction service instance
90  	 * @return
91  	 */
92  	public ITransactionService getService() {
93  		return getTransactionService();
94  	}
95  
96  	public ProcessContext getContext() {
97  		return context;
98  	}
99  
100 	public void setContext(ProcessContext context) {
101 		this.context = context;
102 	}
103 }

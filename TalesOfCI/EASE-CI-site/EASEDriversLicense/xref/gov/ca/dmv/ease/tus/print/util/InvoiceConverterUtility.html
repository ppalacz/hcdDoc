72  	 * Populate invoice.
73  	 *
74  	 * @param elementsList the elements list
75  	 * @param cashElements the cash elements
76  	 * @param debitElements the debit elements
77  	 * @param creditElements the credit elements
78  	 * @param invoice the invoice
79  	 * @param formatter 
80  	 * @param workSetWorkDate the work set work date
81  	 * @param isLeadingZeroes 
82  	 */
83  	public static void populateInvoice(List <String> elementsList,
84  			String[] cashElements, String[] debitElements,
85  			String[] creditElements, Invoice invoice, DecimalFormat formatter,
86  			Date workSetWorkDate, boolean isLeadingZeroes) {
87  		if (cashElements != null) {
88  			addCashPaymentElements(elementsList, cashElements, invoice,
89  					formatter, isLeadingZeroes);
90  		}
91  		if (debitElements != null) {
92  			addDebitPaymentElements(elementsList, debitElements, invoice,
93  					formatter, isLeadingZeroes);
94  		}
95  		if (creditElements != null) {
96  			addCreditPaymentElements(elementsList, creditElements, invoice,
97  					formatter, workSetWorkDate, isLeadingZeroes);
98  		}
99  	}

101 	/***
102 	 * Adds the cash payment elements.
103 	 *
104 	 * @param elementsList the elements list
105 	 * @param cashElements the cash elements
106 	 * @param invoice the invoice
107 	 * @param formatter 
108 	 * @return the list
109 	 */
110 	private static void addCashPaymentElements(List <String> elementsList,
111 			String[] cashElements, Invoice invoice, DecimalFormat formatter,
112 			boolean isLeadingZeroes) {
113 		if (!isNullorEmpty(invoice)) {
114 			if (!isNullorEmpty(invoice.getInvoiceTotal())
115 					&& invoice.getInvoiceTotal().doubleValue() > 0) {
116 				addToElementsList(elementsList, formatter.format(invoice
117 						.getInvoiceTotal()), cashElements[CASH_TOTAL_INDEX]);
118 			}
119 			else {
120 				addToElementsList(elementsList,
121 						IPrintConverterConstants.DL_REC_ZBPTRNV_MSG,
122 						cashElements[CASH_TOTAL_INDEX]);
123 			}
124 			if (!isNullorEmpty(invoice.getCashPayment())) {
125 				addToElementsList(elementsList, formatter.format(invoice
126 						.getCashPayment().getAmount()),
127 						cashElements[CASH_AMOUNT_INDEX]);
128 			}
129 			if (!isNullorEmpty(invoice.getTotalChecksAmount())) {
130 				addToElementsList(elementsList, formatter.format(invoice
131 						.getTotalChecksAmount()),
132 						cashElements[CASH_CHECKS_AMOUNT_INDEX]);
133 			}
134 			if (!isNullorEmpty(invoice.getNetAmount())) {
135 				if (cashElements[CASH_NET_AMOUNT_INDEX] != null) {
136 					if (!isNullorEmpty(invoice.getFeeRefundActionCode())
137 							&& IPrintConverterConstants.FEE_RETURN_ACTION_CODE_C
138 									.equalsIgnoreCase(invoice
139 											.getFeeRefundActionCode().getCode())) {
140 						addToElementsList(
141 								elementsList,
142 								IPrintConverterConstants.FEE_RETURN_ACTION_CODE_C_MSG,
143 								IPrintConverterConstants.DL_REC_ZBPNETAC);
144 						if (!isNullorEmpty(invoice.getNetAmount())) {
145 							addToElementsList(elementsList, formatter
146 									.format(invoice.getNetAmount()),
147 									cashElements[CASH_NET_AMOUNT_INDEX]);
148 						}
149 					}
150 					else if (!isNullorEmpty(invoice.getFeeRefundActionCode())
151 							&& IPrintConverterConstants.FEE_RETURN_ACTION_CODE_H
152 									.equalsIgnoreCase(invoice
153 											.getFeeRefundActionCode().getCode())) {
154 						addToElementsList(
155 								elementsList,
156 								IPrintConverterConstants.FEE_RETURN_ACTION_CODE_H_MSG,
157 								IPrintConverterConstants.DL_REC_ZBPNETAC);
158 						if (!isNullorEmpty(invoice.getNetAmount())) {
159 							addToElementsList(elementsList, formatter
160 									.format(invoice.getNetAmount()),
161 									cashElements[CASH_NET_AMOUNT_INDEX]);
162 						}
163 					}
164 					else if (!isNullorEmpty(invoice.getFeeRefundActionCode())
165 							&& IPrintConverterConstants.FEE_RETURN_ACTION_CODE_R
166 									.equalsIgnoreCase(invoice
167 											.getFeeRefundActionCode().getCode())) {
168 						addToElementsList(
169 								elementsList,
170 								IPrintConverterConstants.FEE_RETURN_ACTION_CODE_R_MSG,
171 								IPrintConverterConstants.DL_REC_ZBPNETAC);
172 						if (!isNullorEmpty(invoice.getNetAmount())) {
173 							addToElementsList(elementsList, formatter
174 									.format(invoice.getNetAmount()),
175 									cashElements[CASH_NET_AMOUNT_INDEX]);
176 						}
177 					}
178 				}
179 			}
180 		}

202 	/***
203 	 * Adds the credit payment elements.
204 	 *
205 	 * @param elementsList the elements list
206 	 * @param creditElements the credit elements
207 	 * @param invoice the invoice
208 	 * @param formatter 
209 	 * @param workSetWorkDate the work set work date
210 	 */
211 	private static void addCreditPaymentElements(List <String> elementsList,
212 			String[] creditElements, Invoice invoice, DecimalFormat formatter,
213 			Date workSetWorkDate, boolean isLeadingZeroes) {
214 		if (!isNullorEmpty(invoice)
215 				&& !isNullorEmpty(invoice.getCreditPayment())) {
216 			addToElementsList(elementsList, formatter.format(invoice
217 					.getCreditPayment().getAmount()),
218 					creditElements[CREDIT_AMOUNT_INDEX]);
219 			if (invoice.getCreditPayment().getAmount().doubleValue() > 0) {
220 				if (creditElements[CREDIT_SALE_TEXT_INDEX] != null) {
221 					addToElementsList(elementsList,
222 							IPrintConverterConstants.DL_REC_ZBPCCSAL_MSG,
223 							creditElements[CREDIT_SALE_TEXT_INDEX]);
224 				}
225 				if (creditElements[CREDIT_NUMBER4_INDEX] != null) {
226 					addToElementsList(
227 							elementsList,
228 							IPrintConverterConstants.DL_REC_ZBPCC_MSG
229 									+ invoice
230 											.getCreditPayment()
231 											.getCreditCardNumber()
232 											.substring(
233 													12,
234 													invoice
235 															.getCreditPayment()
236 															.getCreditCardNumber()
237 															.length()),
238 							creditElements[CREDIT_NUMBER4_INDEX]);
239 				}
240 				if (creditElements[CREDIT_AMOUNT_TEXT_INDEX] != null) {
241 					addToElementsList(elementsList,
242 							IPrintConverterConstants.DL_REC_ZBPTEXT4_MSG,
243 							creditElements[CREDIT_AMOUNT_TEXT_INDEX]);
244 				}
245 				Date targetWorkDate = convertStringToDate(
246 						IPrintConverterConstants.WORKSET_WORKDATE_DEADLINE_DATE,
247 						IPrintConverterConstants.WORKSET_WORKDATE_DEADLINE_DATE_FORMAT);
248 				Calendar workSetWorkDatecal = Calendar.getInstance();
249 				workSetWorkDatecal.setTime(workSetWorkDate);
250 				Calendar targetWorkDateCal = Calendar.getInstance();
251 				targetWorkDateCal.setTime(targetWorkDate);
252 				if (creditElements[CREDIT_FEE_TEXT_INDEX] != null) {
253 					if (workSetWorkDatecal.after(targetWorkDateCal)) {
254 						addToElementsList(
255 								elementsList,
256 								IPrintConverterConstants.DL_REC_ZBPTCCC_MSG_IF_AFTER_WORKDATE,
257 								creditElements[CREDIT_FEE_AMOUNT_TEXT_INDEX]);
258 						addToElementsList(
259 								elementsList,
260 								IPrintConverterConstants.DL_REC_ZBPTEXT5_MSG_BEFORE_AMT
261 										+ IPrintConverterConstants.DL_REC_ZBPTEXT5_AFTER_AMT
262 										+ IPrintConverterConstants.DL_REC_ZBPTEXT5_MSG_AFTER_AMT,
263 								creditElements[CREDIT_FEE_TEXT_INDEX]);
264 					}
265 					else {
266 						addToElementsList(
267 								elementsList,
268 								IPrintConverterConstants.DL_REC_ZBPTCCC_MSG_IF_BEFORE_WORKDATE,
269 								creditElements[CREDIT_FEE_AMOUNT_TEXT_INDEX]);
270 						addToElementsList(
271 								elementsList,
272 								IPrintConverterConstants.DL_REC_ZBPTEXT5_MSG_BEFORE_AMT
273 										+ IPrintConverterConstants.DL_REC_ZBPTEXT5_BEFORE_AMT
274 										+ IPrintConverterConstants.DL_REC_ZBPTEXT5_MSG_AFTER_AMT,
275 								creditElements[CREDIT_FEE_TEXT_INDEX]);
276 					}
277 				}
278 				if (creditElements[CREDIT_FEE_AMOUNT_INDEX] != null) {
279 					if (workSetWorkDatecal.before(targetWorkDateCal)) {
280 						BigDecimal total = invoice.getCreditPayment()
281 								.getAmount().add(CREDIT_FEE);
282 						addToElementsList(elementsList,
283 								formatter.format(total),
284 								creditElements[CREDIT_FEE_AMOUNT_INDEX]);
285 					}
286 					else {
287 						String creditPayment = Double.toString((invoice
288 								.getCreditPayment().getAmount()).doubleValue());
289 						if (creditPayment.length() < 8) {
290 							creditPayment = IPrintConverterConstants.SPACE
291 									+ creditPayment;
292 						}
293 						addToElementsList(elementsList,
294 								formatter.format(invoice.getCreditPayment()
295 										.getAmount()),
296 								creditElements[CREDIT_FEE_AMOUNT_INDEX]);
297 					}
298 				}
299 				if (creditElements[CREDIT_AMOUNT_END_INDEX] != null) {
300 					addToElementsList(elementsList, formatter.format(invoice
301 							.getCreditPayment().getAmount()),
302 							creditElements[CREDIT_AMOUNT_END_INDEX]);
303 				}
304 				// BO not present for credit card batch number and credit card item number.
305 				/*addToElementsList(elementsList, invoice.getCreditPayment()
306 						.getAuthorizationNumber().toString(),
307 						creditElements[CREDIT_BATCH_INDEX]);*/
308 			}
309 			if (creditElements[CREDIT_EXPIRE_INDEX] != null) {
310 				addToElementsList(elementsList, DocumentConverterUtil
311 						.convertDateToString(invoice.getCreditPayment()
312 								.getExpiryDate(),
313 								AbstractDocumentConverter.FORMATTER_MM_YY),
314 						creditElements[CREDIT_EXPIRE_INDEX]);
315 			}
316 			if (creditElements[CREDIT_AUTH_INDEX] != null) {
317 				addToElementsList(elementsList, invoice.getCreditPayment()
318 						.getAuthorizationNumber().toString(),
319 						creditElements[CREDIT_AUTH_INDEX]);
320 			}
321 		}
322 	}

65  	/***
66  	 * Calculate dl fee.
67  	 * 
68  	 * @param criteria
69  	 *            the dl application fee criteria
70  	 * @param dlFeeStaticDataList
71  	 *            the dl fee static data
72  	 * 
73  	 * @return the DriverLicenseFee Information
74  	 */
75  	private DriverLicenseFee calculateDlFee(DlFeeCalculationCriteria criteria,
76  			List <IBusinessObject> dlFeeStaticDataList) {
77  		//Holds the Applicable Driver License Fee
78  		DriverLicenseFee fee = null;
79  		//Iterate the Driver License Static Fee Data
80  		Iterator <IBusinessObject> dlFeeStaticDataIterator = dlFeeStaticDataList
81  				.iterator();
82  		//Temporary Variable
83  		DlApplicationFee dlFee = null;
84  		//Calculate Fee for Primary License Class
85  		while (dlFeeStaticDataIterator.hasNext()) {
86  			dlFee = (DlApplicationFee) dlFeeStaticDataIterator.next();
87  			if (dlFee.getLicenseClassType().equals(
88  					criteria.getPrimaryLicenseClass())) {
89  				//Compute Fee for Primary License Class
90  				fee = this.computeDlFee(criteria, dlFee);
91  				break;
92  			}
93  		}
94  		return fee;
95  	}

263 	 * This method will transform DL Application into DL Fee Calculation Criteria.
264 	 * 
265 	 * @param dlApplication the DL Application
266 	 * @return The Criteria for DL Fee Calculation
267 	 */
268 	private DlFeeCalculationCriteria transform(DlApplication dlApplication) {
269 		DlFeeCalculationCriteria criteria = new DlFeeCalculationCriteria();
270 		String licenseClass = dlApplication.getApplicationType().getCode();
271 		List <CodeSetElement> categoryList = dlApplication.getLicense()
272 				.getExistingCategories();
273 		//License Class
274 		criteria.setPrimaryLicenseClass(licenseClass);
275 		//TODO Find out Renewal
276 		//TODO Find out Renewal by Mail
277 		//TODO Find out Renewal by Indicator
278 		//TODO Need Domain Object for Upgrade Information in DL Application
279 		//TODO Need Indicator for Name Change in DlApplication
280 		//TODO Need Indicator for Duplicate Issuance
281 		//TODO Need Information on Removal of Restrictions
282 		//TODO Indicator for Skill Test
283 		if (!dlApplication.getLicense().getExistingEndorsements().isEmpty()) {
284 			criteria.setAddEndorsement(true);
285 		}
286 		//Add Restrictions
287 		if (!dlApplication.getLicense().getExistingRestrictions().isEmpty()) {
288 			criteria.setAddRestriction(true);
289 		}
290 		//Figure out Categories from License Class
291 		if (licenseClass.equals("A") && !categoryList.isEmpty()) {
292 			//TODO Original
293 			if (dlApplication.isOriginalApplicationIndicator()) {
294 				criteria.setOriginal(true);
295 			}
296 			//TODO Renewal
297 			else {
298 			}
299 		}
300 		//Figure out Categories from License Class
301 		else if (licenseClass.equals("D") && !categoryList.isEmpty()) {
302 			//SET THE APPROPRIATE FLAGS FOR CAT 1, 2, 3 - Same with Original, Upgrade, Renewal et al
303 			for (CodeSetElement category : categoryList) {
304 				if (category.getCode().equals("1")) {
305 					//TODO If Original License
306 					criteria.setCategory1Original(true);
307 					criteria.setOriginal(false);
308 					//TODO If Upgrade
309 					criteria.setCategory1Upgrade(true);
310 					criteria.setUpgrade(false);
311 					//TODO If Upgrade Renewal
312 					criteria.setCategory1UpgradeRenewal(true);
313 				}
314 				else if (category.getCode().equals("2")) {
315 					//TODO If Original License
316 					criteria.setCategory2Original(true);
317 					criteria.setOriginal(false);
318 					//TODO If Upgrade
319 					criteria.setCategory2Upgrade(true);
320 					criteria.setUpgrade(false);
321 					//TODO If Upgrade Renewal
322 					criteria.setCategory3UpgradeRenewal(true);
323 				}
324 				else if (category.getCode().equals("3")) {
325 					//TODO If Original License
326 					criteria.setCategory3Original(true);
327 					criteria.setOriginal(false);
328 					//TODO If Upgrade
329 					criteria.setCategory3Upgrade(true);
330 					criteria.setUpgrade(false);
331 					//TODO If Upgrade Renewal
332 					criteria.setCategory3UpgradeRenewal(true);
333 				}
334 			}
335 		}
336 		return criteria;
337 	}

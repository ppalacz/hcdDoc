46  	public String createMessage(Application application) {
47  		// Get the Application from the Request object.
48  		//		Application application = ((RecordCashierTransactionRequestEcs) recordCashierTransactionRequestEcs)
49  		//				.getApplication();
50  		StringBuilder messageBuffer = new StringBuilder();
51  		addStringToMessageText(messageBuffer, "T", 1);//Add constant "T"
52  		Person applicant = null;
53  		if (EaseUtil.isNotNull(application)
54  				&& (application instanceof DlApplication)) { //DlApplication
55  			DlApplication dlApplication = (DlApplication) application;
56  			applicant = dlApplication.getApplicant();
57  			// Add DL/ID Number
58  			if (EaseUtil.isNotNull(applicant.getDriverLicense())
59  					&& EaseUtil.isNotNull(applicant.getDriverLicense()
60  							.getLicenseNumber()))
61  				addStringToMessageText(messageBuffer, applicant
62  						.getDriverLicense().getLicenseNumber(), 8);
63  			else
64  				addStringToMessageText(messageBuffer, BLANK, 8);
65  		}
66  		else if (EaseUtil.isNotNull(application)
67  				&& application instanceof SalesPersonApplication) { //SalesPersonApplication
68  			SalesPersonApplication spApplication = (SalesPersonApplication) application;
69  			applicant = spApplication.getApplicant();
70  			// Add SP/DSI Number
71  			if (EaseUtil.isNotNull(spApplication.getLicense())
72  					&& EaseUtil.isNotNull(spApplication.getLicense()
73  							.getLicenseNumber()))
74  				addStringToMessageText(messageBuffer, spApplication
75  						.getLicense().getLicenseNumber(), 8);
76  			else
77  				addStringToMessageText(messageBuffer, BLANK, 8);
78  		}
79  		else { //application is not a valid type
80  			addStringToMessageText(messageBuffer, BLANK, 58);
81  			return messageBuffer.toString();
82  		}
83  		// Add Name Corrected Reversed ( Last First Middle, Suf)
84  		if (EaseUtil.isNotNull(applicant.getCurrentName())
85  				&& EaseUtil.isNotNull(applicant.getCurrentName().getFullName(
86  						PersonName.NAME_FORMAT_LAST_FIRST_MIDDLE_SUFFIX)))
87  			addStringToMessageText(messageBuffer, applicant.getCurrentName()
88  					.getFullName(
89  							PersonName.NAME_FORMAT_LAST_FIRST_MIDDLE_SUFFIX),
90  					35);
91  		else
92  			addStringToMessageText(messageBuffer, BLANK, 35);
93  		// Add Birth Date Corrected
94  		if (EaseUtil.isNotNull(applicant.getBirthDate()))
95  			addStringToMessageText(messageBuffer, EcsConverterHelper
96  					.convertDateToString(applicant.getBirthDate(),
97  							EcsConverterHelper.FORMATTER_MM_DD_YYCC), 8);
98  		else
99  			addStringToMessageText(messageBuffer, BLANK, 8);
100 		// Add  Birth Date of Record in Prior Century
101 		if (EaseUtil.isNotNull(applicant.getBirthDate())
102 				&& isBDayInPriorCentury(applicant.getBirthDate()))
103 			addStringToMessageText(messageBuffer, "Y", 1);
104 		else
105 			addStringToMessageText(messageBuffer, "N", 1);
106 		return messageBuffer.toString();
107 	}

109 	private boolean isBDayInPriorCentury(Date date) {
110 		String bdCentury = EcsConverterHelper.convertDateToString(date,
111 				"yyyyMMdd");
112 		String crtCentury = EcsConverterHelper.convertDateToString(new Date(),
113 				"yyyyMMdd");
114 		if (Integer.valueOf(bdCentury.substring(0, 2)) < Integer
115 				.valueOf(crtCentury.substring(0, 2)))
116 			return true;
117 		else
118 			return false;//Should we validate for eroneous dates??????
119 	}

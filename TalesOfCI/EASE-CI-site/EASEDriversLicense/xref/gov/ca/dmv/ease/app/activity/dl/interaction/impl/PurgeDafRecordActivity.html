32  public class PurgeDafRecordActivity extends InteractionActivity {
33  	/*** The Constant serialVersionUID. */
34  	private static final long serialVersionUID = -8768831379603893799L;
35  	private CdaTaskService cdaTaskService;
36  
37  	/* (non-Javadoc)
38  	 * @see gov.ca.dmv.ease.activity.InteractionActivity#preValidate(gov.ca.dmv.ease.app.context.ProcessContext)
39  	 */
40  	@Override
41  	protected void preCondition(ProcessContext processContext) {
42  		super.preCondition(processContext);
43  		DafProcessContext dafProcessContext = (DafProcessContext) processContext;
44  		DafRecordInquiryRequestTs tsRequest = null;
45  		String selectedDafNumber = dafProcessContext.getSelectedDafNumber();
46  		if (selectedDafNumber != null
47  				&& Integer.parseInt(selectedDafNumber) > 0) {
48  			tsRequest = new DafRecordInquiryRequestTs(dafProcessContext
49  					.getUserContext(), Integer.parseInt(selectedDafNumber));
50  			tsRequest.setDafInquiryTypes(DafInquiryTypes.DAF);
51  			DafRecordInquiryResponseTs tsResponse = cdaTaskService
52  					.execute(tsRequest);
53  			if (tsResponse.hasErrors()) {
54  				this.populateErrorMessages(dafProcessContext, tsResponse
55  						.getErrorCollector());
56  			}
57  			else {
58  				dafProcessContext.setTransaction((Transaction) tsResponse
59  						.getTransaction());
60  				dafProcessContext.setDlApplication(((Transaction) tsResponse
61  						.getTransaction()).getDlApplication());
62  			}
63  		}
64  	}
65  
66  	/***
67  	 * @param cdaTaskService the cdaTaskService to set
68  	 */
69  	public void setCdaTaskService(CdaTaskService cdaTaskService) {
70  		this.cdaTaskService = cdaTaskService;
71  	}
72  }

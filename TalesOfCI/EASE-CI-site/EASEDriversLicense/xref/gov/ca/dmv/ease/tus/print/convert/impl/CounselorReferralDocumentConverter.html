100 	/***
101 	 * Populates outgoing XML message for Counselor Referral Document.
102 	 * 
103 	 * @param requestFromClient the request from client
104 	 * 
105 	 * @return the string
106 	 */
107 	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
108 		CounselorReferralDocumentRequestTus request = (CounselorReferralDocumentRequestTus) requestFromClient;
109 		IUserContext currentUserContext = request.getUserContext();
110 		DlApplication dlApplication = request.getDlApplication();
111 		List <String> actionMessages = request.getActionMessages();
112 		Person person = dlApplication.getApplicant();
113 		DriverLicense driverLicense = person.getDriverLicense();
114 		IdCard idCard = person.getIdCard();
115 		Address mailingAddressOfRecord = person
116 				.getExistingAddress(AddressType.MAILING);
117 		Address residentialAddress = person
118 				.getExistingAddress(AddressType.RESIDENTIAL);
119 		IProcessHistory originalProcessHistory = dlApplication
120 				.getOriginalProcessHistory();
121 		IUserContext originalUserContext = originalProcessHistory
122 				.getUserContext();
123 		// Setting xml elements
124 		List <String> elementsList = new ArrayList <String>();
125 		addToElementsList(elementsList, dlApplication.getApplicationTypeText(),
126 				getTypeApplicationTextElement());
127 		if ("CDA".equals(currentUserContext.getTtc())) {
128 			addToElementsList(elementsList, OPTIONAL_TEXT_FOR_CONTINUE_TTC,
129 					getTypeOptionalTextElement());
130 			addToElementsList(elementsList,
131 					getOriginalAppOfficeId(dlApplication),
132 					getOriginalAppOfficeIdElement());
133 			addToElementsList(elementsList, getOriginalApplicationDate(
134 					dlApplication, FORMATTER_MM_DD_YY),
135 					getOriginalAppDateElement());
136 			addToElementsList(elementsList,
137 					getOriginalAppTechId(dlApplication),
138 					getOriginalAppCashierIdElement());
139 			addToElementsList(elementsList,
140 					getOriginalAppCashierSequenceNumber(dlApplication),
141 					getOriginalAppCashierSequenceNumberElement());
142 			addToElementsList(elementsList, getDafRecordNumber(dlApplication),
143 					getDafRecordElement());
144 		}
145 		addToElementsList(elementsList, getDlorIdNumber(person),
146 				getDlIdNumberElement());
147 		addToElementsList(elementsList, getDlExpDateOrText(currentUserContext,
148 				driverLicense), getDlExpDateOrTextElement());
149 		addToElementsList(elementsList, getIdExpDateOrText(currentUserContext,
150 				idCard), getIdExpDateOrTextElement());
151 		addToElementsList(elementsList, getCurrentName(currentUserContext,
152 				person), getCurrentNameElement());
153 		addToElementsList(elementsList, getAkaName(currentUserContext, person),
154 				getAkaNameElement());
155 		addToElementsList(elementsList, DocumentConverterUtil
156 				.getApplicantMailingAddress(mailingAddressOfRecord),
157 				getApplicantMailingAddressElement());
158 		addToElementsList(elementsList, DocumentConverterUtil
159 				.getApplicantMailingCity(mailingAddressOfRecord),
160 				getApplicantMailingCityElement());
161 		addToElementsList(elementsList, DocumentConverterUtil
162 				.getApplicantMailingState(mailingAddressOfRecord, false),
163 				getApplicantMailingStateElement());
164 		addToElementsList(elementsList, DocumentConverterUtil
165 				.getApplicantMailingZip(mailingAddressOfRecord, false),
166 				getApplicantMailingZipElement());
167 		addToElementsList(elementsList,
168 				getApplicantResidentialAddress(residentialAddress),
169 				getApplicantResidenceAddressElement());
170 		addToElementsList(elementsList,
171 				getApplicantResidentialCity(residentialAddress),
172 				getApplicantResidenceCityElement());
173 		addToElementsList(elementsList,
174 				getApplicantResidentialState(residentialAddress),
175 				getApplicantResidenceStateElement());
176 		addToElementsList(elementsList, getGender(person), getGenderElement());
177 		addToElementsList(elementsList, getHairColor(person),
178 				getHairColorElement());
179 		addToElementsList(elementsList, getEyeColor(person),
180 				getEyeColorElement());
181 		addToElementsList(elementsList, getHeightInFeet(person),
182 				getHeightInFeetsElement());
183 		addToElementsList(elementsList, getHeightInInch(person),
184 				getHeightInInchesElement());
185 		addToElementsList(elementsList, getWeight(person), getWeightElement());
186 		addToElementsList(elementsList,
187 				getBirthDate(person, FORMATTER_MM_dd_yy), getBirthDateElement());
188 		addToElementsList(elementsList, getClassOfRecord1(driverLicense),
189 				getClassOfRecord1Element());
190 		addToElementsList(elementsList, getClassOfRecord2(driverLicense),
191 				getClassOfRecord2Element());
192 		addToElementsList(elementsList, getCommercialClassFlag(driverLicense),
193 				getClassOfRecordCommlElement());
194 		addToElementsList(elementsList,
195 				getNonCommLicClassCategory1(driverLicense),
196 				getNonCommlLicenseCategory1Element());
197 		addToElementsList(elementsList,
198 				getNonCommLicClassCategory2(driverLicense),
199 				getNonCommlLicenseCategory2Element());
200 		addActionMessages(actionMessages, currentUserContext, elementsList,
201 				getRecordStatusesElement());
202 		String xmlData = createXmlForSingleDocument(elementsList,
203 				getDocumentTemplateElement());
204 		return xmlData;
205 	}

242 	/***
243 	 * Gets the commercial class flag.
244 	 * 
245 	 * @param driverLicense the driver license
246 	 * 
247 	 * @return the commercial class flag
248 	 */
249 	private String getCommercialClassFlag(DriverLicense driverLicense) {
250 		String commercialClassFlag = BLANK;
251 		if (!isNull(driverLicense)) {
252 			List <CodeSetElement> exisingLicenseClasses = driverLicense
253 					.getExistingLicenseClasses();
254 			for (CodeSetElement exisingLicenseClassCode : exisingLicenseClasses) {
255 				if (!isNullOrBlank(exisingLicenseClassCode)
256 						&& isComercialLicenseClass(exisingLicenseClassCode
257 								.getCode())) {
258 					commercialClassFlag = COMML_CLASS_FLAG;
259 					break;
260 				}
261 			}
262 		}
263 		return commercialClassFlag;
264 	}

420 	/***
421 	 * Adds the action messages.
422 	 * 
423 	 * @param actionMessages the action messages
424 	 * @param currentUserContext the current user context
425 	 * @param elementsList the elements list
426 	 * @param recordStatusMap the record status map
427 	 */
428 	private void addActionMessages(List <String> actionMessages,
429 			IUserContext currentUserContext, List <String> elementsList,
430 			Map <String, String> recordStatusMap) {
431 		if (!isNull(actionMessages)) {
432 			int expectedNumberOfRecordStatuses = 0;
433 			if (actionMessages.size() > 14) {
434 				expectedNumberOfRecordStatuses = 14;
435 			}
436 			else {
437 				expectedNumberOfRecordStatuses = actionMessages.size();
438 			}
439 			for (int i = 0; i < expectedNumberOfRecordStatuses; i++) {
440 				String recordStatusElement = recordStatusMap
441 						.get("RECORD_STATUS" + (i + 1) + "_ELEMENT");
442 				if (getMiscellaneousTtcs().contains(getTtc(currentUserContext))) {// if miscellaneous ttc
443 					if (i > 11) {
444 						continue;
445 					}
446 				}
447 				String actionMessage = actionMessages.get(i);
448 				if (!isNullOrBlank(actionMessage)) {
449 					if (!isNull(recordStatusElement)) {
450 						addToElementsList(elementsList, actionMessage,
451 								recordStatusElement);
452 					}
453 				}
454 			}
455 		}
456 	}

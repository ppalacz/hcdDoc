25  public class CheckModeActivity extends DecisionActivity {
26  	/*** The Constant serialVersionUID. */
27  	private static final long serialVersionUID = 2581418433723337022L;
28  
29  	/* (non-Javadoc)
30  	 * @see gov.ca.dmv.ease.app.activity.impl.DecisionActivity#evaluate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
31  	 * 
32  	 * @return
33  	 * counter_or_mail
34  	 * manual_or_travel
35  	 */
36  	@Override
37  	protected String evaluate(ProcessContext context) {
38  		IUserContext userContext = context.getUserContext();
39  		if (userContext.getOperationalMode() != null) {
40  			String mode = userContext.getOperationalMode().getCode();
41  			if (IEaseModes.TRAVEL_CREW.equals(mode)
42  					|| IEaseModes.MANUAL.equals(mode)) {
43  				return ICdaADecisionActivityConstants.MANUAL_OR_TRAVEL;
44  			}
45  		}
46  		return ICdaADecisionActivityConstants.COUNTER_OR_MAIL;
47  	}
48  }

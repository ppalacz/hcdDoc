39  	/***
40  	 * The purpose of this method is to build the tcode response elements map from the given flat message.
41  	 * 
42  	 * @param message the response string to the host
43  	 * @param tcodeElementNodes the list of tcode node elements
44  	 * 
45  	 * @return the map of response elements 
46  	 */
47  	public static Map <String, String> buildTcodeResponseElementsMap(
48  			String message, List <TcodeElementNode> tcodeElementNodes) {
49  		Map <String, String> tcodeResponseElementsMap = new HashMap <String, String>();
50  		int startPosition = 0;
51  		logger
52  				.info("/***********************************************************************************************************************/");
53  		for (TcodeElementNode tcodeElementNode : tcodeElementNodes) {
54  			int tcodeElelmentLength = tcodeElementNode.getTcodeElementLength();
55  			String tcodeElementValue = message.substring(startPosition,
56  					(startPosition + tcodeElelmentLength)).trim();
57  			tcodeElementNode.setTcodeElementValue(tcodeElementValue);
58  			tcodeResponseElementsMap.put(
59  					tcodeElementNode.getTcodeElementName(), tcodeElementValue);
60  			startPosition += tcodeElelmentLength;
61  			logger.info("\t" + tcodeElementNode.getTcodeElementName() + "("
62  					+ tcodeElementNode.getTcodeElementLength() + ")" + "\t["
63  					+ tcodeElementNode.getTcodeElementValue() + "]");
64  		}
65  		if ((StringUtils.isNotBlank(message))
66  				&& (startPosition < message.length())) {
67  			logger.error("Unparsed String("
68  					+ (message.length() - startPosition) + "): ["
69  					+ message.substring(startPosition, message.length()) + "]");
70  		}
71  		return tcodeResponseElementsMap;
72  	}

85  	/***
86  	 * The purpose of this method is to build the person name from the flat string.
87  	 * 
88  	 * @param strPersonName full person name
89  	 * @param delimiter Delimiter
90  	 * @return the person name object
91  	 */
92  	public static PersonName buildPersonNameFromString(String strPersonName,
93  			String delimiter) {
94  		PersonName personName = null;
95  		if (StringUtils.isNotBlank(strPersonName)) {
96  			personName = new PersonName();
97  			StringTokenizer st = new StringTokenizer(strPersonName, delimiter);
98  			int tokenNumber = 0;
99  			while (st.hasMoreTokens()) {
100 				String strPersonNameToken = st.nextToken();
101 				buildPersonNameFromString(personName,
102 						strPersonNameToken.trim(), tokenNumber);
103 				tokenNumber++;
104 			}
105 		}
106 		return personName;
107 	}

109 	/***
110 	 * The purpose of this method is to build the person name from the person name token.
111 	 * 
112 	 * @param personName the person name object
113 	 * @param strPersonNameToken the person name token
114 	 * @param tokenNumber the token number
115 	 */
116 	private static void buildPersonNameFromString(PersonName personName,
117 			String strPersonNameToken, int tokenNumber) {
118 		if (tokenNumber == 0) {
119 			personName.setLastName(strPersonNameToken);
120 		}
121 		else if (tokenNumber == 1) {
122 			personName.setFirstName(strPersonNameToken);
123 		}
124 		else if (tokenNumber == 2) {
125 			personName.setMiddleName(strPersonNameToken);
126 		}
127 		else if (tokenNumber == 3) {
128 			if (strPersonNameToken.startsWith("2")) {
129 				strPersonNameToken = "II";
130 			}
131 			else if (strPersonNameToken.startsWith("3")) {
132 				strPersonNameToken = "III";
133 			}
134 			else if (strPersonNameToken.startsWith("4")) {
135 				strPersonNameToken = "IV";
136 			}
137 			personName.setSuffix(strPersonNameToken);
138 		}
139 	}

156 		}
157 		fullNameSb.append("@");
158 		/* First Name */
159 		if (StringUtils.isNotBlank(personName.getFirstName())) {
160 			//fullNameSb.append("@");
161 			fullNameSb.append(personName.getFirstName());
162 		}
163 		fullNameSb.append("@");
164 		/* Middle Name */
165 		if (StringUtils.isNotBlank(personName.getMiddleName())) {
166 			//fullNameSb.append("@");
167 			fullNameSb.append(personName.getMiddleName());
168 		}
169 		fullNameSb.append("@");
170 		/* Suffix */
171 		if (StringUtils.isNotBlank(personName.getSuffix())) {
172 			String suffixName = personName.getSuffix();
173 			if (StringUtils.equals(suffixName, "II")) {
174 				suffixName = "2ND";
175 			}
176 			else if (StringUtils.equals(suffixName, "III")) {
177 				suffixName = "3RD";
178 			}
179 			else if (StringUtils.equals(suffixName, "IV")) {
180 				suffixName = "4TH";
181 			}
182 			//fullNameSb.append("@");
183 			fullNameSb.append(suffixName);
184 		}
185 		return fullNameSb.toString();
186 	}

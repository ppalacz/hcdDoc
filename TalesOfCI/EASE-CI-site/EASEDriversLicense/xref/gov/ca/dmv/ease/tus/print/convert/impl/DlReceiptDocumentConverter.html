100 	/***
101 	 * Populate outgoing XML message for Dl Receipt Document.
102 	 *
103 	 * @param requestFromClient the request from client
104 	 * @return the string
105 	 */
106 	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
107 		DlReceiptDocumentRequestTus request = (DlReceiptDocumentRequestTus) requestFromClient;
108 		IUserContext userContext = request.getUserContext();
109 		String ttc = userContext.getTtc();
110 		boolean addBarcode = false;
111 		if (ttc.startsWith("D") || ttc.startsWith("I") || ttc.startsWith("S")) {
112 			addBarcode = true;
113 		}
114 		Application application = request.getApplication();
115 		boolean isCNACondition = application.isCNACondition();
116 		CodeSetElement applicationTypeCodeSetElement = application
117 				.getApplicationType();
118 		Invoice invoice = application.getInvoice();
119 		ApplicationSpecificVariables applicationSpecificVariables = ApplicationSpecificVariables
120 				.getInstance().getApplicationSpecificVariables(application);
121 		Address currentMailingAddress = applicationSpecificVariables.person
122 				.getAddress(AddressType.MAILING);
123 		SocialSecurityDocument socialSecurityDocument = applicationSpecificVariables.person
124 				.getSocialSecurityDocument();
125 		List <String> elementsList = new ArrayList <String>();
126 		List <List <String>> documentsList = new ArrayList <List <String>>();
127 		List <String> doc2ElementsList = null;
128 		boolean isDMVFileCopy = isDMVFileCopy(ttc, invoice);
129 		if (isDMVFileCopy) {
130 			doc2ElementsList = new ArrayList <String>();
131 		}
132 		addHeaderElements(elementsList, doc2ElementsList, ttc,
133 				applicationSpecificVariables.licenseNumber, isCNACondition,
134 				addBarcode);
135 		addNonResidentElement(elementsList, applicationSpecificVariables,
136 				currentMailingAddress.getState(), ttc);
137 		addPersonAndCurrentAddressElements(elementsList, doc2ElementsList,
138 				applicationSpecificVariables.person, currentMailingAddress);
139 		addPreviousClassElements(elementsList, doc2ElementsList,
140 				applicationSpecificVariables.appliedLicenseClasses,
141 				applicationSpecificVariables.newRestrictions, userContext
142 						.getTtc());
143 		addCommercialEndorsementsElements(elementsList, doc2ElementsList,
144 				applicationSpecificVariables.newEndorsements, ttc);
145 		DecimalFormat formatter = new DecimalFormat("####0.00");
146 		InvoiceConverterUtility.populateInvoice(elementsList, CASH_ELEMENTS,
147 				DEBIT_ELEMENTS, CREDIT_ELEMENTS, invoice, formatter,
148 				userContext.getWorkDate(), false);
149 		if (doc2ElementsList != null) {
150 			InvoiceConverterUtility.populateInvoice(doc2ElementsList,
151 					CASH_ELEMENTS, DEBIT_ELEMENTS, CREDIT_ELEMENTS, invoice,
152 					formatter, userContext.getWorkDate(), false);
153 		}
154 		if (invoice != null && invoice.getCreditPayment() != null
155 				&& invoice.getCreditPayment().getAmount().intValue() > 0) {
156 			addToBothElementsLists(elementsList, doc2ElementsList,
157 					IPrintConverterConstants.DL_REC_ZBP_SIGN_MSG,
158 					IPrintConverterConstants.DL_REC_ZBP_SIGN);
159 		}
160 		addSsnResponseElements(elementsList, doc2ElementsList,
161 				applicationSpecificVariables, ttc, socialSecurityDocument);
162 		String dafRecordNumber = getDafRecordNumber(application);
163 		addFooterElements(elementsList, doc2ElementsList, userContext,
164 				isCNACondition, dafRecordNumber);
165 		addNameElements(elementsList, doc2ElementsList, ttc, userContext
166 				.getWorkDate(),
167 				applicationSpecificVariables.isApplicantOrganDonor);
168 		if (applicationTypeCodeSetElement != null) {
169 			addToBothElementsLists(elementsList, doc2ElementsList,
170 					applicationTypeCodeSetElement.getName(),
171 					IPrintConverterConstants.DL_REC_ZBPTYPAT);
172 		}
173 		if (isDlOrIdOrSalesPersonLicenseApplicationTtc(ttc)) {
174 			addToBothElementsLists(elementsList, doc2ElementsList,
175 					convertDateToString(application.getApplicationDate(),
176 							FORMATTER_MM_DD_YY),
177 					IPrintConverterConstants.DL_REC_ZBPOAPDT);
178 		}
179 		String docName = IPrintConverterConstants.DL_REC_DOCUMENT_NAME;
180 		if (!addBarcode) {
181 			docName = IPrintConverterConstants.DL_REC_NO_BARCODE_DOCUMENT_NAME;
182 		}
183 		String xmlData = null;
184 		if (isDMVFileCopy) {
185 			documentsList.add(elementsList);
186 			documentsList.add(doc2ElementsList);
187 			List <String> templateNames = new ArrayList <String>();
188 			templateNames.add(docName);
189 			templateNames
190 					.add(IPrintConverterConstants.DL_REC_NO_BARCODE_DOCUMENT_NAME);
191 			xmlData = DocumentConverterUtil.convertToXml(documentsList,
192 					templateNames);
193 		}
194 		else {
195 			xmlData = createXmlForSingleDocument(elementsList, docName);
196 		}
197 		return xmlData;
198 	}

200 	/***
201 	 * Checks if is dMV file copy.
202 	 *
203 	 * @param ttc the ttc
204 	 * @param invoice the invoice
205 	 * @return true, if is dMV file copy
206 	 */
207 	private boolean isDMVFileCopy(String ttc, Invoice invoice) {
208 		if (invoice != null && invoice.getCreditPayment() != null
209 				&& invoice.getCreditPayment().getAmount().intValue() > 0) {
210 			return true;
211 		}
212 		if ("Q70".equals(ttc) || "Q71".equals(ttc) || "Q72".equals(ttc)
213 				|| "TVC".equals(ttc) || "TVS".equals(ttc) || "04M".equals(ttc)
214 				|| "10Q".equals(ttc) || "12M".equals(ttc) || "13M".equals(ttc)
215 				|| "14M".equals(ttc) || "15M".equals(ttc) || "20M".equals(ttc)
216 				|| "92M".equals(ttc) || "98M".equals(ttc)) {
217 			return true;
218 		}
219 		return false;
220 	}

271 			header2Text = IPrintConverterConstants.DL_REC_MSG_IF_12M;
272 		}
273 		else if (IPrintConverterConstants.DL_REC_TTC_14M.equalsIgnoreCase(ttc)) {
274 			header2Text = IPrintConverterConstants.DL_REC_MSG_IF_14M;
275 		}

281 					.equalsIgnoreCase(ttc))
282 					|| (IPrintConverterConstants.DL_REC_TTC_15M
283 							.equalsIgnoreCase(ttc))) {
284 				header2Text = IPrintConverterConstants.DL_REC_MSG_IF_04M_OR_15M_CNA_VERIF;

287 		addToElementsList(elementsList, header1Text,
288 				IPrintConverterConstants.DL_REC_ZBPHDR2);
289 		if (doc2ElementsList != null) {
290 			addToElementsList(doc2ElementsList, header2Text,
291 					IPrintConverterConstants.DL_REC_ZBPHDR2);

293 		// Setting Application Header, Type Application Text

296 					IPrintConverterConstants.DL_REC_ZBPDISCL_MSG,

299 					IPrintConverterConstants.DL_REC_ZBPAPPLH_MSG,

301 		}

444 	private void addNameElements(List <String> elementsList,
445 			List <String> doc2ElementsList, String ttc, Date workDate,
446 			boolean isApplicantOrganDonor) {
447 		if (isIdentificationCardApplicationTtc(ttc)) {
448 			addToBothElementsLists(elementsList, doc2ElementsList,
449 					IPrintConverterConstants.DL_REC_ZBPTEXT1_MSG_IF_ID_APP,
450 					IPrintConverterConstants.DL_REC_ZBPTEXT1);
451 			addToBothElementsLists(elementsList, doc2ElementsList,
452 					IPrintConverterConstants.DL_REC_ZBPTEXT2_MSG_IF_ID_APP,
453 					IPrintConverterConstants.DL_REC_ZBPTEXT2);
454 			addToBothElementsLists(elementsList, doc2ElementsList,
455 					IPrintConverterConstants.DL_REC_ZBPTEXT3_MSG_IF_DL_APP,
456 					IPrintConverterConstants.DL_REC_ZBPTEXT3);
457 		}
458 		if (isDriverLicenseApplicationTtc(ttc)) {
459 			addToBothElementsLists(elementsList, doc2ElementsList,
460 					IPrintConverterConstants.DL_REC_ZBPTEXT1_MSG_IF_DL_APP,
461 					IPrintConverterConstants.DL_REC_ZBPTEXT1);
462 			addToBothElementsLists(
463 					elementsList,
464 					doc2ElementsList,
465 					IPrintConverterConstants.DL_REC_ZBPTEXT2_MSG_IF_DL_APP1
466 							+ convertDateToString(workDate, FORMATTER_MM_dd_yy)
467 							+ IPrintConverterConstants.DL_REC_ZBPTEXT2_MSG_IF_DL_APP2,
468 					IPrintConverterConstants.DL_REC_ZBPTEXT2);
469 		}

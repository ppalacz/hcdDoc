35  	public void validate(ProcessContext processContext) {
36  		CdaProcessContext cdaProcessContext = (CdaProcessContext) processContext;
37  		IErrorCollector errorCollector = new ErrorCollector();
38  		Action selectedAction = cdaProcessContext.getSelectedAction();
39  		if (selectedAction != null
40  				&& selectedAction.equals(getActionsRegistry().getAction(
41  						IActionsRegistry.ENTER_ACTION))) {
42  			/*Need to pull applicationDate from context.
43  			 *Date applicationDate =  cdaProcessContext.getDlApplication().getApplicationDate();*/
44  			Date applicationDate = new Date();
45  			int tempPermitdays = cdaProcessContext.getTempPermitDays();
46  			/*Need to pull applicationDate from context.
47  			 * Date workDate = cdaProcessContext.getUserContext().getWorkDate();
48  			 */
49  			Date workDate = new Date();
50  			Calendar cal1 = Calendar.getInstance();
51  			Calendar cal2 = Calendar.getInstance();
52  			cal1.setTime(applicationDate);
53  			cal1.add(Calendar.YEAR, 1);
54  			cal2.setTime(workDate);
55  			cal2.add(Calendar.DATE, tempPermitdays);
56  			if (cal1 != null && cal2 != null) {
57  				if (cal2.after(cal1)) {
58  					errorCollector.register(new ApplicationException("0517"));
59  				}
60  			}
61  		}
62  		setErrorCollector(errorCollector);
63  	}
64  }

139 	private void convertSetInstructionPermitClasses() {
140 		//First clear all the values in the list and then add the values from screen
141 		getDriverLicense().setInterimPermits(null);
142 		List<InterimPermit> instructionPermitClasses = new ArrayList<InterimPermit>();
143 		if (this.instructionalPermitClass1 != null
144 				&& this.instructionalPermitClass1.getCode() != null) {
145 			InterimPermit instrClass = new InterimPermit(
146 					this.instructionalPermitClass1,
147 					InterimPermitType.INSTRUCTION_PERMIT);
148 			instructionPermitClasses.add(instrClass);
149 		}
150 
151 		if (this.instructionalPermitClass2 != null
152 				&& this.instructionalPermitClass2.getCode() != null) {
153 			InterimPermit instrClass = new InterimPermit(
154 					this.instructionalPermitClass2,
155 					InterimPermitType.INSTRUCTION_PERMIT);
156 			instructionPermitClasses.add(instrClass);
157 		}
158 		//Set the new instruction permit classes to process context
159 		getDriverLicense().setInterimPermits(instructionPermitClasses);
160 	}

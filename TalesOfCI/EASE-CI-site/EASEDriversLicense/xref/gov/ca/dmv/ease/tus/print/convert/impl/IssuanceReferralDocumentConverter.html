43  	 * Populates outgoing XML message for Issuance Referral Document.
44  	 * 
45  	 * @param requestFromClient the request from client
46  	 * 
47  	 * @return the string
48  	 */
49  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
50  		IUserContext currentUserContext = requestFromClient.getUserContext();
51  		Person applicant = null;
52  		Application application = null;
53  		DlApplication dlApplication = null;
54  		SalesPersonApplication spApplication = null;
55  		String incompleteApplicationReasonCode = BLANK;
56  		IssuanceReferralDocumentRequestTus request = (IssuanceReferralDocumentRequestTus) requestFromClient;
57  		application = request.getApplication();
58  		if (application instanceof SalesPersonApplication) {
59  			spApplication = (SalesPersonApplication) application;
60  			applicant = spApplication.getApplicant();
61  		}
62  		else if (application instanceof DlApplication) {
63  			dlApplication = (DlApplication) application;
64  			applicant = dlApplication.getApplicant();
65  		}
66  		else if (application instanceof DriverRecordUpdateApplication) {
67  			applicant = ((DriverRecordUpdateApplication) application)
68  					.getApplicant();
69  		}
70  		List <String> elementsList = super.getCommonElementsList(
71  				requestFromClient, application, applicant);
72  		if (!EaseUtil.isNullOrBlank(application.getIssuanceReferralMessages())) {
73  			//If Issuance Messages are present then only print these messages. 
74  			List <String> messageDescriptions = new ArrayList <String>();
75  			List <Integer> messageIndicators = new ArrayList <Integer>();
76  			incompleteApplicationReasonCode = getIncompleteAppReasonCode(application);
77  			for (IssuanceReferralMessagesType issuanceReferralMessage : application
78  					.getIssuanceReferralMessages()) {
79  				messageIndicators.add(issuanceReferralMessage.getIndicator());
80  			}
81  			Collections.sort(messageIndicators);
82  			if (EaseUtil.isNotNull(dlApplication)
83  					&& messageIndicators.contains(23)) {
84  				addMsgDescriptionsForCommCheck(messageDescriptions,
85  						dlApplication, applicant);
86  				messageIndicators.remove(new Integer(23)); //Do not use this message indicator again in this converter.
87  			}
88  			addAllMsgDescriptions(messageDescriptions, messageIndicators,
89  					incompleteApplicationReasonCode, applicant);
90  			addMsgDescriptionsToElementList(elementsList, messageDescriptions); //Issuance Messages.
91  		}
92  		addOfficeDetailsToElementList(elementsList, currentUserContext); //Office ID, current date & time.
93  		return createXmlForSingleDocument(elementsList,
94  				IPrintConverterConstants.ISSUANCE_REFERRAL_TEMPLATE);
95  	}

100 	 * @param application the application

104 	private String getIncompleteAppReasonCode(Application application) {
105 		String incompleteApplicationReasonCode = BLANK;
106 		if (application instanceof SalesPersonApplication) {
107 			if (EaseUtil.isNotNull(((SalesPersonApplication) application)
108 					.getNewIncompleteApplicationReasonCode())) {
109 				incompleteApplicationReasonCode = ((SalesPersonApplication) application)
110 						.getNewIncompleteApplicationReasonCode().getCode();
111 			}
112 		}
113 		if (application instanceof DlApplication) {
114 			if (EaseUtil.isNotNull(((DlApplication) application)
115 					.getNewIncompleteApplicationReasonCode())) {
116 				incompleteApplicationReasonCode = ((DlApplication) application)
117 						.getNewIncompleteApplicationReasonCode().getCode();
118 			}
119 		}
120 		return incompleteApplicationReasonCode;
121 	}

123 	/***
124 	 * Checks if is photo printed.
125 	 * 
126 	 * @param applicant the applicant
127 	 * 
128 	 * @return the boolean
129 	 */
130 	private Boolean isPhotoPrinted(Person applicant) {
131 		Boolean photoPrintedIndicator = false;
132 		if (EaseUtil.isNotNull(applicant) && applicant instanceof Person) {
133 			if (EaseUtil.isNotNull(applicant.getDriverLicense())) {
134 				//Driver License Application.
135 				if (EaseUtil.isNotNull(applicant.getDriverLicense()
136 						.getPhotoInformation())) {
137 					photoPrintedIndicator = true;
138 				}
139 			}
140 			else if (EaseUtil.isNotNull(applicant.getId())) {
141 				//ID Card Application.
142 				if (EaseUtil.isNotNull(applicant.getIdCard()
143 						.getPhotoInformation())) {
144 					photoPrintedIndicator = true;
145 				}
146 			}
147 		}
148 		if (EaseUtil.isNotNull(applicant)
149 				&& applicant instanceof VehicleSalesPerson) {
150 			//SalesPerson Application.
151 			VehicleSalesPerson vehicleSalesPerson = (VehicleSalesPerson) applicant;
152 			if (EaseUtil.isNotNull(vehicleSalesPerson.getSalesPersonLicense())) {
153 				if (EaseUtil.isNotNull(vehicleSalesPerson
154 						.getSalesPersonLicense().getPhotoInformation())) {
155 					photoPrintedIndicator = true;
156 				}
157 			}
158 		}
159 		return photoPrintedIndicator;
160 	}

162 	/***
163 	 * Adds the msg descriptions for comm check.
164 	 * 
165 	 * @param messageDescriptions the message descriptions
166 	 * @param dlApplication the dl application
167 	 * @param applicant the applicant
168 	 */
169 	private void addMsgDescriptionsForCommCheck(
170 			List <String> messageDescriptions, DlApplication dlApplication,
171 			Person applicant) {
172 		//For Driver License Applications
173 		if (EaseUtil.isNotNull(dlApplication.isClearedByIssuance())
174 				&& !dlApplication.isClearedByIssuance()) {
175 			//If Issuance Office does not clear the transaction					
176 			messageDescriptions
177 					.add(IPrintConverterConstants.CDL_MANUAL_APP_CHECK_CDLIS_PDPS);
178 		}
179 		else {
180 			if (DocumentConverterUtil
181 					.isAnyCommercialLicenseClassExist(applicant
182 							.getDriverLicense().getAppliedLicenseClasses())) {
183 				//For commercial licenses
184 				messageDescriptions
185 						.add(IssuanceReferralMessagesType.COMMERICAL_CHECK_CDL_OR_NDR
186 								.getDescription());
187 			}
188 			else {
189 				messageDescriptions
190 						.add(IPrintConverterConstants.NON_COMMERCIAL_CHECK_PDPS);
191 			}
192 		}
193 		if (EaseUtil.isNotNull(dlApplication.getCdlisResponseCode())
194 				|| EaseUtil.isNotNull(dlApplication.getPdpsResponseCode())) {
195 			if (EaseUtil.isNotNull(dlApplication.getCdlisResponseCode())
196 					&& IPrintConverterConstants.CDLIS_RESPONSE_CODES_TYPE1
197 							.indexOf(dlApplication.getCdlisResponseCode()
198 									.getCode()) >= 0) {
199 				//If CdlisResponseCode from CDLIS inquiry's response is any of C, L, M, P,T
200 				messageDescriptions.add(dlApplication.getCdlisResponseCode()
201 						.getDescription());
202 			}
203 			else {
204 				if (EaseUtil.isNotNull(dlApplication.getCdlisResponseCode())
205 						&& IPrintConverterConstants.CDLIS_RESPONSE_CODES_TYPE2
206 								.indexOf(dlApplication.getCdlisResponseCode()
207 										.getCode()) >= 0) {
208 					//If CdlisResponseCode from CDLIS inquiry's response is any of A, E, U, I, D, W
209 					messageDescriptions.add(dlApplication
210 							.getCdlisResponseCode().getDescription());
211 				}
212 				if (EaseUtil.isNotNull(dlApplication.getPdpsResponseCode())
213 						&& IPrintConverterConstants.PDPS_RESPONSE_CODES
214 								.indexOf(dlApplication.getPdpsResponseCode()
215 										.getCode()) >= 0) {
216 					//If PdpsResponseCode from CDLIS inquiry's response is any of E, U, I, D, B, S, C, L, M, P, T, W
217 					messageDescriptions.add(dlApplication.getPdpsResponseCode()
218 							.getDescription());
219 				}
220 			}
221 		}
222 	}

224 	/***
225 	 * Adds the all msg descriptions.
226 	 * 
227 	 * @param messageDescriptions the message descriptions
228 	 * @param messageIndicators the message indicators
229 	 * @param incompleteApplicationReasonCode the incomplete application reason code
230 	 * @param applicant the applicant
231 	 */
232 	private void addAllMsgDescriptions(List <String> messageDescriptions,
233 			List <Integer> messageIndicators,
234 			String incompleteApplicationReasonCode, Person applicant) {
235 		for (int indicator : messageIndicators) {
236 			//TODO Mag Stripe
237 			if (indicator == 0) {
238 				//Ignore message 'RECORDS TO BE COMBINED'.
239 				continue;
240 			}
241 			if (indicator == 1 || indicator == 5 || indicator == 10
242 					|| indicator == 11 || indicator == 14) {
243 				if (incompleteApplicationReasonCode == BLANK
244 						|| incompleteApplicationReasonCode == "MG"
245 						|| incompleteApplicationReasonCode == "PH") {
246 					messageDescriptions
247 							.add(getMessageDescriptionForGivenIndicator(indicator));
248 				}
249 			}
250 			else {
251 				if (indicator == 2) {
252 					if (incompleteApplicationReasonCode == BLANK) {
253 						messageDescriptions
254 								.add(getMessageDescriptionForGivenIndicator(indicator));
255 					}
256 				}
257 				else if (indicator == 6) {
258 					messageDescriptions
259 							.add(getMessageDescriptionForGivenIndicator(indicator));
260 				}
261 				else {
262 					if (indicator == 3 || indicator == 4) {
263 						if (incompleteApplicationReasonCode == BLANK
264 								|| isPhotoPrinted(applicant)) {
265 							messageDescriptions
266 									.add(getMessageDescriptionForGivenIndicator(indicator));
267 						}
268 					}
269 					else {
270 						if (indicator == 22) {
271 							if (EaseUtil.isNotNull(applicant)
272 									&& EaseUtil.isNotNull(applicant
273 											.getDriverLicense())) {
274 								if (!DocumentConverterUtil
275 										.isAnyCommercialLicenseClassExist(applicant
276 												.getDriverLicense()
277 												.getAppliedLicenseClasses())) {
278 									//For Non-Commercial change message from CHECK CDLIS/PDPS MATCHES to CHECK PDPS MATCHES
279 									messageDescriptions
280 											.add(getMessageDescriptionForGivenIndicator(
281 													indicator)
282 													.replace(
283 															IPrintConverterConstants.CDLIS_TEXT,
284 															BLANK));
285 								}
286 								else {
287 									messageDescriptions
288 											.add(getMessageDescriptionForGivenIndicator(indicator));
289 								}
290 							}
291 						}
292 						else {
293 							messageDescriptions
294 									.add(getMessageDescriptionForGivenIndicator(indicator));
295 						}
296 					}
297 				}
298 			}
299 			if (indicator == 4 || indicator == 9) {
300 				if (EaseUtil.isNotNull(applicant)
301 						&& EaseUtil.isNotNull(applicant.getDriverLicense())) {
302 					//For Driver License Application.
303 					Integer limitedTermYears = applicant.getDriverLicense()
304 							.getLimitedLicenseTermYears();
305 					if (!EaseUtil.isNullOrBlank(limitedTermYears)) {
306 						switch (limitedTermYears) {
307 						case 1:
308 							messageDescriptions
309 									.add(IPrintConverterConstants.ISSUE_OR_EXTEND_1_YR_LIMITED_TERM);
310 							break;
311 						case 2:
312 							messageDescriptions
313 									.add(IPrintConverterConstants.ISSUE_OR_EXTEND_2_YR_LIMITED_TERM);
314 							break;
315 						case 9:
316 							messageDescriptions
317 									.add(IPrintConverterConstants.END_LIMITED_TERM);
318 							break;
319 						}
320 					}
321 				}
322 			}
323 		}
324 	}

326 	/***
327 	 * Adds the msg descriptions to element list.
328 	 * 
329 	 * @param elementList the element list
330 	 * @param messageDescriptions the message descriptions
331 	 */
332 	public void addMsgDescriptionsToElementList(List <String> elementsList,
333 			List <String> messageDescriptions) {
334 		int count = 1;
335 		for (count = 1; count < messageDescriptions.size() + 1; count++) {
336 			elementsList
337 					.add(messageDescriptions.get(count - 1)
338 							+ FIELD_DELIMITER
339 							+ IPrintConverterConstants.ISSUANCE_REFERRAL_ISSUANCE_MESSAGE
340 									.concat(Integer.toString(count)));
341 			if (count == 9) {
342 				//Print only up to 9 Issuance Referral Messages.
343 				break;
344 			}
345 		}
346 	}

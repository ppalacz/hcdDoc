30  	/***
31  	 * This method calculated Projected Expiration year for DL.
32  	 * VISN_CalculateProjectedExpirationYear.doc
33  	 * 
34  	 * @param ttc the TTC
35  	 * @param originalAppDate the Original Application Date
36  	 * @param dlExpirationDate the DL Expiration Date
37  	 * @param birthDate the Applicant BirthDate
38  	 * @param limitedTermYears the Limited Term Years
39  	 * @param attachments the Attachments
40  	 * @param feeReasonCode the Fee Reason Code
41  	 * @param dlFeeDueYear the DL Fee Due Year
42  	 * 
43  	 * @return the Projected DL Fee Expiration Date
44  	 */
45  	public Date calculateProjectedExpYearForDl(String ttc,
46  			Date originalAppDate, Date dlExpirationDate, Date birthDate,
47  			int limitedTermYears, List <CodeSetElement> attachments,
48  			CodeSetElement feeReasonCode, int dlFeeDueYear) {
49  		Date computedDlExpDate = null;
50  		//Process only if DLA
51  		if (TTC_DLA.equalsIgnoreCase(ttc)) {
52  			//If New Application, then Exit
53  			//OrginialAppDate, dlExpirationDate
54  			if (limitedTermYears > 0 && limitedTermYears < 9) {
55  				//Check if Attachment 10 is Present & 13 is Not Present 
56  				if (checkForAttachments(attachments)) {
57  					//If Fee Reason Code L is Present
58  					if (!EaseUtil.isNullOrBlank(feeReasonCode)
59  							&& L_LIMITED_TERM.equalsIgnoreCase(feeReasonCode
60  									.getCode())) {
61  						//If Fee Due Year Present
62  						if (EaseUtil.isNotNull(dlFeeDueYear)) {
63  							//Invoke Calculate Projected DL Expiration Date
64  							computedDlExpDate = computeDlExpirationDate(originalAppDate, dlExpirationDate, birthDate, limitedTermYears);
65  						}
66  					}
67  				}
68  			}
69  		}
70  		return computedDlExpDate;
71  	}

73  	/***
74  	 * This method computes the Projected DL Expiration Date.
75  	 * 
76  	 * @param originalAppDate the Original Application Date
77  	 * @param dlExpirationDate the DL Expiration Date
78  	 * @param birthDate the Birth Date
79  	 * @param limitedTermYears the Limited Term Years
80  	 * 
81  	 * @return the Projected DL Expiration Date
82  	 */
83  	private Date computeDlExpirationDate(Date originalAppDate,
84  			Date dlExpirationDate, Date birthDate, int limitedTermYears) {
85  		int ddComponent = 0;
86  		int mmComponent = 0;
87  		int yyComponent = 0;
88  		Calendar expirationCal = null;
89  		Calendar originalAppCal = null;
90  		Calendar birthDateCal = Calendar.getInstance();
91  		birthDateCal.setTime(birthDate);
92  		//Get DL Expiration MMDD Component
93  		ddComponent = birthDateCal.get(Calendar.DATE);
94  		mmComponent = birthDateCal.get(Calendar.MONTH);
95  		//If DL Expires on or within 6 Months of Original Application Date
96  		if (!dlExpirationDate.before(originalAppDate)
97  				&& !dlExpirationDate.after(DateUtil.addToSpecifiedDate(
98  						originalAppDate, 0, 6, 0))) {
99  			//Get DL Expiration Year
100 			expirationCal = Calendar.getInstance();
101 			expirationCal.setTime(dlExpirationDate);
102 			yyComponent = expirationCal.get(Calendar.YEAR) + limitedTermYears;
103 		}
104 		//The DL Expires before the Original Application Date or after 6 months of the Original Application Date
105 		else if (dlExpirationDate.before(originalAppDate)
106 				|| dlExpirationDate.after(DateUtil.addToSpecifiedDate(
107 						originalAppDate, 0, 6, 0))) {
108 			//The MMDD of the DL expiration Date is later than the MMDD of the Original Application Date
109 			expirationCal = Calendar.getInstance();
110 			originalAppCal = Calendar.getInstance();
111 			expirationCal.setTime(dlExpirationDate);
112 			originalAppCal.setTime(originalAppDate);
113 			if (expirationCal.get(Calendar.MONTH) > originalAppCal
114 					.get(Calendar.MONTH)) {
115 				yyComponent = originalAppCal.get(Calendar.YEAR)
116 						+ limitedTermYears - 1;
117 			}
118 			else if (expirationCal.get(Calendar.DATE) >= originalAppCal
119 					.get(Calendar.DATE)) {
120 				yyComponent = originalAppCal.get(Calendar.YEAR)
121 						+ limitedTermYears;
122 			}
123 		}
124 		expirationCal = Calendar.getInstance();
125 		expirationCal.set(yyComponent, mmComponent, ddComponent);
126 		return expirationCal.getTime();
127 	}

129 	/***
130 	 * This method checks if Attachment 10 is Present and 13 is Not Present.
131 	 * 
132 	 * @param attachments the Attachments
133 	 * 
134 	 * @return true if Attachment 10 is Present and 13 is Not Present.
135 	 */
136 	private boolean checkForAttachments(List <CodeSetElement> attachments) {
137 		boolean isAttachment13Present = false;
138 		boolean isAttachment10Present = false;
139 		if (!EaseUtil.isNullOrBlank(attachments)) {
140 			for (CodeSetElement attachment : attachments) {
141 				if (ATTACHMENT_10.equalsIgnoreCase(attachment.getCode())) {
142 					isAttachment10Present = true;
143 				}
144 				else if (ATTACHMENT_13.equalsIgnoreCase(attachment.getCode())) {
145 					isAttachment13Present = true;
146 				}
147 			}
148 		}
149 		return isAttachment10Present && !isAttachment13Present;
150 	}

152 	/***
153 	 * This method checks and displays any Errors that occurred during processing of 
154 	 * DL Expiration Date calculations.
155 	 * 
156 	 * @param projectedExpDate the Project Expiration Date
157 	 * @param dlFeeDueYear the DL Fee Due Year
158 	 * @param processDate the Application Process Date
159 	 * 
160 	 * @return the Error Message to Display
161 	 */
162 	public static String checkForErrorMessage(Date projectedExpDate,
163 			int dlFeeDueYear, Date processDate) {
164 		String errorMessage = null;
165 		Calendar expDateCal = Calendar.getInstance();
166 		expDateCal.setTime(projectedExpDate);
167 		//Table 3
168 		if (expDateCal.get(Calendar.YEAR) > dlFeeDueYear) {
169 			errorMessage = LIMITED_TERM_ERROR_MSG;
170 		}
171 		else if (!projectedExpDate.after(processDate)) {
172 			errorMessage = LIMITED_TERM_ERROR_MSG;
173 		}
174 		return errorMessage;
175 	}

31  	/*** The Constant serialVersionUID. */

34  	/***
35  	 * This method validates Process Id.
36  	 * 
37  	 * @param processId
38  	 *            process id
39  	 * @return errorCollector the error collector
40  	 */
41  	public IErrorCollector validate(String processId) {
42  		/*** The VR. */
43  		String vrProcess = "V";
44  		/*** The DL. */
45  		String dlProcess = "D";
46  		IErrorCollector collector = new ErrorCollector();
47  		if (isNullOrBlank(processId)) {
48  			collector.register(new EaseValidationException(SpringUtils
49  					.getMessage(PROCESS_ID_REQUIRED_FIELD_ERROR_MESSAGE)));
50  		}
51  		else if (!processId.toUpperCase().startsWith(vrProcess)
52  				&& !processId.toUpperCase().startsWith(dlProcess)) {
53  			collector.register(new EaseValidationException(SpringUtils
54  					.getMessage(PROCESS_ID_INVALID_ERROR_MESSAGE,
55  							new Object[] { processId })));
56  		}
57  		return collector;
58  	}

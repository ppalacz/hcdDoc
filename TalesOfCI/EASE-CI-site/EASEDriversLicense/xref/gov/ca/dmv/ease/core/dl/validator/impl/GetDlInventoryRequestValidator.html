34  public class GetDlInventoryRequestValidator extends AbstractBaseValidator {
35  	/*** The Constant serialVersionUID. */
36  	private static final long serialVersionUID = 6646295577617173118L;
37  
38  	/***
39  	 * This method validates DlApplication<code>DlApplication</code>.
40  	 * 
41  	 * @param validatable The Validate
42  	 * @return <code>IErrorCollector</code> containing Validation Messages.
43  	 * @see IErrorCollector
44  	 * @see DlApplication
45  	 */
46  	@Override
47  	public void validateValidatable(IValidatable validatable,
48  			IErrorCollector collector) {
49  		if (validatable instanceof DlApplication) {
50  			DlApplication dlApplication = (DlApplication) validatable;
51  			List <CodeSetElement> classCodeSet = dlApplication.getLicense()
52  					.getAppliedLicenseClasses();
53  			if (EaseUtil.isNullOrBlank(classCodeSet)) {
54  				collector
55  						.register(new EaseValidationException(
56  								SpringUtils
57  										.getMessage(DL_APPLICATION_LICENSE_CLASS_CODE_IS_NULL_ERROR_MESSAGE)));
58  			}
59  		}
60  	}
61  }

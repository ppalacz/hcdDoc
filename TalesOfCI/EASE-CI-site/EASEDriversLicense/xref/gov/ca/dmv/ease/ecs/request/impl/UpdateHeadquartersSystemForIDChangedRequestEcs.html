59  	@Override
60  	public FireAndForgetReceipt execute() throws EcsServiceException {
61  		FireAndForgetReceipt updateHeadquartersSystemForIDChangedResponseEcs = null;
62  		IErrorCollector errorCollector = getValidator().validate(this);
63  		if (errorCollector.hasErrors()) {
64  			updateHeadquartersSystemForIDChangedResponseEcs = new FireAndForgetReceipt(
65  					errorCollector);
66  		}
67  		else {
68  			//If the execute method throws EcsResponseIsNullException then it is CNA short term message.
69  			//In the catch block we need to register the error to the error collector.
70  			try {
71  				updateHeadquartersSystemForIDChangedResponseEcs = (FireAndForgetReceipt) super
72  						.execute();
73  			}
74  			catch (EcsResponseIsNullException responseIsNullException) {
75  				errorCollector = new ErrorCollector();
76  				errorCollector.register(new EaseValidationException(
77  						IEcsConstants.CNA_SHORT_TERM_MESSAGE));
78  				updateHeadquartersSystemForIDChangedResponseEcs = new FireAndForgetReceipt(
79  						errorCollector);
80  			}
81  			catch (SystemException sysEx) {
82  				errorCollector = new ErrorCollector();
83  				errorCollector.register(new EaseValidationException(
84  						IEcsConstants.RESOURCE_PROBLEM_MESSAGE));
85  				updateHeadquartersSystemForIDChangedResponseEcs = new FireAndForgetReceipt(
86  						errorCollector);
87  			}
88  		}
89  		return updateHeadquartersSystemForIDChangedResponseEcs;
90  	}

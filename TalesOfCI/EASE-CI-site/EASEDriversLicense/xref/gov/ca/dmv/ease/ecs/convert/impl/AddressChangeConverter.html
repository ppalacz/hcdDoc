37  public class AddressChangeConverter extends AbstractEcsMessageConverter {
38  	/*** The TCode. */
39  	private static final String TCODE = "DL7";
40  	/*** The Requester Prefix Constant. */
41  	private static final String REQ_CODE_PREFIX = "80";
42  
43  	@Override
44  	public String createMessage(IEcsRequest ecsRequest) {
45  		if (ecsRequest == null) {
46  			throw new MessageConversionException();
47  		}
48  		AddressChangeRequestEcs request = (AddressChangeRequestEcs) ecsRequest;
49  		List <Object> tcodeDl7RequestElementsList = this
50  				.buildTcodeDl7RequestElementsList(request);
51  		Object[] tcodeDl7RequestElementsArray = tcodeDl7RequestElementsList
52  				.toArray();
53  		return createMessageFormat(tcodeDl7RequestElementsArray);
54  	}
55  
56  	private List <Object> buildTcodeDl7RequestElementsList(
57  			AddressChangeRequestEcs request) {
58  		AddressChangeApplication addressChange = request.getAddressChange();
59  		Person applicant = addressChange.getApplicant();
60  		List <Object> objectList = new ArrayList <Object>(0);
61  		/* TRANSACTION CODE */
62  		objectList.add(filler(TCODE, 3));
63  		/* TYPE TRANSACTION CODE */
64  		objectList.add(filler(request.getUserContext().getTtc(), 3));
65  		/* TRANSACTION SUB CODE */
66  		objectList.add(filler(request.getTransactionSubCode(), 1));
67  		/* REQ_CODE_PREFIX */
68  		objectList.add(filler(REQ_CODE_PREFIX, 2));
69  		/* OFFICE_ID */
70  		objectList.add(filler(request.getUserContext().getOfficeId(), 3));
71  		/* EMPLOYEE ID */
72  		objectList.add(filler(getEmployeeID(request.getUserContext()), 2));
73  		/* DL NUMBER*/
74  		objectList.add(filler(ConverterHelper.getDlorIdNumber(applicant), 8));
75  		/* 3 POSITION NAME */
76  		objectList.add(ConverterHelper.getFirstThreeCharsOfLastName(applicant));
77  		/*MAIL ADDRESS */
78  		objectList.add(filler(ConverterHelper.getAddress(applicant
79  				.getAddress(AddressType.MAILING)), 35));
80  		/* MAIL CITY */
81  		objectList.add(filler(applicant.getAddress(AddressType.MAILING)
82  				.getCity().trim(), 13));
83  		/* ZIP CODE */
84  		objectList.add(filler(applicant.getAddress(AddressType.MAILING)
85  				.getZip().trim(), 5));
86  		/* RESIDNCE ADDRESS */
87  		objectList.add(filler(ConverterHelper.getAddress(applicant
88  				.getAddress(AddressType.RESIDENTIAL)), 35));
89  		/* RESIDENCE CITY */
90  		objectList.add(filler(applicant.getAddress(AddressType.RESIDENTIAL)
91  				.getCity(), 13));
92  		/* COUNTY CODE */
93  		objectList.add(filler(applicant.getAddress(AddressType.RESIDENTIAL)
94  				.getCounty(), 2));
95  		/* EFFECTIVE DATE */
96  		objectList.add(filler(convertDateToString(applicant
97  				.getNewAddressEffectiveDate(AddressType.MAILING),
98  				EcsConverterHelper.FORMATTER_MM_DD_YY), 6));
99  		/* VOTER_REG_INDICATOR */
100 		objectList.add(filler(addressChange.getVoterRegistationRequestedCode()
101 				.getCode(), 1));
102 		return objectList;
103 	}
104

106 	public IEcsResponse createResponse(String anInboundMessagePayload) {

108 	}
109 }

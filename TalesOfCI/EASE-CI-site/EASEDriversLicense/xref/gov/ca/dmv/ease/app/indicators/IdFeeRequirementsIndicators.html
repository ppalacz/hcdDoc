29  public class IdFeeRequirementsIndicators implements IndicatorsConstants {
30  	/***
31  	 * This method processes ID Fee Indicators.
32  	 * 
33  	 * @param currentTtc the Current TTC
34  	 * @param originalTtc the Original TTC
35  	 * @param feeReasonCode the Fee Reason Code
36  	 * @param feeRecievedDate the Fee Received Date
37  	 * @param birthDate the Birth Date
38  	 * @param processDate the Process Date
39  	 * @param limitedTermFeeDueYear the Limited Term Fee Due Year
40  	 * 
41  	 * @return Map with Key as Result Type, values as Fee Indicator & Fee Reason Code
42  	 * <ul>
43  	 * 	<li>Map Contains 2 Keys</li>
44  	 *  <li>FEE_REQ_INDICATOR <code>IndicatorsConstants.FEE_REQ_INDICATOR_KEY</code> - Contains <b>TRUE</b> or <b>FALSE</b></li>
45  	 *  <li>FEE_REASON_CODE <code>IndicatorsConstants.FEE_REASON_CODE_KEY</code> - Containing Fee Reason Code <code>IFeeReasonCodeConstants</code></li>
46  	 * </ul> 
47  	 */
48  	public Map <String, String> processIdFeeRequirements(String currentTtc,
49  			String originalTtc, CodeSetElement feeReasonCode,
50  			Date feeReceivedDate, Date birthDate, Date processDate,
51  			int limitedTermFeeDueYear) {
52  		Map <String, String> idFeeIndicators = new HashMap <String, String>();
53  		boolean feeIndicator = false;
54  		String feeReasonCodeIndicator = null;
55  		boolean isFeeReasonCodeNotL = (EaseUtil.isNullOrBlank(feeReasonCode))
56  				|| (!EaseUtil.isNullOrBlank(feeReasonCode) && !L_LIMITED_TERM
57  						.equalsIgnoreCase(feeReasonCode.getCode()));
58  		Calendar dateFeeCal = Calendar.getInstance();
59  		dateFeeCal.set(2004, Calendar.JANUARY, 01);
60  		if (TTC_IRT.equalsIgnoreCase(currentTtc)) {
61  			feeIndicator = false;
62  			feeReasonCodeIndicator = N_DL_MAILED_NOT_RETURNED_UNCLAIMED;
63  		}
64  		else if (TTC_IDA.equalsIgnoreCase(currentTtc) && isFeeReasonCodeNotL) {
65  			
66  			Calendar birthCal = Calendar.getInstance();
67  			birthCal.setTime(birthDate);
68  			int ddComponent = birthCal.get(Calendar.DATE);
69  			int mmComponent = birthCal.get(Calendar.MONTH);
70  			int yyComponent = limitedTermFeeDueYear;
71  			birthCal.clear();
72  			birthCal.set(yyComponent, mmComponent, ddComponent);
73  
74  			if(!birthCal.getTime().before(processDate))
75  			feeIndicator = true;
76  			feeReasonCodeIndicator = L_LIMITED_TERM;
77  		}
78  		else if (TTC_IDS.equalsIgnoreCase(currentTtc) && isFeeReasonCodeNotL) {
79  			if (feeReceivedDate.before(dateFeeCal.getTime())) {
80  				feeIndicator = true;
81  				//TODO NEED TO SET FEE REASON CODE
82  				feeReasonCodeIndicator = N_DL_MAILED_NOT_RETURNED_UNCLAIMED;
83  			}
84  			else {
85  				feeIndicator = false;
86  				feeReasonCodeIndicator = N_DL_MAILED_NOT_RETURNED_UNCLAIMED;
87  			}
88  		}
89  		else if ((TTC_IDA.equalsIgnoreCase(originalTtc) || TTC_IDS
90  				.equalsIgnoreCase(originalTtc))
91  				&& isFeeReasonCodeNotL) {
92  			feeIndicator = true;
93  			feeReasonCodeIndicator = N_DL_MAILED_NOT_RETURNED_UNCLAIMED;
94  		}
95  		else {
96  			feeIndicator = false;
97  			feeReasonCodeIndicator = N_DL_MAILED_NOT_RETURNED_UNCLAIMED;
98  		}
99  		//Set Return Values to Map
100 		idFeeIndicators.put(FEE_REQ_INDICATOR_KEY, feeIndicator == true ? TRUE
101 				: FALSE);
102 		idFeeIndicators.put(FEE_REASON_CODE_KEY, feeReasonCodeIndicator);
103 		return idFeeIndicators;
104 	}
105 }

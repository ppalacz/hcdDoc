48  	 * 
49  	 * @return the validate automated receipt response
50  	 */
51  	public ValidateAutomatedReceiptResponseBus execute(
52  			ValidateAutomatedReceiptRequestBus busRequest) {
53  		ValidateAutomatedReceiptResponseBus busResponse = null;
54  		ValidateAutomatedReceiptRequestEcs ecsRequest = null;
55  		IBusResponseFactory responseFactory = new BusResponseFactory();
56  		IErrorCollector errorCollector = new ErrorCollector();
57  		try {
58  			// errorCollector = busRequest.validate();
59  			if (errorCollector.hasErrors()) {
60  				busResponse = responseFactory
61  						.createValidateAutomatedReceiptResponseForBus(errorCollector);
62  			}
63  			else {
64  				ecsRequest = ecsRequestFactory
65  						.createValidateAutomatedReceiptRequestFor(busRequest
66  								.getVouchers(), (UserContext) busRequest
67  								.getUserContext());
68  				ValidateAutomatedReceiptResponseEcs ecsResponse = ecsRequest
69  						.execute();
70  				busResponse = responseFactory
71  						.createValidateAutomatedReceiptResponseForBus(ecsResponse
72  								.getVoucherReturnCodes());
73  			}
74  		}
75  		catch (Exception e) {
76  			// Register exception in error collector and assign it to response
77  			// object.
78  			e.printStackTrace();
79  			errorCollector.register(e);
80  			return busResponse = responseFactory
81  					.createValidateAutomatedReceiptResponseForBus(errorCollector);
82  		}
83  		// process the response
84  		return busResponse;
85  	}

46  	/***
47  	 * This method adds a specified string to the message buffer that holds the
48  	 * message text that will be written to the bridge. Only a specified number
49  	 * of characters is added. and any text value that is to be added to the
50  	 * message buffer that is less than the desired length to be added is padded
51  	 * with blank characters, as needed.
52  	 * 
53  	 * @param messageBuffer
54  	 *            The MessageBuffer to be appended with the desired String value
55  	 *            for a specified length.
56  	 * @param stringToAdd
57  	 *            The String value to be added to the message buffer.
58  	 * @param lengthToAdd
59  	 *            The int value of the length to be added.
60  	 */
61  	protected void addStringToMessageText(StringBuilder messageBuffer,
62  			String stringToAdd, int lengthToAdd) {
63  		if (stringToAdd != null) {
64  			if (stringToAdd.length() >= lengthToAdd) {
65  				messageBuffer.append(stringToAdd.substring(0, lengthToAdd));
66  			}
67  			else {
68  				messageBuffer.append(EcsConverterHelper.filler(stringToAdd,
69  						lengthToAdd, EcsConverterHelper.EMPTY_FILLER,
70  						EcsConverterHelper.PAD_RIGHT));
71  			}
72  		}
73  		else {
74  			messageBuffer.append(EcsConverterHelper.filler(
75  					AbstractEcsMessageConverter.BLANK, lengthToAdd,
76  					EcsConverterHelper.EMPTY_FILLER,
77  					EcsConverterHelper.PAD_RIGHT));
78  		}
79  		return;
80  	}
81  
82  	/***
83  	 * This method adds a numeric int value to the message buffer that holds the
84  	 * message text that will be written to the bridge. Only a specified number
85  	 * of characters is added. A value that is less than the desired length to
86  	 * be added is padded with blank characters on the left, as needed. A value
87  	 * that exceeds the specified number of characters is replace with a zero.
88  	 * 
89  	 * @param messageBuffer
90  	 *            The MessageBuffer to be appended with the desired String value
91  	 *            for a specified length.
92  	 * @param stringToAdd
93  	 *            The String value to be added to the message buffer.
94  	 * @param lengthToAdd
95  	 *            The int value of the length to be added.
96  	 */
97  	protected void addNumberToMessageText(StringBuilder messageBuffer,
98  			int numberToAdd, int lengthToAdd) {
99  		String stringToAdd = String.valueOf(numberToAdd);
100 		if (stringToAdd != null) {
101 			if (stringToAdd.length() >= lengthToAdd) {
102 				messageBuffer.append(stringToAdd.substring(0, lengthToAdd));
103 			}
104 			else {
105 				messageBuffer.append(AbstractEcsMessageConverter.filler(
106 						stringToAdd, lengthToAdd,
107 						EcsConverterHelper.EMPTY_FILLER,
108 						EcsConverterHelper.PAD_LEFT));
109 			}
110 		}
111 		else {
112 			messageBuffer.append(AbstractEcsMessageConverter.filler(BLANK,
113 					lengthToAdd, EcsConverterHelper.EMPTY_FILLER,
114 					EcsConverterHelper.PAD_LEFT));
115 		}
116 		return;
117 	}
118 
119 	/***
120 	 * This method adds a specified decimal number to the message buffer that holds the
121 	 * message text that will be written to the bridge. Only a specified number
122 	 * of characters is added. and any text value that is to be added to the
123 	 * message buffer that is less than the desired length to be added is padded
124 	 * with blank characters, as needed. To be used for dollar amounts especially.
125 	 * 
126 	 * @param messageBuffer
127 	 *            The MessageBuffer to be appended with the desired String value
128 	 *            for a specified length.
129 	 * @param stringToAdd
130 	 *            The String value to be added to the message buffer.
131 	 * @param lengthToAdd
132 	 *            The int value of the length to be added.
133 	 */
134 	protected void addBigDecimalToMessageText(StringBuilder messageBuffer,
135 			BigDecimal numberToAdd, int lengthToAdd) {
136 		String stringToAdd = String.valueOf(numberToAdd);
137 		if (stringToAdd != null) {
138 			if (stringToAdd.length() >= lengthToAdd) {
139 				messageBuffer.append(stringToAdd.substring(0, lengthToAdd));
140 			}
141 			else {
142 				messageBuffer.append(EcsConverterHelper.filler(stringToAdd,
143 						lengthToAdd, EcsConverterHelper.EMPTY_FILLER,
144 						EcsConverterHelper.PAD_LEFT));
145 			}
146 		}
147 		else {
148 			messageBuffer.append(EcsConverterHelper.filler(BLANK, lengthToAdd,
149 					EcsConverterHelper.EMPTY_FILLER,
150 					EcsConverterHelper.PAD_LEFT));
151 		}
152 		return;
153 	}

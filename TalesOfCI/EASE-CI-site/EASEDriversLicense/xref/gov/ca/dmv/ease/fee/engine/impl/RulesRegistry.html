138 	/***
139 	 * Compute fee.
140 	 *
141 	 * @param request the request
142 	 * @param appFees the app fees
143 	 * @return the list
144 	 */
145 	public List <ApplicableFee> computeFee(IFeeRequest request,
146 			List <ApplicableFee> appFees) {
147 		List <TtcFee> fees = new ArrayList <TtcFee>();
148 		String key = request.getMyClassifierString();
149 		if (request instanceof DlaFeeRequest) {
150 			fees = DLARULESREGISTER.get(key);
151 		}
152 		else if (request instanceof CommercialDlDlpFeeRequest) {
153 			fees = COMMERCIALDLPDLRULESREGISTER.get(key);
154 		}
155 		else if (request instanceof NonCommercialDlDlpFeeRequest) {
156 			fees = NONCOMMERCIALDLPDLRULESREGISTER.get(key);
157 		}
158 		else if (request instanceof DlcFeeRequest) {
159 			fees = DLCRULESREGISTER.get(key);
160 		}
161 		else if (request instanceof DleFeeRequest) {
162 			fees = DLERULESREGISTER.get(key);
163 		}
164 		else if (request instanceof DldFeeRequest) {
165 			fees = DLDRULESREGISTER.get(key);
166 		}
167 		if (fees != null) {
168 			for (TtcFee fee : fees) {
169 				ApplicableFee appfee = new ApplicableFee();
170 				appfee.setFeeCode(fee.getFeeCode());
171 				appfee.setDescription(fee.getDescription());
172 				appfee.setFee(this.getFee(fee.getFeeCode(), request
173 						.getEffectiveDate()));
174 				appFees.add(appfee);
175 			}
176 			return appFees;
177 		}
178 		else {
179 			return null;
180 		}
181 	}

227 	/***
228 	 * Construct fees register.
229 	 */
230 	private void constructFeesRegister() {
231 		TreeMap <Date, java.math.BigDecimal> dateRateMap;
232 		for (FeeRate feeRate : this.FEERATES) {
233 			String feeCode = feeRate.getRateCode();
234 			Date feeDate = feeRate.getEffDate();
235 			java.math.BigDecimal fees = feeRate.getFeeVal();
236 			dateRateMap = this.FEESREGISTER.get(feeCode);
237 			if (dateRateMap == null) {
238 				dateRateMap = new TreeMap <Date, java.math.BigDecimal>();
239 			}
240 			dateRateMap.put(feeDate, fees);
241 			this.FEESREGISTER.put(feeCode, dateRateMap);
242 		}
243 	}

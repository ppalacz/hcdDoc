29  	@Override
30  	public boolean evaluate(EliRecord eliRecord, ProcessContext processContext,
31  			String code) {
32  		if (isSingleHit(eliRecord)) {
33  			if (eliRecord.getNonCommercialDriveRetestFee() != null
34  					&& eliRecord.getNonCommercialDriveRetestFee()
35  							.hasCodeEqualTo(N)) {
36  				// If NON-COML"DRIVE RETEST FEE is N.
37  				if (eliRecord.getDriverLicense() == null
38  						|| EaseUtil.isNullOrBlank(eliRecord
39  								.getEchoedDriverLicense()
40  								.getAppliedLicenseClasses())
41  						|| (eliRecord.getEchoedDriverLicense()
42  								.getAppliedLicenseClasses().size() == 1 && !ArrayUtils
43  								.contains(eliRecord.getEchoedDriverLicense()
44  										.getAppliedLicenseClasses(), F))) {
45  					// If inquiry class is not present or it is F.
46  					if (eliRecord.getInvoiceItem() == null
47  							|| eliRecord.getInvoiceItem().getDriverLicenceFee() == null
48  							|| eliRecord
49  									.getInvoiceItem()
50  									.getDriverLicenceFee()
51  									.getDriverLicenseRateStatusForExpectedFeeRateCode(
52  											COML_DL_DRIVE_TEST_FEE_IND) == null
53  							|| isCommDriveTestFeeNotUsed(eliRecord)) {
54  						// If CDL DRIVE TEST FEE USED INDICATOR is N or blank.
55  						return Boolean.TRUE;
56  					}
57  				}
58  			}
59  		}
60  		return Boolean.FALSE;
61  	}

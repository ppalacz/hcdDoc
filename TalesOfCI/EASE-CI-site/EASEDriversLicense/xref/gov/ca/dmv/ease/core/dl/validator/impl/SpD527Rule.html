32  public class SpD527Rule extends AbstractSpRule {
33  	/* (non-Javadoc)
34  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)

36  	public void validate(ProcessContext processContext) {
37  		SpProcessContext spProcessContext = (SpProcessContext) processContext;
38  		SalesPersonApplication spApplication = spProcessContext
39  				.getSalesPersonApplication();
40  		Action selectedAction = spProcessContext.getSelectedAction();
41  		IErrorCollector errorCollector = new ErrorCollector();
42  		if (selectedAction != null
43  				&& selectedAction == getActionsRegistry().getAction(
44  						IActionsRegistry.ENTER_ACTION)) {
45  			String ttc = spProcessContext.getUserContext().getTtc();
46  			
47  			validateFeeReqAndReasonCode(errorCollector, spApplication, ttc);
48  		}
49  		setErrorCollector(errorCollector);
50  	}
51  
52  	/***
53  	 * @param errorCollector
54  	 * @param spProcessContext
55  	 * @param currentTtc
56  	 */
57  	public void validateFeeReqAndReasonCode(IErrorCollector errorCollector,
58  			SalesPersonApplication spApplication, String currentTtc) {
59  		boolean feeRequired = false;
60  		String reasonCode = null;
61  		if (!isNullOrBlank(spApplication)) {
62  			feeRequired = spApplication.isFeeRequired();
63  			if (!isNullOrBlank(spApplication.getFeeRequiredReasonCode())) {
64  				reasonCode = spApplication.getFeeRequiredReasonCode().getCode();
65  			}
66  		}
67  		if (!feeRequired) {
68  			//F004
69  			if (isNullOrBlank(reasonCode)
70  					&& (ITtcConstants.SPD.equals(currentTtc)
71  							|| ITtcConstants.SPO.equals(currentTtc)
72  							|| ITtcConstants.SPR.equals(currentTtc) || ITtcConstants.SPX
73  							.equals(currentTtc))) {
74  				errorCollector
75  						.register(new EaseValidationException(
76  								IFeeAndPaymentErrorMessageConstants.REASON_INVALID_FOR_FEE));
77  			}
78  			else {
79  				//D527
80  				if (ITtcConstants.SPD.equals(currentTtc)
81  						|| ITtcConstants.SPR.equals(currentTtc)
82  						|| ITtcConstants.SPX.equals(currentTtc)) {
83  					//TODO - check for the valid reason code and register the D527 exception
84  					errorCollector
85  							.register(new EaseValidationException(
86  									IDataCollectionErrorMessageConstants.INVALID_FEE_INDICATOR));
87  				}
88  			}
89  		}
90  	}

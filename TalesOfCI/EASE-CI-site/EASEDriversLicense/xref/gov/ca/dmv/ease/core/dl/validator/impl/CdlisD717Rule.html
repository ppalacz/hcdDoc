42  	/***
43  	 * D717. For TTC = DLP When processing a Pending application (DLP) and a
44  	 * name, birth date, gender, or DL # was changed in the initial process on a
45  	 * DL application then the System will remind the user to make the same
46  	 * change to the ID
47  	 * 
48  	 * @param ttc
49  	 *            transaction code
50  	 * @param eliRecord
51  	 *            data from storage
52  	 * @param applicant
53  	 *            applicant data
54  	 * @param changedBirthDate
55  	 *            keyed in birthdate
56  	 * @param changedGender
57  	 *            keyed in gender change
58  	 * @param driverLicense
59  	 *            driver Lic data
60  	 * @param errorCollector
61  	 *            the error collector
62  	 */
63  	public void D717(String ttc,EliRecord eliRecord, Person applicant, Date changedBirthDate, CodeSetElement changedGender, DriverLicense driverLicense,
64  			IErrorCollector errorCollector) {
65  		IdCard idCard = null;
66  			if (!EaseUtil.isNullOrBlank(eliRecord) ) {
67  				idCard = eliRecord.getPerson()
68  						.getIdCard();
69  			}
70  		if ("DLP".equalsIgnoreCase(ttc)
71  				&& idCard != null
72  				&& (!applicant.getBirthDate()
73  						.equals(changedBirthDate))) {
74  			errorCollector
75  					.register(new EaseValidationException(
76  							IDataCollectionErrorMessageConstants.CHANGES_MADE_IN_INITIAL_PROCESS_REQUIRE_CHANGES_ON_ID));
77  		}
78  		if ("DLP".equalsIgnoreCase(ttc)
79  				&& idCard != null
80  				&& (!applicant.getGenderCode()
81  						.equals(changedGender))) {
82  			errorCollector
83  					.register(new EaseValidationException(
84  							IDataCollectionErrorMessageConstants.CHANGES_MADE_IN_INITIAL_PROCESS_REQUIRE_CHANGES_ON_ID));
85  		}
86  		if ("DLP".equalsIgnoreCase(ttc)
87  				&& idCard != null
88  				&& driverLicense != null
89  				&& (!driverLicense.equals(
90  						idCard.getLicenseNumber()))) {
91  			errorCollector
92  					.register(new EaseValidationException(
93  							IDataCollectionErrorMessageConstants.CHANGES_MADE_IN_INITIAL_PROCESS_REQUIRE_CHANGES_ON_ID));
94  		}
95  
96  	}

44  	private static final Logger LOGGER = Logger
45  			.getLogger(CdlisNdrReportDocumentConverter.class);

47  	/***
48  	 * Populate the outgoing XML message for CDLIS-NDR Matches Report Document.
49  	 * 
50  	 * @param requestFromClient the request from client
51  	 * @return the out-bound XML message
52  	 */
53  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
54  		if (requestFromClient == null) {
55  			throw new MessageConversionException(
56  					IPrintServiceErrorMessageConstants.NULL_CDLIS_NDR_MATCHES_REQUEST);
57  		}
58  		CdlisNdrReportDocumentRequestTus request = (CdlisNdrReportDocumentRequestTus) requestFromClient;
59  		IUserContext userContext = request.getUserContext();
60  		DlApplication dlApplication = request.getDlApplication();
61  		Person person = dlApplication.getApplicant();
62  		List <String> elementsList = new ArrayList <String>();
63  		addToElementsList(elementsList,
64  				isTrainingEnvironment(userContext) ? TRAINING_HEADER : BLANK,
65  				IPrintConverterConstants.CDLIS_TRAINING_HEADER_ELEMENT);
66  		addToElementsList(elementsList, dlApplication.getApplicationType()
67  				.getCode(), IPrintConverterConstants.CDLIS_APPLICATION_TYPE);
68  		addToElementsList(elementsList, getDlorIdNumber(person),
69  				IPrintConverterConstants.CDLIS_DRIVER_LICENSE_ID_NUMBER);
70  		addToElementsList(elementsList, getFormattedCurrentName(person),
71  				IPrintConverterConstants.CDLIS_CURRENT_NAME);
72  		addToElementsList(elementsList, convertDateToString(person
73  				.getBirthDate(), FORMATTER_MM_dd_yy),
74  				IPrintConverterConstants.CDLIS_BIRTH_DATE);
75  		addToElementsList(elementsList, person.getSocialSecurityDocument()
76  				.getSsn(), IPrintConverterConstants.CDLIS_SSN);
77  		addToElementsList(elementsList, convertDateToString(dlApplication
78  				.getApplicationDate()),
79  				IPrintConverterConstants.CDLIS_APPLICATION_DATE);
80  		List <CdlisRecord> cdlisRecords = dlApplication.getCdlisRecords();
81  		if (cdlisRecords != null && cdlisRecords.size() > 0) {
82  			for (int count = 0; count < 15; count++) {
83  				CdlisRecord cdlisRecord = cdlisRecords.get(count);
84  				if (count < 9) {
85  					addToElementsList(elementsList, cdlisRecord
86  							.getDesignateAction().getCode(),
87  							IPrintConverterConstants.RECORD_TYPE + "0"
88  									+ ++count);
89  					addToElementsList(elementsList, cdlisRecord
90  							.getCurrentPrimaryDlNumberState().getCode(),
91  							IPrintConverterConstants.STATE_CODE + "0" + count);
92  					addToElementsList(elementsList, cdlisRecord
93  							.getLicenseNumber(),
94  							IPrintConverterConstants.STATE_DL + "0" + count);
95  					count--;
96  				}
97  				else {
98  					addToElementsList(elementsList, cdlisRecord
99  							.getDesignateAction().getCode(),
100 							IPrintConverterConstants.RECORD_TYPE + ++count);
101 					addToElementsList(elementsList, cdlisRecord
102 							.getCurrentPrimaryDlNumberState().getCode(),
103 							IPrintConverterConstants.STATE_CODE + count);
104 					addToElementsList(elementsList, cdlisRecord
105 							.getLicenseNumber(),
106 							IPrintConverterConstants.STATE_DL + count);
107 					count--;
108 				}
109 			}
110 			if (cdlisRecords.size() > 15) {
111 				addToElementsList(elementsList, (cdlisRecords.size() - 15)
112 						+ IPrintConverterConstants.CDLIS_RECORD_MSG,
113 						IPrintConverterConstants.CDLIS_OVERFLOW_MESSAGE);
114 			}
115 		}
116 		addToElementsList(elementsList, userContext.getOfficeId(),
117 				IPrintConverterConstants.CDLIS_OFFICE_ID);
118 		addToElementsList(elementsList, convertDateToString(Calendar
119 				.getInstance().getTime(), FORMATTER_MM_dd_yy),
120 				IPrintConverterConstants.CDLIS_CURRENT_SYSTEM_DATE);
121 		addToElementsList(elementsList, convertDateToString(Calendar
122 				.getInstance().getTime(), FORMATTER_HH_mm),
123 				IPrintConverterConstants.CDLIS_CURRENT_TIME);
124 		String xmlData = createXmlForSingleDocument(elementsList,
125 				IPrintConverterConstants.DOCUMENT_NAME);
126 		LOGGER.info("xml : " + xmlData);
127 		return xmlData;
128 	}

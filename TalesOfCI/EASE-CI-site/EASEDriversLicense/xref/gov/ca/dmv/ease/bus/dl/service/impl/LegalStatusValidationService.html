29  public class LegalStatusValidationService {
30  	
31  	/*** The ecs request factory. */
32  	IEcsRequestFactory ecsRequestFactory;
33  	/*** The ECS request factory */
34  
35  	/***
36  	 * @param ecsRequestFactory the ecsRequestFactory to set
37  	 */
38  	public void setEcsRequestFactory(IEcsRequestFactory ecsRequestFactory) {
39  		this.ecsRequestFactory = ecsRequestFactory;
40  	}
41  
42  	/***
43  	 * This method inquire the legal presence of customers by calling DCS
44  	 * service to validate the legal presence with UNI.
45  	 * 
46  	 * @param busRequest
47  	 *            the validate legal presence request
48  	 * 
49  	 * @return Response returns information associated with the call to UNI by
50  	 *         DCS Service. The Response object contains status of communication
51  	 *         success or failure with CAMVDCS, and if the communication is
52  	 *         successful, contains the Customer and Supporting Document objects
53  	 *         and verification number.
54  	 */
55  	public ValidateLegalPresenceResponseBus execute(
56  			ValidateLegalPresenceRequestBus busRequest) {
57  		ValidateLegalPresenceResponseBus busResponse = null;
58  		LegalPresenceVerificationRequestEcs ecsRequest = null;
59  		LegalPresenceVerificationResponseEcs ecsResponse = null;
60  		IBusDlResponseFactory responseFactory = new BusDlResponseFactory();
61  		IErrorCollector errorCollector = new ErrorCollector();
62  		try {
63  			errorCollector = busRequest.validate();
64  			if (errorCollector.hasErrors()) {
65  				// Set the Validation Messages to Business Utility Service Response
66  				busResponse = responseFactory
67  						.createValidateLegalPresenceResponseForBus(errorCollector);
68  			}
69  			else {
70  				ecsRequest = ecsRequestFactory
71  						.createValidateLegalPresenceRequestFor(busRequest
72  								.getDlApplication(), busRequest
73  								.getUserContext());
74  				ecsResponse = ecsRequest.execute();
75  				if (ecsResponse.hasError()) {
76  					busResponse = responseFactory
77  							.createValidateLegalPresenceResponseForBus(ecsResponse
78  									.getErrorCollector());
79  				}
80  				else {
81  					busResponse = responseFactory
82  							.createValidateLegalPresenceResponseForBus(ecsResponse
83  									.getLpvRecord());
84  				}
85  			}
86  		}
87  		catch (Exception e) {
88  			//Register exception in error collector and assign it to response object.
89  			errorCollector.register(e);
90  			return busResponse = responseFactory
91  					.createValidateLegalPresenceResponseForBus(errorCollector);
92  		}
93  		return busResponse;
94  	}
95  }

28  public class UpdateHomeAddressConfidentialityRequestEcsValidator extends
29  		EcsRequestValidator implements IRequestValidator {
30  	/*** The Constant serialVersionUID. */
31  	private static final long serialVersionUID = -8342637634734527351L;
32  	/*** The Request. */
33  	private static final String REQUEST = "UpdateProofOfInsuranceInformationRequestEcs";
34  	/*** The Converter. */
35  	private static final String CONVERTER = "UpdateHomeAddressConfidentialityConverter";
36  
37  	/* (non-Javadoc)
38  	 * @see gov.ca.dmv.ease.ecs.request.validator.impl.EcsRequestValidator#validateRequest(gov.ca.dmv.ease.ecs.request.IEcsRequest, gov.ca.dmv.ease.fw.error.IErrorCollector)
39  	 */
40  	@Override
41  	protected void validateRequest(IEcsRequest aRequest,
42  			IErrorCollector aCollector) {
43  		if(!EaseUtil.isNotNull(aRequest)) {
44  			addValidationExceptionInErrorCollector(
45  					NULL_REQUEST,
46  					new String[] { REQUEST }, aCollector);
47  			return;
48  		}
49  		if (!(aRequest instanceof UpdateHomeAddressConfidentialityRequestEcs)) {
50  			addValidationExceptionInErrorCollector(
51  					REQUEST_IS_NOT_OF_THE_TYPE_ERROR_MESSAGE,
52  					new String[] { REQUEST }, aCollector);
53  		}
54  		else {
55  			if (!(aRequest.getMessageConverter() instanceof UpdateHomeAddressConfidentialityConverter)) {
56  				this.addValidationExceptionInErrorCollector(
57  						IMPROPER_MESSAGE_CONVERTER_ERROR_MESSAGE,
58  						new String[] { CONVERTER }, aCollector);
59  			}
60  			else {
61  				UpdateHomeAddressConfidentialityRequestEcs requestEcs = (UpdateHomeAddressConfidentialityRequestEcs) aRequest;
62  				IUserContext userContext = requestEcs.getUserContext();
63  				if(!EaseUtil.isNotNull(userContext)) {
64  					addValidationExceptionInErrorCollector(
65  							USER_CONTEXT_IS_NULL, aCollector);
66  					return;
67  				}
68  				if(!EaseUtil.isNotNull(userContext.getOfficeId())) {
69  					addValidationExceptionInErrorCollector(
70  							USER_CONTEXT_OFFICE_ID_IS_NULL, aCollector);
71  					return;
72  				}
73  				if(!EaseUtil.isNotNull(userContext.getTechId())) {
74  					addValidationExceptionInErrorCollector(
75  							USER_CONTEXT_TECH_ID_IS_NULL, aCollector);
76  					return;
77  				}
78  				if(!EaseUtil.isNotNull(requestEcs.getVehicleFileCode())) {
79  					addValidationExceptionInErrorCollector(
80  							FILE_CODE_IS_EMPTY_ERROR_MESSAGE, aCollector);
81  					return;
82  				}
83  				if(!EaseUtil.isNotNull(requestEcs.getVehicleLicenseNumber())) {
84  					addValidationExceptionInErrorCollector(
85  							VEHICLE_LICENSE_NUMBER_IS_EMPTY_ERROR_MESSAGE, aCollector);
86  					return;
87  				}
88  				if(!EaseUtil.isNotNull(requestEcs.getRegisteredOwnerName())) {
89  					addValidationExceptionInErrorCollector(
90  							REGISTERED_OWNER_NAME_IS_EMPTY_ERROR_MESSAGE, aCollector);
91  					return;
92  				}
93  				if(!EaseUtil.isNotNull(requestEcs.getEmployingAgencyName())) {
94  					addValidationExceptionInErrorCollector(
95  							EMPLOYING_AGENCY_NAME_IS_EMPTY_ERROR_MESSAGE, aCollector);
96  					return;
97  				}
98  				if(!EaseUtil.isNotNull(requestEcs.getRequestedFor())) {
99  					addValidationExceptionInErrorCollector(
100 							HOME_ADDRESS_CONFIDENTIALITY_REQUESTED_FOR_CODE_IS_EMPTY_ERROR_MESSAGE, aCollector);
101 					return;
102 				}
103 			}
104 		}
105 	}

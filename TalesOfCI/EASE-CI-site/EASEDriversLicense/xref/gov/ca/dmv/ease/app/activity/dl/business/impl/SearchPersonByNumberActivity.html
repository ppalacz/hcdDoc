36  public class SearchPersonByNumberActivity extends BusinessActivity {
37  	/*** The Constant serialVersionUID. */
38  	private static final long serialVersionUID = -8104641114354433900L;
39  	/*** The issue dl task service. */
40  	private IssueDlTaskService issueDlTaskService;
41  
42  	/***
43  	 * This method creates SearchPersonByNumberRequestTs request class with
44  	 * required business objects from the process context using factory and
45  	 * calls DL Task Service to search person by license number.
46  	 * 
47  	 * Once the response is received from the Task Service it maps the business
48  	 * objects to the process context.
49  	 * 
50  	 */
51  	@Override
52  	protected void invokeService(ProcessContext processContext) {
53  		HqInquiryProcessContext hqInquiryProcessContext = (HqInquiryProcessContext) processContext;
54  		DriverLicense license = hqInquiryProcessContext.getDriverLicense();
55  		IDlTaskServiceRequestFactory reqFactory = new DlTaskServiceRequestFactory();
56  		IErrorCollector errorCollector = new ErrorCollector();
57  		SearchPersonByNumberRequestTs request = reqFactory
58  				.createSearchPersonByNumberRequest(license, processContext
59  						.getUserContext());
60  		SearchPersonByNumberResponseTs response = issueDlTaskService
61  				.execute(request);
62  		if (response.hasErrors()) {
63  			populateErrorMessages(processContext, response.getErrorCollector());
64  			// TODO 'UP','PE','01','03','06','07'
65  			// TODO - Need to check the flag that will be set for DL CNA
66  			hqInquiryProcessContext.setDlAvailable(false);
67  		}
68  		else if (response.getEliRecord() != null) {
69  			// SINGLE HIT
70  			EliRecord eliRecord = response.getEliRecord();
71  			if (!EaseUtil.isNullOrBlank(eliRecord)
72  					&& !EaseUtil.isNullOrBlank(eliRecord.getDriverLicense())
73  					&& hqInquiryProcessContext.getDlaStartedwith()
74  							.equalsIgnoreCase("SearchPersonByNumber")
75  					&& EaseUtil.isNullOrBlank(eliRecord.getDriverLicense()
76  							.getLicenseNumber())) {
77  				errorCollector = new ErrorCollector();
78  				errorCollector.register(new ServiceRequestExecutionException(
79  						"Q632"));
80  				populateErrorMessages(processContext, errorCollector);
81  			}
82  			else {
83  				hqInquiryProcessContext.setEliRecord(eliRecord);
84  				DlaProcessContextHelper.copyEliRecord(hqInquiryProcessContext,
85  						eliRecord);
86  			}
87  		}
88  		else if (response.getEliRecord() == null) {
89  			if (EaseUtil
90  					.isNotBlank(hqInquiryProcessContext.getDlaStartedwith())
91  					&& hqInquiryProcessContext.getDlaStartedwith()
92  							.equalsIgnoreCase("SearchPersonByNumber")) {
93  				errorCollector = new ErrorCollector();
94  				errorCollector.register(new ServiceRequestExecutionException(
95  						"Q632"));
96  				populateErrorMessages(processContext, errorCollector);
97  			}
98  		}
99  		else {
100 			// TODO ' ','CE','NH','ER'
101 			// no place holders for the above error codes
102 		}
103 	}

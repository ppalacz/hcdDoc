44  	@Override
45  	protected void postSubprocessExecute(ProcessContext processContext,
46  			ProcessContext subprocessContext) {
47  		//This need separate sub process activity such as SPInquirySubProcessActivity, time being it is handled in here.....
48  		if (processContext instanceof SpProcessContext) {
49  			SpProcessContext spProcessContext = ((SpProcessContext) processContext);
50  			DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) subprocessContext;
51  			spProcessContext.setDlApplication(dlFeeProcessContext.getDlApplication());
52  			spProcessContext.setSalesPersonApplication(dlFeeProcessContext
53  					.getSalesPersonApplication());
54  		}
55  		else if (processContext instanceof DlaProcessContext) {
56  			DlaProcessContext dlaProcessContext = ((DlaProcessContext) processContext);
57  			DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) subprocessContext;
58  			dlaProcessContext.setDlApplication(dlFeeProcessContext
59  					.getDlApplication());
60  		}
61  		
62  	}

66  	 */
67  	@Override
68  	protected void preSubprocessExecute(ProcessContext processContext,
69  			ProcessContext subProcessContext) {
70  		if (processContext instanceof SpProcessContext) {
71  			SpProcessContext spProcessContext = ((SpProcessContext) processContext);
72  			((DlFeeProcessContext) subProcessContext)
73  					.setSalesPersonApplication(spProcessContext
74  							.getSalesPersonApplication());
75  		}
76  		else if (processContext instanceof DlaProcessContext) {
77  			DlaProcessContext dlaProcessContext = (DlaProcessContext) processContext;
78  			((DlFeeProcessContext) subProcessContext)
79  					.setDlApplication(dlaProcessContext.getDlApplication());

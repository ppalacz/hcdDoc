48  	/***
49  	 * This method validates Person<code>Person</code> and List of <code>DlAgeRequirment</code>.
50  	 * 
51  	 * @param validatable The Validate
52  	 * @param validatables The Validatables
53  	 * @return <code>IErrorCollector</code> containing Validation Messages.
54  	 * @see IErrorCollector
55  	 * @see Person
56  	 * @see DlAgeRequirment
57  	 */
58  	public IErrorCollector validate(IValidatable validatable,
59  			List <IValidatable> validatables) {
60  		IErrorCollector collector = new ErrorCollector();
61  		try {
62  			if (isNullOrBlank(validatable)) {
63  				collector.register(new EaseValidationException(SpringUtils
64  						.getMessage(VALIDATABLE_IS_NULL_ERROR_MESSAGE)));
65  				return collector;
66  			}
67  			else if (isNullOrBlank(validatables)) {
68  				collector.register(new EaseValidationException(SpringUtils
69  						.getMessage(VALIDATABLE_IS_NULL_ERROR_MESSAGE)));
70  				return collector;
71  			}
72  			else if (validatable instanceof Person) {
73  				Person person = (Person) validatable;
74  				List <DlAgeRequirement> dlAgeRequirementList = new ArrayList <DlAgeRequirement>();
75  				for (IValidatable iValidatable : validatables) {
76  					dlAgeRequirementList.add((DlAgeRequirement) iValidatable);
77  				}
78  				// Person is provided
79  				if (isNullOrBlank(person.getCurrentName())) {
80  					collector
81  							.register(new EaseValidationException(
82  									SpringUtils
83  											.getMessage(PERSON_CURRENT_NAME_IS_EMPTY_ERROR_MESSAGE)));
84  				}
85  				if (isNullOrBlank(person.getBirthDate())) {
86  					collector
87  							.register(new EaseValidationException(
88  									SpringUtils
89  											.getMessage(PERSON_BIRTHDATE_IS_NULL_ERROR_MESSAGE)));
90  				}
91  				//Check if Person is above 15.5 Years
92  				boolean isProcess = true;
93  				if (!isNullOrBlank(person.getDriverLicense()
94  						.getAppliedLicenseClasses())) {
95  					//Iterate through all the License Classes
96  					for (CodeSetElement licenseClass : person
97  							.getDriverLicense().getAppliedLicenseClasses()) {
98  						if (isProcess) {
99  							//Iterate through the DL Age Static Data
100 							for (DlAgeRequirement dlAgeReq : dlAgeRequirementList) {
101 								if (isProcess) {
102 									if (dlAgeReq.getLicenseClass()
103 											.equalsIgnoreCase(
104 													licenseClass.getCode())) {
105 										//Fetch the corresponding Age for the License Class and Validate with BirthDate for Minor Check
106 										if ((person.getBirthDate() != null)
107 												&& (DateUtil
108 														.isMinorForSpecifiedAgeLimit(
109 																person
110 																		.getBirthDate(),
111 																dlAgeReq
112 																		.getYear(),
113 																dlAgeReq
114 																		.getMonth()))) {
115 											collector
116 													.register(new EaseValidationException(
117 															SpringUtils
118 																	.getMessage(PERSON_IS_A_MINOR_ERROR_MESSAGE)));
119 											//Set only 1 Error Message in case of DL Application for 2 License Class
120 											isProcess = false;
121 										}
122 									}
123 								}
124 							}
125 						}
126 					}
127 				}
128 			}
129 		}
130 		catch (Exception e) {
131 			//Register exception in error collector and assign it to response object.
132 			collector.register(new PersonInquiryRequestValidationException(e));
133 		}
134 		return collector;
135 	}

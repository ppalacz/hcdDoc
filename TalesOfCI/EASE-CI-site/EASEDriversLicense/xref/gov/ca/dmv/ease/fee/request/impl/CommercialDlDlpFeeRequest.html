132 		String ruleAppFeeRequired = appFeeRequired.type();
133 		String ruleDriveTestFeeRequiredType = CLASSIFIERTYPES.DRIVETESTFEEREQD
134 				.type();
135 		String ruleDriveTestFeeRequired = driveTestFeeRequired.type();
136 		String ruleDriveRetestFeeRequiredType = CLASSIFIERTYPES.DRIVERETESTFEEREQD

139 		String ruleDriveRetestFeeUsedType = CLASSIFIERTYPES.DRIVERETESTFEEUSED
140 				.type();
141 		String ruleDriveRetestFeeUsed = driveRetestFeeUsed.type();
142 		strBuilder.append(ruleLicense1Type).append(EQUALS).append(ruleLicense1)
143 				.append(COLONDELIMITER);
144 		strBuilder.append(ruleLicense2Type).append(EQUALS).append(ruleLicense2)
145 				.append(COLONDELIMITER);
146 		strBuilder.append(ruleOriginalRenewalType).append(EQUALS).append(
147 				ruleOriginalRenewal).append(COLONDELIMITER);
148 		strBuilder.append(ruleAppFeeRequiredType).append(EQUALS).append(
149 				ruleAppFeeRequired).append(COLONDELIMITER);
150 		strBuilder.append(ruleDriveTestFeeRequiredType).append(EQUALS).append(
151 				ruleDriveTestFeeRequired).append(COLONDELIMITER);
152 		strBuilder.append(ruleDriveRetestFeeRequiredType).append(EQUALS)
153 				.append(ruleDriveRetestFeeRequired).append(COLONDELIMITER);
154 		strBuilder.append(ruleDriveRetestFeeUsedType).append(EQUALS).append(
155 				ruleDriveRetestFeeUsed);
156 		String keyClassifier = strBuilder.toString();
157 		return keyClassifier;

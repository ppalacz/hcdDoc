69  	@Override
70  	public String createMessage(IEcsRequest request) {
71  		UpdateMedicalReportRequestEcs updateMedicalReportRequestEcs = (UpdateMedicalReportRequestEcs) request;
72  		IUserContext userContext = updateMedicalReportRequestEcs
73  				.getUserContext();
74  		String dlorIdNumber = updateMedicalReportRequestEcs.getDlOrIdNumber();
75  		String lastName = updateMedicalReportRequestEcs.getLastName();
76  		Date medicalExpirationDate = updateMedicalReportRequestEcs
77  				.getMedicalExpirationDate();
78  		CodeSetElement attachmentCode = updateMedicalReportRequestEcs
79  				.getAttachmentCode();
80  		Object[] values = new Object[21];
81  		values[0] = TCODE; 									  //MUKTCODE	3	TRANSACTION CODE                 
82  		values[1] = SBATCODE_CODE; 							  //MUKTCODE+3	3	3270 Code
83  		values[2] = REQUESTER_CODE; 						  //MUKREQCD	5	REQUESTER CODE        
84  		values[3] = SBAREQCD_CODE; 							  //MUKREQCD+5	3	3270 Code
85  		values[4] = filler(getEmployeeID(userContext), 3); 	  //MUKOPRID	3	OPERATOR ID           
86  		values[5] = SBAOPRID_CODE; 							  //MUKOPRID+3	3	3270 Code
87  		values[6] = filler(dlorIdNumber, 8); 				  //MUKDLNUM	8	DL NUMBER             
88  		values[7] = SBADLNUM_CODE; 							  //MUKDLNUM+8	3	3270 Code
89  		values[8] = filler(lastName.substring(0, 3), 3); 	  //MUK3LNME	3	3 POS LAST NAME       
90  		values[9] = SBA3LNME_CODE; 							  //MUK3LNME+3	3	3270 Code
91  		values[10] = FIRST_INPUT_CODE;						  //MUK1STIC	1	1ST INPUT CODE        
92  		values[11] = SBA1STIC_CODE; 						  //MUK1STIC+1	3	3270 Code
93  		values[12] = FIRST_FIELD_IDENTIFIER; 				  //MUK1STFI	1	1ST FIELD IDENTIFIER  
94  		values[13] = SBA1STFI_CODE; 						  //MUK1STFI+1	3	3270 Code
95  		if (EaseUtil.isNullOrBlank(attachmentCode)) {
96  			values[14] = filler(BLANK, 2); 					  //MUKATTCH	2	ATTACHMENT CODE
97  		}
98  		else {
99  			values[14] = filler(attachmentCode.getCode(), 2); //MUKATTCH	2	ATTACHMENT CODE
100 		}
101 		values[15] = SBAATTCH_CODE; 						  //MUKATTCH+2	3	3270 Code
102 		values[16] = SECOND_INPUT_CODE; 					  //MUK2NDIC	1	2ND INPUT CODE        
103 		values[17] = SBA2NDIC_CODE; 						  //MUK2NDIC+1	3	3270 Code
104 		values[18] = SECOND_FIELD_IDENTIFIER; 				  //MUK2NDFI	1	2ND FIELD IDENTIFIER  
105 		values[19] = SBA2NDFI_CODE; 						  //MUK2NDFI+1	3	3270 Code
106 		values[20] = 
107 		        filler(convertDateToString(medicalExpirationDate), 6); 
108 															  //MUKMDATE	6	MEDICAL DATE          
109 		return createMessageFormat(values);
110 	}

117 		if (isNullOrBlank(responseMessage)) {
118 			throw new MessageConversionException(
119 					"Invalid Response: Unable to process on null message");
120 		}
121 		UpdateMedicalReportResponseEcs response = new UpdateMedicalReportResponseEcs();
122 		try
123 		{
124 			byte[] asciiToLatinBytes = responseMessage.getBytes("8859_1");
125 			String convertedResponse = new String(asciiToLatinBytes, "8859_1");
126 			System.out.println(convertedResponse);
127 			//Set Response Message
128 			response.setResponseMessage(responseMessage.substring(0, 18));
129 			if (responseMessage.length() >= 61) {
130 				response.setMedicalExpirationDate(convertStringToDate(
131 						responseMessage.substring(55, 61),
132 						EcsConverterHelper.FORMATTER_MM_DD_YY));
133 			}
134 		}
135 		catch (Exception e) {
136 			response.getErrorCollector().register(new MessageConversionException("Ascii to UTF Conversion Error"));
137 		}
138 		
139 		return response;

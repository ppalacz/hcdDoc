69  	/***
70  	 * Instantiates a new dl payment receipt converter.
71  	 * 
72  	 * @param userContext the user context
73  	 * @param appln the appln
74  	 * @param invoice the invoice
75  	 */
76  	public DlPaymentReceiptConverter(IUserContext userContext,
77  			DlApplication appln, Invoice invoice) {
78  		// set default template;
79  		setTemplate(new DlPaymentReceiptTemplate());
80  		dlNumber = appln.getLicense().getLicenseNumber();
81  		dlClass = appln.getLicense().getExistingLicenseClasses().get(0).getCode();
82  		Person person = appln.getApplicant();
83  		firstName = person.getCurrentName().getFirstName();
84  		middleName = person.getCurrentName().getMiddleName();
85  		lastName = person.getCurrentName().getLastName();
86  		Address mailingAddress = DummySubjectHelper.getMailingAddress(null);
87  		addressLine1 = mailingAddress.getLine1();
88  		addressLine2 = mailingAddress.getLine2();
89  		amountDue = invoice.getInvoiceTotal().toString();
90  		amountReceivedCash = invoice.getCashPayment().getAmount().toString();
91  		amountReceivedCredit = invoice.getCreditPayment().getAmount()
92  				.toString();
93  		Iterator checkPaymentsItr = invoice.getCheckPayments().iterator();
94  		BigDecimal checkTotal = new BigDecimal(0);
95  		while (checkPaymentsItr.hasNext()) {
96  			CheckPayment checkPayment = (CheckPayment) checkPaymentsItr.next();
97  			checkTotal = new BigDecimal(checkTotal.floatValue()
98  					+ checkPayment.getAmount().floatValue());
99  		}
100 		amountReceivedCheck = checkTotal.toString();
101 		DateFormat dateFormat = new SimpleDateFormat("MMddyy");
102 		DateFormat expDateFormat = new SimpleDateFormat("MM-dd-yy");
103 		workDate = dateFormat.format(userContext.getWorkDate());
104 		officeId = userContext.getOfficeId();
105 		userId = "CJ";
106 		cashierSequenceNumber = "5001";
107 		Calendar cal = Calendar.getInstance();
108 		cal.setTime(userContext.getWorkDate());
109 		cal.add(Calendar.YEAR, 1);
110 		expiryDate = expDateFormat.format(cal.getTime());
111 		receiptVarValueMap.put("#DL_NUMBER", dlNumber);
112 		receiptVarValueMap.put("#CLASS", dlClass);
113 		receiptVarValueMap.put("#AMT_DUE", amountDue);
114 		receiptVarValueMap.put("#FNAME", firstName);
115 		receiptVarValueMap.put("#MNAME", middleName);
116 		receiptVarValueMap.put("#LNAME", lastName);
117 		receiptVarValueMap.put("#CASH_AMT", amountReceivedCash);
118 		receiptVarValueMap.put("#ADDR_LINE1", addressLine1);
119 		receiptVarValueMap.put("#ADDR_LINE2", addressLine2);
120 		receiptVarValueMap.put("#CHECK_AMT", amountReceivedCheck);
121 		receiptVarValueMap.put("#CRDT_AMT", amountReceivedCredit);
122 		receiptVarValueMap.put("#CARD_NUMBER", "0");
123 		receiptVarValueMap.put("#EXPIRY_DATE", expiryDate);
124 		receiptVarValueMap.put("#WORK_DATE", workDate);
125 		receiptVarValueMap.put("#OFFICE_ID", officeId);
126 		receiptVarValueMap.put("#USER_ID", userId);
127 		receiptVarValueMap.put("#CASHIER_SEQ", cashierSequenceNumber);
128 	}
129 }

22  /***
23   * Description: This activity calculates fee related to DL.

32  public class CalculateDlFeeActivity extends BusinessActivity {
33  	/*** The Constant serialVersionUID. */
34  	private static final long serialVersionUID = -2410475715440578643L;
35  	/*** The IssueDlTaskService. */
36  	private IssueDlTaskService issueDlTaskService = null;
37  
38  	/***
39  	 * This method creates CalculateDlFeeRequestTs request class with required
40  	 * business objects from the process context using factory and calls DL Task
41  	 * Service to calculate the DL fee. Once the response is received from the
42  	 * Task Service it maps the business objects to the process context.
43  	 * 
44  	 */
45  	@Override
46  	protected void invokeService(ProcessContext processContext) {
47  		DlFeeProcessContext dlFeeProcessContext = (DlFeeProcessContext) processContext;
48  		CalculateDlFeeResponseTs response = null;
49  		/***
50  		 * 1. Check the TTC for calculating fee
51  		 * 2. Based on the TTC extract the domain objects from process context
52  		 * 3. Create Task Service request and call Task Service
53  		 * 4. Set the fee to invoice in process context
54  		 */
55  		//1. Check the TTC for calculating fee
56  		String strTtc = dlFeeProcessContext.getUserContext().getTtc();
57  		DlApplication dlApplication = dlFeeProcessContext.getDlApplication();
58  		DlTaskServiceRequestFactory dlRequestFactory = new DlTaskServiceRequestFactory();
59  		// 3. Create Task Service request and call Task Service
60  		CalculateDlFeeRequestTs request = dlRequestFactory
61  				.createAssessDlFeeRequest(dlApplication, processContext
62  						.getUserContext());
63  		response = issueDlTaskService.execute(request);
64  		// Check the response for errors
65  		if (response.getErrorCollector().hasErrors()) {
66  			populateErrorMessages(processContext, response.getErrorCollector());
67  		}
68  		// 4. Set the fee to invoice in process context 
69  		else {
70  			Fee fee = response.getFee();
71  			dlFeeProcessContext.getInvoice().getInvoiceItems().clear();
72  			dlFeeProcessContext.getInvoice().addFee(fee);
73  		}
74  		/***
75  		 * FIXME: Temporary fix for all non DLA related ttcs.	
76  		 */
77  		if (!ITtcConstants.DLA.equalsIgnoreCase(strTtc)) {
78  			Fee fee = new Fee();
79  			fee.setAmount(new BigDecimal("28.00"));
80  			response = new CalculateDlFeeResponseTs(fee);
81  			dlFeeProcessContext.getInvoice().getInvoiceItems().clear();
82  			dlFeeProcessContext.getInvoice().addFee(fee);
83  			return;
84  		}
85  	}
86  
87  	/***
88  	 * @param issueDlTaskService the issueDlTaskService to set
89  	 */
90  	public void setIssueDlTaskService(IssueDlTaskService issueDlTaskService) {
91  		this.issueDlTaskService = issueDlTaskService;
92  	}
93  }

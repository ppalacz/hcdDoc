46  	 * @param month

87  	 * Gets the business day.
88  	 * 
89  	 * @param date the date
90  	 * @return the business day
91  	 * @throws Exception the exception
92  	 */
93  	public static Date getBusinessDay(Date date) throws Exception {
94  		DateFormat dateFormat = new SimpleDateFormat("EEEE");
95  		String dayNames[] = new DateFormatSymbols().getWeekdays();
96  		String day = dateFormat.format(date);
97  		if (dayNames[7].equals(day)) {
98  			return addToSpecifiedDate(date, 0, 0, 2);
99  		}
100 		if (dayNames[1].equals(day)) {
101 			return addToSpecifiedDate(date, 0, 0, 1);
102 		}
103 		return date;
104 	}

175 	/***
176 	 * This method finds if the two dates is the same or not.
177 	 * 
178 	 * @param compareDate the compare date
179 	 * @param compareToDate the compare to date
180 	 * @return true if compareDate and compareToDate are the same date.
181 	 */
182 	public static final boolean isSameDate(Date compareDate, Date compareToDate) {
183 		boolean isSameDate = false;
184 		if (compareDate == null && compareToDate == null) {
185 			isSameDate = true;
186 		}
187 		else if (compareDate != null && compareToDate != null) {
188 			DateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
189 			String compareString = dateFormat.format(compareDate);
190 			String compareToString = dateFormat.format(compareToDate);
191 			if (compareString.equals(compareToString)) {
192 				isSameDate = true;
193 			}
194 		}
195 		return isSameDate;
196 	}

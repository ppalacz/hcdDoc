24  public class IsEmptyResponseDecision extends DecisionActivity implements
25  		IDlaDecisionActivityConstantsForBpDefinition {
26  	/*** The Constant serialVersionUID. */
27  	private static final long serialVersionUID = -5816251075994835596L;
28  
29  	/***
30  	 * This method evaluates if response received is empty or not. 
31  	 * If the response is empty it returns true, if not returns false.
32  	 */
33  	@Override
34  	protected String evaluate(ProcessContext context) {
35  		HqInquiryProcessContext processContext = (HqInquiryProcessContext) context;
36  		if (processContext.getEliRecord() != null
37  				&& processContext.getEliRecord().getHitCounter() != null) {
38  			int numberOfHits = processContext.getEliRecord().getHitCounter();
39  			if (numberOfHits == 1) {
40  				//Received single response to ANI inquiry.
41  				return SINGLE_RESPOBSE;
42  			}
43  		}
44  		if (processContext.getAniRecords() != null
45  				&& processContext.getAniRecords().size() == 0) {
46  			return EMPTY_RESPONSE;
47  		}
48  		return MULTIPLE_RESPONSE;
49  	}
50  }

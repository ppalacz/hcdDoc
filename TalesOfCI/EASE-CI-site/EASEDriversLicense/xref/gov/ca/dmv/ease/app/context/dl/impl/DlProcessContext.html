170 	/***
171 	 * Empty constructor.
172 	 */
173 	public DlProcessContext() {
174 		this(new DlApplication());
175 		dlApplication.setNewLicenseNumberRequired(false);
176 		Person person = new Person();
177 		person.setCurrentName(new PersonName());
178 		List <PersonName> personNames = new ArrayList <PersonName>();
179 		personNames.add(new PersonName());
180 		person.setPersonNames(personNames);
181 		person.setDriverLicense(new DriverLicense());
182 		person.getDriverLicense().setAppliedLicenseClasses(
183 				new ArrayList <CodeSetElement>());
184 		dlApplication.setApplicant(person);
185 		SocialSecurityDocument ssnDocument = new SocialSecurityDocument();
186 		ssnDocument.setSsnServiceVerificationCode(new CodeSetElement());
187 		LegalPresenceDocument legalPresenceDocument = new LegalPresenceDocument();
188 		legalPresenceDocument.setVerificationCode(new CodeSetElement());
189 		List <Document> list = new ArrayList <Document>();
190 		list.add(ssnDocument);
191 		list.add(legalPresenceDocument);
192 		person.setDocuments(list);
193 		Address mailingAddress = new Address(AddressType.MAILING);
194 		Address residenceAddress = new Address(AddressType.RESIDENTIAL);
195 		person.addAddress(mailingAddress);
196 		person.addAddress(residenceAddress);
197 		person.setIdCard(new IdCard());
198 		Invoice invoice = new Invoice();
199 		List <InvoiceItem> items = new ArrayList <InvoiceItem>();
200 		invoice.setInvoiceItems(items);
201 		List <Payment> payments = new ArrayList <Payment>();
202 		invoice.setPayments(payments);
203 		dlApplication.setInvoice(invoice);
204 	}

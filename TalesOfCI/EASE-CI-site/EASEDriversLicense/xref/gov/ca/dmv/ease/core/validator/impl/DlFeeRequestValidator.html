34  public class DlFeeRequestValidator extends AbstractBusValidator {
35  	/*** The Constant serialVersionUID. */
36  	private static final long serialVersionUID = 6966979131462259652L;
37  
38  	/***
39  	 * This method validates DlApplication <code>DlApplication</code>.
40  	 * 
41  	 * @param validatables The DlApplication to validate
42  	 * @return <code>IErrorCollector</code> containing Validation Messages.
43  	 * @see IErrorCollector
44  	 * @see DlApplication
45  	 */
46  	@Override
47  	public void validateValidatable(IValidatable validatable,
48  			IErrorCollector collector) {
49  		try {
50  			if (validatable instanceof DlApplication) {
51  				DlApplication dlApplication = (DlApplication) validatable;
52  				//Check if License Class is Null
53  				if (isNullOrBlank(dlApplication.getLicense())
54  						|| isNullOrBlank(dlApplication.getLicense()
55  								.getAppliedLicenseClasses())) {
56  					collector
57  							.register(new EaseValidationException(
58  									SpringUtils
59  											.getMessage(DL_APPLICATION_LICENSE_CLASS_IS_NULL_ERROR_MESSAGE)));
60  				}
61  				else {
62  					List <CodeSetElement> classtypes = dlApplication
63  							.getLicense().getAppliedLicenseClasses();
64  					if (!isNullOrBlank(classtypes)) {
65  						for (CodeSetElement classCode : classtypes) {
66  							if (isNullOrBlank(classCode)) {
67  								collector
68  										.register(new EaseValidationException(
69  												SpringUtils
70  														.getMessage(DL_APPLICATION_LICENSE_CLASS_NAME_IS_NULL_ERROR_MESSAGE)));
71  							}
72  						}
73  					}
74  					else {
75  						collector
76  								.register(new EaseValidationException(
77  										SpringUtils
78  												.getMessage(DL_APPLICATION_LICENSE_CLASS_CODE_IS_NULL_ERROR_MESSAGE)));
79  					}
80  				}
81  			}
82  		}
83  		catch (Exception e) {
84  			collector.register(e);
85  		}
86  	}
87  }

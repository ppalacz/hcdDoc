61  	 * Return true, if waiving for drive testResults.
62  	 * 
63  	 * @param driveAndLawTestResults
64  	 * @return a boolean
65  	 */
66  	public boolean isTestResultWaived(
67  			List <DriveAndLawTestResult> driveAndLawTestResults) {
68  		boolean isWaived = false;
69  		String driveTestResultCode = null;
70  		for (DriveAndLawTestResult driveAndLawTestResult : driveAndLawTestResults) {
71  			driveTestResultCode = driveAndLawTestResult.getTestResultCode()
72  					.getCode();
73  			if (IDrivingTestResultsConstants.WAIVED
74  					.equalsIgnoreCase(driveTestResultCode)) {
75  				isWaived = true;
76  				return isWaived;
77  			}
78  		}
79  		return isWaived;
80  	}

364 	/***
365 	 * Return the age of a person.
366 	 * 
367 	 * @param birthDate
368 	 * @return current age.
369 	 */
370 	private int calculateAge(Date birthDate) {
371 		int age = 0;
372 		if (birthDate != null) {
373 			Calendar currentDate = Calendar.getInstance();
374 			Calendar dob = Calendar.getInstance();
375 			dob.setTime(birthDate);
376 			int currentYear = currentDate.get(Calendar.YEAR);
377 			int dobYear = dob.get(Calendar.YEAR);
378 			age = currentYear - dobYear;
379 			int currentMonth = currentDate.get(Calendar.MONTH);
380 			int dobMonth = dob.get(Calendar.MONTH);
381 			if (dobMonth > currentMonth) {
382 				age--;
383 			}
384 			else if (currentMonth == dobMonth) {
385 				int currentDay = currentDate.get(Calendar.DAY_OF_MONTH);
386 				int dobDay = dob.get(Calendar.DAY_OF_MONTH);
387 				if (dobDay > currentDay) {
388 					age--;
389 				}
390 			}
391 		}
392 		return age;
393 	}

139 	 * Adds the test results to elements list.
140 	 * 
141 	 * @param elementsList the elements list
142 	 * @param driveAndLawTestResult the drive and law test result
143 	 */
144 	private void addTestResultsToElementsList(List<String> elementsList,
145 			List <CommercialTestResult> driveAndLawTestResults, String resultLabel) {
146 		int count = 0;
147 		if(!EaseUtil.isNullOrBlank(driveAndLawTestResults)) {
148 			for(CommercialTestResult driveAndLawTestResult : driveAndLawTestResults) {
149 				count = count + 1;
150 				addToElementsList(elementsList, getCodeText(
151 						driveAndLawTestResult.getTestResultCode()), 
152 						resultLabel + count); //Add Test Result Code.
153 				if (count == 3) {
154 					break;
155 				}
156 			}
157 			while(count < 3) {
158 					addToElementsList(elementsList, BLANK, resultLabel + count);
159 					count = count + 1;
160 			}
161 		}
162 		else {
163 			for (count = 1; count <= 3; count++) {
164 				addToElementsList(elementsList, BLANK, resultLabel + count);
165 			}
166 		}
167 	}

170 	/***
171 	 * Adds the drive test results to elements list.
172 	 * 
173 	 * @param elementsList the elements list
174 	 * @param driveAndLawTestResults the drive and law test results
175 	 * @param classLabel the class label
176 	 * @param resultLabel the result label
177 	 * @param techIdLabel the tech id label
178 	 * @param failureDateLabel the failure date label
179 	 */
180 	private void addDriveTestResultsToElementsList(List<String> elementsList,
181 			List <CommercialTestResult> driveAndLawTestResults, String classLabel, 
182 			String resultLabel, String techIdLabel, String failureDateLabel) {
183 		int count = 0;
184 		boolean failureDateAdded = false;
185 		if(!EaseUtil.isNullOrBlank(driveAndLawTestResults)) {
186 			for(CommercialTestResult driveAndLawTestResult : driveAndLawTestResults) {
187 				count = count + 1;
188 				addToElementsList(elementsList, getCodeText(
189 						driveAndLawTestResult.getLicenseClassCode()), 
190 						classLabel + count); //Add License Class Code.
191 				addToElementsList(elementsList, getCodeText(
192 						driveAndLawTestResult.getTestResultCode()), 
193 						resultLabel + count); //Add Test Result Code.
194 				addToElementsList(elementsList, driveAndLawTestResult.getExaminerID(), 
195 						techIdLabel + count); //Add Examiner Id.
196 				if(!failureDateAdded && EaseUtil.isNotNull(
197 						driveAndLawTestResult.getTestFailureDate())) {
198 					addToElementsList(elementsList, convertDateToString(
199 							driveAndLawTestResult.getTestFailureDate(), 
200 									FORMATTER_MM_dd), failureDateLabel); //Add Failure Date.
201 					failureDateAdded = true;
202 				}
203 				if (count == 3) {
204 					break;
205 				}
206 			}
207 			while (count < 3) {
208 				addToElementsList(elementsList, BLANK, 
209 						classLabel + count); //Add License Class Code.
210 				addToElementsList(elementsList, BLANK, 
211 						resultLabel + count); //Add Test Result Code.
212 				addToElementsList(elementsList, BLANK, 
213 						techIdLabel + count); //Add Examiner Id.
214 				if(!failureDateAdded ) {
215 					addToElementsList(elementsList, BLANK, failureDateLabel); //Add Failure Date.
216 					failureDateAdded = true;
217 				}
218 				count = count + 1;
219 			}
220 		}
221 		else {
222 			for (count = 1; count <= 3; count++) {
223 				addToElementsList(elementsList, BLANK, 
224 						classLabel + count); //Add License Class Code.
225 				addToElementsList(elementsList, BLANK, 
226 						resultLabel + count); //Add Test Result Code.
227 				addToElementsList(elementsList, BLANK, 
228 						techIdLabel + count); //Add Examiner Id.
229 			}
230 			addToElementsList(elementsList, BLANK, failureDateLabel); //Add Failure Date.
231 		}
232 	}

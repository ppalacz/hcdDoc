40  	/***
41  	 * This method validates the DlCdlisReceiptRequest request for its
42  	 * DlApplication, Invoice objects.
43  	 * 
44  	 * @param dlAppln
45  	 * @param invoice
46  	 * 
47  	 * @return ErrorCollector
48  	 */
49  	public IErrorCollector validate(DriverLicense driverLicense, Invoice invoice) {
50  		IErrorCollector collector = new ErrorCollector();
51  		try {
52  			if (EaseUtil.isNullOrBlank(driverLicense)) {
53  				collector.register(new EaseValidationException(SpringUtils
54  						.getMessage(INVALID_DRIVER_LICENSE_ERROR_MESSAGE)));
55  			}
56  			else if (EaseUtil.isNullOrBlank(driverLicense
57  					.getAppliedLicenseClasses())) {
58  				collector.register(new EaseValidationException(SpringUtils
59  						.getMessage(INVALID_DRIVER_LICENSE_ERROR_MESSAGE)));
60  			}
61  			if (EaseUtil.isNullOrBlank(invoice)) {
62  				collector.register(new EaseValidationException(SpringUtils
63  						.getMessage(INVALID_INVOICE_ERROR_MESSAGE)));
64  			}
65  		}
66  		catch (Exception e) {
67  			collector.register(e);
68  		}
69  		return collector;
70  	}

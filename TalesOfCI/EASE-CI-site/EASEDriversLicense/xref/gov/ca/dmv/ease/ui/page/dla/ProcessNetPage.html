478 		List <Voucher> listVouchers = null;
479 		if (getDlFeeProcessContext().getInvoice() != null) {
480 			listVouchers = getDlFeeProcessContext().getInvoice()
481 					.getVoucherPayments();
482 		}
483 		if (listVouchers == null) {
484 			listVouchers = new ArrayList <Voucher>();
485 		}
486 		for (int index = listVouchers.size(); index <= 5; index++) {
487 			listVouchers.add(new Voucher());
488 		}

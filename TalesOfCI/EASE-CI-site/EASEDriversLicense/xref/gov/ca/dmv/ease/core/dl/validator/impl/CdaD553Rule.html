37  	public void validate(ProcessContext processContext) {
38  		CdaProcessContext cdaProcessContext = (CdaProcessContext) processContext;
39  		IErrorCollector errorCollector = new ErrorCollector();
40  		List <CommercialTestResult> commercialPreTripTestResults = cdaProcessContext
41  				.getDriverLicense().getCommercialTestResultGroup()
42  				.getCommercialTestResults(
43  						CommercialTestResultType.PRETRIP_TEST_RESULT);
44  		if (isPreTripClassA(commercialPreTripTestResults)
45  				&& !isAppClassA(cdaProcessContext.getDriverLicense())) {
46  			errorCollector.register(new ApplicationException("D553"));
47  		}
48  		if (isPreTripClassB(commercialPreTripTestResults)
49  				&& !isAppClassA(cdaProcessContext.getDriverLicense())
50  				&& !isAppClassB(cdaProcessContext.getDriverLicense())) {
51  			errorCollector.register(new ApplicationException("D553"));
52  		}
53  	}

55  	/***
56  	 * Check if the pre trip class is A.
57  	 * 
58  	 * @param commercialPreTripTestResults
59  	 * @return boolean
60  	 */
61  	private boolean isPreTripClassA(
62  			List <CommercialTestResult> commercialPreTripTestResults) {
63  		boolean isPreTripClassA = false;
64  		if (commercialPreTripTestResults != null
65  				&& commercialPreTripTestResults.size() > 0) {
66  			String testClassCode = commercialPreTripTestResults.get(
67  					commercialPreTripTestResults.size() - 1)
68  					.getLicenseClassCode().getCode();
69  			if (IDlClassTypeConstants.COMMERCIAL_APP_CLASS_A
70  					.equalsIgnoreCase(testClassCode)) {
71  				isPreTripClassA = true;
72  				return isPreTripClassA;
73  			}
74  		}
75  		return isPreTripClassA;
76  	}

44  	 * Populates outgoing XML message for Non Commercial Test Results Document
45  	 * 
46  	 * @param requestFromClient the request from client
47  	 * 
48  	 * @return the string
49  	 */
50  	public String convertToDocumentBinding(PrintServiceRequest requestFromClient) {
51  		IUserContext userContext = requestFromClient.getUserContext();
52  		NonCommercialTestResultsDocumentRequestTus request = (NonCommercialTestResultsDocumentRequestTus) requestFromClient;
53  		DlApplication dlApplication = request.getDlApplication();
54  		Person person = dlApplication.getApplicant();
55  		DriverLicense driverLicense = person.getDriverLicense();
56  		List <CodeSetElement> newAttachments = driverLicense
57  				.getNewAttachments();
58  		DriveAndLawTestResultGroup resultsGroup = driverLicense
59  				.getDriveAndLawTestResultGroup();
60  		List <String> elementsList = super.getCommonElementsList(dlApplication,
61  				userContext, person); //Set all common elements of test results document.
62  		addToElementsList(elementsList, getFeeDueYear(driverLicense),
63  				IPrintConverterConstants.NONCOMM_TEST_RESULTS_FEE_DUE); //FEE DUE YEAR
64  		addPhotoTakenIndToElementsList(elementsList, driverLicense); //PHOTO TAKEN INDICATOR
65  		addAppDateToElementsList(elementsList, dlApplication); //APPLICATION DATE
66  		addOutOfStateLicToElementsList(elementsList, person); //OUT OF STATE LIC & LICENSE LOCATION
67  		addRestOrProvToElementsList(elementsList, dlApplication); // RESTRICTION/PROVISIONAL
68  		addMinorAndTrainingCodesToElementsList(elementsList, driverLicense,
69  				dlApplication); //MINOR CODE, DRIVER ED, DIRVER TRNG, MC TRNG
70  		addInstructionPermitDetailsToElementsList(elementsList, driverLicense); // IP ISSUE DATE, IP CLASS 1 & 2, IP COMM IND
71  		addTempLicenseDetailsToElementsList(elementsList, driverLicense); // TL ISSUE DATE, IP CLASS 1 & 2, TL COMM IND
72  		if (EaseUtil.isNotNull(resultsGroup)) {
73  			//Add all test results
74  			addToElementsList(
75  					elementsList,
76  					convertDateToString(resultsGroup.getLawTestFailureDate(),
77  							FORMATTER_MM_dd),
78  					IPrintConverterConstants.NONCOMM_TEST_RESULTS_LAW_TEST_FAIL_DT); //LAW TEST FAIL DT
79  			addTestResultGroupToElementsList(elementsList, resultsGroup); //LAW, DRIVE, W/P, SIGNS RESULTS
80  			addToElementsList(elementsList, getCodeText(resultsGroup
81  					.getVisionTestResultCode()),
82  					IPrintConverterConstants.NONCOMM_TEST_RESULTS_VISION); //VISON TEST
83  			addToElementsList(elementsList, getCodeText(resultsGroup
84  					.getOralTestResultCode()),
85  					IPrintConverterConstants.NONCOMM_TEST_RESULTS_ORAL); //ORAL TEST INDICATOR
86  		}
87  		addToElementsList(elementsList, getCodeText(driverLicense
88  				.getForeignLanguageCode()),
89  				IPrintConverterConstants.NONCOMM_TEST_RESULTS_FOREIGN); //FOREIGN LANG TEST
90  		if (EaseUtil.isNotNull(driverLicense.getLimitedLicenseTermYears())) {
91  			addToElementsList(elementsList, driverLicense
92  					.getLimitedLicenseTermYears().toString(),
93  					IPrintConverterConstants.NONCOMM_TEST_RESULTS_LT_YRS); //# LIMITED TERM YEARS
94  		}
95  		addNewAttachmentsToElementList(elementsList, newAttachments); //APP ATTCHMENTS
96  		addToElementsList(elementsList, convertDateToString(person
97  				.getMedicalExpirationDate(), FORMATTER_MM_dd_yy),
98  				IPrintConverterConstants.NONCOMM_TEST_RESULTS_MED_EXP_DT); //MEDICAL EXAM EXP DT
99  		return createXmlForSingleDocument(elementsList,
100 				IPrintConverterConstants.NONCOMM_TEST_RESULTS_DOC_NAME);
101 	}

104 	 * Adds the photo taken ind to element list.
105 	 * 
106 	 * @param elementsList the elements list
107 	 * @param driverLicense the driver license
108 	 */
109 	private void addPhotoTakenIndToElementsList(List <String> elementsList,
110 			DriverLicense driverLicense) {
111 		String requiredIndText = BLANK;
112 		if (EaseUtil.isNotNull(driverLicense)) {
113 			if (EaseUtil.isNotNull(driverLicense.getPhotoInformation())
114 					&& !EaseUtil.isNullOrBlank(driverLicense
115 							.getPhotoInformation().getPhotoTakenDate())) {
116 				requiredIndText = BLANK; //Photo is taken as photo taken date is present.
117 			}
118 			else {
119 				requiredIndText = IPrintConverterConstants.PHOTO_NOT_TAKEN_INDICATOR;
120 			}
121 		}
122 		else {
123 			requiredIndText = IPrintConverterConstants.PHOTO_NOT_TAKEN_INDICATOR;
124 		}
125 		addToElementsList(elementsList, requiredIndText,
126 				IPrintConverterConstants.TEST_RESULTS_PHOTO_TAKEN_IND); //PHOTO TAKEN INDICATOR
127 	}

159 	/***
160 	 * Adds the out of state lic to elements list.
161 	 * 
162 	 * @param elementsList the elements list
163 	 * @param person the person
164 	 */
165 	private void addOutOfStateLicToElementsList(List <String> elementsList,
166 			Person person) {
167 		String licText = BLANK;
168 		String licLoc = BLANK;
169 		if (!EaseUtil.isNullOrBlank(person.getPreviousLicenses())) {
170 			OutOfStateLicense osLicense = person.getPreviousLicenses().get(0);
171 			if (EaseUtil.isNotNull(osLicense.getIssuedByState())) {
172 				//Out of state license state code or country name
173 				if (osLicense.getIssuedByState().getCode().equalsIgnoreCase(
174 						IPrintConverterConstants.STATE_CODE_OT)) {
175 					//If State Code is OTHER then use country name abbrv. 
176 					licText = osLicense.getIssuedByCountry();
177 				}
178 				else {
179 					licText = osLicense.getIssuedByState().getCode();
180 				}
181 			}
182 			if (EaseUtil.isNotNull(osLicense.getLicenseLocation())) {
183 				licLoc = osLicense.getLicenseLocation().getCode(); //licesne loc code
184 			}
185 		}
186 		addToElementsList(elementsList, licText,
187 				IPrintConverterConstants.NONCOMM_TEST_RESULTS_OUT_OF_STATE_LIC);
188 		addToElementsList(elementsList, licLoc,
189 				IPrintConverterConstants.NONCOMM_TEST_RESULTS_LIC_LOC);
190 	}

282 	/***
283 	 * Adds the ip or temp details to elements list.
284 	 * 
285 	 * @param elementsList the elements list
286 	 * @param permits the permits
287 	 * @param issueDateLabel the issue date label
288 	 * @param class1Lable the class1 lable
289 	 * @param class2Label the class2 label
290 	 * @param commIndLabel the comm ind label
291 	 */
292 	private void addIpOrTempDetailsToElementsList(List <String> elementsList,
293 			List <InterimPermit> permits, String issueDateLabel,
294 			String class1Lable, String class2Label, String commIndLabel) {
295 		String issueDate = BLANK;
296 		String class1 = BLANK;
297 		String class2 = BLANK;
298 		String commInd = BLANK;
299 		if (!EaseUtil.isNullOrBlank(permits)) {
300 			issueDate = convertDateToString(
301 					permits.get(0).getPermitIssueDate(), FORMATTER_MM_dd); //Issue Date
302 			class1 = permits.get(0).getPermitLicenseClassCode().getCode(); // Class 1
303 			if (permits.size() == 2) {
304 				class2 = permits.get(1).getPermitLicenseClassCode().getCode(); // Class 2
305 			}
306 			if (ConverterHelper.isComercialLicenseClass(class1)
307 					|| ConverterHelper.isComercialLicenseClass(class2)) {
308 				commInd = IPrintConverterConstants.COMMERCIAL_INDICATOR; // Commercial Indicator
309 			}
310 		}
311 		addToElementsList(elementsList, issueDate, issueDateLabel);
312 		addToElementsList(elementsList, class1, class1Lable);
313 		addToElementsList(elementsList, class2, class2Label);
314 		addToElementsList(elementsList, commInd, commIndLabel);
315 	}

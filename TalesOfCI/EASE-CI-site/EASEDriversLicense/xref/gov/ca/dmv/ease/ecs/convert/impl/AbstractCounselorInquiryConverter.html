38  public abstract class AbstractCounselorInquiryConverter extends
39  		AbstractEcsMessageConverter {
40  	/*** The converter TCode. */
41  	protected String tCode = BLANK;
42  
43  	/* (non-Javadoc)
44  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createMessage(gov.ca.dmv.ease.ecs.request.IEcsRequest)
45  	 */
46  	@Override
47  	public String createMessage(IEcsRequest requestFromClient) {
48  		CounselorInquiryRequestEcs request = (CounselorInquiryRequestEcs) requestFromClient;
49  		IUserContext currentUserContext = request.getUserContext();
50  		String driverLicenseNumber = request.getDriverLicenseNumber();
51  		Object[] message = new Object[] { tCode, currentUserContext.getTtc(),
52  				currentUserContext.getOfficeId(),
53  				currentUserContext.getTechId(), driverLicenseNumber,
54  				getDataLengthField() };
55  		return createMessageFormat(message);
56  	}
57  
58  	/* (non-Javadoc)
59  	 * @see gov.ca.dmv.ease.ecs.convert.impl.AbstractEcsMessageConverter#createResponse(java.lang.String)
60  	 */
61  	@Override
62  	public IEcsResponse createResponseFromMultiplePayloads(
63  			List <String> inboundMessagePayloads) {
64  		StringBuilder strBuff = new StringBuilder();
65  		CounselorInquiryResponseEcs counselorInquiryResponseEcs = new CounselorInquiryResponseEcs();
66  		if (EaseUtil.isNullOrBlank(inboundMessagePayloads)) {
67  			throw new MessageConversionException(
68  					"Invalid Response: Unable to process null message");
69  		}
70  		for (String anInboundMessagePayload : inboundMessagePayloads) {
71  			//If TCode is DLD - remove 3270 character sets from the return data. 
72  			if (tCode.equalsIgnoreCase(IEcsConverterConstants.TCODE_DLD)) {
73  				String formattedScreen = null;
74  				try {
75  					String ebcidic = new String(anInboundMessagePayload
76  							.getBytes("Cp1047"));
77  					String screen = Decoder3270.decode(ebcidic);
78  					formattedScreen = Decoder3270.formatScreen(screen);
79  					strBuff.append(formattedScreen).append(LINE_SEPARATOR);
80  				}
81  				catch (UnsupportedEncodingException e) {
82  					throw new MessageConversionException(e);
83  				}
84  				String formatResponse = StringUtils
85  						.trimTrailingWhitespace(removeDuplicateHeader(strBuff));
86  				counselorInquiryResponseEcs
87  						.setInquiryReturnData(formatResponse);
88  			}
89  			//If TCode is DLC - send the return data as it is for printing.
90  			else {
91  				//Fetching Counselor Inquiry Response Code from response string
92  				String returnCode = anInboundMessagePayload.substring(
93  						IEcsConverterConstants.int_22,
94  						IEcsConverterConstants.int_24);
95  				//Fetching Counselor Inquiry Return Data from response string
96  				String returnData = null;
97  				if (anInboundMessagePayload.length() >= IEcsConverterConstants.int_25) {
98  					returnData = anInboundMessagePayload
99  							.substring(IEcsConverterConstants.int_24);
100 				}
101 				CodeSetElement codeSetElement = new CodeSetElement();
102 				//Fetch code set element from database for return code from back end. 
103 				codeSetElement = getCodeSetElement(
104 						IEcsConverterConstants.DLC_DLD_RETURN_CODE, returnCode);
105 				//If code set element not found in database then set to long term cna return code
106 				if (EaseUtil.isNullOrBlank(codeSetElement)) {
107 					codeSetElement = getCodeSetElement(
108 							IEcsConverterConstants.DLC_DLD_RETURN_CODE,
109 							IEcsConverterConstants.LONG_TERM_CNA_RETURN_CODE);
110 				}
111 				counselorInquiryResponseEcs
112 						.setInquiryResponseCode(codeSetElement);
113 				counselorInquiryResponseEcs.setInquiryReturnData(returnData);
114 			}
115 		}
116 		return counselorInquiryResponseEcs;
117 	}
118 
119 	private String getDataLengthField() {
120 		if (tCode.equalsIgnoreCase(IEcsConverterConstants.TCODE_DFT)) {
121 			return filler(BLANK, 3);
122 		}
123 		else {
124 			return IEcsConverterConstants.DATA_LENGTH_FIELD;
125 		}
126 	}

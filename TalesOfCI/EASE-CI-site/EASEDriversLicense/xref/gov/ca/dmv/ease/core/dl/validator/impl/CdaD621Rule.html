35  public class CdaD621Rule extends AbstractValidateRule {
36  	/* (non-Javadoc)
37  	 * @see gov.ca.dmv.ease.validator.IValidationRule#validate(gov.ca.dmv.ease.app.context.impl.ProcessContext)
38  	 */
39  	public void validate(ProcessContext processContext) {
40  		CdaProcessContext cdaProcessContext = (CdaProcessContext) processContext;
41  		IErrorCollector errorCollector = new ErrorCollector();
42  		Action selectedAction = cdaProcessContext.getSelectedAction();
43  		if (selectedAction != null
44  				&& selectedAction.equals(getActionsRegistry().getAction(
45  						IActionsRegistry.ENTER_ACTION))) {
46  			validate(cdaProcessContext.getDriverLicense(), cdaProcessContext.getDriveAndLawTestResultGroup(), errorCollector);
47  		}
48  		setErrorCollector(errorCollector);
49  	}
50  	
51  	/***
52  	 * 
53  	 * @param driverLicense
54  	 * @param driveAndLawTestResultGroup
55  	 * @param errorCollector
56  	 */
57  	public void validate(DriverLicense driverLicense,
58  			DriveAndLawTestResultGroup driveAndLawTestResultGroup,
59  			IErrorCollector errorCollector) {
60  		List<DrivingRestriction> existingRestrictions = driverLicense
61  				.getExistingRestrictions();
62  		List<DrivingRestriction> newRestrictions = driverLicense
63  				.getNewRestrictions();
64  		List<DrivingRestriction> restrictions = new ArrayList<DrivingRestriction>();
65  		if (newRestrictions != null)
66  			restrictions.addAll(newRestrictions);
67  		if (existingRestrictions != null)
68  			restrictions.addAll(existingRestrictions);
69  		if (restrictions != null && restrictions.size() > 0) {
70  			for (Iterator<DrivingRestriction> iterator = restrictions
71  					.iterator(); iterator.hasNext();) {
72  				String visionTestResultCode = null;
73  				if (driveAndLawTestResultGroup != null
74  						&& driveAndLawTestResultGroup.getVisionTestResultCode() != null) {
75  					visionTestResultCode = driveAndLawTestResultGroup
76  							.getVisionTestResultCode().getCode();
77  				}
78  				DrivingRestriction drivingRestriction = iterator.next();
79  				String restrictionCode = null;
80  				if (drivingRestriction.getRestrictionTypeCode().getCode() != null) {
81  					restrictionCode = drivingRestriction
82  							.getRestrictionTypeCode().getCode();
83  				}
84  				if ("01".equalsIgnoreCase(restrictionCode)
85  						&& ("P".equalsIgnoreCase(visionTestResultCode) || "S"
86  								.equalsIgnoreCase(visionTestResultCode))) {
87  					errorCollector.register(new ApplicationException("D621"));
88  				}
89  			}
90  		}
91  	}

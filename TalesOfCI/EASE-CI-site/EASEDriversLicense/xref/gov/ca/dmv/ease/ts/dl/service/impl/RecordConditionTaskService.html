28  public class RecordConditionTaskService extends TaskService {
29  	/***
30  	 * 
31  	 */
32  	private static final long serialVersionUID = -4100131484761901255L;
33  	private RecordConditionService recordConditionService;
34  
35  	public RecordConditionResponseTs execute(RecordConditionRequestTs tsRequest) {
36  		RecordConditionResponseTs tsResponse = null;
37  		RecordConditionRequestBus busRequest = null;
38  		RecordConditionResponseBus busResponse = null;
39  		IErrorCollector collector = new ErrorCollector();
40  		try {
41  			//collector = tsRequest.validate();
42  			if (collector.hasErrors()) {
43  				tsResponse = new RecordConditionResponseTs(collector);
44  			}
45  			else {
46  				busRequest = new RecordConditionRequestBus(tsRequest
47  						.getUserContext(), tsRequest.getTcode(), tsRequest
48  						.getProcessContext());
49  				busResponse = recordConditionService.execute(busRequest);
50  				if (busResponse.hasErrors()) {
51  					tsResponse = new RecordConditionResponseTs(busResponse
52  							.getErrorCollector());
53  				}
54  				else {
55  					tsResponse = new RecordConditionResponseTs(busResponse
56  							.getRecordConditions());
57  				}
58  			}
59  		}
60  		catch (Exception e) {
61  			// Register exception in error collector and assign it to response
62  			// object.
63  			collector.register(e);
64  			tsResponse = new RecordConditionResponseTs(collector);
65  		}
66  		return tsResponse;
67  	}
68  
69  	/***
70  	 * @param recordConditionService the recordConditionService to set
71  	 */
72  	public void setRecordConditionService(
73  			RecordConditionService recordConditionService) {
74  		this.recordConditionService = recordConditionService;
75  	}
76  }

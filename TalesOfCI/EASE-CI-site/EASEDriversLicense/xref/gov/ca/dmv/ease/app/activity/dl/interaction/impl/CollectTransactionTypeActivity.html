31  public class CollectTransactionTypeActivity extends InteractionActivity {
32  	/*** The Constant serialVersionUID. */
33  	private static final long serialVersionUID = 2080866235849675160L;
34  
35  	/* (non-Javadoc)
36  	 * @see gov.ca.dmv.ease.app.activity.InteractionActivity#enter(gov.ca.dmv.ease.app.context.ProcessContext)
37  	 */
38  	@Override
39  	public void executeAuthorized(ProcessContext processContext) {
40  		//FIXME - this method must not be public - it should not increase visibility from the superclass
41  		IErrorCollector errorCollector = new ErrorCollector();
42  		String menuSelection = processContext.getUserContext().getTtc();
43  		List <String> userRoles = processContext.getUserContext().getRoles();
44  		BusinessProcess businessProcess = getProcessRegistry().getProcess(
45  				menuSelection);
46  		if (businessProcess == null) {
47  			errorCollector.register(new ApplicationException("EASE002"));
48  			this.populateErrorMessages(processContext, errorCollector);
49  			return;
50  		}
51  		boolean isAuthorizedUser = false;
52  		List <String> securityRoles = businessProcess.getSecurityRoles();
53  		if ((securityRoles != null) && (userRoles != null)) {
54  			Iterator <String> iter = securityRoles.iterator();
55  			while (iter.hasNext()) {
56  				if (userRoles.contains(iter.next().trim())) {
57  					isAuthorizedUser = true;
58  					break;
59  				}
60  			}
61  		}
62  		if (!isAuthorizedUser) {
63  			errorCollector.register(new ApplicationException("EASE001"));
64  			this.populateErrorMessages(processContext, errorCollector);
65  			return;
66  		}
67  		ChildContext childContext = (ChildContext) businessProcess
68  				.getProcessContext();
69  		childContext.startIn(processContext, true);
70  	}
71  }

71  					&& commercialPreTripTestResults.size() > 0) {
72  				if (EaseUtil.isNotNull(commercialPreTripTestResults)
73  						&& commercialPreTripTestResults.size() > 0
74  						&& isStarPresent(commercialPreTripTestResults)) {
75  					errorCollector
76  							.register(new ApplicationException(
77  									IDataCollectionErrorMessageConstants.ERASE_STAR_OR_PRESS_CORRECTIONS));
78  				}
79  			}
80  			List <CommercialTestResult> commercialPerformanceTestResults = commercialTestResultGroup
81  					.getCommercialTestResults(CommercialTestResultType.PERFORMANCE_TEST_RESULT);
82  			if (EaseUtil.isNotNull(commercialPerformanceTestResults)
83  					&& commercialPerformanceTestResults.size() > 0
84  					&& isStarPresent(commercialPerformanceTestResults)) {
85  				errorCollector
86  						.register(new ApplicationException(
87  								IDataCollectionErrorMessageConstants.ERASE_STAR_OR_PRESS_CORRECTIONS));
88  			}
89  		}
90  	}

92  	/***
93  	 * @param errorCollector
94  	 * @param commercialTestResultGroup
95  	 */
96  	private void validate(CommercialTestResultGroup commercialTestResultGroup,
97  			CommercialTestResultType commercialTestResultType,
98  			IErrorCollector errorCollector) {
99  		List <CommercialTestResult> commercialGeneralTestResults = commercialTestResultGroup
100 				.getCommercialTestResults(commercialTestResultType);
101 		if (EaseUtil.isNotNull(commercialGeneralTestResults)
102 				&& commercialGeneralTestResults.size() > 0) {
103 			for (CommercialTestResult commercialGeneralTestResult : commercialGeneralTestResults) {
104 				String testResultCode = commercialGeneralTestResult
105 						.getTestResultCode().getCode();
106 				if (IGeneralTestConstants.CORRECTION
107 						.equalsIgnoreCase(testResultCode)) {
108 					errorCollector
109 							.register(new ApplicationException(
110 									IDataCollectionErrorMessageConstants.ERASE_STAR_OR_PRESS_CORRECTIONS));
111 				}
112 			}
113 		}
114 	}

116 	/***
117 	 * Return true if star present.
118 	 * 
119 	 * @param commercialTestResults
120 	 * @return  a boolean
121 	 */
122 	private boolean isStarPresent(
123 			List <CommercialTestResult> commercialTestResults) {
124 		boolean isStarPresent = false;
125 		for (CommercialTestResult commercialTestResult : commercialTestResults) {
126 			if (IGeneralTestConstants.CORRECTION
127 					.equalsIgnoreCase(commercialTestResult.getExaminerID())
128 					|| IGeneralTestConstants.CORRECTION
129 							.equalsIgnoreCase(commercialTestResult
130 									.getTestResultCode().getCode())
131 					|| IGeneralTestConstants.CORRECTION
132 							.equalsIgnoreCase(commercialTestResult
133 									.getLicenseClassCode().getCode())) {
134 				isStarPresent = true;
135 			}
136 		}
137 		return isStarPresent;
138 	}
